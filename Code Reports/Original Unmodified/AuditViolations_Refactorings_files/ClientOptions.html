58 /**
59  * Defines how available client options are displayed on the Setting
60  * dialog from File/Preferences Also contains several Comparators used
61  * for display purposes.
62  *
63  * Most available client options and their default values are defined
64  * in the file <code>base/client-options.xml</code> in the FreeCol
65  * data directory. They are overridden by the player's personal
66  * settings in the file <code>options.xml</code> in the user
67  * directory. Note that some options are generated and added dynamically.
68  *
69  * Each option should be given an unique identifier (defined as a
70  * constant in this class). In general, the options are called
71  * something like "model.option.UNIQUENAME". Since the options must
72  * also be represented by the GUI, they following two keys must be
73  * added to the file <code>FreeColMessages.properties</code>:
74  *
75  * <ul><li>model.option.UNIQUENAME.name</li>
76  * <li>model.option.UNIQUENAME.shortDescription</li></ul>
77  */

464     private class MessageSourceComparator implements Comparator<ModelMessage> {
465         private final Game game;
466
467         // sort according to message source
468
469         private MessageSourceComparator(Game game) {
470             this.game = game;
471         }
472
473         @Override
474         public int compare(ModelMessage message1, ModelMessage message2) {
475             String sourceId1 = message1.getSourceId();
476             String sourceId2 = message2.getSourceId();
477             if (Utils.equals(sourceId1, sourceId2)) {
478                 return messageTypeComparator.compare(message1, message2);
479             }
480             FreeColGameObject source1 = game.getMessageSource(message1);
481             FreeColGameObject source2 = game.getMessageSource(message2);
482             int base = getClassIndex(source1) - getClassIndex(source2);
483             if (base == 0) {
484                 if (source1 instanceof Colony) {
485                     return getColonyComparator().compare((Colony) source1, (Colony) source2);
486                 }
487             }
488             return base;
489         }
490
491         private int getClassIndex(Object object) {
492             if (object instanceof Player) {
493                 return 10;
494             } else if (object instanceof Colony) {
495                 return 20;
496             } else if (object instanceof Europe) {
497                 return 30;
498             } else if (object instanceof Unit) {
499                 return 40;
500             } else if (object instanceof FreeColGameObject) {
501                 return 50;
502             } else {
503                 return 1000;
504             }
505         }
506     }

511                     - m2.getMessageType().ordinal();

527      *
528      * @param save The <code>FreeColSaveGame</code> to read the options from.
529      * @return True if the options were loaded without error.
530      */
531     private boolean load(FreeColSavegameFile save) {
532         if (save == null) return false;
533         boolean ret = false;
534         try (
535             FreeColXMLReader xr = save.getClientOptionsFreeColXMLReader();
536         ) {
537             ret = load(xr);
538         } catch (IOException|XMLStreamException ex) {
539             logger.log(Level.WARNING, "Load OptionGroup(" + getId()
540                 + ") from file " + save.getPath() + " crashed", ex);
541             return false;
542         }
543         logger.info("Load OptionGroup(" + getId() + ") from " + save.getPath()
544             + ((ret) ? " succeeded" : " failed"));
545         return ret;
546     }
