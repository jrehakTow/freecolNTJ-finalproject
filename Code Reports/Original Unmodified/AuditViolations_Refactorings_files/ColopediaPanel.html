187     private void select(String id) {
188         DefaultMutableTreeNode node = nodeMap.get(id);
189         if (node == null) {
190             logger.warning("Unable to find node with id '" + id + "'.");
191         } else {
192             TreePath oldPath = tree.getSelectionPath();
193             if (oldPath != null && oldPath.getParentPath() != null) {
194                 tree.collapsePath(oldPath.getParentPath());
195             }
196             TreePath newPath = new TreePath(node.getPath());
197             tree.scrollPathToVisible(newPath);
198             tree.expandPath(newPath);
199             showDetails((ColopediaTreeItem) node.getUserObject());
200         }
