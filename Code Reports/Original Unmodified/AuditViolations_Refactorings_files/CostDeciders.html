185             if (cost != ILLEGAL_MOVE && cost != Map.INFINITY && tile != null) {

190                     tiles: for (Tile t : tile.getSurroundingTiles(1)) {

195                                     && u.isOffensiveUnit())) {
196                                 this.movesLeft = 0;
197                                 this.newTurns++;
198                                 break tiles;

211     /**
212      * Gets a composite cost decider composed of two or more
213      * individual cost deciders.  The result/s are determined by the
214      * cost decider which returns the highest cost, with an
215      * ILLEGAL_MOVE result dominating.
216      *
217      * @param cds A series (two minimum) of <code>CostDecider</code>s
218      *     to compose.
219      * @return A new <code>CostDecider</code> composed of the argument
220      *     cost deciders.
221      */
222     public static CostDecider getComposedCostDecider(final CostDecider... cds) {
223         if (cds.length < 2) {
224             throw new IllegalArgumentException("Short CostDecider list");
225         }
226
227         return new CostDecider() {
228
229             private final CostDecider[] costDeciders = cds;
230             private int ret = -1;
231             private int index = -1;
232
233             @Override
234             public int getCost(Unit unit, Location oldLocation,
235                                Location newLocation, int movesLeft) {
236                 for (int i = 0; i < costDeciders.length; i++) {
237                     int cost = costDeciders[i].getCost(unit, oldLocation,
238                                                        newLocation, movesLeft);
239                     if (cost == ILLEGAL_MOVE || cost == Map.INFINITY) {
240                         index = i;
241                         return ILLEGAL_MOVE;
242                     }
243                     if (cost > ret) {
244                         index = i;
245                         ret = cost;
246                     }
247                 }
248                 return ret;
249             }
250
251             @Override
252             public int getMovesLeft() {
253                 return (index < 0) ? 0 : costDeciders[index].getMovesLeft();
254             }
255
256             @Override
257             public int getNewTurns() {
258                 return (index < 0) ? 0 : costDeciders[index].getNewTurns();
259             }
260         };
