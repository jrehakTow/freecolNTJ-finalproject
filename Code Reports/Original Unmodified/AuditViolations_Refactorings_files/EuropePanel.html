787     /**
788      * Unload the contents of the currently selected carrier.
789      */
790     private void unloadAction() {
791         Unit unit = getSelectedUnit();
792         if (unit != null && unit.isCarrier()) {
793             Iterator<Goods> goodsIterator = unit.getGoodsIterator();
794             while (goodsIterator.hasNext()) {
795                 Goods goods = goodsIterator.next();
796                 if (getMyPlayer().canTrade(goods.getType())) {
797                     igc().sellGoods(goods);
798                 } else {
799                     igc().payArrears(goods.getType());
800                 }
801             }
802             Iterator<Unit> unitIterator = unit.getUnitIterator();
803             while (unitIterator.hasNext()) {
804                 Unit newUnit = unitIterator.next();
805                 igc().leaveShip(newUnit);
806             }
807             cargoPanel.update();
808             docksPanel.update();
809         }
810         requestFocus();
811     }

813     /**
814      * A unit sets sail for the new world.
815      */
816     private void sailAction() {
817         Unit unit = getSelectedUnit();
818         if (unit != null && unit.isNaval()) {
819             UnitLabel unitLabel = getSelectedUnitLabel();
820             toAmericaPanel.add(unitLabel, true);
821         }
822         requestFocus();
823     }

841     /**
842      * {@inheritDoc}
843      */
844     @Override
845     public void actionPerformed(ActionEvent ae) {
846         final String command = ae.getActionCommand();
847         EuropeAction act = EuropeAction.valueOf(command);
848         switch (act) {
849         case EXIT:
850             exitAction();
851             break;
852         case PURCHASE:
853             getGUI().showPurchasePanel();
854             break;
855         case RECRUIT:
856             getGUI().showRecruitPanel();
857             break;
858         case SAIL:
859             sailAction();
860             break;
861         case TRAIN:
862             getGUI().showTrainPanel();
863             break;
864         case UNLOAD:
865             unloadAction();
866             break;
867         default:
868             super.actionPerformed(ae);
869             break;
870         }
