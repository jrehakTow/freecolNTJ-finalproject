100     /**
101      * Gets a list of this Nation's starting units at the given
102      * difficulty.
103      *
104      * @param key The value of the expert-starting-units field.
105      * @return A list of <code>AbstractUnit</code>s to start with.
106      */
107     public List<AbstractUnit> getStartingUnits(String key) {
108         Map<String, AbstractUnit> result = new HashMap<>();
109         Map<String, AbstractUnit> defaultMap = startingUnitMap.get(null);
110         Map<String, AbstractUnit> difficultyMap = startingUnitMap.get(key);
111         if (defaultMap != null) {
112             result.putAll(defaultMap);
113         }
114         if (difficultyMap != null) {
115             result.putAll(difficultyMap);
116         }
117         return new ArrayList<>(result.values());
118     }

122      *
123      * @param id The unit identifier.
124      * @param unit The <code>AbstractUnit</code> to add.
125      * @param expert Is this an expert unit?
126      */
127     private void addStartingUnit(String id, AbstractUnit unit, boolean expert) {
128         String exTag = (expert) ? Boolean.TRUE.toString() : null;
129         Map<String, AbstractUnit> units = startingUnitMap.get(exTag);
130         if (units == null) {
131             units = new HashMap<>();
132             startingUnitMap.put(exTag, units);
133         }
134         units.put(id, unit);
135     }

181     private void writeUnit(FreeColXMLWriter xw, String id,

183                            boolean expert) throws XMLStreamException {
