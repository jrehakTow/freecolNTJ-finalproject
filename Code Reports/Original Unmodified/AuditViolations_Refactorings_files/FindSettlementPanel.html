158     private void updateSearch(DisplayListOption displayListOption) {
159         DefaultListModel<Settlement> model
160             = new DefaultListModel<>();
161         Object selected = this.settlementList.getSelectedValue();
162
163         for (Player player : getGame().getLivePlayers(null)) {
164             boolean ok;
165             switch (displayListOption) {
166             case ONLY_NATIVES:
167                 ok = player.isIndian();
168                 break;
169             case ONLY_EUROPEAN:
170                 ok = player.isEuropean();
171                 break;
172             case ALL:
173                 ok = true;
174                 break;
175             default:
176                 ok = false;
177                 break;
178             }
179             if (ok) {
180                 for (Settlement s : player.getSettlements()) {
181                     model.addElement(s);
182                 }
183             }
184         }
185
186         this.settlementList.setModel(model);
187         this.settlementList.setSelectedValue(selected, true);
188         if (this.settlementList.getSelectedIndex() < 0) {
189             this.settlementList.setSelectedIndex(0);
190         }
191     }

193     private void selectSettlement() {
194         Settlement settlement = this.settlementList.getSelectedValue();
195         if (settlement instanceof Colony
196             && settlement.getOwner() == getMyPlayer()) {
197             getGUI().removeFromCanvas(FindSettlementPanel.this);
198             getGUI().showColonyPanel((Colony)settlement, null);
199         } else if (settlement instanceof IndianSettlement) {
200             getGUI().removeFromCanvas(FindSettlementPanel.this);
201             getGUI().showIndianSettlementPanel((IndianSettlement)settlement);
202         }
203     }
