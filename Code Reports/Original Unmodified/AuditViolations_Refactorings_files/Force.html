68     /**
69      * Create a new Force.
70      *
71      * @param specification The <code>Specification</code> for this object.
72      * @param units A list of <code>AbstractUnit</code>s defining the force.
73      * @param ability An optional ability name required of the units
74      *     in the force.
75      */
76     public Force(Specification specification, List<AbstractUnit> units,
77                  String ability) {
78         this(specification);
79         for (AbstractUnit unit : units) {
80             UnitType unitType = unit.getType(specification);
81             if (ability == null || unitType.hasAbility(ability)) {
82                 if (unitType.hasAbility(Ability.NAVAL_UNIT)) {
83                     navalUnits.add(unit);
84                 } else {
85                     landUnits.add(unit);
86                 }
87             } else {
88                 logger.warning("Found unit lacking required ability \""
89                     + ability + "\": " + unit);
90             }
91         }
92         updateSpaceAndCapacity();
93     }

114     /**
115      * Update the space and capacity variables.
116      */
117     public final void updateSpaceAndCapacity() {
118         final Specification spec = getSpecification();
119         this.capacity = sum(this.navalUnits,
120             nu -> nu.getType(spec).canCarryUnits(),
121             nu -> nu.getType(spec).getSpace() * nu.getNumber());
122         this.spaceRequired = sum(this.landUnits,
123             lu -> lu.getType(spec).getSpaceTaken() * lu.getNumber());
124     }

164     /**
165      * Adds units to this Force.
166      *
167      * @param au The addition to this Force.
168      */
169     public void add(AbstractUnit au) {
170         final Specification spec = getSpecification();
171         final UnitType unitType = au.getType(spec);
172         final int n = au.getNumber();
173         boolean added = false;
174         if (unitType.hasAbility(Ability.NAVAL_UNIT)) {
175             for (AbstractUnit refUnit : navalUnits) {
176                 if (spec.getUnitType(refUnit.getId()) == unitType) {
177                     refUnit.setNumber(refUnit.getNumber() + n);
178                     if (unitType.canCarryUnits()) {
179                         this.capacity += unitType.getSpace() * n;
180                     }
181                     added = true;
182                     break;
183                 }
184             }
185             if (!added) navalUnits.add(au);
186         } else {
187             for (AbstractUnit refUnit : landUnits) {
188                 if (spec.getUnitType(refUnit.getId()) == unitType
189                     && refUnit.getRoleId().equals(au.getRoleId())) {
190                     refUnit.setNumber(refUnit.getNumber() + n);
191                     spaceRequired += unitType.getSpaceTaken() * n;
192                     added = true;
193                     break;
194                 }
195             }
196             if (!added) this.landUnits.add(au);
197         }
198         updateSpaceAndCapacity();
199     }
