316     /**
317      * Loads the client options.
318      * There are several sources:
319      *   1) Base options (data/base/client-options.xml)
320      *   2) Standard action manager actions
321      *   3) Saved game
322      *   4) User options
323      *
324      * The base and action manager options are definitive, so they can
325      * just be added/loaded.  The others are from sources that may be
326      * out of date (i.e. options can be in the wrong group, or no longer
327      * exist), so they must be merged cautiously.
328      *
329      * @param savedGameFile An optional saved game <code>File</code>
330      *     to load options from.
331      * @return The loaded <code>ClientOptions</code>.
332      */
333     private ClientOptions loadClientOptions(File savedGameFile) {
334         ClientOptions clop = new ClientOptions();
335         logger.info("Load default client options.");
336         clop.load(FreeColDirectories.getBaseClientOptionsFile());
337
338         if (actionManager != null) {
339             logger.info("Load client options from the action manager.");
340             clop.add(actionManager);
341         }
342
343         if (savedGameFile != null) {
344             try {
345                 FreeColSavegameFile fcsf
346                     = new FreeColSavegameFile(savedGameFile);
347                 logger.info("Merge client options from saved game: "
348                     + savedGameFile.getPath());
349                 clop.merge(fcsf);
350             } catch (IOException ioe) {
351                 logger.log(Level.WARNING, "Could not open saved game "
352                     + savedGameFile.getPath(), ioe);
353             }
354         }
355
356         final File userOptions = FreeColDirectories.getClientOptionsFile();
357         if (userOptions != null && userOptions.exists()) {
358             logger.info("Merge client options from user options file: "
359                 + userOptions.getPath());
360             clop.merge(userOptions);
361         }
362
363         //logger.info("Final client options: " + clop.toString());
364         return clop;
365     }

738     /**
739      * If currently in a game, displays a quit dialog and if desired,
740      * logs out of the current game.
741      *
742      * When the game is clear, show the new game panel.
743      *
744      * Called from the New action, often from the button on the MainPanel,
745      * and IGC.victory()
746      *
747      * @param prompt If true, prompt to confirm stopping the game.
748      */
749     public void newGame(boolean prompt) {
750         Specification specification = null;
751         if (getGame() != null) {
752             if (isMapEditor()) {
753                 specification = getGame().getSpecification();
754             } else if (!prompt || gui.confirmStopGame()) {
755                 getConnectController().quitGame(true);
756                 FreeColSeed.incrementFreeColSeed();
757             } else {
758                 return;
759             }
760         }
761         gui.removeInGameComponents();
762         gui.showNewPanel(specification);
763     }

808     /**
809      * Quits the application without any questions.
810      */
811     public void quit() {
812         getConnectController().quitGame(isSinglePlayer());
813         try { // delete outdated autosave files
814             long validPeriod = 1000L * 24L * 60L * 60L // days to ms
815                 * clientOptions.getInteger(ClientOptions.AUTOSAVE_VALIDITY);
816             long timeNow = System.currentTimeMillis();
817             File autoSave = FreeColDirectories.getAutosaveDirectory();
818             String[] flist;
819             if (validPeriod != 0L && autoSave != null
820                 && (flist = autoSave.list()) != null) {
821                 for (String f : flist) {
822                     if (!f.endsWith("." + FreeCol.FREECOL_SAVE_EXTENSION)) continue;
823                     // delete files which are older than user option allows
824                     File saveGameFile = new File(autoSave, f);
825                     if (saveGameFile.lastModified() + validPeriod < timeNow) {
826                         saveGameFile.delete();
827                     }
828                 }
829             }
830         } catch (Exception e) {
831             logger.log(Level.WARNING, "Failed to delete autosave", e);
832         }
833         try {
834             gui.quit();
835         } catch (Exception e) {
836             System.exit(1);
837         }
838         System.exit(0);
839     }
