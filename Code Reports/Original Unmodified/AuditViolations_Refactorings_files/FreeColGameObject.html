36 /**
37  * The superclass of all game objects in FreeCol.
38  *
39  * All FreeColGameObjects need to be able to refer to the game they belong
40  * to.  Therefore, the game attribute must not be null, except in the special
41  * case where a Game is being initially created.
42  *
43  * Most FreeColGameObjects are intended to be accessible by identifier (@see
44  * Game#getFreeColObject) but some are not, and should override isInternable
45  * to return false.
46  */

118     /**
119      * Sets the unique identifier of this object and registers it in its
120      * <code>Game</code> with that identifier, i.e. "intern" this object.
121      *
122      * @param newId The unique identifier of this object.
123      */
124     public final void internId(final String newId) {
125         final Game game = getGame();
126         if (game != null && newId != null && isInternable()) {
127             final String oldId = getId();
128             if (!newId.equals(oldId)) {
129                 if (oldId != null) {
130                     game.removeFreeColGameObject(oldId, "override");
131                 }
132                 setId(newId);
133                 if (newId != null) {
134                     game.setFreeColGameObject(newId, this);
135                 }
136             }
137         } else {
138             setId(newId);
139         }
140     }
