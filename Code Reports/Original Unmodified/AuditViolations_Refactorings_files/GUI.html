514     /**
515      * Check if an attack results in a transition from peace or cease fire to
516      * war and, if so, warn the player.
517      *
518      * @param attacker The potential attacking <code>Unit</code>.
519      * @param target The target <code>Tile</code>.
520      * @return True to attack, false to abort.
521      */
522     public boolean confirmHostileAction(Unit attacker, Tile target) {
523         if (attacker.hasAbility(Ability.PIRACY)) {
524             // Privateers can attack and remain at peace
525             return true;
526         }
527
528         Player enemy;
529         if (target.hasSettlement()) {
530             enemy = target.getSettlement().getOwner();
531         } else if (target == attacker.getTile()) {
532             // Fortify on tile owned by another nation
533             enemy = target.getOwner();
534             if (enemy == null) return true;
535         } else {
536             Unit defender = target.getDefendingUnit(attacker);
537             if (defender == null) {
538                 logger.warning("Attacking, but no defender - will try!");
539                 return true;
540             }
541             if (defender.hasAbility(Ability.PIRACY)) {
542                 // Privateers can be attacked and remain at peace
543                 return true;
544             }
545             enemy = defender.getOwner();
546         }
547
548         String messageId;
549         switch (attacker.getOwner().getStance(enemy)) {
550         case WAR:
551             logger.finest("Player at war, no confirmation needed");
552             return true;
553         case CEASE_FIRE:
554             messageId = "confirmHostile.ceaseFire";
555             break;
556         case ALLIANCE:
557             messageId = "confirmHostile.alliance";
558             break;
559         case UNCONTACTED: case PEACE: default:
560             messageId = "confirmHostile.peace";
561             break;
562         }
563         return confirm(attacker.getTile(), StringTemplate
564             .template(messageId)
565             .addStringTemplate("%nation%", enemy.getNationLabel()),
566             attacker, "confirmHostile.yes", "cancel");
567     }
