455     /**
456      * Returns the appropriate BufferedImage for a FreeColObject.
457      * Please, use a more specific method!
458      *
459      * @param display The FreeColObject to display.
460      * @param scale How much the image should be scaled.
461      * @return The appropriate BufferedImage.
462      */
463     public BufferedImage getObjectImage(FreeColObject display, float scale) {
464         try {
465             final float combinedScale = scaleFactor * scale;
466             final Dimension size = scaleDimension(ICON_SIZE, combinedScale);
467             BufferedImage image;
468             if (display instanceof Goods) {
469                 display = ((Goods)display).getType();
470             } else if (display instanceof Player) {
471                 display = ((Player)display).getNation();
472             }
473
474             if (display instanceof Unit) {
475                 Unit unit = (Unit)display;
476                 image = getUnitImage(unit, size);
477             } else if (display instanceof UnitType) {
478                 UnitType unitType = (UnitType)display;
479                 image = getUnitImage(unitType, size);
480             } else if (display instanceof Settlement) {
481                 Settlement settlement = (Settlement)display;
482                 image = getSettlementImage(settlement, size);
483             } else if (display instanceof LostCityRumour) {
484                 image = getMiscImage(ImageLibrary.LOST_CITY_RUMOUR, size);
485             } else if (display instanceof GoodsType) {
486                 FreeColSpecObjectType type = (FreeColSpecObjectType)display;
487                 image = getIconImage(type);
488             } else if (display instanceof Nation) {
489                 FreeColSpecObjectType type = (FreeColSpecObjectType)display;
490                 image = getMiscIconImage(type, size);
491             } else if (display instanceof BuildingType) {
492                 BuildingType type = (BuildingType)display;
493                 image = getBuildingImage(type, size);
494             } else {
495                 logger.warning("could not find image of unknown type for " + display);
496                 return null;
497             }
498
499             if (image == null) {
500                 logger.warning("could not find image for " + display);
501                 return null;
502             }
503             return image;
504         } catch (Exception e) {
505             logger.log(Level.WARNING, "could not find image", e);
506             return null;
507         }
508     }
