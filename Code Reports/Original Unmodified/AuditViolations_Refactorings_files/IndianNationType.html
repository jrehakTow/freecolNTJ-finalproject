177     /**
178      * Generates choices for skill that could be taught from a settlement on
179      * a given Tile.
180      *
181      * @param tile The <code>Tile</code> where the settlement will be located.
182      * @return A random choice set of skills.
183      */
184     public List<RandomChoice<UnitType>> generateSkillsForTile(Tile tile) {
185         final List<RandomChoice<UnitType>> skills = getSkills();
186         final Map<GoodsType, Integer> scale = toMap(getSkills().stream(),
187             rc -> rc.getObject().getExpertProduction(), rc -> 1);
188
189         for (Tile t: tile.getSurroundingTiles(1)) {
190             for (Entry<GoodsType, Integer> entry : scale.entrySet()) {
191                 GoodsType goodsType = entry.getKey();
192                 scale.put(goodsType, entry.getValue()
193                           + t.getPotentialProduction(goodsType, null));
194             }
195         }
196
197         return toList(map(skills, rc -> {
198                     UnitType ut = rc.getObject();
199                     int scaleValue = scale.get(ut.getExpertProduction());
200                     return new RandomChoice<>(ut, rc.getProbability() * scaleValue);
201                 }));
202     }
