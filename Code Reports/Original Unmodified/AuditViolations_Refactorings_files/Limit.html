29 /**
30  * The <code>Limit</code> class encapsulates a limit on the
31  * availability of FreeColObjects.  It consists of a left hand side,
32  * an operator and a right hand side, and can be used to limit the
33  * number of units of a particular type (e.g. wagon trains) to the
34  * number of a player's colonies, for example. The left hand side must
35  * apply to the object on which a limit is to be placed.

272      *
273      * @param lhs The left hand side <code>Integer</code>.
274      * @param rhs The right hand side <code>Integer</code>.
275      * @return The result of the evaluation.
276      */
277     private boolean evaluate(Integer lhs, Integer rhs) {
278         if (lhs == null || rhs == null) return true;
279         switch (operator) {
280         case EQ: return lhs == rhs;
281         case LT: return lhs <  rhs;
282         case GT: return lhs >  rhs;
283         case LE: return lhs <= rhs;
284         case GE: return lhs >= rhs;
285         default: break;
286         }
287         return false;
288     }
