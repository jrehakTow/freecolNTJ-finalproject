169     /**
170      * Chooses a type of Lost City Rumour.  The type of rumour depends
171      * on the exploring unit, as well as player settings.
172      *
173      * The scouting outcome is based on three factors: good/bad percent
174      * rumour difficulty option, expert scout or not, DeSoto or not.
175      *
176      * @param unit The <code>Unit</code> exploring (optional).
177      * @param random A random number source.
178      * @return The type of rumour.
179      *
180      * FIXME: Make RumourType a FreeColSpecObjectType and move all the
181      * magic numbers in here to the specification.
182      */
183     public RumourType chooseType(Unit unit, Random random) {
184         final Specification spec = getSpecification();
185         final Tile tile = getTile();
186         final boolean allowLearn = unit != null
187             && !unit.getType().getUnitTypesLearntInLostCity().isEmpty();
188
189         // Base bad and good chances are difficulty options.
190         int percentBad = spec.getInteger(GameOptions.BAD_RUMOUR);
191         int percentGood = spec.getInteger(GameOptions.GOOD_RUMOUR);
192
193         // Expert scouts have a beneficial modifier that works on both
194         // percentages
195         if (unit != null) {
196             float mod = unit.applyModifiers(1.0f, getGame().getTurn(),
197                                             Modifier.EXPLORE_LOST_CITY_RUMOUR);
198             percentBad = Math.round(percentBad * mod);
199             percentGood = Math.round(percentGood * mod);
200         }
201
202         // DeSoto forces all good results.
203         if (unit != null
204             && unit.getOwner().hasAbility(Ability.RUMOURS_ALWAYS_POSITIVE)) {
205             percentBad = 0;
206             percentGood = 100;
207         }
208
209         // Neutral is what is left.
210         int percentNeutral = Math.max(0, 100 - percentBad - percentGood);
211
212         // Add all possible events to a RandomChoice List
213         List<RandomChoice<RumourType>> c = new ArrayList<>();
214
215         // The GOOD
216         if (allowLearn) {
217             c.add(new RandomChoice<>(RumourType.LEARN,
218                     30 * percentGood));
219             c.add(new RandomChoice<>(RumourType.TRIBAL_CHIEF,
220                     30 * percentGood));
221             c.add(new RandomChoice<>(RumourType.COLONIST,
222                     20 * percentGood));
223         } else {
224             c.add(new RandomChoice<>(RumourType.TRIBAL_CHIEF,
225                     50 * percentGood));
226             c.add(new RandomChoice<>(RumourType.COLONIST,
227                     30 * percentGood));
228         }
229         if (unit == null
230             || unit.getOwner().getPlayerType() == Player.PlayerType.COLONIAL) {
231             c.add(new RandomChoice<>(RumourType.FOUNTAIN_OF_YOUTH,
232                         2 * percentGood));
233         }
234         c.add(new RandomChoice<>(RumourType.MOUNDS,
235                 8 * percentGood));
236         c.add(new RandomChoice<>(RumourType.RUINS,
237                 6 * percentGood));
238         c.add(new RandomChoice<>(RumourType.CIBOLA,
239                 4 * percentGood));
240
241         // The BAD
242         if (tile.getOwner() != null && tile.getOwner().isIndian()) {
243             // If the tile is native-owned, allow burial grounds rumour.
244             c.add(new RandomChoice<>(RumourType.BURIAL_GROUND,
245                     25 * percentBad));
246             c.add(new RandomChoice<>(RumourType.EXPEDITION_VANISHES,
247                     75 * percentBad));
248         } else {
249             c.add(new RandomChoice<>(RumourType.EXPEDITION_VANISHES,
250                     100 * percentBad));
251         }
252
253         // The NEUTRAL
254         c.add(new RandomChoice<>(RumourType.NOTHING,
255                 100 * percentNeutral));
256
257         return RandomChoice.getWeightedRandom(logger, "Choose rumour", c,
258                                               random);
259     }

261     /**
262      * Get the message for a "nothing" rumour.
263      *
264      * @param player The <code>Player</code> to generate the message for.
265      * @param mounds Is this rumour a result of exploring "strange mounds"?
266      * @param random A pseudo-random number source.
267      * @return A suitable message.
268      */
269     public ModelMessage getNothingMessage(Player player, boolean mounds,
270                                           Random random) {
271         final Game game = getGame();
272         String key;
273         if (mounds) {
274             key = RumourType.NOTHING.getAlternateDescriptionKey("mounds");
275             return new ModelMessage(ModelMessage.MessageType.LOST_CITY_RUMOUR,
276                                     key, player);
277         } else if (game.getTurn().getYear() % 100 == 12
278             && randomInt(logger, "Mayans?", random, 4) == 0) {
279             int years = MAYAN_PROPHESY_YEAR - game.getTurn().getYear();
280             return new ModelMessage(ModelMessage.MessageType.LOST_CITY_RUMOUR,
281                                     "model.lostCityRumour.nothing.mayans",
282                                     player)
283                 .addAmount("%years%", years);
284         }
285         int i;
286         if (rumourNothing < 0) {
287             i = 0;
288             for (;;) {
289                 key = Messages.descriptionKey("model.lostCityRumour.nothing." + i);
290                 if (!Messages.containsKey(key)) break;
291                 i++;
292             }
293             rumourNothing = i;
294         }
295         i = randomInt(logger, "Nothing rumour", random, rumourNothing);
296         key = Messages.descriptionKey("model.lostCityRumour.nothing." + i);
297         return new ModelMessage(ModelMessage.MessageType.LOST_CITY_RUMOUR,
298                                 key, player);
299     }
