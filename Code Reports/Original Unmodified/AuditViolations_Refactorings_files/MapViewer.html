662     /**
663      * Position the map so that the supplied tile is displayed at the center.
664      *
665      * @param pos The <code>Tile</code> to center at.
666      */
667     private void positionMap(Tile pos) {
668         final Game game = getGame();
669         int x = pos.getX(), y = pos.getY();
670         int leftColumns = getLeftColumns(), rightColumns = getRightColumns();
671
672         /*
673           PART 1
674           ======
675           Calculate: bottomRow, topRow, bottomRowY, topRowY
676           This will tell us which rows need to be drawn on the screen (from
677           bottomRow until and including topRow).
678           bottomRowY will tell us at which height the bottom row needs to be
679           drawn.
680         */
681         alignedTop = false;
682         alignedBottom = false;
683         if (y < topRows) {
684             alignedTop = true;
685             // We are at the top of the map
686             bottomRow = (size.height / (halfHeight)) - 1;
687             if ((size.height % (halfHeight)) != 0) {
688                 bottomRow++;
689             }
690             topRow = 0;
691             bottomRowY = bottomRow * (halfHeight);
692             topRowY = 0;
693         } else if (y >= (game.getMap().getHeight() - bottomRows)) {
694             alignedBottom = true;
695             // We are at the bottom of the map
696             bottomRow = game.getMap().getHeight() - 1;
697
698             topRow = size.height / (halfHeight);
699             if ((size.height % (halfHeight)) > 0) {
700                 topRow++;
701             }
702             topRow = game.getMap().getHeight() - topRow;
703
704             bottomRowY = size.height - tileHeight;
705             topRowY = bottomRowY - (bottomRow - topRow) * (halfHeight);
706         } else {
707             // We are not at the top of the map and not at the bottom
708             bottomRow = y + bottomRows - 1;
709             topRow = y - topRows;
710             bottomRowY = topSpace + (halfHeight) * bottomRows;
711             topRowY = topSpace - topRows * (halfHeight);
712         }
713
714         /*
715           PART 2
716           ======
717           Calculate: leftColumn, rightColumn, leftColumnX
718           This will tell us which columns need to be drawn on the screen (from
719           leftColumn until and including rightColumn).
720           leftColumnX will tell us at which x-coordinate the left
721           column needs to be drawn (this is for the Tiles where y&1 == 0;
722           the others should be halfWidth more to the right).
723         */
724
725         alignedLeft = false;
726         alignedRight = false;
727         if (x < leftColumns) {
728             // We are at the left side of the map
729             leftColumn = 0;
730
731             rightColumn = size.width / tileWidth - 1;
732             if ((size.width % tileWidth) > 0) {
733                 rightColumn++;
734             }
735
736             leftColumnX = 0;
737             alignedLeft = true;
738         } else if (x >= (game.getMap().getWidth() - rightColumns)) {
739             // We are at the right side of the map
740             rightColumn = game.getMap().getWidth() - 1;
741
742             leftColumn = size.width / tileWidth;
743             if ((size.width % tileWidth) > 0) {
744                 leftColumn++;
745             }
746
747             leftColumnX = size.width - tileWidth - halfWidth -
748                 leftColumn * tileWidth;
749             leftColumn = rightColumn - leftColumn;
750             alignedRight = true;
751         } else {
752             // We are not at the left side of the map and not at the right side
753             leftColumn = x - leftColumns;
754             rightColumn = x + rightColumns;
755             leftColumnX = (size.width - tileWidth) / 2
756                 - leftColumns * tileWidth;
757         }
758     }
