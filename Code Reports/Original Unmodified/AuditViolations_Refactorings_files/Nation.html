248     /**

250      *
251      * @param game The current <code>Game</code>.
252      * @param random A pseudo-random number source.
253      * @return A player name key, or an empty string on failure.
254      */
255     public static String getRandomNonPlayerNationNameKey(Game game,
256                                                          Random random) {
257         int nations = EUROPEAN_NATIONS.size();
258         int start = randomInt(logger, "Random nation", random, nations);
259         for (int index = 0; index < nations; index++) {
260             String nationId = "model.nation."
261                 + EUROPEAN_NATIONS.get((start + index) % nations);
262             if (game.getPlayerByNationId(nationId) == null) {
263                 return Messages.nameKey(nationId);
264             }
265         }
266         // this should never happen
267         return "";
268     }
