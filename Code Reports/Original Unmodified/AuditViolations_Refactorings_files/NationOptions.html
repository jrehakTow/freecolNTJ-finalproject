114     /**
115      * Creates a new <code>NationOptions</code> instance.
116      *
117      * @param specification The <code>Specification</code> to refer to.
118      */
119     public NationOptions(Specification specification) {
120         super(specification);
121         
122         this.nationalAdvantages = FreeCol.getAdvantages();
123         if (specification != null) {
124             int counter = 0, maxEuropeans = FreeCol.getEuropeanCount();
125             for (Nation nation : specification.getNations()) {
126                 if (nation.isUnknownEnemy() || nation.getType().isREF()) {
127                     continue;
128                 } else if (nation.getType().isEuropean()
129                     && nation.isSelectable()) {
130                     if (counter < maxEuropeans) {
131                         nations.put(nation, NationState.AVAILABLE);
132                         counter++;
133                     } else {
134                         nations.put(nation, NationState.NOT_AVAILABLE);
135                     }
136                 } else {
137                     nations.put(nation, NationState.AI_ONLY);
138                 }
139             }
140         }
141     }
