132     private void buildTree(OptionGroup group, DefaultMutableTreeNode parent) {
133         for (Option option : group.getOptions()) {
134             if (option instanceof OptionGroup) {

136                 DefaultMutableTreeNode branch

188     private void addOptionUI(Option option, boolean editable) {
189         OptionUI ui = optionUIs.get(option.getId());
190         if (ui == null) {
191             ui = OptionUI.getOptionUI(gui, option, editable);
192             if (ui == null) {
193                 logger.warning("Unknown option type: " + option.toString());
194                 return;
195             }
196             if (ui instanceof FreeColActionUI) {
197                 ((FreeColActionUI)ui).setOptionGroupUI(this);
198             }
199             if (option.getId() != null) {
200                 optionUIs.put(option.getId(), ui);
201             }
202         }
203         JLabel label = ui.getJLabel();
204         if (label == null) {
205             detailPanel.add(ui.getComponent(), "newline, span");
206         } else {
207             detailPanel.add(label);
208             detailPanel.add(ui.getComponent());
209         }
210         if (group.isEditable()) {
211             optionUpdaters.add(ui);
212         }
213     }
