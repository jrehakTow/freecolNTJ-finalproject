145     /**
146      * Create a new <code>RearrangeColonyMessage</code> with the
147      * supplied colony.  Add changes with addChange().
148      *
149      * @param colony The <code>Colony</code> that is rearranging.
150      * @param workers A list of worker <code>Unit</code>s to rearrange.
151      * @param scratch A scratch <code>Colony</code> laid out as required.
152      */
153     public RearrangeColonyMessage(Colony colony, List<Unit> workers,
154                                   Colony scratch) {
155         super(getTagName());
156
157         this.colonyId = colony.getId();
158         this.unitChanges.clear();
159         for (Unit u : workers) {
160             Unit su = scratch.getCorresponding(u);
161             if (u.getLocation().getId().equals(su.getLocation().getId())
162                 && u.getWorkType() == su.getWorkType()
163                 && u.getRole() == su.getRole()
164                 && u.getRoleCount() == su.getRoleCount()) continue;
165             addChange(u,
166                 (Location)colony.getCorresponding((FreeColObject)su.getLocation()),
167                 su.getWorkType(), su.getRole(), su.getRoleCount());
168         }
169     }

222      * Handle a "rearrangeColony"-message.
223      *
224      * @param server The <code>FreeColServer</code> handling the request.
225      * @param player The <code>Player</code> rearrangeing the colony.
226      * @param connection The <code>Connection</code> the message is from.
227      * @return An update <code>Element</code> with the rearranged colony,
228      *     or an error <code>Element</code> on failure.
229      */
230     public Element handle(FreeColServer server, Player player,
231                           Connection connection) {
232         final ServerPlayer serverPlayer = server.getPlayer(connection);
233
234         Colony colony;
235         try {
236             colony = player.getOurFreeColGameObject(this.colonyId, Colony.class);
237         } catch (Exception e) {
238             return serverPlayer.clientError(e.getMessage())
239                 .build(serverPlayer);
240         }
241
242         if (isEmpty()) {
243             return serverPlayer.clientError("Empty rearrangement list.")
244                 .build(serverPlayer);
245         }
246         int i = 0;
247         for (UnitChange uc : unitChanges) {
248             if (uc.unit == null) {
249                 return serverPlayer.clientError("Invalid unit " + i)
250                     .build(serverPlayer);
251             }
252             if (uc.loc == null) {
253                 return serverPlayer.clientError("Invalid location " + i)
254                     .build(serverPlayer);
255             }
256             if (uc.role == null) {
257                 return serverPlayer.clientError("Invalid role " + i)
258                     .build(serverPlayer);
259             }
260             if (uc.roleCount < 0) {
261                 return serverPlayer.clientError("Invalid role count " + i)
262                     .build(serverPlayer);
263             }
264         }
265
266         // Rearrange can proceed.
267         return server.getInGameController()
268             .rearrangeColony(serverPlayer, colony, this.unitChanges)
269             .build(serverPlayer);
270     }
