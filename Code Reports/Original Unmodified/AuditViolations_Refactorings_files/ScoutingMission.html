401     @Override
402     public Mission doMission(LogBuilder lb) {
403         lb.add(tag);
404         String reason = invalidReason();
405         if (isTargetReason(reason)) {
406             return retargetMission(reason, lb);
407         } else if (reason != null) {
408             return lbFail(lb, false, reason);
409         }
410
411         // Go to the target.
412         final AIUnit aiUnit = getAIUnit();
413         final Unit unit = getUnit();
414         Direction d;
415         Unit.MoveType mt = travelToTarget(getTarget(),
416             CostDeciders.avoidSettlementsAndBlockingUnits(), lb);
417         switch (mt) {
418         case MOVE: // Arrived at a colony
419             break;
420
421         case MOVE_HIGH_SEAS: case MOVE_NO_MOVES: case MOVE_ILLEGAL:
422             return lbWait(lb);
423
424         case MOVE_NO_REPAIR:
425             return lbFail(lb, false, AIUNITDIED);
426
427         case MOVE_NO_ACCESS_EMBARK:
428             return this;
429
430         case MOVE_NO_TILE:
431             moveRandomly(tag, null);
432             return lbDodge(lb);
433
434         case ATTACK_UNIT:
435             // Could be adjacent to the destination but it is
436             // temporarily blocked by another unit.  Make a random
437             // (directed if possible) move and try again.
438             moveRandomly(tag, unit.getTile()
439                 .getDirection(getTarget().getTile()));
440             return lbDodge(lb);
441
442         case ENTER_INDIAN_SETTLEMENT_WITH_SCOUT:
443             d = unit.getTile().getDirection(getTarget().getTile());
444             assert d != null && getTarget() instanceof IndianSettlement;
445             if (AIMessage.askScoutSpeakToChief(aiUnit,
446                                                (IndianSettlement)getTarget())) {
447                 lbDone(lb, true, "speak-with-chief at ", getTarget());
448             } else {
449                 lbFail(lb, true, "unexpected failure to speak at ", getTarget());
450             }
451             break;
452
453         case EXPLORE_LOST_CITY_RUMOUR:
454             d = unit.getTile().getDirection(getTarget().getTile());
455             assert d != null;
456             if (AIMessage.askMove(aiUnit, d)) {
457                 lbDone(lb, true, "explore at ", getTarget());
458             } else {
459                 lbFail(lb, true, "unexpected failure at ", getTarget());
460             }
461             break;
462
463         default:
464             return lbMove(lb, mt);
465         }
466         if (unit.isDisposed()) return this;
467
468         // Retarget on failure or complete, but do not retarget from
469         // one colony to another, just drop equipment and invalidate
470         // the mission.
471         lbAt(lb);
472         Location completed = getTarget();
473         Location newTarget = findTarget(aiUnit, 20, false);
474         if (newTarget == null
475             || (completed instanceof Colony && newTarget == completed)) {
476             if (completed instanceof Colony && canScoutNatives(aiUnit)) {
477                 aiUnit.equipForRole(getSpecification().getDefaultRole());
478             }
479             return lbFail(lb, false, ", found no targets");
480         }
481         setTarget(newTarget);
482         return lbRetarget(lb);
483     }
