40 /**
41  * The networking server in which new clients can connect and methods
42  * like <code>sendToAll</code> are kept.
43  *
44  * <br><br>
45  *
46  * When a new client connects to the server a new {@link Connection}
47  * is made, with {@link net.sf.freecol.server.control.UserConnectionHandler}
48  * as the control object.
49  *
50  * @see net.sf.freecol.common.networking
51  */

53
54     private static final Logger logger = Logger.getLogger(Server.class.getName());
55
56     /** Backlog for socket. */
57     private static final int BACKLOG_DEFAULT = 10;

59     /** The public "well-known" socket to which clients may connect. */
60     private final ServerSocket serverSocket;
61
62     /** A hash of Connection objects, keyed by the Socket they relate to. */
63     private final HashMap<Socket, Connection> connections = new HashMap<>();
64
65     /**
66      * Whether to keep running the main loop that is awaiting new
67      * client connections.
68      */
69     private boolean running = true;

71     /** The owner of this <code>Server</code>. */
72     private final FreeColServer freeColServer;
73
74     /** The name of the host for the public socket. */
75     private final String host;

77     /** The TCP port that is beeing used for the public socket. */
78     private final int port;

81     private final Object shutdownLock = new Object();

84     /**
85      * Creates a new network server. Use {@link #run server.start()} to start
86      * listening for new connections.
87      *
88      * @param freeColServer The owner of this <code>Server</code>.
89      * @param host The name of the host for the public socket.
90      * @param port The TCP port to use for the public socket.
91      * @throws IOException if the public socket cannot be created.
92      */
93     public Server(FreeColServer freeColServer, String host,
94                   int port) throws IOException {
95         super(FreeCol.SERVER_THREAD + "Server");
96
97         this.freeColServer = freeColServer;
98         this.host = host;
99         this.port = port;
100         this.serverSocket = new ServerSocket(port, BACKLOG_DEFAULT,

175     /**
176      * Sends a network message to all connections with an optional exception.
177      *
178      * @param message The <code>DOMMessage</code> to send.
179      * @param exceptConnection An optional <code>Connection</code> not
180      *     to send to.
181      */
182     public void sendToAll(DOMMessage message, Connection exceptConnection) {
183         Element element = message.toXMLElement();
184         for (Connection c : new ArrayList<>(connections.values())) {
185             if (c == exceptConnection) continue;
186             if (c.isAlive()) {
187                 try {
188                     c.sendAndWait(element);
189                 } catch (IOException e) {
190                     logger.log(Level.WARNING, "Unable to send to: " + c, e);
191                 }
192             } else {
193                 logger.log(Level.INFO, "Reap dead connection: " + c);
194                 removeConnection(c);
195             }
196         }
197     }

208     /**
209      * Start the thread processing.  Contains the loop that is waiting
210      * for new connections to the public socket.  When a new client
211      * connects to the server a new {@link Connection} is made, with
212      * {@link net.sf.freecol.server.control.UserConnectionHandler} as
213      * the control object.
214      */
215     @Override
216     public void run() {
217         // This method's entire body is synchronized to shutdownLock.
218         // The reason why this is done is to prevent the shutdown
219         // method from finishing before this thread is finished
220         // working.  We have to do this because the
221         // ServerSocket::close method keeps the server alive for
222         // several milliseconds *even after the close method is
223         // finished*.  Because of this a new server can't be created
224         // on the same port as this server right after closing this
225         // server.
226         //
227         // Now that the shutdown method 'hangs' until the entire
228         // server thread is finished you can be certain that the
229         // ServerSocket is REALLY closed after execution of shutdown.
230         synchronized (this.shutdownLock) {
231             while (this.running) {
232                 Socket clientSocket = null;
233                 try {
234                     clientSocket = serverSocket.accept();
235
236                     logger.info("Got client connection from "
237                         + clientSocket.getInetAddress()
238                         + ":" + clientSocket.getPort());
239                     Connection connection = new Connection(clientSocket,
240                         this.freeColServer.getUserConnectionHandler(),
241                         FreeCol.SERVER_THREAD);
242                     addConnection(connection);
243                 } catch (IOException e) {
244                     if (this.running) {
245                         logger.log(Level.WARNING, "Connection failed: ", e);
246                     }
247                 }
248             }
249         }
250     }
251
252     /**
253      * Shuts down the server thread.
254      */
255     public void shutdown() {
256         this.running = false;
257  
258         try {
259             this.serverSocket.close();
260             logger.fine("Closed server socket.");
261         } catch (IOException e) {
262             logger.log(Level.WARNING, "Could not close the server socket!", e);
263         }
264
265         synchronized (this.shutdownLock) {
266             // See run() above.
267             logger.fine("Wait for Server.run to complete.");
268         }
269
270         for (Connection c : this.connections.values()) {
271             if (c.isAlive()) c.close();
272         }
273         this.connections.clear();
274
275         this.freeColServer.removeFromMetaServer();
276         logger.fine("Server shutdown.");
277     }
