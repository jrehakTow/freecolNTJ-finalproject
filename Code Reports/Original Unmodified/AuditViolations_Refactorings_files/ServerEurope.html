81      */
82     @Override
83     public boolean equipForRole(Unit unit, Role role, int roleCount) {
84         if (!unit.roleIsAvailable(role)) return false;
85
86         // Get the change in goods
87         List<AbstractGoods> required = unit.getGoodsDifference(role, roleCount);
88
89         // Check the pricing
90         int price = priceGoods(required);
91         if (price < 0 || !unit.getOwner().checkGold(price)) return false;
92
93         // Sell any excess
94         final ServerPlayer owner = (ServerPlayer)getOwner();
95         for (AbstractGoods ag : required) {
96             if (ag.getAmount() >= 0) continue;
97             if (owner.canTrade(ag.getType(), Market.Access.EUROPE)) {
98                 int rm = owner.sell(null, ag.getType(), -ag.getAmount());
99                 if (rm > 0) {
100                     owner.addExtraTrade(new AbstractGoods(ag.getType(), rm));
101                 }
102             }
103         }
104         // Buy what is needed
105         for (AbstractGoods ag : required) {
106             if (ag.getAmount() <= 0) continue;
107             int m = owner.buy(null, ag.getType(), ag.getAmount());
108             if (m > 0) {
109                 owner.addExtraTrade(new AbstractGoods(ag.getType(), -m));
110             }
111         }
112
113         unit.changeRole(role, roleCount);
114         return true;
115     }

120      *
121      * @param random A pseudo-random number source.
122      */
123     public void initializeMigration(Random random) {
124         final Specification spec = getGame().getSpecification();
125         UnitType unitType;
126
127         if (spec.hasOption(GameOptions.IMMIGRANTS)) {
128             UnitListOption option
129                 = (UnitListOption)spec.getOption(GameOptions.IMMIGRANTS);
130             for (AbstractUnit au : option.getOptionValues()) {
131                 unitType = au.getType(spec);
132                 addRecruitable(au.getType(spec));
133             }
134         } else {
135             // @compat 0.10.3
136             for (int index = 0;; index++) {
137                 String optionId = "model.option.recruitable.slot" + index;
138                 String unitTypeId;
139                 if (spec.hasOption(optionId)
140                     && (unitTypeId = spec.getString(optionId)) != null
141                     && (unitType = spec.getUnitType(unitTypeId)) != null
142                     && addRecruitable(unitType)) continue;
143                 break; // Failed
144             }
145             // end @compat
146         }
147
148         // Fill out to the full amount of recruits if the above failed
149         List<RandomChoice<UnitType>> recruits = generateRecruitablesList();

165     /**
166      * Extract the recruitable at a given slot, and replace it with
167      * the given new recruitable type.
168      *
169      * Note that we shift the old units down, because the AI always
170      * recruits from the lowest slot.
171      *
172      * @param slot The slot to recruit with.
173      * @param random A pseudo-random number source.
174      * @return The recruited <code>UnitType</code>.
175      */
176     public UnitType extractRecruitable(int slot, Random random) {
177         // An invalid slot is normal when the player has no control over
178         // recruit type.
179         final int count = MigrationType.getMigrantCount();
180         int index = (MigrationType.specificMigrantSlot(slot))
181             ? MigrationType.migrantSlotToIndex(slot)
182             : randomInt(logger, "Choose emigrant", random, count);
183         UnitType result = recruitables.get(index);
184         for (int i = index; i < count-1; i++) {
185             recruitables.set(i, recruitables.get(i+1));
186         }
187         recruitables.set(count-1, RandomChoice.getWeightedRandom(logger,
188                 "Replace recruit", generateRecruitablesList(), random));
189         return result;
190     }

193      * Generate a weighted list of unit types recruitable by this player.
194      *
195      * @return A weighted list of recruitable unit types.
196      */
197     public List<RandomChoice<UnitType>> generateRecruitablesList() {
198         final Player owner = getOwner();
199         return transform(getSpecification().getUnitTypeList(),
200             ut -> ut.isRecruitable()
201                 && owner.hasAbility(Ability.CAN_RECRUIT_UNIT, ut),
202             ut -> new RandomChoice<>(ut, ut.getRecruitProbability()),
203             Collectors.toList());
204     }

212     public boolean replaceRecruits(Random random) {
213         List<RandomChoice<UnitType>> recruits = generateRecruitablesList();
214         boolean result = false;
215         int i = 0;
216         for (UnitType ut : recruitables) {
217             if (hasAbility(Ability.CAN_RECRUIT_UNIT, ut)) continue;
218             UnitType newType = RandomChoice.getWeightedRandom(logger,
219                 "Replace recruit", recruits, random);
220             recruitables.set(i, newType);
221             result = true;
222             i++;
