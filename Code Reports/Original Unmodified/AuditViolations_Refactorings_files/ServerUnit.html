539     /**
540      * If a unit moves, check if an opposing naval unit slows it down.
541      * Note that the unit moves are reduced here.
542      *
543      * @param newTile The <code>Tile</code> the unit is moving to.
544      * @param random A pseudo-random number source.
545      * @return Either an enemy unit that causes a slowdown, or null if none.
546      */
547     private Unit getSlowedBy(Tile newTile, Random random) {
548         Player player = getOwner();
549         Game game = getGame();
550         CombatModel combatModel = game.getCombatModel();
551         boolean pirate = hasAbility(Ability.PIRACY);
552         Unit attacker = null;
553         double attackPower = 0, totalAttackPower = 0;
554
555         if (!isNaval() || getMovesLeft() <= 0) return null;
556         for (Tile tile : newTile.getSurroundingTiles(1)) {
557             // Ships in settlements do not slow enemy ships, but:
558             // FIXME: should a fortress slow a ship?
559             Player enemy;
560             if (tile.isLand()
561                 || tile.getColony() != null
562                 || tile.getFirstUnit() == null
563                 || (enemy = tile.getFirstUnit().getOwner()) == player) continue;
564             for (Unit enemyUnit : tile.getUnitList()) {
565                 if ((pirate || enemyUnit.hasAbility(Ability.PIRACY)
566                      || (enemyUnit.isOffensiveUnit() && player.atWarWith(enemy)))
567                     && enemyUnit.isNaval()
568                     && combatModel.getOffencePower(enemyUnit, this) > attackPower) {
569                     attackPower = combatModel.getOffencePower(enemyUnit, this);
570                     totalAttackPower += attackPower;
571                     attacker = enemyUnit;
572                 }
573             }
574         }
575         if (attacker != null) {
576             double defencePower = combatModel.getDefencePower(attacker, this);
577             double totalProbability = totalAttackPower + defencePower;
578             if (randomInt(logger, "Slowed", random,
579                     (int)Math.round(totalProbability + 1)) < totalAttackPower) {
580                 int diff = Math.max(0,
581                     (int)Math.round(totalAttackPower - defencePower));
582                 int moves = Math.min(9, 3 + diff / 3);
583                 setMovesLeft(getMovesLeft() - moves);
584                 logger.info(getId() + " slowed by " + attacker.getId()
585                     + " by " + Integer.toString(moves) + " moves.");
586             } else {
587                 attacker = null;
588             }
589         }
590         return attacker;
591     }

858      * @param newTile The <code>Tile</code> to move to.
859      * @param random A pseudo-random number source.
860      * @param cs A <code>ChangeSet</code> to update.
861      */
862     public void csMove(Tile newTile, Random random, ChangeSet cs) {
863         final ServerPlayer serverPlayer = (ServerPlayer)getOwner();
864
865         // Plan to update tiles that could not be seen before but will
866         // now be within the line-of-sight.
867         final Location oldLocation = getLocation();
868         List<Tile> oldTiles = (oldLocation.getTile() == null)
869             ? Collections.<Tile>emptyList()
870             : oldLocation.getTile().getSurroundingTiles(1, getLineOfSight());
871         List<Tile> newTiles = collectNewTiles(newTile);
872
873         // Update unit state.
874         setState(UnitState.ACTIVE);
