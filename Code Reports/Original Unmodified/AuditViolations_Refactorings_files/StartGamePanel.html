236     /**
237      * Check that the user has not specified degenerate victory conditions
238      * that are automatically true.
239      *
240      * @return True if the victory conditions are sensible.
241      */
242     private boolean checkVictoryConditions() {
243         Specification spec = getSpecification();
244         if (singlePlayerGame
245             && spec.getBoolean(GameOptions.VICTORY_DEFEAT_EUROPEANS)
246             && !spec.getBoolean(GameOptions.VICTORY_DEFEAT_REF)) {
247             int n = count(getGame().getNationOptions().getNations().entrySet(),
248                 e -> e.getKey().getType().isEuropean()
249                     && !e.getKey().isUnknownEnemy()
250                     && e.getValue() != NationState.NOT_AVAILABLE);
251             if (n == 0) {
252                 getGUI().showInformationMessage("info.noEuropeans");
253                 return false;
254             }
255         }
256         return true;
257     }

294     public void actionPerformed(ActionEvent ae) {
295         final String command = ae.getActionCommand();
296         final FreeColClient fcc = getFreeColClient();
297         final SwingGUI gui = getGUI();
298         try {
299             switch (Integer.parseInt(command)) {
300             case START:
301                 int row = table.getSelectedRow();
302                 int col = table.getSelectedColumn();
303                 if (row > -1 && col > -1){
304                     table.getCellEditor(row, col).stopCellEditing();
305                 }
306                 if (!checkVictoryConditions()) break;
307
308                 // The ready flag was set to false for single player
309                 // mode in order to allow the player to change
310                 // whatever he wants.
311                 if (singlePlayerGame) {
312                     getMyPlayer().setReady(true);
313                 }
314
315                 fcc.getPreGameController().requestLaunch();
316                 break;
317             case CANCEL:
318                 fcc.getConnectController().quitGame(true);
319                 gui.removeFromCanvas(this);
320                 gui.showNewPanel();
321                 break;
322             case READY:
323                 fcc.getPreGameController().setReady(readyBox.isSelected());
324                 refreshPlayersTable();
325                 break;
326             case CHAT:
327                 if (!chat.getText().isEmpty()) {
328                     fcc.getPreGameController().chat(chat.getText());
329                     displayChat(getMyPlayer().getName(), chat.getText(),
330                                 false);
331                     chat.setText("");
332                 }
333                 break;
334             case GAME_OPTIONS:
335                 OptionGroup go = gui.showGameOptionsDialog(fcc.isAdmin(), true);
336                 if (go != null) {
337                     fcc.getGame().setGameOptions(go);
338                     fcc.getPreGameController().updateGameOptions();
339                 }
340                 break;
341             case MAP_GENERATOR_OPTIONS:
342                 OptionGroup mgo = gui.showMapGeneratorOptionsDialog(fcc.isAdmin());
343                 if (mgo != null) {
344                     fcc.getGame().setMapGeneratorOptions(mgo);
345                     fcc.getPreGameController().updateMapGeneratorOptions();
346                 }
347                 break;
348             default:
349                 super.actionPerformed(ae);
350             }
351         } catch (NumberFormatException nfe) {
352             logger.warning("Invalid ActionEvent, not a number: " + command);
353         }
354     }
