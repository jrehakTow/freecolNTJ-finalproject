153     /**
154      * Get a list of the maximum abstract goods that could be loaded
155      * at this stop.  That is, a list of all the cargo goods types
156      * normally with amount equal to one CARGO_SIZE, but with
157      * duplicates removed and amounts accumulated.
158      *
159      * @return A list of <code>AbstractGoods</code> to load.
160      */
161     public List<AbstractGoods> getCompactCargo() {
162         List<AbstractGoods> result = new ArrayList<>();
163         for (GoodsType type : getCargo()) {
164             AbstractGoods ag = AbstractGoods.findByType(type, result);
165             if (ag != null) {
166                 ag.setAmount(ag.getAmount() + GoodsContainer.CARGO_SIZE);
167             } else {
168                 result.add(new AbstractGoods(type, GoodsContainer.CARGO_SIZE));
169             }
170         }
171         return result;
172     }

197     /**
198      * Is there work for a unit to do at this stop?
199      *
200      * @param unit The <code>Unit</code> to test.
201      * @param turns Account for production from this many turns.
202      * @return True if this unit should load or unload cargo at the stop.
203      */
204     public boolean hasWork(Unit unit, int turns) {
205         // Look for goods to load.
206         List<AbstractGoods> stopGoods = getCompactCargo();
207         // There is space on the unit to load some more of this goods
208         // type, so return true if there is some available at the stop.
209         if (any(stopGoods.stream()
210                 .filter(ag -> unit.getGoodsCount(ag.getType()) < ag.getAmount()),
211                 ag -> getExportAmount(ag.getType(), turns) > 0)) return true;
212
213         // Look for goods to unload.
214         if (any(unit.getCompactGoodsList().stream()
215                 .filter(ag -> !AbstractGoods.containsType(ag.getType(), stopGoods)),
216                 ag -> getImportAmount(ag.getType(), turns) > 0)) return true;
217
218         return false;
219     }
