97         boolean ret = false;

99             newLoc = unit.getLocation();
100             if (colony != null) {
101                 newType = unit.getType();
102                 newRole = unit.getRole();
103                 newRoleCount = unit.getRoleCount();
104                 newWork = unit.getWorkType();
105                 newWorkAmount = (newWork == null) ? 0
106                     : getAmount(newLoc, newWork);
107             }
108             newMovesLeft = unit.getMovesLeft();
109         }
110
111         FreeColGameObject oldFcgo = (FreeColGameObject)loc;
112         FreeColGameObject newFcgo = (FreeColGameObject)newLoc;
113         if (loc != newLoc) {
114             oldFcgo.firePropertyChange(change(oldFcgo), unit, null);
115             if (newLoc != null) {
116                 newFcgo.firePropertyChange(change(newFcgo), null, unit);
117             }
118             ret = true;
119         }
120         if (colony != null) {

129                 ret = true;
130             }
131             if (work != newWork) {
132                 if (work != null && oldFcgo != null && workAmount != 0) {

134                 }
135                 if (newWork != null && newFcgo != null && newWorkAmount != 0) {
136                     newFcgo.firePropertyChange(newWork.getId(),
137                                                0, newWorkAmount);
138                 }
139                 ret = true;
140             } else if (workAmount != newWorkAmount) {
141                 newFcgo.firePropertyChange(newWork.getId(),
142                                            workAmount, newWorkAmount);
143                 ret = true;
144             }
145         }
146         if (role != newRole && newRole != null) {
147             unit.firePropertyChange(Unit.ROLE_CHANGE, role, newRole);
148             ret = true;
149         } else if (roleCount != newRoleCount && newRoleCount >= 0) {

151             ret = true;
152         }
153         if (unit.getGoodsContainer() != null) {
154             ret |= unit.getGoodsContainer().fireChanges();
155         }

157             unit.firePropertyChange(Unit.CARGO_CHANGE, null, unit);
158             ret = true;
159         }
160         if (movesLeft != newMovesLeft) {
161             unit.firePropertyChange(Unit.MOVE_CHANGE, movesLeft, newMovesLeft);
162             ret = true;
163         }
164         return ret;
165     }

193     public int compareTo(UnitWas uw) {
194         // Order by decreasing capacity of the location the unit is to
195         // be moved to, so that if we traverse a sorted list of
196         // UnitWas we minimize the chance of a unit being moved to a
197         // full location.
198         //
199         // Unfortunately this also tends to move units that need
200         // equipment first, leading to failures to rearm, so it is
201         // best to make two passes anyway.  See revertAll().  However
202         // we can still try our best by using the amount of equipment
203         // the unit needs as a secondary criterion (favouring the
204         // least equipped).
205         List<Role> roles = this.unit.getAvailableRoles(null);
206         int cmp = ((UnitLocation)uw.loc).getUnitCapacity()
207             - ((UnitLocation)this.loc).getUnitCapacity();
208         if (cmp == 0) {
209             cmp = roles.indexOf(this.role) - roles.indexOf(uw.role);
210         }
211         return cmp;
212     }
