473             } else if (player.owns(tile)) { // Already ours, clear users
474                 Colony colony = (Colony)tile.getOwningSettlement();
475                 if (colony != null) {
476                     logger.warning("Building on colony tile: " + tile);
477                     return lbFail(lb, false, "building on colony tile ", tile);
478                 }
479             } else {
480                 // Not our tile, so claim it first.  Fail if someone
481                 // has claimed the tile and will not sell.  Otherwise
482                 // try to buy it or steal it.
483                 int price = player.getLandPrice(tile);
484                 boolean fail = price < 0;
485                 if (price > 0 && !player.checkGold(price)) {
486                     if (randomInt(logger, "Land gold?", getAIRandom(), 4) == 0) {
487                         lb.add(", ");
488                         euaip.cheatGold(price, lb);
489                         lb.add(" to buy ", tile);
490                     }
491                 }
492                 if (price >= 0) {
493                     fail = !AIMessage.askClaimLand(tile, aiUnit,
494                         ((price == 0) ? 0 : (player.checkGold(price)) ? price
495                             : Constants.STEAL_LAND))
496                         || !player.owns(tile);
497                 }
498                 if (fail) return retargetMission("tile-claim-at-" + tile, lb);
499                 lb.add(", claimed colony tile");
500             }
501
502             // Log the colony values so we can improve things
503             if (logger.isLoggable(Level.FINE)) {
504                 LogBuilder l2 = new LogBuilder(64);
505                 l2.add(tag, " score-at-foundation ", tile, ":");
506                 for (Double d : player.getAllColonyValues(tile)) {
507                     l2.add(" ", d);
508                 }
509                 l2.log(logger, Level.FINE);
510             }
511             
512             // Clear to build the colony.
513             if (AIMessage.askBuildColony(aiUnit, Player.ASSIGN_SETTLEMENT_NAME)
514                 && tile.getColony() != null) {
515                 Colony colony = tile.getColony();
516                 AIColony aiColony = aiMain.getAIColony(colony);
517                 aiColony.requestRearrange();
518                 Mission m = euaip.getWorkInsideColonyMission(aiUnit, aiColony);
519                 return lbDone(lb, m != null, colony);
520             }
521             return lbFail(lb, false, "build at ", tile);
522         }
523     }
