332         /**

334          *
335          * Note that we must copy attackers and defenders because a
336          * successful attacker can move, any an unsuccessful
337          * participant can die, and unsuccessful defenders can be
338          * captured.  Furthermore for defenders, insufficient
339          * information is serialized when a unit is inside a
340          * settlement, but if unscoped too much is disclosed.  So we
341          * make a copy and neuter it.
342          *
343          * We have to remember if the defender was in a settlement
344          * because by the time serialization occurs the settlement
345          * might have been destroyed.

348          * exception to the normal visibility rules.
349          *
350          * @param see The visibility of this change.
351          * @param attacker The <code>Unit</code> that is attacking.
352          * @param defender The <code>Unit</code> that is defending.
353          * @param success Did the attack succeed.
354          */
355         public AttackChange(See see, Unit attacker, Unit defender,
356                             boolean success) {
357             super(see);
358             Game game = attacker.getGame();
359             this.defenderInSettlement = defender.getTile().hasSettlement();
360             this.attacker = attacker.copy(game, Unit.class);
361             this.attacker.setLocationNoUpdate(this.attacker.getTile());
362             this.defender = defender.copy(game, Unit.class);
363             this.defender.setLocationNoUpdate(this.defender.getTile());
364             this.defender.setWorkType(null);
365             this.defender.setState(Unit.UnitState.ACTIVE);            
366             this.success = success;
367         }
