848      * <code>BuildableType</code>.
849      *
850      * @param buildable The <code>BuildableType</code> to build.
851      * @param needed The <code>AbstractGoods</code> needed to continue
852      *     the build.
853      * @return The number of turns to build the buildable (which may
854      *     be zero, UNDEFINED if no useful work is being done, negative
855      *     if some requirement is or will block completion (value is
856      *     the negation of (turns-to-blockage + 1), and if the needed
857      *     argument is supplied it is set to the goods deficit).
858      */
859     public int getTurnsToComplete(BuildableType buildable,
860                                   AbstractGoods needed) {
861         final List<AbstractGoods> required = buildable.getRequiredGoods();
862         int turns = 0, satisfied = 0, failing = 0, underway = 0;
863         
864         ProductionInfo info = productionCache.getProductionInfo(buildQueue);
865         for (AbstractGoods ag : required) {
866             final GoodsType type = ag.getType();
867             final int amountNeeded = ag.getAmount();
868             final int amountAvailable = getGoodsCount(type);
869             if (amountAvailable >= amountNeeded) {
870                 satisfied++;
871                 continue;
872             }
873             int production = productionCache.getNetProductionOf(type);
874             if (info != null) {
875                 AbstractGoods consumption = AbstractGoods.findByType(type,
876                     info.getConsumption());
877                 if (consumption != null) {
878                     // add the amount the build queue itself will consume
879                     production += consumption.getAmount();
880                 }
881             }
882             if (production <= 0) {
883                 failing++;
884                 if (needed != null) {
885                     needed.setType(type);
886                     needed.setAmount(amountNeeded - amountAvailable);
887                 }
888                 continue;
889             }
890
891             underway++;
892             int amountRemaining = amountNeeded - amountAvailable;
893             int eta = amountRemaining / production;
894             if (amountRemaining % production != 0) eta++;
895             turns = Math.max(turns, eta);
896         }
897
898         return (satisfied + underway == required.size()) ? turns // Will finish
899             : (failing == required.size()) ? UNDEFINED // Not even trying
900             : -(turns + 1); // Blocked by something
901     }
