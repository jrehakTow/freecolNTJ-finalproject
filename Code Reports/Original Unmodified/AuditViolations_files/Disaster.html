33 /**
34  * This class describes disasters that can happen to a Colony, such as
35  * flooding, disease or Indian raids.
36  */
37 public class Disaster extends FreeColSpecObjectType {
38
39     /**
40      * Bankruptcy occurs if upkeep is enabled and a player is unable
41      * to pay for the maintenance of all buildings.
42      */
43     public static final String BANKRUPTCY = "model.disaster.bankruptcy";
44
45     /** Whether to apply one, many or all applicable disasters. */
46     public static enum Effects { ONE, SEVERAL, ALL };
47
48     /** Whether this disaster is natural.  Defaults to false. */
49     private boolean natural = false;
50
51     /** The number of effects of this disaster. Defaults to <code>ONE</code>. */
52     private Effects numberOfEffects = Effects.ONE;
53
54     /** The effects of this disaster. */
55     private List<RandomChoice<Effect>> effects = null;
56
57
58     /**
59      * Create a new disaster.
60      *
61      * @param id The object identifier.
62      * @param specification The <code>Specification</code> to refer to.
63      */
64     public Disaster(String id, Specification specification) {
65         super(id, specification);
66     }
67
68
69     /**
70      * Is this a natural disaster?
71      *
72      * @return True if this is a natural disaster.
73      */
74     public final boolean isNatural() {
75         return natural;
76     }
77
78     /**
79      * Get the number of effects.
80      *
81      * @return The <code>Effects</code> to apply.
82      */
83     public final Effects getNumberOfEffects() {
84         return numberOfEffects;
85     }
86
87     /**
88      * Get the random choice list of effects.
89      *
90      * @return A list of random <code>Effect</code> choices.
91      */
92     public final List<RandomChoice<Effect>> getEffects() {
93         return (effects == null)
94             ? Collections.<RandomChoice<Effect>>emptyList()
95             : effects;
96     }
97
98     /**
99      * Add an effect.
100      *
101      * @param effect The <code>Effect</code> to add.
102      */
103     private void addEffect(Effect effect) {
104         if (effects == null) effects = new ArrayList<>();
105         effects.add(new RandomChoice<>(effect, effect.getProbability()));
106     }

111     private static final String EFFECT_TAG = "effect";
112     private static final String EFFECTS_TAG = "effects";
113     private static final String NATURAL_TAG = "natural";

115
116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     protected void writeAttributes(FreeColXMLWriter xw) throws XMLStreamException {
121         super.writeAttributes(xw);
122
123         xw.writeAttribute(NATURAL_TAG, natural);
124
125         xw.writeAttribute(EFFECTS_TAG, numberOfEffects);
126     }

128     /**
129      * {@inheritDoc}
130      */
131     @Override
132     protected void writeChildren(FreeColXMLWriter xw) throws XMLStreamException {
133         super.writeChildren(xw);
134
135         for (RandomChoice<Effect> choice : getEffects()) {
136             choice.getObject().toXML(xw);
137         }
138     }

140     /**
141      * {@inheritDoc}
142      */
143     @Override
144     protected void readAttributes(FreeColXMLReader xr) throws XMLStreamException {
145         super.readAttributes(xr);
146
147         final Specification spec = getSpecification();
148
149         Disaster parent = xr.getType(spec, EXTENDS_TAG, Disaster.class, this);
150
151         natural = xr.getAttribute(NATURAL_TAG, parent.natural);
152
153         numberOfEffects = (xr.hasAttribute(EFFECTS_TAG))
154             ? xr.getAttribute(EFFECTS_TAG, Effects.class, Effects.ONE)
155             : parent.numberOfEffects;
156     }
157
158     /**
159      * {@inheritDoc}
160      */
161     @Override
162     protected void readChildren(FreeColXMLReader xr) throws XMLStreamException {
163         // Clear containers.
164         if (xr.shouldClearContainers()) {
165             effects = null;
166         }
167
168         final Specification spec = getSpecification();
169         Disaster parent = xr.getType(spec, EXTENDS_TAG, Disaster.class, this);
170
171         if (parent != this && !parent.getEffects().isEmpty()) {
172             if (effects == null) effects = new ArrayList<>();
173             for (RandomChoice<Effect> choice : parent.getEffects()) {
174                 Effect effect = new Effect(choice.getObject());
175                 effect.getFeatureContainer().replaceSource(parent, this);
176                 addEffect(effect);
177             }
178         }
179
180         super.readChildren(xr);
181     }
182
183     /**
184      * {@inheritDoc}
185      */
186     @Override
187     protected void readChild(FreeColXMLReader xr) throws XMLStreamException {
188         final Specification spec = getSpecification();
189         final String tag = xr.getLocalName();
190
191         if (EFFECT_TAG.equals(tag)) {
192             Effect effect = new Effect(xr, spec);
193             effect.getFeatureContainer().replaceSource(null, this);
194             addEffect(effect);
195
196         } else {
197             super.readChild(xr);
198         }
199     }
