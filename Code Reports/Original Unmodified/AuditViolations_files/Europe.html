60     /**
61      * Migration handling.
62      *
63      * Migration routines operate on:
64      * - "indexes" which refer to a valid member of the recruitables
65      *   list, and must be [0, RECRUIT_COUNT)
66      * - "slots", where slot zero means "pick a random migrant" and the
67      *   other slots in [1, RECRUIT_COUNT] refer to the index == slot-1
68      * The following constant should be used when the random choice
69      * behaviour is desired.
70      */
71     public static enum MigrationType {
72         NORMAL,     // Unit decided to migrate
73         RECRUIT,    // Player is paying
74         FOUNTAIN,   // As a result of a Fountain of Youth discovery
75         SURVIVAL;   // Emergency autorecruit in server

218     /**
219      * Gets the current price for a recruit.
220      *
221      * @return The current price of the recruit in this <code>Europe</code>.
222      */
223     public int getRecruitPrice() {
224         if (!owner.isColonial()) return -1;
225         int required = owner.getImmigrationRequired();
226         int immigration = owner.getImmigration();
227         int difference = Math.max(required - immigration, 0);
228         return Math.max((recruitPrice * difference) / required,
229                         recruitLowerCap);

232     /**
233      * Get any immigration produced in Europe.
234      *
235      * Col1 penalizes immigration by -4 per unit in Europe per turn,
236      * but there is a +2 player bonus, which we might as well add
237      * here.  Total immigration per turn can not be negative, but that
238      * is handled in ServerPlayer.
239      *
240      * @param production The current total colony production.
241      * @return Immigration produced this turn in Europe.
242      */
243     public int getImmigration(int production) {
244         final Specification spec = getSpecification();
245         int n = count(getUnitList(), u -> u.isPerson());
246         n *= spec.getInteger(GameOptions.EUROPEAN_UNIT_IMMIGRATION_PENALTY);
247         n += spec.getInteger(GameOptions.PLAYER_IMMIGRATION_BONUS);
248         // Do not allow total production to be negative.
249         if (n + production < 0) n = -production;
250         return n;
251     }

355         Player player = getOwner();
356         Market market = player.getMarket();
357         int price = 0;
358         for (AbstractGoods ag : goods) {
359             if (ag.getAmount() <= 0) continue;
360             GoodsType goodsType = ag.getType();
361             // Refuse to trade in boycotted goods
362             if (!player.canTrade(goodsType)) {
363                 price = -1;
364                 break;
365             }
366             price += market.getBidPrice(goodsType, ag.getAmount());
367         }
368         return price;
369     }
