96     /**
97      * Converts the data from the given input stream to an 
98      * uncompressed and indented text to the output stream.
99      * Both streams are closed by this method.
100      * 
101      * <br><br>
102      * 
103      * Savegame compression is automatically detected, so using
104      * this method on an uncompressed savegame creates an
105      * indented version of that savegame.
106      * 
107      * @param in The input stream.
108      * @param out The output stream.
109      * 
110      * @throws IOException if thrown while reading or writing the streams. 
111      */
112     public void convertToXML(InputStream in, OutputStream out) throws IOException {
113         try {
114             in = new BufferedInputStream(in);
115             out = new BufferedOutputStream(out);
116             
117             // Automatically detect savegame compression:
118             in.mark(10);
119             byte[] buf = new byte[5];
120             in.read(buf, 0, 5);
121             in.reset();
122             if (!"<?xml".equals(new String(buf, "UTF-8"))) {
123                 in =  new BufferedInputStream(new GZIPInputStream(in));
124             }
125
126             // Support for XML comments has not been added:
127             int indent = 0;
128             int i;      
129             while ((i = in.read()) != -1) {
130                 char c = (char) i;
131                 if (c == '<') {
132                     i = in.read();
133                     char b = (char) i;
134                     if (b == '/') {
135                         indent -= 4;
136                     }
137                     for (int h=0; h<indent; h++) {
138                         out.write(' ');
139                     }
140                     out.write(c);
141                     if (b != '\n' && b != '\r') {
142                         out.write(b);
143                     }
144                     if (b != '/' && b != '?') {
145                         indent += 4;
146                     }
147                 } else if (c == '/') {
148                     out.write(c);
149                     i = in.read();
150                     c = (char) i;
151                     if (c == '>') {
152                         indent -= 4;
153                         out.write(c);
154                         out.write('\n');
155                     }
156                 } else if (c == '>') {
157                     out.write(c);
158                     out.write('\n');
159                 } else if (c != '\n' && c != '\r') {
160                     out.write(c);
161                 }           
162             }
163
164         } finally {
165             in.close();
166             out.close();
167         }
168     }
