71      * Choose the number of columns for the OptionPane buttons.
72      *
73      * @param nButtons The number of buttons.
74      * @return A suitable number of columns.
75      */
76     private int getColumns(int nButtons) {
77         return (nButtons > 21) ? 4
78             :  ((nButtons % 4) == 0 && nButtons > 12) ? 4
79             :  ((nButtons % 3) == 0 && nButtons > 6)  ? 3
80             :  ((nButtons % 2) == 0 && nButtons > 4)  ? 2
81             :  (nButtons > 5)  ? 2
82             :  1;
83     }

85     /**
86      * Prepare the new buttons for this component.
87      *
88      * @param buttons An array of objects provided to define the buttons.
89      */
90     private void prepareButtons(Object[] buttons) {
91         if (this.newButtons != null) return;
92
93         final String okLabel = Messages.message("ok");
94         final String cancelLabel = Messages.message("cancel");
95         this.newButtons = new JButton[buttons.length];
96
97         int maxWidth = 0, maxHeight = 0;
98         for (int i = 0; i < buttons.length; i++) {
99             JButton b;
100             if (buttons[i] instanceof ChoiceItem) {
101                 ChoiceItem ci = (ChoiceItem)buttons[i];
102                 String label = ci.toString();
103                 Icon icon = ci.getIcon();
104                 b = (icon == null) ? new JButton(label)
105                     : (label == null || label.isEmpty()) ? new JButton(icon)
106                     : new JButton(label, icon);
107                 b.setName("OptionPane.button." + label);
108                 if (ci.isOK()) this.okIndex = i;
109                 else if (ci.isCancel()) this.cancelIndex = i;
110             } else if (buttons[i] instanceof Icon) {
111                 b = new JButton((Icon)buttons[i]);
112                 b.setName("OptionPane.button.withIcon");
113             } else {
114                 String label = buttons[i].toString();
115                 b = new JButton(label);
116                 b.setName("OptionPane.button." + label);
117                 if (okLabel.equals(label)) this.okIndex = i;
118                 else if (cancelLabel.equals(label)) this.cancelIndex = i;
119             }
120             maxWidth = Math.max(maxWidth, b.getMinimumSize().width);
121             maxHeight = Math.max(maxHeight, b.getMinimumSize().height);
122             ActionListener buttonListener = createButtonActionListener(i);
123             if (buttonListener != null) b.addActionListener(buttonListener);
124             this.newButtons[i] = b;
125         }
126         if (maxWidth > 0) {
127             Dimension dimension = new Dimension(maxWidth, maxHeight);
128             for (int i = 0; i < buttons.length; i++) {
129                 if (buttons[i] instanceof Icon) continue;
130                 newButtons[i].setPreferredSize(dimension);
131                 newButtons[i].setMinimumSize(dimension);
132             }
133         }
134     }

147         JPanel bottom;

161     /**
162      * {@inheritDoc}
163      */
164     @Override
165     protected void addButtonComponents(Container container, Object[] buttons,
166                                        int initialIndex) {
167         if (buttons == null) return;
168
169         if (0 <= initialIndex && initialIndex < buttons.length) {
170             JButton b = newButtons[initialIndex];
171             this.initialFocusComponent = b;
172             b.addHierarchyListener((HierarchyEvent e) -> {
173                     if ((e.getChangeFlags() & HierarchyEvent.PARENT_CHANGED) != 0) {
174                         JButton button = (JButton)e.getComponent();
175                         JRootPane root = SwingUtilities.getRootPane(button);
176                         if (root != null) root.setDefaultButton(button);
177                     }
178                 });
179         }
180
181         if (okIndex >= 0) {
182             // If OK is present this is a confirm-dialog.  Put everything
183             // in the same span especially the Cancel.
184             if (cancelIndex >= 0) {
185                 container.add(newButtons[cancelIndex], "tag cancel");
186             }
187             container.add(newButtons[okIndex], "tag ok");
188             for (int i = 0; i < buttons.length; i++) {
189                 if (i == okIndex || i == cancelIndex) continue;
190                 container.add(newButtons[i]);
191             }
192         } else {
193             // This must be a choice dialog.  The wrap argument to the
194             // MigLayout constructor will do the work for us.
195             for (int i = 0; i < buttons.length; i++) {
196                 if (i == cancelIndex) continue;
197                 container.add(newButtons[i]);
198             }
199             if (cancelIndex >= 0) {
200                 container.add(newButtons[cancelIndex],
201                               "newline 20, tag cancel");
202             }
203         }
204         // The righthand button gets truncated, so add some extra space.
205         Dimension prefer = container.getPreferredSize();
206         prefer = new Dimension((int)(prefer.getWidth() + 100),
207                                (int)prefer.getHeight());
208         container.setMinimumSize(prefer);
209         container.setPreferredSize(prefer);
210     }
