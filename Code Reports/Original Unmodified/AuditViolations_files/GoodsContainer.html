167     /**
168      * Adds goods by type and amount to this goods container.
169      *
170      * Note: negative amounts are allowed.
171      *
172      * @param type The <code>GoodsType</code> to add.
173      * @param amount The amount of goods to add.
174      * @return True if the addition succeeds.
175      */
176     public boolean addGoods(GoodsType type, int amount) {
177         int oldAmount = getGoodsCount(type);
178         int newAmount = oldAmount + amount;
179
180         if (newAmount < 0) {
181             throw new IllegalStateException("Operation would leave "
182                 + newAmount + " goods of type " + type
183                 + " in Location " + parent);
184         } else if (newAmount == 0) {
185             synchronized (this.storedGoods) {
186                 this.storedGoods.remove(type);
187             }
188         } else {
189             synchronized (this.storedGoods) {
190                 this.storedGoods.put(type, newAmount);
191             }
192         }
193         return true;
194     }

216     /**
217      * Removes goods by type and amount from this goods container.
218      *
219      * @param type The <code>GoodsType</code> to remove.
220      * @param amount The amount of goods to remove.
221      * @return The <code>Goods</code> actually removed, which may have a
222      *     lower actual amount, or null if nothing removed.
223      */
224     public Goods removeGoods(GoodsType type, int amount) {
225         int oldAmount = getGoodsCount(type);
226         if (oldAmount <= 0) return null;
227
228         int newAmount = oldAmount - amount;
229         Goods removedGoods;
230         if (newAmount > 0) {
231             removedGoods = new Goods(getGame(), null, type, amount);
232             synchronized (this.storedGoods) {
233                 this.storedGoods.put(type, newAmount);
234             }
235         } else {
236             removedGoods = new Goods(getGame(), null, type, oldAmount);
237             synchronized (this.storedGoods) {
238                 this.storedGoods.remove(type);
239             }
240         }
241         return removedGoods;
242     }

352         List<Goods> totalGoods = new ArrayList<>();
353         synchronized (this.storedGoods) {
354             for (GoodsType goodsType : this.storedGoods.keySet()) {
355                 int amount = this.storedGoods.get(goodsType);
356                 while (amount > 0) {
357                     totalGoods.add(new Goods(getGame(), parent, goodsType,
358                             ((amount >= CARGO_SIZE) ? CARGO_SIZE : amount)));
359                     amount -= CARGO_SIZE;
360                 }
361             }
362         }

496     /**
497      * Write a storage container to a stream.
498      *
499      * @param xw The <code>FreeColXMLWriter</code> to write to.
500      * @param tag The element tag.
501      * @param storage The storage container.
502      * @exception XMLStreamException if there is a problem writing to
503      *     the stream.
504      */
505     private void writeStorage(FreeColXMLWriter xw, String tag,
506                               Map<GoodsType, Integer> storage) throws XMLStreamException {
507         if (storage.isEmpty()) return;
508
509         xw.writeStartElement(tag);
510
511         for (GoodsType goodsType : toSortedList(storage.keySet())) {
512

514

560      *
561      * @param xr The <code>FreeColXMLReader</code> to read from.
562      * @param storage The storage container.
563      * @exception XMLStreamException if there is a problem reading from
564      *     the stream.
565      */
566     private void readStorage(FreeColXMLReader xr,
567         Map<GoodsType, Integer> storage) throws XMLStreamException {
568         final Specification spec = getGame().getSpecification();
569
570         while (xr.nextTag() != XMLStreamConstants.END_ELEMENT) {
571             String tag = xr.getLocalName();
572
573             if (Goods.getTagName().equals(tag)) {
574                 GoodsType goodsType = xr.getType(spec, TYPE_TAG,
575                     GoodsType.class, (GoodsType)null);
576
577                 int amount = xr.getAttribute(AMOUNT_TAG, 0);
578
579                 storage.put(goodsType, amount);
580
581             } else {
582                 throw new XMLStreamException("Bogus GoodsContainer tag: "
583                     + tag);
584             }
585             xr.closeTag(tag);
586         }
