521         if (this.mode != newMode) {
522             switch (this.mode = newMode) {
523             case END:
524                 this.mapEditorPanel.setVisible(false);
525                 this.tileInfoPanel.setVisible(false);
526                 this.unitInfoPanel.setVisible(false);
527                 this.endTurnPanel.setVisible(true);
528                 break;
529             case MAP:
530                 this.endTurnPanel.setVisible(false);
531                 this.tileInfoPanel.setVisible(false);
532                 this.unitInfoPanel.setVisible(false);
533                 this.mapEditorPanel.setVisible(true);
534                 break;
535             case TILE:
536                 this.endTurnPanel.setVisible(false);
537                 this.mapEditorPanel.setVisible(false);
538                 this.unitInfoPanel.setVisible(false);
539                 this.tileInfoPanel.setVisible(true);
540                 break;
541             case UNIT:
542                 this.endTurnPanel.setVisible(false);
543                 this.mapEditorPanel.setVisible(false);
544                 this.tileInfoPanel.setVisible(false);
545                 this.unitInfoPanel.setVisible(true);
546                 break;
547             case NONE: default:
548                 this.endTurnPanel.setVisible(false);
549                 this.mapEditorPanel.setVisible(false);
550                 this.tileInfoPanel.setVisible(false);
551                 this.unitInfoPanel.setVisible(false);
552                 break;
553             }
554         }
