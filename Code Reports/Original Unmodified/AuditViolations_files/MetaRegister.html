32 /**
33  * The <code>MetaRegister</code> stores information about running servers.
34  * Each server has it's own {@link MetaItem} object.
35  */
36 public final class MetaRegister {
37
38     private static final Logger logger = Logger.getLogger(MetaRegister.class.getName());
39
40     /** The current list of servers. */
41     private final ArrayList<MetaItem> items = new ArrayList<>();
42     
43     
44     /**
45      * Gets the server entry with the diven address and port.
46      *
47      * @param address The IP-address of the server.
48      * @param port The port number of the server.
49      * @return The server entry or <code>null</code> if the given
50      *     entry could not be found.
51      */
52     private MetaItem getItem(String address, int port) {
53         int index = indexOf(address, port);
54         if (index >= 0) {
55             return items.get(index);
56         } else {
57             return null;
58         }
59     }
60
61     /**
62      * Gets the index of the server entry with the diven address and port.
63      *
64      * @param address The IP-address of the server.
65      * @param port The port number of the server.
66      * @return The index or <code>-1</code> if the given entry could
67      *     not be found.
68      */
69     private int indexOf(String address, int port) {
70         for (int i = 0; i < items.size(); i++) {
71             if (address.equals(items.get(i).getAddress())
72                 && port == items.get(i).getPort()) {
73                 return i;
74             }
75         }
76         return -1;
77     }

79     /**
80      * Updates a given <code>MetaItem</code>.
81      *
82      * @param mi The <code>MetaItem</code> that should be updated.
83      * @param name The name of the server.
84      * @param address The IP-address of the server.
85      * @param port The port number in which clients may connect.
86      * @param slotsAvailable Number of players that may conncet.
87      * @param currentlyPlaying Number of players that are currently connected.
88      * @param isGameStarted <i>true</i> if the game has started.
89      * @param version The version of the server.
90      * @param gameState The current state of the game:
91      *     {@link net.sf.freecol.server.FreeColServer.GameState#STARTING_GAME},
92      *     {@link net.sf.freecol.server.FreeColServer.GameState#IN_GAME} or
93      *     {@link net.sf.freecol.server.FreeColServer.GameState#ENDING_GAME}.
94      */
95     private void updateServer(MetaItem mi, String name, String address,
96                               int port, int slotsAvailable,
97                               int currentlyPlaying, boolean isGameStarted,
98                               String version, int gameState) {
99         mi.update(name, address, port, slotsAvailable, currentlyPlaying,
100                   isGameStarted, version, gameState);
101         logger.info("Server updated:" + mi.toString());
102     }

107     /**
108      * Adds a new server with the given attributes.
109      *
110      * @param name The name of the server.
111      * @param address The IP-address of the server.
112      * @param port The port number in which clients may connect.
113      * @param slotsAvailable Number of players that may conncet.
114      * @param currentlyPlaying Number of players that are currently connected.
115      * @param isGameStarted <i>true</i> if the game has started.
116      * @param version The version of the server.
117      * @param gameState The current state of the game.
118      * @exception IOException if the connection fails.
119      */
120     public synchronized void addServer(String name, String address, int port,
121                                        int slotsAvailable, int currentlyPlaying,
122                                        boolean isGameStarted, String version,
123                                        int gameState) throws IOException {
124         MetaItem mi = getItem(address, port);
125         if (mi == null) { // Check connection before adding the server:
126             try (
127                 Connection mc = new Connection(address, port, null,
128                                                FreeCol.METASERVER_THREAD);
129             ) {
130                 mc.disconnect();
131             } catch (IOException ioe) {
132                 logger.log(Level.WARNING, "Server rejected disconnect.", ioe);
133                 throw ioe;
134             }
135             mi = new MetaItem();
136             mi.update(name, address, port, slotsAvailable, currentlyPlaying,
137                       isGameStarted, version, gameState);
138             items.add(mi);
139             logger.info("Server added:" + address + ":" + port);
140         } else {
141             updateServer(mi, name, address, port, slotsAvailable,
142                 currentlyPlaying, isGameStarted, version, gameState);
143         }
144     }

155     /**
156      * Removes servers that have not sent an update for some time.
157      */
158     public synchronized void removeDeadServers() {
159         logger.info("Removing dead servers.");
160
161         long time = System.currentTimeMillis() - MetaServer.REMOVE_OLDER_THAN;
162         for (int i=0; i<items.size(); i++) {
163             if (items.get(i).getLastUpdated() < time) {
164                 logger.info("Removing: " + items.get(i));
165                 items.remove(i);
166             }
167         }
168     }

179             items.remove(index);

181         } else {

183         }
