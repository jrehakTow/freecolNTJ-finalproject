459         String reason = invalidMissionReason(aiUnit);
460         return (reason != null)
461             ? reason
462             : (loc instanceof Tile)
463             ? invalidTileReason(aiUnit, (Tile)loc)
464             : (loc instanceof Colony)
465             ? ((aiUnit.getUnit().getLocation() instanceof Tile)
466                 ? invalidColonyReason(aiUnit, (Colony)loc)
467                 : invalidTargetReason(loc, aiUnit.getUnit().getOwner()))
468             : Mission.TARGETINVALID;
469     }

543     public Mission doMission(LogBuilder lb) {
544         lb.add(tag);
545         final AIUnit aiUnit = getAIUnit();
546
547         // Check for completion and tileImprovement failure up front.
548         if (tileImprovementPlan != null) {
549             if (tileImprovementPlan.isComplete()) {
550                 lbDone(lb, true, tileImprovementPlan.getType(),
551                        " at ", getTarget());
552                 setTarget(null);
553             } else if (!tileImprovementPlan.validate()) {
554                 lbFail(lb, true, "abandoned invalid plan at ",
555                        getTarget(), "/", tileImprovementPlan);
556                 setTarget(null);
557             }
558         }
559
560         /*if (retarget) {
561             newTarget = findTarget(aiUnit, 10, unit.isInEurope());
562             if (newTarget == null) {

564                 return lbFail(lb, false, " found no target");
565             }
566             setTarget(newTarget);
567             lb.add(", retargeting ", newTarget, "/", tileImprovementPlan);
568         }*/
569
570         String reason = invalidReason();
571         if (isTargetReason(reason)) {
572             return retargetMission(reason, lb);
573         } else if (reason != null) {
574             return lbFail(lb, false, reason);
575         }
576
577         final Unit unit = getUnit();
578         final Player player = unit.getOwner();
579         final EuropeanAIPlayer aiPlayer = getEuropeanAIPlayer();
580         final CostDecider costDecider
581             = CostDeciders.avoidSettlementsAndBlockingUnits();
582         Location newTarget;
583
584         if (!hasTools()) { // Get tools first.
585             // Go there and clear target on arrival.
586             Unit.MoveType mt = travelToTarget(getTarget(), costDecider, lb);
587             switch (mt) {
588             case MOVE: // Arrived
589                 break;
590
591             case MOVE_HIGH_SEAS: case MOVE_NO_MOVES: case MOVE_ILLEGAL:
592                 return lbWait(lb);

594             case MOVE_NO_REPAIR:

597             case MOVE_NO_ACCESS_EMBARK: case MOVE_NO_TILE:
598                 return this;
599
600             default:
601                 return lbMove(lb, mt);
602             }
603
604             // Try to equip
605             lbAt(lb);
606             if (aiUnit.equipForRole(getSpecification().getPioneerRole())
607                 && hasTools()) {
608                 lb.add(", equips");
609                 newTarget = findTarget(aiUnit, 10, false);
610                 if (newTarget == null) {
611                     return lbFail(lb, false, "no pioneering target");
612                 }
613             } else {
614                 lb.add(", fails to equip");
615                 newTarget = findTarget(aiUnit, 10, false);
616                 if (newTarget == null
617                     || Map.isSameLocation(newTarget, getTarget())) {
618                     return lbFail(lb, false, "no tools target");
619                 }
620             }
621             setTarget(newTarget);
622             return lbRetarget(lb);
623         }
624
625         // Going to an intermediate colony?
626         if (getTarget() instanceof Colony
627             && invalidTargetReason(getTarget(), player) == null) {
628             Unit.MoveType mt = travelToTarget(getTarget(), costDecider, lb);
629             switch (mt) {
630             case MOVE:
631                 break;
632
633             case MOVE_HIGH_SEAS:
634                 return lbWait(lb);
635
636             case MOVE_NO_REPAIR:
637                 return lbFail(lb, false, AIUNITDIED);
638
639             case MOVE_NO_MOVES: case MOVE_NO_TILE: case MOVE_ILLEGAL:
640                 return this;
641
642             default:
643                 return lbMove(lb, mt);
644             }
645
646             lbAt(lb);
647             newTarget = findTarget(aiUnit, 10, false);
648             if (newTarget == null) {
649                 return lbFail(lb, false, "found no pioneering target");
650             }
651             setTarget(newTarget);
652             return lbRetarget(lb);
653         }
654
655         // Check for threats.
656         // The code below is very conservative.  When enabled it
657         // reduces the number of completed improvements by a factor of
658         // 4 -- 5, which is unacceptable.  Therefore, disabled for
659         // now.  FIXME: do something better.
660         /*
661         int turnsNeeded = DEFAULT_THREAT_TURNS;
662         if (unit.getWorkImprovement() != null) {

664         }
665         if (unit.isInDanger(turnsNeeded, 0.25f)) {
666             if (unit.getTile().getColony() != null) {

678
679         // Going to a tile to perform an improvement.
680         Tile tile = getTarget().getTile();
681         for (;;) {
682             Unit.MoveType mt = travelToTarget(getTarget(), costDecider, lb);
683             switch (mt) {
684             case MOVE: // Arrived
685                 break;
686                 
687             case MOVE_HIGH_SEAS:
688                 return lbWait(lb);
689                 
690             case MOVE_NO_REPAIR:
691                 return lbFail(lb, false, AIUNITDIED);
692
693             case MOVE_NO_MOVES: case MOVE_NO_TILE: case MOVE_ILLEGAL:
694                 return this;
695                 
696             case MOVE_NO_ATTACK_CIVILIAN:
697                 // Might be a temporary blockage due to an occupying
698                 // unit at the target.  Move randomly and retry if
699                 // adjacent.
700                 Direction d = unit.getTile().getDirection(tile);
701                 if (d != null) moveRandomly(tag, d);
702                 continue;
703
704             default:
705                 return lbMove(lb, mt);
706             }
707             break;
708         }
709
710         // Take control of the land before proceeding to improve.
711         lbAt(lb);
712         if (!player.owns(tile)) {
713             // FIXME: Better choice whether to pay or steal.
714             // Currently always pay if we can, steal if we can not.
715             boolean fail;
716             int price = player.getLandPrice(tile);
717             if (price < 0) {
718                 fail = true;
719             } else {
720                 if (price > 0 && !player.checkGold(price)) {
721                     price = Constants.STEAL_LAND;
722                 }
723                 fail = !AIMessage.askClaimLand(tile, aiUnit, price)
724                     || !player.owns(tile); // Failed to claim ownership
725             }
726             if (fail) {
727                 aiPlayer.removeTileImprovementPlan(tileImprovementPlan);
728                 tileImprovementPlan.dispose();
729                 lb.add(", land claim failed at ", tile);
730                 if ((newTarget = findTarget(aiUnit, 10, false)) == null) {
731                     return lbFail(lb, false, "no alternate target");
732                 }
733                 setTarget(newTarget);
734                 return lbRetarget(lb);
735             }
736         }
737
738         // Work on the improvement
739         if (unit.getState() == UnitState.IMPROVING) {
740             return lbWait(lb, ", improving ", tileImprovementPlan);
741         } else if (unit.checkSetState(UnitState.IMPROVING)) {
742             aiPlayer.removeTileImprovementPlan(tileImprovementPlan);
743             if (AIMessage.askChangeWorkImprovementType(aiUnit,
744                     tileImprovementPlan.getType())) {
745                 return lbWait(lb, ", began ", tileImprovementPlan);
746             } else {
747                 tileImprovementPlan.dispose();
748                 return lbFail(lb, false, "to change work type at ", tile);
749             }
750         }
751         // Probably just out of moves.
752         return lbWait(lb, ", waiting to improve at ", tile);
753     }
