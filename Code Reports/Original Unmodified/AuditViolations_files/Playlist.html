26 /**
27  * Represent a set of sounds that will be presented to a SoundPlayer in a

29  */

32     /**
33     * This const represents a way in which successive sounds can be played.
34     */
35     public static final int PLAY_ALL = 0,   // play all sounds in the Playlist once
36                             REPEAT_ALL = 1, // play all sounds in the Playlist until the end of times
37                             PLAY_ONE = 2,   // play one sound in the Playlist once
38                             REPEAT_ONE = 3; // play one sound in the Playlist until the end of times
39
40     /**
41     * This const represents a way in which a sound will be picked from the list.
42     */
43     public static final int FORWARDS = 0,   // order of the sounds is determined by order in Playlist
44                             BACKWARDS = 1,  // order of the sounds is determined by reverse order in Playlist
45                             SHUFFLE = 2;    // order of the sounds is randomly determined (each sound has the same chance of being picked, but eventually all sounds will be played as much as any other sound)
46

48     private int num; // '-1' means we haven't started yet, '-2' means this Playlist is exhausted
49     private int repeatMode;
50     private int pickMode;
51     private final int[] playedSounds;

75     * @param soundFiles The sounds that will make up this Playlist. The order of the sounds may be important.
76     * @param repeatMode Determines how, which and if songs will be repeated. Should be one of {PLAY_ALL, REPEAT_ALL, PLAY_ONE, REPEAT_ONE}.
77     * @param pickMode The way in which sounds will be picked from the list. Should be one of {FORWARDS, BACKWARDS, SHUFFLE}.
78     */
79     public Playlist(File[] soundFiles, int repeatMode, int pickMode) {
80         if (soundFiles.length == 0) {
81             throw new IllegalArgumentException("It's not possible to create an empty Playlist.");
82         }
83
84         this.soundFiles = soundFiles;
85         this.repeatMode = repeatMode;
86         this.pickMode = pickMode;
87         num = -1;
88
89         if (pickMode == SHUFFLE) {
90             playedSounds = new int[soundFiles.length];
91             for (int i = 0; i < playedSounds.length; i++) {
92                 playedSounds[i] = Integer.MAX_VALUE;
93             }
94         }
95         else {
96             playedSounds = null;
97         }
98     }

125     * Returns the next sound file on this Playlist or null if there is no such sound.
126     * @return The next sound file on this Playlist or null if there is no such sound.
127     */
128     public File next() {
129         if (num == -2) {
130             // Playlist exhausted.
131         }
132         else if (num == -1) {
133             // Let's pick a first sound.
134             if (pickMode == FORWARDS) {
135                 num = 0;
136             }
137             else if (pickMode == BACKWARDS) {
138                 num = soundFiles.length - 1;
139             }
140             else { // SHUFFLE mode
141                 num = (int)(Math.random() * soundFiles.length);
142                 playedSounds[0] = num;
143             }
144         }
145         else if (repeatMode == PLAY_ONE) {
146             num = -2;
147         }
148         else if (repeatMode == REPEAT_ONE) {
149             // num stays the same
150         }
151         else if (pickMode == SHUFFLE) {
152             if (playedSounds[playedSounds.length - 1] != Integer.MAX_VALUE) {
153                 if (repeatMode == PLAY_ALL) {
154                     num = -2;
155                 }
156                 else { // (repeatMode == REPEAT_ALL)
157                     for (int i = 1; i < playedSounds.length; i++) {
158                         playedSounds[i] = Integer.MAX_VALUE;
159                     }
160                     num = (int)(Math.random() * soundFiles.length);
161                     playedSounds[0] = num;
162                 }
163             }
164             else {
165                 int i = 0;
166                 for (; i < playedSounds.length; i++) {
167                     if (playedSounds[i] == Integer.MAX_VALUE) {
168                         break;
169                     }
170                 }
171
172                 int tmp = (int)(Math.random() * (soundFiles.length - i));
173                 for (int j = 0; j < i; j++) {
174                     if (tmp < playedSounds[j]) {
175                         num = tmp;
176                         break;
177                     }
178                     else {
179                         tmp++;
180                     }
181                 }
182
183                 playedSounds[i] = num;
184                 Arrays.sort(playedSounds);
185             }
186         } else {
187             switch (repeatMode) {
188             case PLAY_ALL:
189                 if (pickMode == FORWARDS) {
190                     num++;
191                     if (num == soundFiles.length) {
192                         num = -2;
193                     }
194                 } else { // (pickMode == BACKWARDS)
195                     num--;
196                     if (num == -1) {
197                         num = -2;
198                     }
199                 }
200                 break;
201             case REPEAT_ALL:
202                 if (pickMode == FORWARDS) {
203                     num++;
204                     if (num == soundFiles.length) {
205                         num = 0;
206                     }
207                 } else { // (pickMode == BACKWARDS)
208                     num--;
209                     if (num == -1) {
210                         num = soundFiles.length - 1;
211                     }
212                 }
213                 break;
214             default:
215                 break;
216             }
217         }
218         
219         if ((num >= 0) && (num < soundFiles.length)) {
220             return soundFiles[num];
221         } else {
222             return null;
223         }
224     }

228     /**
229     * Returns <i>false</i> if this Playlist is exhausted, <i>true</i> otherwise.
230     * @return <i>false</i> if this Playlist is exhausted, <i>true</i> otherwise.
231     */
232     public boolean hasNext() {
233         if (repeatMode == PLAY_ALL) {
234             if (num == -1) {
235                 return true;
236             }
237             else if (num == -2) {
238                 return false;
239             }
240             else {
241                 if (((pickMode == FORWARDS) && (num == soundFiles.length - 1))
242                         || ((pickMode == BACKWARDS) && (num == 0))
243                         || ((pickMode == SHUFFLE) && (playedSounds[playedSounds.length - 1] != Integer.MAX_VALUE))) {
244                     return false;
245                 }
246                 else {
247                     return true;
248                 }
249             }
250         }
251         else if (repeatMode == PLAY_ONE) {
252             if (num == -1) {
253                 return true;
254             }
255             else {
256                 return false;
257             }
258         }
259         else {
260             return true;
261         }
262     }
