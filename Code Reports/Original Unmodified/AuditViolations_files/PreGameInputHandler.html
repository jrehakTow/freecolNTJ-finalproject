163     private Element login(Element element) {
164         final FreeColClient fcc = getFreeColClient();
165         final LoginMessage message = new LoginMessage(new Game(), element);
166         Game game = message.getGame();
167         fcc.setGame(game);
168
169         fcc.setSinglePlayer(message.isSinglePlayer());
170
171         final String user = message.getUserName();
172         Player player = game.getPlayerByName(user);
173         if (player == null) {
174             logger.warning("New game does not contain player: " + user);
175             StringTemplate st = StringTemplate.template("server.noSuchPlayer")
176                 .addName("%player%", user);
177             getGUI().showErrorMessage(st);
178             return null;
179         }
180         fcc.setMyPlayer(player);
181         fcc.addSpecificationActions(game.getSpecification());
182         logger.info("FreeColClient logged in as " + user
183             + "/" + player.getId());
184
185         final boolean currentPlayer = message.isCurrentPlayer();
186         if (currentPlayer) game.setCurrentPlayer(player);
187
188         if (message.getStartGame()) fcc.getPreGameController().startGame();
189         return null;
190     }

284      *
285      * Wait until map is received from server, sometimes this

293     private Element startGame(@SuppressWarnings("unused") Element element) {
294         new Thread(FreeCol.CLIENT_THREAD + "Starting game") {
295                 @Override
296                 public void run() {
297                     for (;;) {
298                         Game game = getGame();
299                         if (game != null && game.getMap() != null) break;
300                         try {
301                             Thread.sleep(200);
302                         } catch (Exception ex) {}
303                     }
304                     
305                     SwingUtilities.invokeLater(() -> {
306                             getFreeColClient().getPreGameController()
307                                 .startGame();
308                         });
309                 }
310             }.start();
311         return null;
312     }
