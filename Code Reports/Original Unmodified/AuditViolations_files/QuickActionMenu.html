296     private boolean addWorkItems(final UnitLabel unitLabel) {
297         final Unit unit = unitLabel.getUnit();
298         if (unit.isCarrier()) return false;
299
300         final UnitType unitType = unit.getType();
301         final GoodsType expertGoods = unitType.getExpertProduction();
302         final Colony colony = unit.getLocation().getColony();
303         final Specification spec = freeColClient.getGame().getSpecification();
304         final WorkLocation current = unit.getWorkLocation();
305         final int bonusChange = (current != null) ? 0
306             : colony.governmentChange(colony.getUnitCount() + 1);
307         final int bonus = colony.getProductionBonus();
308
309         Map<JMenuItem, Integer> items = new HashMap<>();
310         Map<JMenuItem, Integer> extras = new HashMap<>();
311         JMenuItem expertOwned = null;
312         JMenuItem expertUnowned = null;
313         for (GoodsType type : spec.getGoodsTypeList()) {
314             int bestOwnedProd = bonus + bonusChange,
315                 bestUnownedProd = bonus + bonusChange;
316             WorkLocation bestOwned = null, bestUnowned = null;
317             for (WorkLocation wl : colony.getAllWorkLocations()) {
318                 int prod = 0;
319                 switch (wl.getNoAddReason(unit)) {
320                 case NONE:
321                     prod = wl.getPotentialProduction(type, unitType);
322                     if (bestOwnedProd < prod) {
323                         bestOwnedProd = prod;
324                         bestOwned = wl;
325                     }
326                     break;
327                 case ALREADY_PRESENT:
328                     prod = wl.getPotentialProduction(type, unitType);
329                     if (bestOwnedProd < prod) {
330                         bestOwnedProd = prod;
331                         bestOwned = (unit.getWorkType() == type) ? null : wl;
332                     }
333                     break;
334                 case CLAIM_REQUIRED:
335                     prod = wl.getPotentialProduction(type, unitType);
336                     if (bestUnownedProd < prod) {
337                         bestUnownedProd = prod;
338                         bestUnowned = wl;
339                     }
340                     break;
341                 default:
342                     break;
343                 }
344             }
345             if (bestOwned != null && bestOwnedProd > 0) {
346                 JMenuItem ji = makeProductionItem(type, bestOwned,
347                     bestOwnedProd, unitLabel, false);
348                 if (type == expertGoods) {
349                     expertOwned = ji;
350                 } else {
351                     items.put(ji, bestOwnedProd);
352                 }
353             }
354             if (bestUnowned != null && bestUnownedProd > bestOwnedProd
355                 && bestUnownedProd > 0) {
356                 JMenuItem ji = makeProductionItem(type, bestUnowned,
357                     bestUnownedProd, unitLabel, true);
358                 if (type == expertGoods) {
359                     expertUnowned = ji;
360                 } else {
361                     extras.put(ji, bestUnownedProd);
362                 }
363             }
364         }
365
366         JMenu container = Utility.localizedMenu("quickActionMenu.changeWork");
367         List<JMenuItem> owned = descendingList(items);
368         if (expertOwned != null) owned.add(0, expertOwned);
369         for (JMenuItem j : owned) container.add(j);
370         List<JMenuItem> unowned = descendingList(extras);
371         if (expertUnowned != null) unowned.add(0, expertUnowned);
372         if (!unowned.isEmpty()) {
373             if (!owned.isEmpty()) container.addSeparator();
374             for (JMenuItem j : unowned) container.add(j);
375         }
376         if (container.getItemCount() > 0) this.add(container);
377
378         if (current != null && unit.getWorkType() != null) {
379             JMenuItem ji = Utility.localizedMenuItem("showProductionModifiers");
380             ji.addActionListener((ActionEvent ae) -> {
381                     gui.showWorkProductionPanel(unit);
382                 });
383             this.add(ji);
384         }
385         return !(owned.isEmpty() && unowned.isEmpty() && current == null);
386     }

549     /**
550      * Nasty hack to get menu item to change roles.
551      *
552      * Hacky because we are continuing to express this in terms of equipment
553      * changes despite the point of the role cutover was to get rid of
554      * equipment types.  However, its time to release, and we should avoid
555      * string changes.  Get rid of this post 0.11.0-release.
556      *
557      * @param unitLabel The <code>UnitLabel</code> to create items for.
558      * @param from The starting <code>Role</code>.
559      * @param fromCount The starting role count.
560      * @param to The new <code>Role</code>.
561      * @param toCount The new role count.
562      * @param price An optional price to charge for the change.
563      * @return A suitable menu item.
564      */
565     private JMenuItem createRoleItem(final UnitLabel unitLabel,
566                                      final Role from, final int fromCount, 
567                                      final Role to, final int toCount,
568                                      final int price) {
569         // Get the text
570         String key = "model.role.change." + to.getSuffix();
571         if (!Messages.containsKey(key)) {
572             // Fall back to the full "from"."to" key
573             key = "model.role.change." + from.getSuffix()
574                 + "." + to.getSuffix();
575         }
576         String text = Messages.message(key);
577         if (price > 0) {
578             text += " ("
579                 + Messages.message(StringTemplate.template("goldAmount")
580                     .addAmount("%amount%", price))
581                 + ")";
582         }
583
584         // Get an icon
585         AbstractGoods change = null;
586         List<AbstractGoods> need = Role.getGoodsDifference(from, fromCount,
587                                                            to, toCount);
588         switch (need.size()) {
589         case 0:
590             break;
591         case 1:
592             change = need.get(0);
593             break;
594         default:
595             change = find(need, ag -> ag.getAmount() > 0);
596             break;
597         }
598         Icon icon = (change == null) ? null : new ImageIcon(
599             gui.getImageLibrary().getSmallIconImage(change.getType()));
600
601         JMenuItem item = new JMenuItem(text, icon);
602         final InGameController igc = freeColClient.getInGameController();
603         item.addActionListener((ActionEvent ae) -> {
604                 igc.equipUnitForRole(unitLabel.getUnit(), to, toCount);
605                 unitLabel.updateIcon();
606                 // FIXME: fix the PCL handling so this can go away
607                 if (parentPanel instanceof ColonyPanel) {
608                     ((ColonyPanel)parentPanel).update();
609                 }
610             });
611         return item;

614     /**
615      * Add menu items for role manipulation for a unit.
616      *
617      * Note "clear speciality" is here too to keep it well separated from
618      * other items.
619      *
620      * @param unitLabel The <code>UnitLabel</code> specifying the unit.
621      * @return True if menu items were added and a separator is now needed.
622      */
623     private boolean addRoleItems(final UnitLabel unitLabel) {
624         final Unit unit = unitLabel.getUnit();
625         final Role role = unit.getRole();
626         final int roleCount = unit.getRoleCount();
627         boolean separatorNeeded = false;
628
629         UnitLocation uloc = (unit.isInEurope()) ? unit.getOwner().getEurope()
630             : unit.getSettlement();
631         if (uloc == null) return false;
632         for (Role r : unit.getAvailableRoles(null)) {
633             if (r == role) continue;
634             JMenuItem newItem;
635             if (r.isDefaultRole()) { // Always valid
636                 newItem = createRoleItem(unitLabel, role, roleCount, r, 0, 0);
637             } else {
638                 newItem = null;
639                 for (int count = r.getMaximumCount(); count > 0; count--) {
640                     List<AbstractGoods> req = unit.getGoodsDifference(r, count);
641                     int price = uloc.priceGoods(req);
642                     if (price < 0) continue;
643                     if (unit.getOwner().checkGold(price)) {
644                         newItem = createRoleItem(unitLabel, role, roleCount,
645                                                  r, count, price);
646                         break;
647                     }
648                 }
649             }
650             if (newItem != null) {
651                 this.add(newItem);
652                 separatorNeeded = true;
653             }
654         }
655
656         UnitType newUnitType = unit.getType()
657             .getTargetType(ChangeType.CLEAR_SKILL, unit.getOwner());
658         if (newUnitType != null) {
659             if (separatorNeeded) this.addSeparator();
660             JMenuItem menuItem = Utility.localizedMenuItem("quickActionMenu.clearSpeciality",
661                 new ImageIcon(
662                     gui.getImageLibrary().getTinyUnitImage(newUnitType)));
663             menuItem.setActionCommand(UnitAction.CLEAR_SPECIALITY.toString());
664             menuItem.addActionListener(unitLabel);
665             this.add(menuItem);
666             if (unit.getLocation() instanceof Building
667                 && !((Building)unit.getLocation()).canAddType(newUnitType)) {
668                 menuItem.setEnabled(false);
669             }
670             separatorNeeded = true;
671         }
672         return separatorNeeded;
673     }
