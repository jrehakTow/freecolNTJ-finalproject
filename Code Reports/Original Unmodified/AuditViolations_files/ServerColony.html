438         checkBuildQueueIntegrity(true);
439
440         // If a build queue is empty, check that we are not producing
441         // any goods types useful for BuildableTypes, except if that
442         // type is the input to some other form of production.  (Note:
443         // isBuildingMaterial is also true for goods used to produce
444         // role-equipment, hence neededForBuildableType).  Such
445         // production probably means we forgot to reset the build
446         // queue.  Thus, if hammers are being produced it is worth
447         // warning about, but not if producing tools.
448         for (BuildQueue<?> queue : queues) {
449             if (queue.isEmpty()) {
450                 for (GoodsType g : spec.getGoodsTypeList()) {
451                     if (g.isBuildingMaterial()
452                         && !g.isRawMaterial()
453                         && !g.isBreedable()
454                         && getAdjustedNetProductionOf(g) > 0
455                         && neededForBuildableType(g)) {
456                         cs.addMessage(See.only(owner),
457                             new ModelMessage(MessageType.BUILDING_COMPLETED,
458                                              "model.colony.notBuildingAnything",
459                                              this)
460                                 .addName("%colony%", getName()));
461                         break;
462                     }
463                 }
464             }
465         }
466
467         // Update SoL.
468         updateSoL();
469         if (sonsOfLiberty / 10 != oldSonsOfLiberty / 10) {
470             cs.addMessage(See.only(owner),
471                 new ModelMessage(MessageType.SONS_OF_LIBERTY,
472                                  ((sonsOfLiberty > oldSonsOfLiberty)
473                                      ? "model.colony.soLIncrease"
474                                      : "model.colony.soLDecrease"),
475                                  this, spec.getGoodsType("model.goods.bells"))
476                     .addAmount("%oldSoL%", oldSonsOfLiberty)
477                     .addAmount("%newSoL%", sonsOfLiberty)
478                     .addName("%colony%", getName()));
479
480             ModelMessage govMgtMessage = checkForGovMgtChangeMessage();
481             if (govMgtMessage != null) {
482                 cs.addMessage(See.only(owner), govMgtMessage);
483             }
484         }
485         updateProductionBonus();
486         // We have to wait for the production bonus to stabilize
487         // before checking for completion of training.  This is a rare
488         // case so it is not worth reordering the work location calls
489         // to csNewTurn.
490         for (WorkLocation workLocation : getCurrentWorkLocations()) {
491             if (workLocation.canTeach()) {
492                 ServerBuilding building = (ServerBuilding)workLocation;
493                 for (Unit teacher : building.getUnitList()) {
494                     building.csCheckTeach(teacher, cs);
495                 }
496             }
497         }
498
499         // Try to update minimally.
500         if (tileDirty) {
501             cs.add(See.perhaps(), tile);
502         } else {
503             cs.add(See.only(owner), this);
504         }
505         lb.add(", ");
506     }

558      * Eject units to any available work location.
559      *
560      * Called on building type changes, see below and
561      * @see ServerPlayer#csDamageBuilding
562      *
563      * -til: Might change the visible colony size.
564      *
565      * @param workLocation The <code>WorkLocation</code> to eject from.

571         unit: for (Unit u : units) {
572             for (WorkLocation wl : getAvailableWorkLocations()) {

574                 u.setLocation(wl);//-vis: safe/colony
575                 continue unit;
576             }
577             u.setLocation(getTile());//-vis: safe/colony
578         }
579         if (getOwner().isAI()) {
580             firePropertyChange(REARRANGE_WORKERS, true, false);
581         }
582         return true;

586      * Builds a building from a build queue.
587      *
588      * @param buildQueue The <code>BuildQueue</code> to build from.
589      * @param cs A <code>ChangeSet</code> to update.
590      * @return True if the build succeeded.
591      */
592     private boolean csBuildBuilding(BuildQueue<? extends BuildableType> buildQueue,
593                                     ChangeSet cs) {
594         BuildingType type = (BuildingType) buildQueue.getCurrentlyBuilding();
595         Tile copied = getTile().getTileToCache();
596         BuildingType from = type.getUpgradesFrom();
597         boolean success;
598         if (from == null) {
599             success = buildBuilding(new ServerBuilding(getGame(), this, type));//-til
600         } else {
601             Building building = getBuilding(from);
602             List<Unit> eject = building.upgrade();//-til
603             success = eject != null;
604             if (success) {
605                 ejectUnits(building, eject);//-til
606                 if (!eject.isEmpty()) getTile().cacheUnseen(copied);//+til
607             } else {
608                 cs.addMessage(See.only((ServerPlayer)owner),
609                               getUnbuildableMessage(type));
610             }
611         }
612         if (success) {
613             cs.addMessage(See.only((ServerPlayer) owner),
614                 new ModelMessage(MessageType.BUILDING_COMPLETED,
615                                  "model.colony.buildingReady", this)
616                     .addName("%colony%", getName())
617                     .addNamed("%building%", type));
618             if (owner.isAI()) {
619                 firePropertyChange(REARRANGE_WORKERS, true, false);
620             }
621             logger.info("New building in " + getName()
622                 + ": " + type.getSuffix());
623         }
624         return success;
625     }

627     /**
628      * Removes a buildable from a build queue, and updates the queue so that
629      * a valid buildable is now being built if possible.
630      *
631      * @param queue The <code>BuildQueue</code> to update.
632      * @param cs A <code>ChangeSet</code> to update.
633      * @return The next buildable that can be built, or null if nothing.
634      */
635     private BuildableType csNextBuildable(BuildQueue<? extends BuildableType> queue,
636                                           ChangeSet cs) {
637         Specification spec = getSpecification();
638         ServerPlayer owner = (ServerPlayer) getOwner();
639         BuildableType buildable;
640         boolean invalidate = false;
641
642         while ((buildable = queue.getCurrentlyBuilding()) != null) {
643             switch (getNoBuildReason(buildable, null)) {
644             case LIMIT_EXCEEDED:
645                 // Expected when a player builds its last available wagon
646                 // and there is nothing else in the build queue.
647                 break;
648             case NONE:
649                 return buildable;
650             case NOT_BUILDING:
651                 for (GoodsType goodsType : spec.getGoodsTypeList()) {
652                     if (goodsType.isBuildingMaterial()
653                         && !goodsType.isStorable()
654                         && getTotalProductionOf(goodsType) > 0) {
655                         // Production is idle
656                         cs.addMessage(See.only(owner),
657                             new ModelMessage(MessageType.WARNING,
658                                              "model.colony.cannotBuild",
659                                              this)
660                                 .addName("%colony%", getName()));
661                     }
662                 }
663                 return null;
664                 
665             case POPULATION_TOO_SMALL:
666                 cs.addMessage(See.only(owner),
667                     new ModelMessage(MessageType.WARNING,
668                                      "model.colony.buildNeedPop",
669                                      this)
670                         .addName("%colony%", getName())
671                         .addNamed("%building%", buildable));
672                 break;
673             default: // Are there other warnings to send?
674                 logger.warning("Unexpected build failure at " + getName()
675                     + " for " + buildable
676                     + ": " + getNoBuildReason(buildable, null));
677                 cs.addMessage(See.only(owner),
678                               getUnbuildableMessage(buildable));
679                 break;
680             }
681             queue.remove(0);
682             invalidate = true;
683         }
684         if (invalidate) invalidateCache();
685         return null;
686     }

717      *
718      * -vis: Owner and new owner
