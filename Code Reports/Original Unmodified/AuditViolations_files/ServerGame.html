459                     unit.setLocation(strongestAIPlayer.getHighSeas());//-vis

461                 } else if (unit.getLocation() instanceof Tile) {
462                     Tile tile = unit.getTile();
463                     if (!tiles.contains(tile)) tiles.add(tile);
464                 }
465             }
466         }
467
468         StringTemplate loser = weakestAIPlayer.getNationLabel();
469         StringTemplate winner = strongestAIPlayer.getNationLabel();
470         cs.addMessage(See.all(),
471             new ModelMessage(ModelMessage.MessageType.FOREIGN_DIPLOMACY,
472                              "model.game.spanishSuccession", strongestAIPlayer)
473                 .addStringTemplate("%loserNation%", loser)
474                 .addStringTemplate("%nation%", winner));
475         cs.addGlobalHistory(this,
476             new HistoryEvent(getTurn(),
477                 HistoryEvent.HistoryEventType.SPANISH_SUCCESSION, null)
478                    .addStringTemplate("%loserNation%", loser)
479                    .addStringTemplate("%nation%", winner));
480         setSpanishSuccession(true);
481         cs.addPartial(See.all(), this, "spanishSuccession");
482         cs.add(See.perhaps(), tiles);
483         
484         weakest.csKill(cs);//+vis(weakest)
485         strongest.invalidateCanSeeTiles();//+vis(strongest)
486
487         // Trace fail where not all units are transferred
488         for (FreeColGameObject fcgo : getFreeColGameObjects()) {
489             if (fcgo instanceof Ownable
490                 && ((Ownable)fcgo).getOwner() == weakest) {
491                 throw new RuntimeException("Lurking " + weakest.getId()
492                     + " fcgo: " + fcgo);
493             }
494         }
495
496         return weakest;
497     }
