441      * @param propagate If true, propagate the alarm change upward to the
442      *     owning player.
443      * @param cs A <code>ChangeSet</code> to update.
444      * @return True if the alarm changed and the tile added.
445      */
446     public boolean csModifyAlarm(Player player, int add, boolean propagate,
447                                  ChangeSet cs) {
448         Tile copied = getTile().getTileToCache();
449         boolean change = csChangeAlarm(player, add, propagate, cs);//-til
450         if (change) {
451             getTile().cacheUnseen(copied);//+til
452             cs.add(See.perhaps(), this);
453         }
454         return change;
455     }

457     /**
458      * Changes the missionary for this settlement and updates other players.
459      *
460      * +vis: Handles the visibility implications.
461      * +til: Handles the tile appearance change.
462      *
463      * @param missionary The new missionary for this settlement.
464      * @param cs A <code>ChangeSet</code> to update.
465      */
466     public void csChangeMissionary(Unit missionary, ChangeSet cs) {
467         final Unit old = getMissionary();
468         if (missionary == old) return;
469
470         final Tile tile = getTile();
471         final ServerPlayer newOwner = (missionary == null) ? null
472             : (ServerPlayer)missionary.getOwner();
473         tile.cacheUnseen(newOwner);//+til
474
475         if (old != null) {
476             final ServerPlayer oldOwner = (ServerPlayer)old.getOwner(); 
477             setMissionary(null);//-vis(oldOwner),-til
478             tile.updateIndianSettlement(oldOwner);
479             cs.addRemove(See.only(oldOwner), null, old);//-vis(oldOwner)
480             old.dispose();
481             cs.add(See.only(oldOwner), tile);
482             oldOwner.invalidateCanSeeTiles();//+vis(oldOwner)
483         }
484
485         if (missionary != null) {
486             setMissionary(missionary);//-vis(newOwner)
487             // Take the missionary off the map, and give it a fake
488             // location at the settlement, bypassing the normal
489             // validity checks.
490             missionary.setLocation(null);//-vis(newOwner)
491             missionary.setLocationNoUpdate(this);//-vis(newOwner),-til
492             missionary.setMovesLeft(0);
493             cs.add(See.only(newOwner), missionary);
494             setConvertProgress(0);
495             csChangeAlarm(newOwner, ALARM_NEW_MISSIONARY, true, cs);//-til
496             tile.updateIndianSettlement(newOwner);
497             
498             int radius = getMissionaryLineOfSight();
499             for (Tile t : tile.getSurroundingTiles(1, radius)) {
500                 if (newOwner.exploreTile(t) || !newOwner.canSee(t)) {
501                     cs.add(See.only(newOwner), t);
502                 }
503             }
504             cs.add(See.perhaps().always(newOwner), tile);
505             newOwner.invalidateCanSeeTiles();//+vis(newOwner)
506         }
507     }

511      *
512      * @param destroy If true, the settlement is destroyed, if false the

538             }
539         }
