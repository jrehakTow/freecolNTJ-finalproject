82      * Calculates the odds of success in combat.
83      *
84      * @param attacker The attacker.
85      * @param defender The defender.
86      * @param lb An optional <code>LogBuilder</code> to log to.
87      * @return The combat odds.
88      */
89     private CombatOdds calculateCombatOdds(FreeColGameObject attacker,
90                                            FreeColGameObject defender,
91                                            LogBuilder lb) {
92         if (attacker == null || defender == null) {
93             if (lb != null) lb.add(" odds=unknowable");
94             return new CombatOdds(CombatOdds.UNKNOWN_ODDS);
95         }
96
97         if (lb != null) lb.add(" attacker=", attacker, " ");
98         double attackPower = getOffencePower(attacker, defender, lb);
99         if (lb != null) lb.add(" defender=", defender, " ");
100         double defencePower = getDefencePower(attacker, defender, lb);
101         if (attackPower == 0.0 && defencePower == 0.0) {
102             if (lb != null) lb.add(" odds=unknown");
103             return new CombatOdds(CombatOdds.UNKNOWN_ODDS);
104         }
105         double victory = attackPower / (attackPower + defencePower);
106         if (lb != null) lb.add(" odds=", victory);
107         return new CombatOdds(victory);
108     }

467         } else {
468             throw new IllegalArgumentException("Bogus combat");
469         }
470
471         // @compat 0.11.0
472         // Any modifier with the default modifier index needs to be fixed
473         for (Modifier r : result) {
474             if (r.getModifierIndex() == Modifier.DEFAULT_MODIFIER_INDEX) {
475                 r.setModifierIndex(Modifier.GENERAL_COMBAT_INDEX);
476             }
477         }
478         // end @compat 0.11.0
479
480         return result;
481     }

516     /**
517      * Add all the defensive modifiers that apply to a land attack.
518      *
519      * @param attacker The attacker.
520      * @param defender The defender <code>Unit</code>.
521      * @param result The set of modifiers to add to.
522      */
523     private void addLandDefensiveModifiers(FreeColGameObject attacker,
524                                            Unit defender,
525                                            Set<Modifier> result) {
526         final Specification spec = defender.getSpecification();
527         final Tile tile = defender.getTile();
528         final Settlement settlement = (tile == null) ? null
529             : tile.getSettlement();
530
531         if (tile != null) {
532             boolean disableFortified = false;
533

536
537             if (settlement == null) {
538                 // PF#73 demonstrated that tile modifiers do not apply
539                 // for colonies
540                 result.addAll(tile.getType().getDefenceModifiers());
541
542                 // Artillery in the Open penalty

544                     && defender.getState() != Unit.UnitState.FORTIFIED) {
545                     result.addAll(spec.getModifiers(Modifier.ARTILLERY_IN_THE_OPEN));
546                 }
547
548             } else { // In settlement
549                 // Settlement defence bonus

557                 }

563                 }

566                     Building stockade = ((Colony)settlement).getStockade();

568                         disableFortified |= hasStrongDefenceModifier(stockade.getType());
569                     }
570                 }
571             }
572
573             // Fortify bonus
574             if (defender.getState() == Unit.UnitState.FORTIFIED
575                 && !disableFortified) {
576                 result.addAll(spec.getModifiers(Modifier.FORTIFIED));
577             }
578         }
579     }

581     /**
582      * Generates a result of a unit attacking.
583      * Takes care to only call the pseudo-random source *once*.
584      *
585      * @param random A pseudo-random number source.
586      * @param attacker The attacker.
587      * @param defender The defender.
588      * @return The results of the combat.
589      */
590     @Override
591     public List<CombatResult> generateAttackResult(Random random,
592         FreeColGameObject attacker, FreeColGameObject defender) {
593         LogBuilder lb = new LogBuilder(256);
594         lb.add("Combat");
595         ArrayList<CombatResult> crs = new ArrayList<>();
596         CombatOdds odds = calculateCombatOdds(attacker, defender, lb);
597         double r = randomDouble(logger, "AttackResult", random);
598         lb.add(" random(1.0)=", r);
599         boolean great = false; // Great win or loss?
600         String action;
601
602         if (combatIsAttack(attacker, defender)) {
603             Unit attackerUnit = (Unit) attacker;
604             Unit defenderUnit = (Unit) defender;
605             action = "Attack";
606
607             // For random double 0 <= r < 1.0:
608             // Partition this range into wins < odds.win and losses above.
609             // Within the 0 <= r < odds.win range, partition the first 10%
610             // to be great wins and the rest to be ordinary wins.
611             //   r < 0.1 * odds.win  => great win
612             //   else r < odds.win   => win
613             // Within the odds.win <= r < 1.0 range, partition the first
614             // 20% to be evasions (if defender has the evadeAttack ability),
615             // the next 70% to be ordinary losses, and the rest great losses.
616             //   r < odds.win + 0.2 * (1.0 - odds.win) = 0.8 * odds.win + 0.2
617             //     => evade
618             //   else r < odds.win + (0.2 + 0.7) * (1.0 - odds.win)
619             //     = 0.1 * odds.win + 0.9 => loss
620             //   else => great loss
621             // ...and beached ships always lose.
622             if (r < odds.win || defenderUnit.isBeached()) {
623                 great = r < 0.1 * odds.win; // Great Win
624                 crs.add(CombatResult.WIN);
625                 resolveAttack(attackerUnit, defenderUnit, great,
626                     // Rescale to 0 <= r < 1
627                     r / (0.1 * odds.win), crs);
628             } else if (r < 0.8 * odds.win + 0.2
629                     && defenderUnit.hasAbility(Ability.EVADE_ATTACK)) {
630                 crs.add(CombatResult.NO_RESULT);
631                 crs.add(CombatResult.EVADE_ATTACK);
632             } else {
633                 great = r >= 0.1 * odds.win + 0.9; // Great Loss
634                 crs.add(CombatResult.LOSE);
635                 resolveAttack(defenderUnit, attackerUnit, great,
636                     // Rescaling to 0 <= r < 1
637                     // (rearrange: 0.8 * odds.win + 0.2 <= r < 1.0)
638                     (1.25 * r - 0.25 - odds.win)/(1.0 - odds.win), crs);
639             }
640
641         } else if (combatIsBombard(attacker, defender)) {
642             Unit defenderUnit = (Unit) defender;
643             if (!defenderUnit.isNaval()) {
644                 // One day we might want:
645                 //   crs.add(CombatResult.SLAUGHTER_UNIT_BOMBARD);
646                 throw new IllegalStateException("Bombard of non-naval");
647             }
648             action = "Bombard";
649
650             // The bombard succeeds.
651             if (r <= odds.win) {
652                 crs.add(CombatResult.WIN);
653
654                 // Great wins occur at most in 1 in 3 of successful bombards,
655                 // Good defences reduce this proportion.
656                 double offencePower = getOffencePower(attacker, defender);
657                 double defencePower = getDefencePower(attacker, defender);
658                 double diff = Math.max(3.0, defencePower * 2.0 - offencePower);
659                 great = r < odds.win / diff;
660
661                 // Sink the defender on great wins or lack of repair
662                 // location, otherwise just damage.
663                 if (great || defenderUnit.getRepairLocation() == null) {
664                     crs.add(CombatResult.SINK_SHIP_BOMBARD);
665                 } else {
666                     crs.add(CombatResult.DAMAGE_SHIP_BOMBARD);
667                 }
668
669             // The bombard fails but this is not a win for the
670             // defender, just an evasion, as it is not currently given
671             // an opportunity to return fire.
672             } else {
673                 crs.add(CombatResult.NO_RESULT);
674                 crs.add(CombatResult.EVADE_BOMBARD);
675             }
676
677         } else {
678             throw new IllegalStateException("Bogus combat");
679         }
680
681         // Log the results so that we have a solid record of combat
682         // determinations for debugging and investigation of user
683         // `I just lost N combats' complaints.
684         lb.add(" great=", great, " ", action);
685         for (CombatResult cr : crs) lb.add(" ", cr);
686         lb.log(logger, Level.INFO);
687
688         return crs;
689     }
690
691     /**
692      * Resolve all the consequences of a normal attack.
693      *
694      * @param winner The winning <code>Unit</code>.
695      * @param loser The losing <code>Unit</code>.
696      * @param great True if this is a great win/loss.
697      * @param r A "residual" random value (for convert/burn mission).
698      * @param crs A list of <code>CombatResult</code>s to add to.
699      */
700     private void resolveAttack(Unit winner, Unit loser, boolean great,
701                                double r, List<CombatResult> crs) {
702         Player loserPlayer = loser.getOwner();
703         Tile tile = loser.getTile();
704         Player winnerPlayer = winner.getOwner();
705         boolean attackerWon = crs.get(0) == CombatResult.WIN;
706         boolean loserMustDie = loser.hasAbility(Ability.DISPOSE_ON_COMBAT_LOSS);
707
708         if (loser.isNaval()) {
709             // Naval victors get to loot the defenders hold.  Sink the
710             // loser on great win/loss, lack of repair location, or
711             // beached.
712             if (winner.isNaval() && winner.canCaptureGoods()
713                 && !loser.getGoodsList().isEmpty()) {
714                 crs.add(CombatResult.LOOT_SHIP);
715             }
716             if (great || loserMustDie
717                 || loser.getRepairLocation() == null
718                 || loser.isBeached()) {
719                 crs.add(CombatResult.SINK_SHIP_ATTACK);
720             } else {
721                 crs.add(CombatResult.DAMAGE_SHIP_ATTACK);
722             }
723
724         } else { // loser is land unit
725             // Autoequip the defender?
726             Role autoRole = (attackerWon) ? loser.getAutomaticRole() : null;
727             if (autoRole != null) crs.add(CombatResult.AUTOEQUIP_UNIT);
728
729             // Special handling for settlements
730             boolean done = false;
731             Settlement settlement = tile.getSettlement();
732             if (settlement instanceof Colony) {
733                 final Colony colony = (Colony)settlement;
734                 // A Colony falls to Europeans when the last defender
735                 // is unarmed.  Natives will pillage if possible but
736                 // otherwise proceed to kill colonists incrementally
737                 // until the colony falls for lack of survivors.
738                 // Ships in a falling colony will be damaged or sunk
739                 // if they have no repair location.
740                 if (!loser.isDefensiveUnit() && autoRole == null) {
741                     List<Unit> ships = colony.getTile().getNavalUnits();
742                     final CombatResult shipResult = (ships.isEmpty()) ? null
743                         : (ships.get(0).getRepairLocation() == null)
744                         ? CombatResult.SINK_COLONY_SHIPS
745                         : CombatResult.DAMAGE_COLONY_SHIPS;
746
747                     if (winnerPlayer.isEuropean()) {
748                         if (loserMustDie) {
749                             crs.add(CombatResult.SLAUGHTER_UNIT);
750                         }
751                         if (shipResult != null) crs.add(shipResult);
752                         crs.add(CombatResult.CAPTURE_COLONY);
753                         done = true;
754
755                     } else if (!great && colony.canBePillaged(winner)) {
756                         crs.add(CombatResult.PILLAGE_COLONY);
757                         done = true;
758
759                     } else if (colony.getUnitCount() > 1
760                         || loser.getLocation() == tile) {
761                         loserMustDie = true;
762                         done = false; // Treat as ordinary combat
763
764                     } else {
765                         crs.add(CombatResult.SLAUGHTER_UNIT);
766                         if (shipResult != null) crs.add(shipResult);
767                         crs.add(CombatResult.DESTROY_COLONY);
768                         done = true;
769                     }
770                 }
771  
772             } else if (settlement instanceof IndianSettlement) {
773                 final IndianSettlement is = (IndianSettlement)settlement;
774                 // Attacking and defeating the defender of a native
775                 // settlement with a mission may yield converts but
776                 // also may provoke the burning of all missions.
777                 // Native settlements fall when there are no units
778                 // present either in-settlement or on the settlement
779                 // tile.
780                 int lose = 0;
781                 if (loserMustDie) {
782                     // Add death of loser before any convert captures,
783                     // or the RNG might randomly decide to convert the
784                     // unit that is then slaughtered.
785                     crs.add(CombatResult.SLAUGHTER_UNIT);
786                     lose++;
787                     // For now, no usual unit combat actions can proceed,
788                     // which means we can not expect to capture equipment
789                     // from settlements without untangling this dependency.
790                     done = true;
791                 }
792                 if (attackerWon) {
793                     if (r < winner.getConvertProbability()) {
794                         if (is.getUnitCount() + tile.getUnitCount() > lose
795                             && is.hasMissionary(winnerPlayer)
796                             && !combatIsAmphibious(winner, loser)) {
797                             crs.add(CombatResult.CAPTURE_CONVERT);
798                             lose++;
799                         }
800                     } else if (r >= 1.0 - winner.getBurnProbability()) {
801                         if (any(loserPlayer.getIndianSettlements(),
802                                 s -> s.hasMissionary(winnerPlayer))) {
803                             crs.add(CombatResult.BURN_MISSIONS);
804                         }
805                     }
806                 }
807                 if (settlement.getUnitCount() + tile.getUnitCount() <= lose) {
808                     crs.add(CombatResult.DESTROY_SETTLEMENT);
809                     done = true;
810                 }
811             }
812
813             if (!done) {
814                 final Role loserRole = loser.getRole();
815                 // First check if the loser was automatically armed, and
816                 // if so see if the winner can capture that equipment,
817                 // which may kill or demote the loser.
818                 if (autoRole != null) {
819                     crs.add((winner.canCaptureEquipment(autoRole) != null)
820                         ? CombatResult.CAPTURE_AUTOEQUIP
821                         : CombatResult.LOSE_AUTOEQUIP);
822                     if (loserMustDie) {
823                         crs.add(CombatResult.SLAUGHTER_UNIT);
824                     } else if (loser.hasAbility(Ability.DEMOTE_ON_ALL_EQUIPMENT_LOST)) {
825                         crs.add(CombatResult.DEMOTE_UNIT);
826                     }
827
828                 // Some losers are just doomed (e.g. seasonedScout), do not
829                 // check for capture/demote/lose-equipment.
830                 } else if (loserMustDie) {
831                     crs.add(CombatResult.SLAUGHTER_UNIT);
832
833                 // Then check if the user had other offensive
834                 // role-equipment, that can be captured or lost, which
835                 // may kill or demote the loser.
836                 } else if (loserRole.isOffensive()) {
837                     crs.add((winner.canCaptureEquipment(loserRole) != null)
838                         ? CombatResult.CAPTURE_EQUIP
839                         : CombatResult.LOSE_EQUIP);
840                     if (loserMustDie
841                         || loser.losingEquipmentKillsUnit()) {
842                         crs.add(CombatResult.SLAUGHTER_UNIT);
843                     } else if (loser.losingEquipmentDemotesUnit()) {
844                         crs.add(CombatResult.DEMOTE_UNIT);
845                     }
846
847                 // But some can be captured.
848                 } else if (loser.hasAbility(Ability.CAN_BE_CAPTURED)
849                     && winner.hasAbility(Ability.CAPTURE_UNITS)
850                     && !combatIsAmphibious(winner, loser)) {
851                     // Demotion on capture is handled by capture routine.
852                     crs.add(CombatResult.CAPTURE_UNIT);
853
854                 // Or losing just causes a demotion.
855                 } else if (loser.getTypeChange(ChangeType.DEMOTION,
856                                                loserPlayer) != null) {
857                     crs.add(CombatResult.DEMOTE_UNIT);
858
859                 // But finally, the default is to kill them.
860                 } else {
861                     crs.add(CombatResult.SLAUGHTER_UNIT);
862                 }
863             }
864         }
865
866         // Promote great winners or with automatic promotion, if possible.
867         UnitTypeChange promotion = winner.getType()
868             .getUnitTypeChange(ChangeType.PROMOTION, winnerPlayer);
869         if (promotion != null
870             && (winner.hasAbility(Ability.AUTOMATIC_PROMOTION)
871                 || (great
872                     && (100 * (r - Math.floor(r))
