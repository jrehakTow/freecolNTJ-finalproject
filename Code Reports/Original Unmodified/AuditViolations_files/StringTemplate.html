34 /**
35  * The <code>StringTemplate</code> represents a non-localized string
36  * that can be localized by looking up its value in a message bundle
37  * or similar Map. The StringTemplate may contain variables (keys)
38  * delimited by the '%' character, such as "%amount%" that will be
39  * replaced with a string or a StringTemplate. If the StringTemplate
40  * contains replacement values but no keys, then it is considered a
41  * "label" StringTemplate, and its value will be used to join the
42  * replacement values.
43  *
44  * @version 1.0
45  */
46
47 public class StringTemplate extends FreeColObject {

49     /**
50      * The type of this StringTemplate, either NAME, a proper name
51      * that must not be localized (e.g. "George Washington"), or KEY,
52      * a string that must be localized (e.g. "model.goods.food.name"),
53      * or TEMPLATE, a key with replacements to apply to the localized
54      * value of the key, or LABEL, a separator string that will be
55      * used to join the replacement values.
56      */
