290     /**
291      * Gets the production types available for the given combination
292      * of colony center tile and production level.  If the production
293      * level is null, all production levels will be returned.
294      *
295      * @param unattended Whether the production is unattended.
296      * @param level The production level.
297      * @return A list of <code>ProductionType</code>s.
298      */
299     public List<ProductionType> getAvailableProductionTypes(boolean unattended,
300                                                             String level) {
301         List<ProductionType> good = new ArrayList<>(),
302             better = new ArrayList<>();
303         for (ProductionType productionType : productionTypes) {
304             if (productionType.getUnattended() != unattended) continue;
305             if (productionType.appliesExactly(level)) {
306                 better.add(productionType);
307             } else if (productionType.appliesTo(level)) {
308                 good.add(productionType);
309             }
310         }
311         return (!better.isEmpty()) ? better : good;
312     }

331     /**
332      * Get the base production of a given goods type for an optional
333      * unit type.
334      * 
335      * @param productionType An optional <code>ProductionType</code> to use,
336      *     if null the best available one is used.
337      * @param goodsType The <code>GoodsType</code> to produce.
338      * @param unitType An optional <code>UnitType</code> that is to do
339      *     the work, if null the unattended production is considered.
340      * @return The amount of goods produced.
341      */
342     public int getBaseProduction(ProductionType productionType,
343                                  GoodsType goodsType, UnitType unitType) {
344         if (goodsType == null) return 0;
345         if (productionType == null) {
346             productionType = ProductionType.getBestProductionType(goodsType,
347                 getAvailableProductionTypes(unitType == null));
348         }
349         if (productionType == null) return 0;
350         AbstractGoods best = productionType.getOutput(goodsType);
351         return (best == null) ? 0 : best.getAmount();
352     }

551     protected void readChild(FreeColXMLReader xr) throws XMLStreamException {
552         final Specification spec = getSpecification();
553         final String tag = xr.getLocalName();
554
555         if (DISASTER_TAG.equals(tag)) {
556             Disaster d = xr.getType(spec, ID_ATTRIBUTE_TAG,
557                                     Disaster.class, (Disaster)null);
558             if (d != null) {
559                 addDisaster(d, xr.getAttribute(PROBABILITY_TAG, 100));
560             }
561             xr.closeTag(DISASTER_TAG);
562
563         } else if (GEN_TAG.equals(tag)) {
564             humidity[0] = xr.getAttribute(HUMIDITY_MIN_TAG, 0);
565             humidity[1] = xr.getAttribute(HUMIDITY_MAX_TAG, 100);
566             temperature[0] = xr.getAttribute(TEMPERATURE_MIN_TAG, -20);
567             temperature[1] = xr.getAttribute(TEMPERATURE_MAX_TAG, 40);
568             altitude[0] = xr.getAttribute(ALTITUDE_MIN_TAG, 0);
569             altitude[1] = xr.getAttribute(ALTITUDE_MAX_TAG, 0);
570             // @compat 0.11.3
571             if (xr.hasAttribute(OLD_HUMIDITY_MIN_TAG)) {
572                 humidity[0] = xr.getAttribute(OLD_HUMIDITY_MIN_TAG, 0);
573             }
574             if (xr.hasAttribute(OLD_HUMIDITY_MAX_TAG)) {
575                 humidity[1] = xr.getAttribute(OLD_HUMIDITY_MAX_TAG, 100);
576             }
577             if (xr.hasAttribute(OLD_TEMPERATURE_MIN_TAG)) {
578                 temperature[0] = xr.getAttribute(OLD_TEMPERATURE_MIN_TAG, -20);
579             }
580             if (xr.hasAttribute(OLD_TEMPERATURE_MAX_TAG)) {
581                 temperature[1] = xr.getAttribute(OLD_TEMPERATURE_MAX_TAG, 40);
582             }
583             if (xr.hasAttribute(OLD_ALTITUDE_MIN_TAG)) {
584                 altitude[0] = xr.getAttribute(OLD_ALTITUDE_MIN_TAG, 0);
585             }
586             if (xr.hasAttribute(OLD_ALTITUDE_MAX_TAG)) {
587                 altitude[1] = xr.getAttribute(OLD_ALTITUDE_MAX_TAG, 0);
588             }
589             // end @compat 0.11.3
590             xr.closeTag(GEN_TAG);
591
592         } else if (PRODUCTION_TAG.equals(tag)
593             && xr.getAttribute(DELETE_TAG, false)) {
594             productionTypes.clear();
595             xr.closeTag(PRODUCTION_TAG);
596
597         } else if (PRODUCTION_TAG.equals(tag)
598                    // @compat 0.10.6
599                    && xr.getAttribute(GOODS_TYPE_TAG, (String)null) == null
600                    // end @compat 0.10.6
601                    ) {
602             productionTypes.add(new ProductionType(xr, spec));
603
604         // @compat 0.10.6
605         } else if (PRODUCTION_TAG.equals(tag)
606             || PRIMARY_PRODUCTION_TAG.equals(tag)
607             || SECONDARY_PRODUCTION_TAG.equals(tag)) {
608             GoodsType type = xr.getType(spec, GOODS_TYPE_TAG,
609                                         GoodsType.class, (GoodsType)null);
610             int amount = xr.getAttribute(VALUE_TAG, 0);
611             AbstractGoods goods = new AbstractGoods(type, amount);
612             String tileProduction = xr.getAttribute(TILE_PRODUCTION_TAG,
613                                                     (String)null);
614             // CAUTION: this only works if the primary production is
615             // defined before the secondary production
616             if (PRIMARY_PRODUCTION_TAG.equals(tag)) {
617                 productionTypes.add(new ProductionType(goods, true,
618                                                        tileProduction));
619             } else if (SECONDARY_PRODUCTION_TAG.equals(tag)) {
620                 String level = spec.getDifficultyLevel();
621                 List<ProductionType> unattendedTypes
622                     = getAvailableProductionTypes(true, level);
623                 for (ProductionType productionType : unattendedTypes) {
624                     if (tileProduction == null
625                         || tileProduction.equals(productionType.getProductionLevel())) {
626                         productionType.getOutputs().add(goods);
627                     }
628                 }
629             } else {
630                 productionTypes.add(new ProductionType(goods, false,
631                                                        tileProduction));
632             }
633             xr.closeTag(tag);
634             // end @compat 0.10.6
635
636         } else if (RESOURCE_TAG.equals(tag)) {
637             addResourceType(xr.getType(spec, TYPE_TAG, ResourceType.class,
638                                        (ResourceType)null),
639                             xr.getAttribute(PROBABILITY_TAG, 100));
640             xr.closeTag(RESOURCE_TAG);
641
642         } else if (Modifier.getTagName().equals(tag)) {
643             // @compat 0.10.7
644             // the tile type no longer contains the base production modifier
645             String id = xr.getAttribute(ID_ATTRIBUTE_TAG, null);
646             if (id.startsWith("model.goods.")) {
647                 xr.closeTag(Modifier.getTagName());
648             } else {
649                 super.readChild(xr);
650             }
651             // end @compat
652         } else {
653             super.readChild(xr);
654         }
