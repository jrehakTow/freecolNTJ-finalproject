44     private static final int ANIMATION_DELAY = 33;

54      * @param freeColClient The enclosing <code>FreeColClient</code>.

56      * @param sourceTile The <code>Tile</code> the unit is moving from.
57      * @param destinationTile The <code>Tile</code> the unit is moving to.
58      */

60                              Tile sourceTile, Tile destinationTile) {
61         super(freeColClient);

69     /**
70      * Do the animation.
71      */
72     public void animate() {
73         final int movementSpeed
74             = getFreeColClient().getAnimationSpeed(unit.getOwner());
75         final SwingGUI gui = (SwingGUI)getGUI();
76         final Point srcP = gui.getTilePosition(sourceTile);
77         final Point dstP = gui.getTilePosition(destinationTile);
78         
79         if (srcP == null || dstP == null || movementSpeed <= 0) return;
80
81         float scale = gui.getMapScale();
82         final int movementRatio = (int)(Math.pow(2, movementSpeed + 1) * scale);
83         final Rectangle r1 = gui.getTileBounds(sourceTile);
84         final Rectangle r2 = gui.getTileBounds(destinationTile);
85         final Rectangle bounds = r1.union(r2);
86
87         gui.executeWithUnitOutForAnimation(unit, sourceTile,
88             (JLabel unitLabel) -> {
89                 final int labelWidth = unitLabel.getWidth();
90                 final int labelHeight = unitLabel.getHeight();
91                 final Point srcPoint = gui.calculateUnitLabelPositionInTile(labelWidth, labelHeight, srcP);
92                 final Point dstPoint = gui.calculateUnitLabelPositionInTile(labelWidth, labelHeight, dstP);
93                 final double xratio = ImageLibrary.TILE_SIZE.width
94                     / (double)ImageLibrary.TILE_SIZE.height;
95                 final int stepX = (srcPoint.getX() == dstPoint.getX()) ? 0
96                     : (srcPoint.getX() > dstPoint.getX()) ? -1 : 1;
97                 final int stepY = (srcPoint.getY() == dstPoint.getY()) ? 0
98                     : (srcPoint.getY() > dstPoint.getY()) ? -1 : 1;
99
100                 // Painting the whole screen once to get rid of
101                 // disposed dialog-boxes.
102                 gui.paintImmediatelyCanvasInItsBounds();
103
104                 int dropFrames = 0;
105                 Point point = srcPoint;
106                 while (!point.equals(dstPoint)) {
107                     long time = System.currentTimeMillis();
108
109                     point.x += stepX * xratio * movementRatio;
110                     point.y += stepY * movementRatio;
111                     if ((stepX < 0 && point.x < dstPoint.x)
112                         || (stepX > 0 && point.x > dstPoint.x)) {
113                         point.x = dstPoint.x;
114                     }
115                     if ((stepY < 0 && point.y < dstPoint.y)
116                         || (stepY > 0 && point.y > dstPoint.y)) {
117                         point.y = dstPoint.y;
118                     }
119                     if (dropFrames <= 0) {
120                         unitLabel.setLocation(point);
121                         gui.paintImmediatelyCanvasIn(bounds);
122                             
123                         int timeTaken = (int)(System.currentTimeMillis() - time);
124                         final int waitTime = ANIMATION_DELAY - timeTaken;
125                         if (waitTime > 0) {
126                             try {
127                                 Thread.sleep(waitTime);
128                             } catch (InterruptedException ex) {
129                                 //ignore
130                             }
131                             dropFrames = 0;
132                         } else {
133                             dropFrames = timeTaken / ANIMATION_DELAY - 1;
134                         }
135                     } else {
136                             dropFrames--;
137                     }
138                 }
139             });
140     }
