499                 // Goods being destroyed.
500                 aio.dispose();
501                 continue;
502             }
503             if (aio instanceof Wish) {
504                 if (!((Wish)aio).shouldBeStored()) continue;
505             }
506
507             try {
508                 if (aio.getId() == null) {
509                     logger.warning("Null AI identifier for: "
510                         + aio.getClass().getName());
511                 } else {
512                     aio.toXML(xw);
513                 }
514             } catch (Exception e) {
515                 logger.log(Level.WARNING, "Failed to write AI object: " + aio,
516                     e);
517             }
518         }
519     }

543     /**
544      * {@inheritDoc}
545      */
546     @Override
547     protected void readChild(FreeColXMLReader xr) throws XMLStreamException {
548         final String tag = xr.getLocalName();
549         final String oid = xr.readId();
550
551         try {
552             Wish wish = null;
553
554             // The AI data is quite shallow, so we can get away with
555             // fixing up forward references just with this simple
556             // lookup.  AIObjects that can be forward referenced must
557             // ensure they complete initialization somewhere in their
558             // serialization read* routines.
559             AIObject aio;
560             if (oid != null && (aio = getAIObject(oid)) != null) {
561                 aio.readFromXML(xr);
562
563             // @compat 0.10.1
564             } else if (COLONIAL_AI_PLAYER_TAG.equals(tag)) {
565                 new EuropeanAIPlayer(this, xr);
566             // end @compat
567
568             } else if (AIColony.getTagName().equals(tag)) {
569                 new AIColony(this, xr);
570
571             } else if (AIGoods.getTagName().equals(tag)) {
572                 new AIGoods(this, xr);
573
574             } else if (AIPlayer.getTagName().equals(tag)) {
575                 Player p = getGame().getFreeColGameObject(oid, Player.class);
576                 if (p != null) {
577                     if (p.isIndian()) {
578                         new NativeAIPlayer(this, xr);
579                     } else if (p.isREF()) {
580                         new REFAIPlayer(this, xr);
581                     } else if (p.isEuropean()) {
582                         new EuropeanAIPlayer(this, xr);
583                     } else {
584                         throw new RuntimeException("Bogus AIPlayer: " + p);
585                     }
586                 }
587
588             } else if (AIUnit.getTagName().equals(tag)) {
589                 new AIUnit(this, xr);
590
591             } else if (GoodsWish.getTagName().equals(tag)
592                 // @compat 0.10.3
593                 || GOODS_WISH_TAG.equals(tag)
594                 // end @compat
595                        ) {
596                 wish = new GoodsWish(this, xr);
597
598             } else if (TileImprovementPlan.getTagName().equals(tag)
599                 // @compat 0.10.3
600                 || OLD_TILE_IMPROVEMENT_PLAN_TAG.equals(tag)
601                 // end @compat
602                        ) {
603                 new TileImprovementPlan(this, xr);
604
605             } else if (WorkerWish.getTagName().equals(tag)) {
606                 wish = new WorkerWish(this, xr);
607             
608             } else {
609                 super.readChild(xr);
610             }
611             
612             if (wish != null) {
613                 AIColony ac = wish.getDestinationAIColony();
614                 if (ac != null) ac.addWish(wish);
615             }
616
617         } catch (Exception e) {
618             logger.log(Level.WARNING, "Exception reading AIObject: "
619                        + tag + ", id=" + oid, e);
620             // We are hosed.  Try to resynchronize at the end of the tag
621             // or aiMain.
622             final String mainTag = getTagName();
623             while (xr.nextTag() != XMLStreamConstants.END_ELEMENT
624                 || !(xr.atTag(tag) || xr.atTag(mainTag)));
625         }
626     }
