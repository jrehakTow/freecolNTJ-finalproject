163     @Override
164     public void actionPerformed(ActionEvent ae) {
165         final String url = ae.getActionCommand();
166         if (SITE_URL.equals(url) || PROJECT_URL.equals(url)
167             || MANUAL_URL.equals(url)) {
168             String os = System.getProperty("os.name");
169             // FIXME: move this to OS utilities
170             String[] cmd = null;
171             if (os == null) {
172                 // error, the operating system could not be determined
173                 return;
174             } else if (os.toLowerCase().contains("mac")) {
175                 // Apple Macintosh, Safari is the main browser
176                 cmd = new String[] { "open" , "-a", "Safari", url };
177             } else if (os.toLowerCase().contains("windows")) {
178                 // Microsoft Windows, use the default browser
179                 cmd = new String[] { "rundll32.exe",
180                     "url.dll,FileProtocolHandler", url};
181             } else if (os.toLowerCase().contains("linux")) {
182                 // GNU Linux, use xdg-utils to launch the default
183                 // browser (portland.freedesktop.org)
184                 cmd = new String[] { "xdg-open", url};
185             } else {
186                 cmd = new String[] { "firefox", url};
187             }
188             try {
189                 Runtime.getRuntime().exec(cmd);
190             } catch (IOException x) {
191                 // couldn't start browser
192             }
193         } else {
194             super.actionPerformed(ae);
195         }
196     }
