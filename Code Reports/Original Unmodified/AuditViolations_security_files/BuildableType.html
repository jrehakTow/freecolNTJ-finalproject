307     @Override
308     protected void readChild(FreeColXMLReader xr) throws XMLStreamException {
309         final Specification spec = getSpecification();
310         final String tag = xr.getLocalName();
311
312         if (REQUIRED_ABILITY_TAG.equals(tag)) {
313             String id = xr.readId();
314             addRequiredAbility(id, xr.getAttribute(VALUE_TAG, true));
315             spec.addAbility(id);
316             xr.closeTag(REQUIRED_ABILITY_TAG);
317
318         } else if (REQUIRED_GOODS_TAG.equals(tag)) {
319             GoodsType type = xr.getType(spec, ID_ATTRIBUTE_TAG,
320                                         GoodsType.class, (GoodsType)null);
321             int amount = xr.getAttribute(VALUE_TAG, 0);
322             addRequiredGoods(new AbstractGoods(type, amount));
323             xr.closeTag(REQUIRED_GOODS_TAG);
324
325         } else if (Limit.getTagName().equals(tag)) {
326             Limit limit = new Limit(xr, spec);
327             if (limit.getLeftHandSide().getType() == null) {
328                 limit.getLeftHandSide().setType(getId());
329             }
330             addLimit(limit);
331
332         } else {
333             super.readChild(xr);
334         }
