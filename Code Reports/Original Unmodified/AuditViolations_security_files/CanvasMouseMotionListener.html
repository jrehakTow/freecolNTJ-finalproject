33 /**
34  * Listens to the mouse being moved at the level of the Canvas.
35  */
36 public final class CanvasMouseMotionListener extends AbstractCanvasListener
37     implements MouseMotionListener {
38
39     private static final Logger logger = Logger.getLogger(CanvasMouseMotionListener.class.getName());
40
41     /** Number of pixels that must be moved before a goto is enabled. */
42     private static final int DRAG_THRESHOLD = 16;
43
44     /**
45      * Temporary variable for checking if we need to recalculate the
46      * path when dragging units.
47      */
48     private Tile lastTile;
49     
50
51     /**
52      * Creates a new listener for mouse movement.
53      *
54      * @param freeColClient The <code>FreeColClient</code> for the game.
55      * @param canvas The <code>Canvas</code> to listen on.
56      */
57     public CanvasMouseMotionListener(FreeColClient freeColClient,
58                                      Canvas canvas) {
59         super(freeColClient, canvas);
60     }
61
62
63     /**
64      * Invoked when the mouse has been moved.
65      *
66      * @param e The MouseEvent that holds all the information.
67      */
68     @Override
69     public void mouseMoved(MouseEvent e) {
70         if (e.getY() >= AUTO_SCROLL_SPACE) {
71             performAutoScrollIfActive(e);
72         }
73
74         if (canvas.isGotoStarted()) {
75             if (canvas.getActiveUnit() == null) {
76                 canvas.stopGoto();
77             }
78
79             Tile tile = canvas.convertToMapTile(e.getX(), e.getY());
80
81             if (tile != null) {
82                 if (lastTile != tile) {
83                     Unit active = canvas.getActiveUnit();
84                     lastTile = tile;
85                     if (active != null && active.getTile() != tile) {
86                         PathNode dragPath = active.findPath(tile);
87                         canvas.setGotoPath(dragPath);
88                     } else {
89                         canvas.setGotoPath(null);
90                     }
91                 }
92             }
93         }
94     }
95
96     /**
97      * Invoked when the mouse has been dragged.
98      *
99      * @param e The MouseEvent that holds all the information.
100      */
101     @Override
102     public void mouseDragged(MouseEvent e) {
103
104         performDragScrollIfActive(e);
105
106         Tile tile = canvas.convertToMapTile(e.getX(), e.getY());
107         if (tile != null
108             && ((e.getModifiers() & MouseEvent.BUTTON1_MASK)
109                 == MouseEvent.BUTTON1_MASK)) {
110             // only perform the goto for the left mouse button
111             if (canvas.isGotoStarted()) {
112                 Unit active = canvas.getActiveUnit();
113                 if (active == null) {
114                     canvas.stopGoto();
115                 } else if (lastTile != tile) {
116                     lastTile = tile;
117                     PathNode dragPath = active.findPath(tile);
118                     canvas.setGotoPath(dragPath);
119                 }
120             } else {
121                 // Only start a goto if the drag is 16 pixels or more
122                 Point dragPoint = canvas.getDragPoint();
123                 int deltaX = Math.abs(e.getX() - dragPoint.x);
124                 int deltaY = Math.abs(e.getY() - dragPoint.y);
125                 if (deltaX >= DRAG_THRESHOLD || deltaY >= DRAG_THRESHOLD) {
126                     canvas.startGoto();
127                 }
128             }
129         }
130     }
