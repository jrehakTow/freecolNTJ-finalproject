135     public void update(BuildableType buildable) {
136         removeAll();
137         final ImageLibrary lib = ((SwingGUI)freeColClient.getGUI())
138             .getTileImageLibrary();
139         final Font font = FontLibrary.createFont(FontLibrary.FontType.NORMAL,
140             FontLibrary.FontSize.SMALLER, lib.getScaleFactor());
141
142         if (buildable == null) {
143             String clickToBuild = Messages.message(getDefaultLabel());
144             int breakingPoint = getBreakingPoint(clickToBuild);
145             if (breakingPoint > 0) {
146                 JLabel label0 = new JLabel(
147                     clickToBuild.substring(0, breakingPoint));
148                 label0.setFont(font);
149                 add(label0, "span, align center");
150                 JLabel label1 = new JLabel(
151                     clickToBuild.substring(breakingPoint + 1));
152                 label1.setFont(font);
153                 add(label1, "span, align center");
154             } else {
155                 JLabel label = new JLabel(clickToBuild);
156                 label.setFont(font);
157                 add(label, "span, align center");
158             }
159         } else {
160             int turns = colony.getTurnsToComplete(buildable);
161             Image image = lib.getSmallBuildableImage(
162                 buildable, colony.getOwner());
163             add(new JLabel(new ImageIcon(image)), "spany");
164             JLabel label0 = Utility.localizedLabel(buildable.getCurrentlyBuildingLabel());
165             label0.setFont(font);
166             add(label0);
167             JLabel label1 = Utility.localizedLabel(StringTemplate
168                 .template("constructionPanel.turnsToComplete")
169                 .addName("%number%", Turn.getTurnsText(turns)));
170             label1.setFont(font);
171             add(label1);
172
173             for (AbstractGoods ag : buildable.getRequiredGoods()) {
174                 int amountNeeded = ag.getAmount();
175                 int amountAvailable = colony.getGoodsCount(ag.getType());
176                 int amountProduced = colony.getAdjustedNetProductionOf(ag.getType());
177                 add(new FreeColProgressBar(ag.getType(), 0,
178                                            amountNeeded, amountAvailable, amountProduced),
179                     "height 20:");
180             }
181         }
182
183         revalidate();
184         repaint();
185     }
