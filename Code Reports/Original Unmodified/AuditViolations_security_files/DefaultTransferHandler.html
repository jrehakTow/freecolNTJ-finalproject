484         final Unit unit = unitLabel.getUnit();
485         if (!unit.hasAbility(Ability.CAN_BE_EQUIPPED)
486             || unit.getRole().hasAbility(Ability.ESTABLISH_MISSION)) {
487             // Do not equip missionaries.  The test below will succeed
488             // when dragging incompatible goods (anything:-) because
489             // there is no actual missionary equipment.
490             return false;
491         }
492
493         for (Role role : unit.getAvailableRoles(null)) {
494             if (role.isDefaultRole()) continue;
495             List<AbstractGoods> required = unit.getGoodsDifference(role, 1);
496             int count;
497             if (required.size() == 1
498                 && required.get(0).getType() == goods.getType()
499                 && (count = Math.min(role.getMaximumCount(),
500                         goods.getAmount() / required.get(0).getAmount())) > 0
501                 && (role != unit.getRole() || count != unit.getRoleCount())) {
502                 freeColClient.getInGameController()
503                     .equipUnitForRole(unit, role, count);
504                 unitLabel.updateIcon();
505                 return true;
506             }
507         }
508         return false;
509     }
