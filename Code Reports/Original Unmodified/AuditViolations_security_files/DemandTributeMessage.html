36 /**
37  * The message sent when demanding tribute from a native settlement.
38  */
39 public class DemandTributeMessage extends DOMMessage {
40
41     public static final String TAG = "demandTribute";
42     private static final String DIRECTION_TAG = "direction";
43     private static final String UNIT_TAG = "unit";
44
45     /** The identifier of the object demanding tribute. */
46     private final String unitId;
47
48     /** The direction the demand is made. */
49     private final String directionString;
50
51
52     /**
53      * Create a new <code>DemandTributeMessage</code> with the
54      * supplied unit and direction.
55      *
56      * @param unit The <code>Unit</code> that is demanding.
57      * @param direction The <code>Direction</code> the unit is looking.
58      */
59     public DemandTributeMessage(Unit unit, Direction direction) {
60         super(getTagName());
61
62         this.unitId = unit.getId();
63         this.directionString = String.valueOf(direction);
64     }
65
66     /**
67      * Create a new <code>DemandTributeMessage</code> from a
68      * supplied element.
69      *
70      * @param game The <code>Game</code> this message belongs to.
71      * @param element The <code>Element</code> to use to create the message.
72      */
73     public DemandTributeMessage(Game game, Element element) {
74         super(getTagName());
75
76         this.unitId = getStringAttribute(element, UNIT_TAG);
77         this.directionString = getStringAttribute(element, DIRECTION_TAG);
78     }

81     /**
82      * Handle a "demandTribute"-message.
83      *
84      * @param server The <code>FreeColServer</code> handling the message.
85      * @param player The <code>Player</code> that sent the message.
86      * @param connection The <code>Connection</code> message was received on.
87      * @return An <code>Element</code> to update the originating
88      *     player with the result of the demand.
89      */
90     public Element handle(FreeColServer server, Player player,
91                           Connection connection) {
92         final ServerPlayer serverPlayer = server.getPlayer(connection);
93
94         Unit unit;
95         try {
96             unit = player.getOurFreeColGameObject(this.unitId, Unit.class);
97         } catch (Exception e) {
98             return serverPlayer.clientError(e.getMessage())
99                 .build(serverPlayer);
100         }
101         if (unit.isArmed()
102             || unit.hasAbility(Ability.DEMAND_TRIBUTE)) {
103             ; // ok
104         } else {
105             return serverPlayer.clientError("Unit is neither armed"
106                 + " nor able to demand tribute: " + unitId)
107                 .build(serverPlayer);
108         }
109
110         Tile tile;
111         try {
112             tile = unit.getNeighbourTile(this.directionString);
113         } catch (Exception e) {
114             return serverPlayer.clientError(e.getMessage())
115                 .build(serverPlayer);
116         }
117
118         ServerIndianSettlement settlement
119             = (ServerIndianSettlement)tile.getIndianSettlement();
120         if (settlement == null) {
121             return serverPlayer.clientError("There is native settlement at: "
122                 + tile.getId())
123                 .build(serverPlayer);
124         }
125
126         MoveType type = unit.getMoveType(tile);
127         if (type != MoveType.ATTACK_SETTLEMENT
128             && type != MoveType.ENTER_INDIAN_SETTLEMENT_WITH_SCOUT) {
129             return serverPlayer.clientError("Unable to demand tribute at: "
130                 + settlement.getName() + ": " + type.whyIllegal())
131                 .build(serverPlayer);
132         }
133
134         // Do the demand
135         return server.getInGameController()
136             .demandTribute(serverPlayer, unit, settlement)
137             .build(serverPlayer);
138     }
