275     /**
276      * Applies a collection of modifiers to the given float value.
277      *
278      * @param number The number to modify.
279      * @param turn An optional applicable <code>Turn</code>.
280      * @param mods The <code>Modifier</code>s to apply.
281      * @return The modified number.
282      */
283     public static float applyModifiers(float number, Turn turn,
284                                        Collection<Modifier> mods) {
285         if (mods == null || mods.isEmpty()) return number;
286         List<Modifier> modifiers = new ArrayList<>(mods);
287         Collections.sort(modifiers);
288         float result = number;
289         for (Modifier m : modifiers) {
290             float value = m.getValue(turn);
291             if (value == Modifier.UNKNOWN) return value;
292             result = m.apply(result, value);
293         }
294         return result;
295     }
