109     static Font createMainFont(String fontName, float scaleFactor) {
110         final float defaultSize = 12f * scaleFactor;
111         mainFont = null;

203      * 
204      * @param fontType How the font should look like.
205      * @param fontSize Its relative size.
206      * @param style The font style for choosing plain, bold or italic.
207      * @param scaleFactor The applied scale factor.
208      * @return The created Font.
209      */
210     public static Font createFont(FontType fontType, FontSize fontSize,
211                                   int style, float scaleFactor) {
212         float scaledSize = calcScaledSize(fontSize, scaleFactor);
213         String fontKey = getFontKey(fontType);
214         Font font = (fontKey == null)
215             ? mainFont
216             : ResourceManager.getFont(fontKey);
217         font = font.deriveFont(style, scaledSize);
218         return font;
219     }

234      */
235     public static Font createCompatibleFont(String string, FontType fontType,
236                                             FontSize fontSize,
237                                             int style, float scaleFactor) {
238         // TODO: Consider testing the normal font for compatibility and try
239         //       some or all other available fonts for complete/longest match:
240         //       header/simple->main->normal->simple/header->emergency
241         float scaledSize = calcScaledSize(fontSize, scaleFactor);
242         String fontKey = getFontKey(fontType);
243         Font font = null;
244         if(fontType != FontType.NORMAL) {
245             font = ResourceManager.getFont(fontKey);
246             if(font.canDisplayUpTo(string) != -1)
247                 font = null;
248         }
249         if(font == null) {
250             fontKey = getFontKey(FontType.NORMAL);
251             font = (fontKey == null)
252                 ? mainFont
253                 : ResourceManager.getFont(fontKey);
254         }
255         font = font.deriveFont(style, scaledSize);
256         return font;
257     }

259     private static float calcScaledSize(FontSize fontSize, float scaleFactor) {
260         float pixelSize;
261         switch(fontSize) {
262             default:
263                 logger.warning("Unknown FontSize");
264             case TINY:
265                 pixelSize = 12f;
266                 break;
267             case SMALLER:
268                 pixelSize = 16f;
269                 break;
270             case SMALL:
271                 pixelSize = 24f;
272                 break;
273             case MEDIUM:
274                 pixelSize = 36f;
275                 break;
276             case BIG:
277                 pixelSize = 48f;
278         }
279         return pixelSize * scaleFactor;
280     }
