139     /**
140      * Configures the debug modes.
141      *
142      * @param optionValue The command line option.
143      * @return True if the modes were set as specified.
144      */
145     public static boolean setDebugModes(String optionValue) {
146         if (optionValue == null) return false;
147         if (optionValue.isEmpty()) return true;
148         // @compat 0.10.x
149         try {
150             int i = Integer.parseInt(optionValue);
151             switch (i) {
152             case 3:
153                 enableDebugMode(DebugMode.COMMS);
154                 // Fall through
155             case 2:
156                 enableDebugMode(DebugMode.INIT);
157                 // Fall through
158             case 1:
159                 enableDebugMode(DebugMode.MENUS);
160                 return true;
161             default:
162                 return false;
163             }
164         } catch (NumberFormatException nfe) {}
165         // end @compat
166
167         for (String s : optionValue.split(",")) {
168             try {
169                 DebugMode mode = Enum.valueOf(DebugMode.class,
170                                               s.toUpperCase(Locale.US));
171                 enableDebugMode(mode);
172             } catch (Exception e) {
173                 logger.warning("Unrecognized debug mode: " + optionValue);
174                 return false;
175             }
176         }
177         return true;
178     }

252      * Try to complete a debug run if one is happening.
253      *
254      * @param freeColClient The <code>FreeColClient</code> for the game.
255      * @param force Force early completion of a run.
256      * @return True if a debug run was completed.
257      */
258     public static boolean finishDebugRun(FreeColClient freeColClient,
259                                          boolean force) {
260         if (getDebugRunTurns() < 0) return false; // Not a debug run
261         if (getDebugRunTurns() > 0 && !force) return false; // Still going
262         // Zero => signalEndDebugRun was called
263         setDebugRunTurns(-1);
264
265         if (getDebugRunSave() != null) {
266             FreeColServer fcs = freeColClient.getFreeColServer();
267             if (fcs != null) {
268                 try {
269                     fcs.saveGame(new File(".", getDebugRunSave()),
270                                  freeColClient.getClientOptions(), null);
271                 } catch (IOException e) {}
272             }
273             freeColClient.quit();
274         }
275         return true;
276     }
