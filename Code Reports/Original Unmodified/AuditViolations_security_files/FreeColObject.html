954      * Newly created objects will prefer to refer to other newly

963      * @return The copied object, or null on error.
964      */
965     public <T extends FreeColObject> T copy(Game game, Class<T> returnClass) {

968             FreeColXMLReader xr = new FreeColXMLReader(new StringReader(this.serialize()));
969         ) {
970             ret = xr.copy(game, returnClass);
971         } catch (Exception e) {
972             logger.log(Level.WARNING, "Failed to copy: " + getId(), e);
973         }
974         return ret;
975     }
