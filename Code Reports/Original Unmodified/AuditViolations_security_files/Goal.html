40 /**
41  * A Goal is used to encapsulate a specific part of the
42  * decision-making process of an AI.
43  *
44  * Using a top-down approach, every {@link AIPlayer} has a set of Goals which,
45  * in turn, may have further subgoals. In combination, this tree of goals

47  *
48  * Units (each one wrapped in an {@link AIUnit} object) will be moved between
49  * existing Goal objects.

421      * or is responsible to sanitize this result itself before trying
422      * to {@link #yieldUnit(UnitType,AIObject)} based on it.
423      */
424     public float getYieldedUnitWeight(UnitType ut, AIObject o) {
425         //weights should normally be in range [0;1]
426         //if there is a matching unit, this will be overwritten
427         float unitWeight = 99f;
428
429         Iterator<AIUnit> uit = getOwnedAIUnitsIterator();
430         while (uit.hasNext()) {
431             AIUnit u = uit.next();
432             //all units in one goal have the same weight, so no need to compare
433             if (u.getUnit().getType().equals(ut)) {
434                 unitWeight = getAbsoluteWeight();
435             }
436         }
437         //check subgoals
438         Iterator<Goal> git = getSubGoalIterator();
439         while (git!=null && git.hasNext()) {
440             Goal g = git.next();
441             float newWeight = g.getYieldedUnitWeight(ut, o);
442             if (newWeight < unitWeight) {
443                 unitWeight = newWeight;
444             }
445         }
446         return unitWeight;
447     }

449     /**
450      * Removes a unit from the goal, potentially from a subgoal,
451      * and yields it to the caller.
452      *
453      * Returned unit should be the one with minimum absolute weight,
454      * see {@link #getYieldedUnitWeight(UnitType,AIObject)}.
455      *
456      * @param ut The {@link UnitType} wanted by the parent
457      * @param o The {@link AIObject} (should be AIPlayer or another Goal) calling this
458      * @return The {@link AIUnit} with minimal absolute weight.
459      * Note that this may be null if no matching unit is found!
460      */
461     public AIUnit yieldUnit(UnitType ut, AIObject o) {
462         float unitWeight = 99f;
463         AIUnit yieldedUnit = null;
464         boolean isOwnUnit = false;
465
466
467         Iterator<AIUnit> uit = getOwnedAIUnitsIterator();
468         while (uit.hasNext()) {
469             AIUnit u = uit.next();
470             //all units in one goal have the same weight, so no need to compare
471             if (u.getUnit().getType().equals(ut)) {
472                 unitWeight = getAbsoluteWeight();
473                 yieldedUnit = u;
474                 isOwnUnit = true;
475             }
476         }
477         //check subgoals
478         Iterator<Goal> git = getSubGoalIterator();
479         while (git!=null && git.hasNext()) {
480             Goal g = git.next();
481             float newWeight = g.getYieldedUnitWeight(ut, o);
482             if (newWeight < unitWeight) {
483                 unitWeight = newWeight;
484                 yieldedUnit = g.yieldUnit(ut, o);
485                 isOwnUnit = false;
486             }
487         }
488         if (isOwnUnit) {
489             removeUnit(yieldedUnit);
490             needsPlanning = true;
491         }
492         return yieldedUnit;
493     }

498      * This may be the case if the Unit has been removed from the game between turns.

509                                + u.getGoal().getGoalDescription());

511             }

515             //}
516         }
517     }

630 /* How this is supposed to work:
631  *
632  * Assuming the AIPlayer has a set of goals...
633  * At the start of a turn, it will call
634  *
635  *    setNeedsPlanningRecursive(true);
636  *
637  * for all its goals. This will set all goals in a state of needing a check.
638  * After that, it will iterate through its goals in a WHILE-loop:
639  *
640  *    boolean furtherPlanning = true;
641  *    while (furtherPlanning) {
642  *        FOR ALL GOALS g DO {
643  *            g.doPlanning();
644  *        }
645  *
646  *        SATISFY_UNIT_REQUESTS();
647  *        HANDLE_REMAINING_REQUESTS();
648  *
649  *        furtherPlanning = false;
650  *        FOR ALL GOALS g DO {
651  *            furtherPlanning = (furtherPlanning || g.needsPlanning());
652  *        }
653  *    }
654  *
655  * The first FOR-loop will recursively reach all existing goals and plan()
656  * each of them once, bottom-up.
657  *
658  * After that, unit requests from the goals will have piled up.
659  * SATISFY_UNIT_REQUESTS() will try to satisfy as many as possible,
660  * in order of importance, by using the units that were created between turns
661  * or that have been given back by other goals.
662  *
663  * HANDLE_REMAINING_REQUESTS() will try to do something about remaining requests,
664  * for example by buying/building new units, setting up new goals etc.
665  *
666  * This, as well as goals called later during this process moving a unit to a former goal,
667  * may have set one or more of the goals to needsPlanning=true. In the second FOR-loop,
668  * we check whether this is the case, to eventually repeat the process.
669  *
670  * Eventually, the WHILE-loop will exit. Any units in the goals will have
671  * been used for this turn. There may still be some excess units that haven't been
672  * requested by any Goal. The AIPlayer may now choose to create a new Goal for these,
673  * add them to an existing Goal (which will make use of them next turn),
674  * or just keep them unused to have something to deal out next turn.
675  */
