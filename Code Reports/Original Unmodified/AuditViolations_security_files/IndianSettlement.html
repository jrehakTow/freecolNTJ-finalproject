699     /**
700      * Gets the amount of gold this <code>IndianSettlment</code>
701      * is willing to pay for the given <code>Goods</code>.
702      *
703      * It is only meaningful to call this method from the server,
704      * since the settlement's {@link GoodsContainer} is hidden from
705      * the clients.  The AI uses it though so it stays here for now.
706      * Note that it takes no account of whether the native player
707      * actually has the gold.
708      *
709      * FIXME: this is rancid with magic numbers.
710      *
711      * @param type The type of <code>Goods</code> to price.
712      * @param amount The amount of <code>Goods</code> to price.
713      * @return The price.
714      */
715     public int getPriceToBuy(GoodsType type, int amount) {
716         if (amount > GoodsContainer.CARGO_SIZE) {
717             throw new IllegalArgumentException("Amount > "
718                 + GoodsContainer.CARGO_SIZE);
719         }
720
721         int price = 0;
722         if (type.isMilitaryGoods()) {
723             // Might return zero if a surplus is present
724             price = getMilitaryGoodsPriceToBuy(type, amount);
725         }
726         if (price == 0) {
727             price = getNormalGoodsPriceToBuy(type, amount);
728         }
729
730         // Apply wanted bonus
731         final int wantedBase = 100; // Granularity for wanted bonus
732         final int wantedBonus // Premium paid for wanted goods types
733             = (type == wantedGoods[0]) ? 150
734             : (type == wantedGoods[1]) ? 125
735             : (type == wantedGoods[2]) ? 110
736             : 100;
737         // Do not simplify with *=, we want the integer truncation.
738         price = wantedBonus * price / wantedBase;
739
740         logger.finest("Full price(" + amount + " " + type + ")"
741                       + " -> " + price);
742         return price;
743     }

745     /**
746      * Price some goods according to the amount present in the settlement.
747      *
748      * @param type The type of goods for sale.
749      * @param amount The amount of goods for sale.
750      * @return A price for the goods.
751      */
752     private int getNormalGoodsPriceToBuy(GoodsType type, int amount) {
753         final int tradeGoodsAdd = 20; // Fake additional trade goods present
754         final int capacity = getGoodsCapacity();
755         int current = getGoodsCount(type);
756
757         // Increase effective stock if its raw material is produced here.
758         GoodsType rawType = type.getInputType();
759         if (rawType != null) {
760             int rawProduction = getMaximumProduction(rawType);
761             int add = (rawProduction < 5) ? 10 * rawProduction
762                 : (rawProduction < 10) ? 5 * rawProduction + 25
763                 : (rawProduction < 20) ? 2 * rawProduction + 55
764                 : 100;
765             // Decrease bonus in proportion to current stock, up to capacity.
766             add = add * Math.max(0, capacity - current) / capacity;
767             current += add;
768         } else if (type.isTradeGoods()) {
769             // Small artificial increase of the trade goods stored.
770             current += tradeGoodsAdd;
771         }
772
773         // Only interested in the amount of goods that keeps the
774         // total under the threshold.
775         int retain = Math.min(getWantedGoodsAmount(type), capacity);
776         int valued = (retain <= current) ? 0
777             : Math.min(amount, retain - current);
778
779         // Unit price then is maximum price plus the bonus for the
780         // settlement type, reduced by the proportion of goods present.
781         int unitPrice = (GOODS_BASE_PRICE + getType().getTradeBonus())
782             * Math.max(0, capacity - current) / capacity;
783
784         // But farmed goods are always less interesting.
785         // and small settlements are not interested in building.
786         if (type.isFarmed() || type.isRawBuildingMaterial()) unitPrice /= 2;
787
788         // Only pay for the portion that is valued.
789         return (unitPrice < 0) ? 0 : valued * unitPrice;
790     }

792     /**
793      * Calculates how much of the given goods type this settlement
794      * wants and should retain.
795      *
796      * @param type The <code>GoodsType</code>.
797      * @return The amount of goods wanted.
798      */
799     protected int getWantedGoodsAmount(GoodsType type) {
800         if (getUnitCount() <= 0) return 0;
801
802         final Specification spec = getSpecification();
803         final UnitType unitType = getFirstUnit().getType();
804         final List<Role> militaryRoles = Role.getAvailableRoles(getOwner(),
805             unitType, spec.getMilitaryRoles());
806
807         if (type.isMilitaryGoods()) { // Retain enough goods to fully arm
808             return sum(getOwnedUnits(),
809                        u -> !militaryRoles.contains(u.getRole()),
810                        u -> AbstractGoods.getCount(type, 
811                            u.getGoodsDifference(militaryRoles.get(0), 1)));
812         }
813
814         int consumption = getConsumptionOf(type);
815         if (type == spec.getPrimaryFoodType()) {
816             // Food is perishable, do not try to retain that much
817             return Math.max(40, consumption * 3);
818         }
819         if (type.isTradeGoods() || type.isNewWorldLuxuryType()
820             || type.isRefined()) {
821             // Aim for 10 years supply, resupply is doubtful
822             return Math.max(80, consumption * 20);
823         }
824         // Just keep some around
825         return 2 * getUnitCount();
826     }

828     /**
829      * Price some goods that have military value to the settlement.
830      *
831      * @param type The type of goods for sale.
832      * @param amount The amount of goods for sale.
833      * @return A price for the goods.
834      */
835     private int getMilitaryGoodsPriceToBuy(GoodsType type, int amount) {
836         final int full = GOODS_BASE_PRICE + getType().getTradeBonus();
837         int required = getWantedGoodsAmount(type);
838         if (required == 0) return 0; // Do not pay military price
839
840         // If the settlement can use more than half of the goods on offer,
841         // then pay top dollar for the lot.  Otherwise only pay the premium
842         // price for the part they need and refer the remaining amount to
843         // the normal goods pricing.
844         int valued = Math.max(0, required - getGoodsCount(type));
845         int price = (valued > amount / 2) ? full * amount
846             : valued * full + getNormalGoodsPriceToBuy(type, amount - valued);
847         logger.finest("Military price(" + amount + " " + type + ")"
848                       + " valued=" + valued
849                       + " -> " + price);
850         return price;
851     }
