124     private static BufferedImage getSkin(FreeColClient freeColClient) {
125         Player player = freeColClient.getMyPlayer();
126         String key = player.isRebel() ? "image.skin.InformationPanel.rebel"
127             : "image.skin.InformationPanel." + player.getNationResourceKey();
128         if(!ResourceManager.hasImageResource(key))
