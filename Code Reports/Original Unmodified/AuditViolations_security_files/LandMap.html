167     /**
168      * Generate the land map using the given generator type.
169      *
170      * @param type The generator type.
171      * @param minimumNumberOfTiles The minimum land tiles to generate.
172      * @param preferredDistanceToEdge The preferred distance to the map edge.
173      * @param random A pseudo random number source.
174      */
175     private void generate(int type, int preferredDistanceToEdge,
176                           int minimumNumberOfTiles, Random random) {
177         switch (type) {
178         case MapGeneratorOptions.LAND_GENERATOR_CLASSIC:
179             createClassicLandMap(preferredDistanceToEdge, minimumNumberOfTiles,
180                                  random);
181             break;
182         case MapGeneratorOptions.LAND_GENERATOR_CONTINENT:
183             // Create one landmass of 75%, start it somewhere near the
184             // center, then fill up with small islands.
185             addPolarRegions();
186             int contsize = (minimumNumberOfTiles * 75) / 100;
187             addLandMass(contsize, contsize, width/2, height/4
188                         + randomInt(logger, "Landmass", random, height/2),
189                         preferredDistanceToEdge, random);
190             while (numberOfLandTiles < minimumNumberOfTiles) {
191                 addLandMass(15, 25, -1, -1, preferredDistanceToEdge, random);
192             }
193             break;
194         case MapGeneratorOptions.LAND_GENERATOR_ARCHIPELAGO:
195             // Create 5 islands of 10% each, then delegate to the Islands
196             // generator.
197             addPolarRegions();
198             int archsize = (minimumNumberOfTiles * 10) / 100;
199             for (int i = 0; i < 5; i++) {
200                 addLandMass(archsize - 5, archsize + 5, -1, -1,
201                             preferredDistanceToEdge, random);
202             }
203             // Fall through
204         case MapGeneratorOptions.LAND_GENERATOR_ISLANDS:
205             // Create islands of 25..75 tiles.
206             addPolarRegions();
207             while (numberOfLandTiles < minimumNumberOfTiles) {
208                 int s = randomInt(logger, "Island", random, 50) + 25;
209                 addLandMass(25, s, -1, -1,
210                             preferredDistanceToEdge, random);
211             }
212             break;
213         }
214         cleanMap();
215     }

217     /**
218      * Create the standard FreeCol land map.
219      *
220      * @param preferredDistanceToEdge The nominal edge clearance.
221      * @param minimumNumberOfTiles Lower bound for the tiles to create.
222      * @param random A pseudo-random number source.
223      */
224     private void createClassicLandMap(int preferredDistanceToEdge,
225                                       int minimumNumberOfTiles,
226                                       Random random) {
227         int x, y;
228         while (numberOfLandTiles < minimumNumberOfTiles) {
229             int failCounter = 0;
230             do {
231                 x = (randomInt(logger, "ClassicW", random,
232                                width - preferredDistanceToEdge * 4))
233                     + preferredDistanceToEdge * 2;
234                 y = (randomInt(logger, "ClassicH", random,
235                                height - preferredDistanceToEdge * 4))
236                     + preferredDistanceToEdge * 2;
237                 failCounter++;
238                 // If landmass% is set too high, this loop may fail to
239                 // find a free tile.  Decrease necessary minimum over
240                 // time, so that this process will eventually succeed.
241                 if (failCounter > 100) {
242                     failCounter = 0;
243                     minimumNumberOfTiles--;
244                     break;
245                 }
246             } while (map[x][y]);
247             setLand(x, y, preferredDistanceToEdge, random);
248         }
249         addPolarRegions();
250     }

352     /**
353      * Determines, based on position, number of adjacent land tiles
354      * and some random factor, whether a given map position should be
355      * set to land.  This is called for all valid map positions
356      * adjacent to a position that has been set to land by
357      * #setLand(int,int), and may recursively call setLand for the
358      * current position.
359      *
360      * @param x The x coordinate to grow land at.
361      * @param y The y coordinate to grow land at.
362      * @param preferredDistanceToEdge The preferred distance to the map edge.
363      * @param random A pseudo random number source.
364      * @return The number of tiles set to land.
365      */
366     private int growLand(int x, int y, int preferredDistanceToEdge,
367                          Random random) {
368         if (map[x][y]) return 0; // Already land
369
370         // Generate a comparison value:
371         // Only if the number of adjacent land tiles is bigger than this value,
372         // this tile will be set to land.
373         // This value is part random, part based on position, that is:
374         // -1 in the center of the map, and growing to
375         // preferredDistanceToEdge (*2 for pole ends) at the maps edges.
376         int r = randomInt(logger, "Grow", random, 8)
377             + Math.max(-1,
378                 (1 + Math.max(preferredDistanceToEdge - Math.min(x, width-x),
379                     2 * preferredDistanceToEdge - Math.min(y, height-y))));
380
381         Position p = new Position(x, y);
382         int sum = count(Direction.values(), d -> {
383                 Position n = new Position(p, d);
384                 return n.isValid(width, height) && map[n.getX()][n.getY()];
385             });
386         return (sum > r) ? setLand(x, y, preferredDistanceToEdge, random) : 0;
387     }

391      * size up to maxSize, and adds it to the current map if it is at
392      * least minSize.
393      *
394      * @param minSize Minimum number of tiles in the land mass.
395      * @param maxSize Maximum number of tiles in the land mass.
396      * @param x Optional starting x coordinate (chosen randomly if negative).
397      * @param y Optional starting y coordinate (chosen randomly if negative).
398      * @param preferredDistanceToEdge The preferred distance to the map edge.
399      * @param random A pseudo random number source.
400      * @return The number of tiles added.
401      */
402     private int addLandMass(int minSize, int maxSize, int x, int y,
403                             int preferredDistanceToEdge, Random random) {
404         int size = 0;
405         boolean[][] newLand = new boolean[width][height];
406
407         // Pick a starting position that is sea without neighbouring land.
408         if (x < 0 || y < 0) {
409             do {
410                 x = randomInt(logger, "LandW", random, 
411                               width - preferredDistanceToEdge * 2)
412                     + preferredDistanceToEdge;
413                 y = randomInt(logger, "LandH", random,
414                               height - preferredDistanceToEdge * 2)
415                     + preferredDistanceToEdge;
416             } while (map[x][y] || !isSingleTile(x, y));
417         }
418
419         newLand[x][y] = true;
420         size++;
421
422         // Add all valid neighbour positions to list
423         Position p = new Position(x, y);
424         List<Position> l = newPositions(p, preferredDistanceToEdge);
425
426         // Get a random position from the list,
427         // set it to land,
428         // add its valid neighbours to the list
429         int enough = minSize + randomInt(logger, "LandSize", random,
430                                          maxSize - minSize + 1);
431         while (size < enough && !l.isEmpty()) {
432             int i = randomInt(logger, "Lsiz", random, l.size());
433             p = l.remove(i);
434
435             if (!newLand[p.getX()][p.getY()]) {
436                 newLand[p.getX()][p.getY()] = true;
437                 size++;
438                 l.addAll(newPositions(p, preferredDistanceToEdge));
439             }
440         }
441
442         // Add generated land to map if sufficiently large
443         if (size >= minSize) {
444             for (x = 0; x < width; x++) {
445                 for (y = 0; y < height; y++) {
446                     if (newLand[x][y] == true) {
447                         map[x][y] = true;
448                         numberOfLandTiles++;
449                     }
450                 }
451             }
452         }
453         return (size >= minSize) ? size : 0;
454     }
