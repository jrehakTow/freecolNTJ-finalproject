88      * Constructs a new Message with data from the given InputSource. The
89      * constructor to use if this is an INCOMING message.
90      * 
91      * @param inputSource The <code>InputSource</code> to get the XML-data
92      *            from.
93      * @exception IOException if thrown by the <code>InputSource</code>.
94      * @exception SAXException if thrown during parsing.
95      */
96     private Message(InputSource inputSource) throws SAXException, IOException {
97         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
98         Document tempDocument = null;
99         boolean dumpMsgOnError
100             = FreeColDebugger.isInDebugMode(FreeColDebugger.DebugMode.COMMS);
101         if (dumpMsgOnError) {
102             inputSource.setByteStream(new BufferedInputStream(inputSource.getByteStream()));
103
104             inputSource.getByteStream().mark(1000000);
105         }
106
107         try {
108             DocumentBuilder builder = factory.newDocumentBuilder();
109             tempDocument = builder.parse(inputSource);
110         } catch (ParserConfigurationException pce) {
111             // Parser with specified options can't be built
112             logger.log(Level.WARNING, "Parser error", pce);
113         } catch (IOException|SAXException ex) {
114             throw ex;
115         } catch (ArrayIndexOutOfBoundsException e) {
116             // Xerces throws ArrayIndexOutOfBoundsException when it barfs on
117             // some FreeCol messages. I'd like to see the messages upon which
118             // it barfs
119             if (dumpMsgOnError) {
120                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
121                 inputSource.getByteStream().reset();
122                 while (true) {
123                     int i = inputSource.getByteStream().read();
124                     if (-1 == i) {
125                         break;
126                     }
127                     baos.write(i);
128                 }
129                 logger.severe(baos.toString("UTF-8"));
130             }
131             throw e;
132         }
133         document = tempDocument;
134     }
