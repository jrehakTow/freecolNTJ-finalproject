188     /**
189      * Find out if a tile contains a suitable target for seek-and-destroy.
190      *
191      * FIXME: Package for access by a test only - necessary?
192      *
193      * @param attacker The attacking <code>Unit</code>.
194      * @param tile The <code>Tile</code> to attack into.
195      * @return True if an attack can be launched.
196      */
197     public boolean isTargetValidForSeekAndDestroy(Unit attacker, Tile tile) {
198         // Insist the attacker exists.
199         if (attacker == null) return false;
200
201         Player attackerPlayer = attacker.getOwner();
202
203         // Determine the defending player.
204         Settlement settlement = tile.getSettlement();
205         Unit defender = tile.getDefendingUnit(attacker);
206         Player defenderPlayer = (settlement != null) ? settlement.getOwner()
207             : (defender != null) ? defender.getOwner()
208             : null;
209
210         // Insist there be a defending player.
211         if (defenderPlayer == null) return false;
212
213         // Can not attack our own units.
214         if (attackerPlayer == defenderPlayer) return false;
215
216         // If European, do not attack if not at war.
217         // If native, do not attack if not at war and at least content.
218         // Otherwise some attacks are allowed even if not at war.
219         boolean atWar = attackerPlayer.atWarWith(defenderPlayer);
220         if (attackerPlayer.isEuropean()) {
221             if (!atWar) return false;
222         } else if (attackerPlayer.isIndian()) {
223             if (!atWar && attackerPlayer.getTension(defenderPlayer)
224                 .getLevel().compareTo(Tension.Level.CONTENT) <= 0) {
225                 return false;
226             }
227         }
228         return attacker.canAttack(defender);
229     }

235      * Log the missions of this player.
236      *
237      * @param reasons A map of reasons for the current mission by unit.
238      * @param lb A <code>LogBuilder</code> to log to.

240     protected void logMissions(java.util.Map<Unit, String> reasons,
241                                LogBuilder lb) {
242         List<AIUnit> units = getAIUnits();
243         Collections.sort(units, aiUnitLocationComparator);
244         for (AIUnit aiu : units) {
245             Unit u = aiu.getUnit();
246             String reason = reasons.get(u);
247             if (reason == null) reason = "OMITTED";
248             String ms = "NONE";
249             Location target = null;
250             if (aiu.hasMission()) {
251                 Mission m = aiu.getMission();
252                 ms = lastPart(m.getClass().toString(), ".");
253                 ms = ms.substring(0, ms.length() - "Mission".length());
254                 target = m.getTarget();
255             }
256
257             lb.add("\n  @",
258                 String.format("%-30s%-10s%-40s%-16s",
259                     chop(u.getLocation().toShortString(), 30),
260                     chop(reason, 10),
261                     chop(u.toShortString(), 40),
262                     chop(ms, 16)));
263             if (target != null) lb.add("->", target);
264         }
265     }

352     /**
353      * Makes every unit perform their mission.
354      *
355      * @param aiUnits A list of <code>AIUnit</code>s to perform missions.
356      * @param lb A <code>LogBuilder</code> to log to.
357      * @return A list of <code>AIUnit</code>s that have moves left.
358      */
359     protected List<AIUnit> doMissions(List<AIUnit> aiUnits, LogBuilder lb) {
360         lb.add("\n  Do Missions:");
361         List<AIUnit> result = new ArrayList<>();
362         for (AIUnit aiu : aiUnits) {
363             final Unit unit = aiu.getUnit();
364             if (unit == null || unit.isDisposed()) continue;
365             lb.add("\n  ", unit, " ");
366             try {
367                 aiu.doMission(lb);
368             } catch (Exception e) {
369                 logger.log(Level.WARNING, "doMissions failed for: " + aiu, e);
370             }
371             if (!unit.isDisposed() && unit.getMovesLeft() > 0) result.add(aiu);
372         }
373         return result;
374     }
