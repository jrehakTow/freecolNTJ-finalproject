383     /**

385      * name from the list of default names if possible.
386      *

389      * @return A suitable name.
390      */
391     public static String getRegionName(Player player, Region region) {
392         if (region.hasName()) {
393             return Messages.message(region.getLabel());
394         }
395         final net.sf.freecol.common.model.Map map = player.getGame().getMap();
396         String name;
397         int index;
398
399         // Try national names first.
400         final String prefix = makeRegionKey(player, region.getType());
401         requireRegionNames(player, region.getType());
402         synchronized (regionNameLock) {
403             List<String> names = regionNames.get(prefix);
404             while (!names.isEmpty()) {
405                 name = names.remove(0);
406                 if (map.getRegionByName(name) == null) return name;
407             }
408         }
409             
410         // There are a bunch of extra rivers not attached to a specific
411         // nation at model.other.region.river.*.
412         if (region.getType() == Region.RegionType.RIVER) {
413             requireRiverNames();
414             synchronized (riverNameLock) {
415                 while (!riverNames.isEmpty()) {
416                     name = riverNames.remove(0);
417                     if (map.getRegionByName(name) == null) return name;
418                 }
419             }
420         }
421
422         // Fall back to generic names.
423         synchronized (regionNameLock) {
424             index = regionIndex.get(prefix);
425         }
426         StringTemplate nn = player.getNationLabel();
427         do {
428             name = Messages.message(StringTemplate.label(" ")
429                 .addStringTemplate(nn)
