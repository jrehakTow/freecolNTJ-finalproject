473     /**
474      * Brings gifts to nice players with nearby colonies.

476      * @param randoms An array of random percentages.
477      * @param lb A <code>LogBuilder</code> to log to.
478      */
479     private void bringGifts(int[] randoms, LogBuilder lb) {
480         final Player player = getPlayer();
481         final CostDecider cd = CostDeciders.numberOfLegalTiles();
482         final int giftProbability = getSpecification()
483             .getInteger(GameOptions.GIFT_PROBABILITY);
484         int randomIdx = 0;
485         lb.mark();
486
487         for (IndianSettlement is : player.getIndianSettlements()) {
488             // Do not bring gifts all the time.
489             if (randoms[randomIdx++] >= giftProbability) continue;
490
491             // Check if the settlement has anything to give.
492             Goods gift = is.getRandomGift(getAIRandom());
493             if (gift == null) continue;
494
495             // Check if there are available units, and if there are already
496             // enough missions in operation.
497             List<Unit> availableUnits = new ArrayList<>();
498             int alreadyAssignedUnits = 0;
499             for (Unit ou : is.getOwnedUnits()) {
500                 AIUnit aiu = getAIUnit(ou);
501                 if (aiu == null) {
502                     continue;
503                 } else if (aiu.hasMission(IndianBringGiftMission.class)) {
504                     alreadyAssignedUnits++;
505                 } else if (Mission.invalidNewMissionReason(aiu) == null) {
506                     availableUnits.add(ou);
507                 }
508             }
509             if (alreadyAssignedUnits > MAX_NUMBER_OF_GIFTS_BEING_DELIVERED) {
510                 lb.add(is.getName(), " has ", alreadyAssignedUnits, 
511                        " already, ");
512                 continue;
513             } else if (availableUnits.isEmpty()) {
514                 lb.add(is.getName(), " has no gift units, ");
515                 continue;
516             }
517             // Pick a random available capable unit.
518             Unit unit = null;
519             AIUnit aiUnit = null;
520             Tile home = is.getTile();
521             while (unit == null && !availableUnits.isEmpty()) {
522                 Unit u = availableUnits.get(randomInt(logger, "Gift unit",
523                         getAIRandom(), availableUnits.size()));
524                 availableUnits.remove(u);

526                 if (IndianBringGiftMission.invalidReason(aiUnit) == null
527                     && u.findPath(u.getTile(), home, null, cd) != null) {
528                     unit = u;
529                 }
530             }
531             if (unit == null) {
532                 lb.add(is.getName(), " found no gift unit, ");
533                 continue;
534             }
535
536             // Collect nearby colonies.  Filter out ones which are uncontacted,
537             // unreachable or otherwise unsuitable.  Score the rest on alarm
538             // and distance.
539             List<RandomChoice<Colony>> nearbyColonies = new ArrayList<>();
540             for (Tile t : home.getSurroundingTiles(MAX_DISTANCE_TO_BRING_GIFTS)) {
541                 Colony c = t.getColony();
542                 PathNode path;
543                 if (c == null
544                     || !is.hasContacted(c.getOwner())
545                     || IndianBringGiftMission.invalidReason(aiUnit, c) != null

547                                              null, cd)) == null) continue;

549                 nearbyColonies.add(new RandomChoice<>(c,
550                         1000000 / alarm / path.getTotalTurns()));
551             }
552
553             // If there are any suitable colonies, pick a random one
554             // to send a gift to.
555             if (nearbyColonies.isEmpty()) {
556                 lb.add(is.getName(), " found no gift colonies, ");
557                 continue;
558             }
559             Colony target = RandomChoice.getWeightedRandom(logger,
560                 "Choose gift colony", nearbyColonies, getAIRandom());

566             Mission m = new IndianBringGiftMission(getAIMain(), aiUnit, target);
567             lb.add(m, " gift from ", is.getName(),
568                    " to ", target.getName(), ", ");
569         }
570         if (lb.grew("\n  Gifts: ")) lb.shrink(", ");
571     }

587         for (IndianSettlement is : player.getIndianSettlements()) {
588             // Do not demand tribute all of the time.
589             if (randoms[randomIdx++] >= demandProbability) continue;
590
591             // Check if there are available units, and if there are already
592             // enough missions in operation.

594             int alreadyAssignedUnits = 0;

610                 lb.add(is.getName(), " has no demand units, ");
611                 continue;
612             }
613             // Pick a random available capable unit.
614             Tile home = is.getTile();
615             Unit unit = null;
616             AIUnit aiUnit = null;
617             while (unit == null && !availableUnits.isEmpty()) {
618                 Unit u = availableUnits.get(randomInt(logger, "Demand unit",
619                         getAIRandom(), availableUnits.size()));
620                 availableUnits.remove(u);
621                 aiUnit = getAIUnit(u);
622                 if (IndianDemandMission.invalidReason(aiUnit) == null
623                     && u.findPath(u.getTile(), home, null, cd) != null) {
624                     unit = u;
625                 }
626             }

628                 lb.add(is.getName(), " found no demand unit, ");
629                 continue;
630             }
631
632             // Collect nearby colonies.  Filter out ones which are unreachable
633             // or with which the settlement is on adequate terms.

635             for (Tile t : home.getSurroundingTiles(MAX_DISTANCE_TO_MAKE_DEMANDS)) {

651             // Sometimes a random one, sometimes the weakest, sometimes the
652             // most annoying.
653             if (nearbyColonies.isEmpty()) {
654                 lb.add(is.getName(), " found no demand colonies, ");
655                 continue;
656             }
657             Colony target = RandomChoice.getWeightedRandom(logger,
658                 "Choose demand colony", nearbyColonies, getAIRandom());
659             if (target == null) {
660                 lb.add(is.getName(), " found no demand target, ");
661                 continue;
662             }
663
664             // Send the unit.
665             Mission m = new IndianDemandMission(getAIMain(), aiUnit, target);
666             lb.add("At ", is.getName(), " ", m,
667                    " will demand of ", target, ", ");
668         }
669         if (lb.grew("\n  Tribute: ")) lb.shrink(", ");
670     }

821      * {@inheritDoc}
822      */
823     @Override
824     public int buyProposition(Unit unit, Settlement settlement,
825                               Goods goods, int gold) {
826         logger.finest("Entering method buyProposition");
827         Specification spec = getSpecification();
828         IndianSettlement is = (IndianSettlement) settlement;
829         Player buyer = unit.getOwner();
830         String goldKey = "tradeGold#" + goods.getType().getId()
831             + "#" + goods.getAmount() + "#" + settlement.getId();
832         String hagglingKey = "tradeHaggling#" + unit.getId();
833         int price;
834         Integer registered = sessionRegister.get(goldKey);
835         if (registered == null) {
836             price = is.getPriceToSell(goods);
837             switch (is.getAlarm(buyer).getLevel()) {
838             case HAPPY: case CONTENT:
839                 break;
840             case DISPLEASED:
841                 price *= 2;
842                 break;
843             default:
844                 return Constants.NO_TRADE_HOSTILE;
845             }
846             Set<Modifier> modifiers = new HashSet<>();
847             if (is.hasMissionary(buyer)
848                 && spec.getBoolean(GameOptions.ENHANCED_MISSIONARIES)) {
849                 Unit u = is.getMissionary();
850                 modifiers.addAll(u.getMissionaryTradeModifiers(false));
851             }
852             if (unit.isNaval()) {
853                 modifiers.addAll(getShipTradePenalties(false));
854             }
855             price = (int)FeatureContainer.applyModifiers((float)price,
856                 getGame().getTurn(), modifiers);
857             sessionRegister.put(goldKey, price);
858             return price;
859         }
860         price = registered;
861         if (price < 0 || price == gold) return price;
862         if (gold < (price * 9) / 10) {
863             logger.warning("Cheating attempt: sending offer too low");
864             sessionRegister.put(goldKey, -1);
865             return Constants.NO_TRADE;
866         }
867
868         int haggling = 1;
869         if (sessionRegister.containsKey(hagglingKey)) {
870             haggling = sessionRegister.get(hagglingKey);
871         }
872         if (randomInt(logger, "Haggle-buy", getAIRandom(), 3 + haggling) >= 3) {
873             sessionRegister.put(goldKey, -1);
874             return Constants.NO_TRADE_HAGGLE;
875         }
876         sessionRegister.put(goldKey, gold);
877         sessionRegister.put(hagglingKey, haggling + 1);
878         return gold;
879     }
