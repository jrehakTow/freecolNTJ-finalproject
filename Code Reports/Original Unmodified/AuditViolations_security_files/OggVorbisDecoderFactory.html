340         /**
341          * Decode the PCM data.
342          *
343          * @param samples The number of samples to decode.
344          * @return The number of bytes waiting in the conversion buffer to
345          *     be written.
346          */
347         private int decodePacket(int samples) {
348             int range = (samples < convBuf.length) ? samples : convBuf.length;
349             for (int i = 0; i < orbisInfo.channels; i++) {
350                 int sampleIndex = i * 2;
351                 for (int j = 0; j < range; j++) {
352                     // Retrieve the PCM
353                     int value = (int)(pcmData[0][i][pcmi[i] + j] * 32767.0f);
354                     // Clip to signed 16 bit
355                     if (value > 32767) value = 32767;
356                     else if (value < -32768) value = -32768;
357                     // Stuff into the conversion buffer, little endian
358                     convBuf[sampleIndex] = (byte)(value);
359                     convBuf[sampleIndex + 1] = (byte)(value >>> 8);
360                     // Jump forward (interleaving channels)
361                     sampleIndex += 2 * (orbisInfo.channels);
362                 }
363             }
364             return range;
365         }
