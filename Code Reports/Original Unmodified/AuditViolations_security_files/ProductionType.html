304      *
305      * @param goodsType An optional <code>GoodsType</code> to restrict the

309      */
310     public static ProductionType getBestProductionType(GoodsType goodsType,
311         Collection<ProductionType> types) {
312         ProductionType best = null;
313         int bestSum = 0;
314         for (ProductionType pt : types) {
315             int sum = pt.getOutputs().stream()
316                 .filter(o -> goodsType == null || goodsType == o.getType())
317                 .mapToInt(AbstractGoods::getAmount).sum();
318             if (bestSum < sum) {
319                 bestSum = sum;
320                 best = pt;
321             }
322         }
323         return best;
324     }
