111         final int groupBy = options.getInteger(ClientOptions.MESSAGES_GROUP_BY);

117
118         Object source = this;
119         ModelMessage.MessageType type = null;
120         for (ModelMessage message : messages) {
121             // Add headline if the grouping changed
122             switch (groupBy) {
123             case ClientOptions.MESSAGES_GROUP_BY_SOURCE:
124                 FreeColGameObject messageSource = game.getMessageSource(message);
125                 if (messageSource != source) {
126                     source = messageSource;
127                     reportPanel.add(getHeadline(messageSource), "newline 20, skip");

130             case ClientOptions.MESSAGES_GROUP_BY_TYPE:
131                 if (message.getMessageType() != type) {
132                     type = message.getMessageType();

134                         message.getMessageType(), FontLibrary.FontSize.SMALL);
135                     reportPanel.add(headline, "newline 20, skip, span");

138             default:

140             }
141             
142             JComponent component = new JLabel();

144             final ImageLibrary lib = getImageLibrary();
145             if (messageDisplay != null) {
146                 Image image = lib.getObjectImage(messageDisplay, 1f);

150                     JButton button = Utility.getLinkButton(null, icon,
151                         messageDisplay.getId());
152                     button.addActionListener(this);
153                     component = button;
154                 } else if (messageDisplay instanceof Unit) {
155                     JButton button = Utility.getLinkButton(null, icon,
156                         ((Unit)messageDisplay).up().getId());
157                     button.addActionListener(this);
158                     component = button;
159                 } else { // includes Player

163

237     private JComponent getHeadline(FreeColGameObject source) {
238         String text;
239         String commandId = null;
240         if (source == null) {
241             text = "";
242         } else if (source instanceof Player) {
243             Player player = (Player) source;
244             StringTemplate template = StringTemplate
245                 .template("report.turn.playerNation")
246                 .addName("%player%", player.getName())
247                 .addStringTemplate("%nation%", player.getNationLabel());
248             text = Messages.message(template);
249         } else if (source instanceof Europe) {
250             Europe europe = (Europe) source;
251             text = Messages.getName(europe);
252             commandId = europe.getId();
253         } else if (source instanceof Market) {
254             Market market = (Market) source;
255             StringTemplate template = market.getOwner().getMarketName();
256             text = Messages.message(template);
257             commandId = getMyPlayer().getEurope().getId();
258         } else if (source instanceof Colony) {
259             final Colony colony = (Colony) source;
260             text = colony.getName();
261             commandId = colony.getId();
262         } else if (source instanceof Unit) {
263             final Unit unit = (Unit) source;
264             text = unit.getDescription(Unit.UnitLabelType.NATIONAL);
265             commandId = unit.getLocation().getId();
266         } else if (source instanceof Tile) {
267             final Tile tile = (Tile) source;
268             StringTemplate template = tile.getLocationLabelFor(getMyPlayer());
269             text = Messages.message(template);
270             commandId = tile.getId();
271         } else if (source instanceof Nameable) {
272             text = ((Nameable) source).getName();
273         } else {
274             text = source.toString();
275         }
276
277         Font font = FontLibrary.createCompatibleFont(text,
278             FontLibrary.FontType.HEADER, FontLibrary.FontSize.SMALL);
279         JComponent headline;
280         if(commandId != null) {
281             JButton button = new JButton(text);
282             button.addActionListener(this);
283             button.setActionCommand(commandId);
284             headline = button;
285             headline.setForeground(Utility.LINK_COLOR);
286         } else {
287             headline = new JLabel(text);
288         }
289         headline.setFont(font);
290         headline.setOpaque(false);
291         headline.setBorder(Utility.blankBorder(5, 0, 0, 0));
292         return headline;
293     }

295     private void insertMessage(StyledDocument document, ModelMessage message,
296                                Player player) throws BadLocationException {
297         for (Object o : message.splitLinks(player)) {
298             if (o instanceof String) {
299                 document.insertString(document.getLength(), (String)o,
300                                       document.getStyle("regular"));
301             } else if (o instanceof JButton) {
302                 JButton b = (JButton)o;
303                 b.addActionListener(this);
304                 StyleConstants.setComponent(document.getStyle("button"), b);
305                 document.insertString(document.getLength(), " ",
306                                       document.getStyle("button"));
307             }
308         }
