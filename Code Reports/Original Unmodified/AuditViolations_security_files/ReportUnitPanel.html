82     /**
83      * The constructor that will add the items to this panel.
84      *
85      * @param freeColClient The <code>FreeColClient</code> for the game.
86      * @param key the report name key
87      * @param showColonies Whether to show colonies with no selected units.
88      */
89     public ReportUnitPanel(FreeColClient freeColClient, String key,
90                            boolean showColonies) {
91         super(freeColClient, key);
92
93         this.showColonies = showColonies;
94         reportPanel.setLayout(new MigLayout("fillx, wrap 12", "", ""));
95
96         gatherData();
97         addREFUnits();
98         addOwnUnits();
99
100         reportPanel.add(new JSeparator(JSeparator.HORIZONTAL),
101                         "newline, span, growx, wrap 40");
102
103         // Colonies first, sorted according to user preferences
104         for (Colony colony : freeColClient.getMySortedColonies()) {
105             handleLocation(colony, colony.getName(), inColonies.get(colony));
106         }
107
108         // Europe next
109         Europe europe = getMyPlayer().getEurope();
110         if (europe != null) {
111             handleLocation(europe, Messages.getName(europe), inEurope);
112         }
113
114         // Finally all other locations, sorted alphabetically.
115         List<String> otherNames = new ArrayList<>(inLocations.keySet());
116         Collections.sort(otherNames);
117         for (Entry<String, List<Unit>> e : mapEntriesByKey(inLocations)) {
118             handleLocation(null, e.getKey(), e.getValue());
119         }
120
121         revalidate();
122         repaint();
123     }

126     protected int getCount(String key, UnitType type) {
127         TypeCountMap<UnitType> map = units.get(key);
128         return (map == null) ? 0 : map.getCount(type);

131     protected void incrementCount(String key, UnitType type, int number) {
132         TypeCountMap<UnitType> map = units.get(key);
133         if (map == null) {
134             map = new TypeCountMap<>();
135             units.put(key, map);
136         }
137         map.incrementCount(type, number);
138     }

140     protected void addUnit(Unit unit, String key) {
141         if (unit.getLocation() == null) {
142             return; // Can not happen.
143         } else if (unit.isInEurope()) {
144             inEurope.add(unit);
145         } else {
146             Colony colony = unit.getLocation().getColony();
147             if (colony == null) {
148                 String locationName = getLocationLabelFor(unit);
149                 List<Unit> unitList = inLocations.get(locationName);
150                 if (unitList == null) {
151                     unitList = new ArrayList<>();
152                     inLocations.put(locationName, unitList);
153                 }
154                 unitList.add(unit);
155             } else {
156                 List<Unit> unitList = inColonies.get(colony);
157                 if (unitList == null) {
158                     unitList = new ArrayList<>();
159                     inColonies.put(colony, unitList);
160                 }
161                 unitList.add(unit);
162             }
163         }
164         incrementCount(key, unit.getType(), 1);
165     }

167     protected void handleLocation(Location location, String locationName,
168                                   List<Unit> unitList) {
169         if ((unitList == null || unitList.isEmpty()) && !showColonies) {
170             return;
171         }
172
173         JComponent component;
174         if (location == null) {
175             component = new JLabel(locationName);
176         } else {
177             JButton button = Utility.getLinkButton(locationName, null, location.getId());
178             button.addActionListener(this);
179             component = button;
180         }
181         reportPanel.add(component, "newline, span, split 2");
182
183         reportPanel.add(new JSeparator(JSeparator.HORIZONTAL), "growx");
184
185         if (unitList == null || unitList.isEmpty()) {
186             reportPanel.add(Utility.localizedLabel("none"), "sg");
187         } else {
188             for (Unit u : toSortedList(unitList, Unit.typeRoleComparator)) {
189                 JButton unitButton = getUnitButton(u);
190                 if (u.isCarrier()) {
191                     reportPanel.add(unitButton, "newline, sg");
192                     for (Goods goods : u.getGoodsList()) {
193                         GoodsLabel goodsLabel = new GoodsLabel(getGUI(), goods);
194                         reportPanel.add(goodsLabel);
195                     }
196                     for (Unit unitLoaded : u.getUnitList()) {
197                         UnitLabel unitLoadedLabel
198                             = new UnitLabel(getFreeColClient(), unitLoaded, true);
199                         reportPanel.add(unitLoadedLabel);
200                     }
201                 } else {
202                     reportPanel.add(unitButton, "sg");
203                 }
204             }
205         }
206     }
