43 /**
44  * This class is responsible for drawing the Roads on a tile.
45  */
46 public final class RoadPainter {
47     // Helper variables for displaying the map.
48     private int tileHeight, tileWidth, halfHeight, halfWidth;
49
50     // roads
51     private final EnumMap<Direction, Point2D.Float> corners =
52         new EnumMap<>(Direction.class);
53     private final EnumMap<Direction, List<Direction>> prohibitedRoads =
54         new EnumMap<>(Direction.class);
55     private Stroke roadStroke = new BasicStroke(2);
56
57     public RoadPainter(Dimension tileSize) {
58         tileHeight = tileSize.height;
59         tileWidth = tileSize.width;
60         halfHeight = tileHeight/2;
61         halfWidth = tileWidth/2;
62
63         int dy = tileHeight/16;
64
65         // corners
66         corners.put(Direction.N,  new Point2D.Float(halfWidth, 0));
67         corners.put(Direction.NE, new Point2D.Float(0.75f * tileWidth, 0.25f * tileHeight));
68         corners.put(Direction.E,  new Point2D.Float(tileWidth, halfHeight));
69         corners.put(Direction.SE, new Point2D.Float(0.75f * tileWidth, 0.75f * tileHeight));
70         corners.put(Direction.S,  new Point2D.Float(halfWidth, tileHeight));
71         corners.put(Direction.SW, new Point2D.Float(0.25f * tileWidth, 0.75f * tileHeight));
72         corners.put(Direction.W,  new Point2D.Float(0, halfHeight));
73         corners.put(Direction.NW, new Point2D.Float(0.25f * tileWidth, 0.25f * tileHeight));
74
75         // road pairs to skip drawing when doing 3 or 4 exit point tiles
76         //  don't put more than two directions in each list,
77         //  otherwise a 3-point tile may not draw any roads at all!
78         prohibitedRoads.put(Direction.N,  Arrays.asList(Direction.NW, Direction.NE));
79         prohibitedRoads.put(Direction.NE, Arrays.asList(Direction.N, Direction.E));
80         prohibitedRoads.put(Direction.E,  Arrays.asList(Direction.NE, Direction.SE));
81         prohibitedRoads.put(Direction.SE, Arrays.asList(Direction.E, Direction.S));
82         prohibitedRoads.put(Direction.S,  Arrays.asList(Direction.SE, Direction.SW));
83         prohibitedRoads.put(Direction.SW, Arrays.asList(Direction.S, Direction.W));
84         prohibitedRoads.put(Direction.W,  Arrays.asList(Direction.SW, Direction.NW));
85         prohibitedRoads.put(Direction.NW, Arrays.asList(Direction.W, Direction.N));
86
87         roadStroke = new BasicStroke(dy / 2.0f);
88     }
89
90     /**
91      * Draws all roads on the given Tile.
92      *
93      * @param g The <code>Graphics</code> to draw the road upon.
94      * @param tile a <code>Tile</code>
95      */
96     public void displayRoad(Graphics2D g, Tile tile) {
97         Color oldColor = g.getColor();
98         g.setColor(ResourceManager.getColor("color.map.road"));
99         g.setStroke(roadStroke);
100         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
101                            RenderingHints.VALUE_ANTIALIAS_ON);
102         GeneralPath path = new GeneralPath();
103         Map map = tile.getMap();
104         int x = tile.getX();
105         int y = tile.getY();
106         List<Point2D.Float> points = new ArrayList<>(8);
107         List<Direction> directions = Direction.allDirections.stream()
108             .filter((Direction direction) -> {
109                     Tile borderingTile = map.getTile(direction.step(x, y));
110                     TileImprovement r;
111                     return (borderingTile != null
112                         && (r = borderingTile.getRoad()) != null
113                         && r.isComplete());
114                 })
115             .peek((Direction direction) -> points.add(corners.get(direction)))
116             .collect(Collectors.toList());
117
118         switch(points.size()) {
119         case 0:
120             path.moveTo(0.35f * tileWidth, 0.35f * tileHeight);
121             path.lineTo(0.65f * tileWidth, 0.65f * tileHeight);
122             path.moveTo(0.35f * tileWidth, 0.65f * tileHeight);
123             path.lineTo(0.65f * tileWidth, 0.35f * tileHeight);
124             break;
125         case 1:
126             path.moveTo(halfWidth, halfHeight);
127             path.lineTo(points.get(0).getX(), points.get(0).getY());
128             break;
129         case 2:
130             path.moveTo(points.get(0).getX(), points.get(0).getY());
131             path.quadTo(halfWidth, halfHeight, points.get(1).getX(), points.get(1).getY());
132             break;
133         case 3:
134         case 4: {
135             Direction pen = directions.get(directions.size() - 1);
136             Point2D p = corners.get(pen);
137             path.moveTo(p.getX(), p.getY());
138             for (Direction d : directions) {
139                 p = corners.get(d);
140                 if(prohibitedRoads.get(pen).contains(d)) {
141                     path.moveTo(p.getX(), p.getY());
142                 } else {
143                     path.quadTo(halfWidth, halfHeight, p.getX(), p.getY());
144                 }
145                 pen = d;
146             }
147             break;
148         }
149         default:
150             for (Point2D p : points) {
151                 path.moveTo(halfWidth, halfHeight);
152                 path.lineTo(p.getX(), p.getY());
153             }
154         }
155         g.draw(path);
156         g.setColor(oldColor);
157         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
158     }
