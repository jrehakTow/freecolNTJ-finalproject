116     public Element handle(FreeColServer server, Player player,
117                           Connection connection) {
118         final ServerPlayer serverPlayer = server.getPlayer(connection);
119
120         Unit unit;
121         try {
122             unit = player.getOurFreeColGameObject(this.unitId, Unit.class);
123         } catch (Exception e) {
124             return serverPlayer.clientError(e.getMessage())
125                 .build(serverPlayer);
126         }
127         if (!unit.hasAbility(Ability.SPEAK_WITH_CHIEF)) {
128             return serverPlayer.clientError("Unit lacks ability to speak to chief: "
129                 + this.unitId)

131         }
132
133         IndianSettlement is;
134         try {
135             is = (IndianSettlement)unit
136                 .getAdjacentIndianSettlementSafely(this.settlementId);
137         } catch (Exception e) {
138             return serverPlayer.clientError(e.getMessage())
139                 .build(serverPlayer);
140         }
141
142         MoveType type = unit.getMoveType(is.getTile());
143         if (type != MoveType.ENTER_INDIAN_SETTLEMENT_WITH_SCOUT) {
144             return serverPlayer.clientError("Unable to enter "
145                 + is.getName() + ": " + type.whyIllegal())
146                 .build(serverPlayer);
147         }
148
149         // Valid request, do the scouting.

153     }
