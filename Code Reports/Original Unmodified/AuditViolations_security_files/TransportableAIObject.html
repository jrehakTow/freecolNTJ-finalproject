182     public boolean dropTransport() {
183         AIUnit transport = getTransport();
184         if (transport != null) {
185             if (!transport.isDisposed()
186                 && getLocation() != transport.getUnit()) {
187                 TransportMission tm
188                     = transport.getMission(TransportMission.class);
189                 if (tm != null) tm.removeTransportable(this);
190             }
191             setTransport(null);
192         }
193         return getTransport() == null;
194     }

219         if (old != null) {

221
222             TransportMission tm = old.getMission(TransportMission.class);
223             if (tm != null) tm.removeTransportable(this);
224         }

226         if (aiCarrier != null) {
227             //TransportMission tm = aiCarrier.getMission(TransportMission.class);
