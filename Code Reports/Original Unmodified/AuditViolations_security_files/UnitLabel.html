368     @Override
369     public void actionPerformed(ActionEvent ae) {

372         final InGameController igc = freeColClient.getInGameController();
373         String[] args = ae.getActionCommand().split("/");
374         GoodsType gt;
375         switch (Enum.valueOf(UnitAction.class,
376                              args[0].toUpperCase(Locale.US))) {
377         case ASSIGN:
378             igc.assignTeacher(unit,
379                               game.getFreeColGameObject(args[1], Unit.class));
380             break;
381         case WORK_COLONYTILE:
382             if (args.length < 3) break;
383             ColonyTile colonyTile
384                 = game.getFreeColGameObject(args[1], ColonyTile.class);
385             if (args.length >= 4 && "!".equals(args[3])) {
386                 // Claim tile if needed
387                 if (!igc.claimTile(colonyTile.getWorkTile(),
388                                    unit.getColony())) break;
389             }
390             if (colonyTile != unit.getLocation()) igc.work(unit, colonyTile);
391             if ((gt = spec.getGoodsType(args[2])) != null
392                 && unit.getWorkType() != gt) {
393                 igc.changeWorkType(unit, gt);
394             }
395             break;
396         case WORK_BUILDING:
397             if (args.length < 3) break;
398             Building building
399                 = game.getFreeColGameObject(args[1], Building.class);
400             if (building != unit.getLocation()) igc.work(unit, building);
401             if ((gt = spec.getGoodsType(args[2])) != null
402                 && unit.getWorkType() != gt) {
403                 igc.changeWorkType(unit, gt);
404             }
405             break;
406         case ACTIVATE_UNIT:
407             igc.changeState(unit, Unit.UnitState.ACTIVE);
408             gui.setActiveUnit(unit);
409             break;
410         case FORTIFY:

413         case SENTRY:

416         case COLOPEDIA:

428         case ASSIGN_TRADE_ROUTE:
