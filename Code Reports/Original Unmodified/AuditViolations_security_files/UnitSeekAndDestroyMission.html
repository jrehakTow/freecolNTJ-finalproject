459         String reason = invalidReason();
460         if (isTargetReason(reason)) {
461             Colony colony;
462             Mission m;
463             if (Mission.TARGETOWNERSHIP.equals(reason)
464                 && getTarget() instanceof Colony
465                 && (colony = (Colony)getTarget()) != null
466                 && getPlayer().owns(colony)
467                 && (m = getAIPlayer().getDefendSettlementMission(aiUnit,
468                         colony)) != null) {
469                 return lbDone(lb, true, " capturing colony ", colony.getName());
470             }            
471             return retargetMission(reason, lb);
472         } else if (reason != null) {
473             return lbFail(lb, false, reason);
474         }
475
476         // Is there a target-of-opportunity?
477         final Unit unit = getUnit();
478         Location nearbyTarget = (unit.isOnCarrier()) ? null
479             : findTarget(aiUnit, 1, false);
480         if (nearbyTarget != null) {
481             if (getTarget() == null) {
482                 setTarget(nearbyTarget);
483                 return lbRetarget(lb);
484             }
485             if (nearbyTarget == getTarget()) {
486                 nearbyTarget = null;
487             } else {
488                 Tile now = unit.getTile();
489                 Tile nearbyTile = nearbyTarget.getTile();
490                 Tile targetTile = getTarget().getTile();
491                 if (now != null && nearbyTile != null && targetTile != null
492                     && (now.getDistanceTo(nearbyTile)
493                         >= now.getDistanceTo(targetTile))) {
494                     nearbyTarget = null;
495                 } else {
496                     lb.add(", found target of opportunity ", nearbyTarget);
497                 }
498             }
499         }
500
501         // Go to the target.
502         Location currentTarget = (nearbyTarget != null) ? nearbyTarget
503             : getTarget();
504         // Note avoiding other targets by choice of cost decider.
505         Unit.MoveType mt = travelToTarget(currentTarget,
506             CostDeciders.avoidSettlementsAndBlockingUnits(), lb);
507         switch (mt) {
508         case MOVE_HIGH_SEAS: case MOVE_NO_MOVES: case MOVE_ILLEGAL:
509             return lbWait(lb);
510
511         case MOVE_NO_REPAIR:
512             return lbFail(lb, false, AIUNITDIED);
513
514         case MOVE_NO_ACCESS_EMBARK: case MOVE_NO_TILE:
515             return this;
516
517         case ATTACK_UNIT: case ATTACK_SETTLEMENT:
518             Tile unitTile = unit.getTile();
519             Settlement settlement = unitTile.getSettlement();
520             if (settlement != null && settlement.getUnitCount() < 2) {
521                 // Do not risk attacking out of a settlement that
522                 // might collapse.  Defend instead.
523                 Mission m = getAIPlayer()
524                     .getDefendSettlementMission(aiUnit, settlement);
525                 return lbDone(lb, m != null, " desperate defence of ",
526                               settlement);
527             }
528             Direction d = unitTile.getDirection(currentTarget.getTile());
529             assert d != null;
530             AIMessage.askAttack(aiUnit, d);
531             return lbAttack(lb, currentTarget);
532
533         default:
534             return lbMove(lb, mt);
535         }
536     }
