181      * Make a suitable link button for a given key.
182      *
183      * Colonies and Europe-like objects are obvious, locations and units
184      * are dependent on the message source.
185      * TODO: Are there more useful possibilities?
186      *
187      * @param key The message key to make a link for.
188      * @param val The text for the link.
189      * @param player The <code>Player</code> to make a link for.
190      * @param source The message source <code>FreeColGameObject</code>.
191      * @return A <code>JButton</code> for the link, or null if no good
192      *     choice found.
193      */
194     public static JButton getMessageButton(String key, String val,
195         Player player, FreeColGameObject source) {
196         FreeColGameObject link = null;
197         if ("%colony%".equals(key) || key.endsWith("Colony%")) {
198             Settlement settlement = player.getGame().getSettlementByName(val);
199             link = (settlement == null) ? null
200                 : (player.owns(settlement)) ? settlement
201                 : settlement.getTile();
202         } else if ("%europe%".equals(key) || "%market%".equals(key)) {
203             link = player.getEurope();
204         } else if ("%location%".equals(key) || key.endsWith("Location%")) {
205             if (source instanceof Location) {
206                 link = source.getLinkTarget(player);
207             }
208         } else if ("%unit%".equals(key) || key.endsWith("Unit%")) {
209             if (source instanceof Unit) {
210                 link = source.getLinkTarget(player);
211             }
212         }
213         return (link == null) ? null
214             : getLinkButton(val, null, link.getId());
215     }
