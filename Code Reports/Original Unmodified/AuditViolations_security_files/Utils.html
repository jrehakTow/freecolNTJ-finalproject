55     /**
56      * Check if two objects are equal but also checks for null.
57      *
58      * @param <T> The object type.
59      * @param one First object to compare
60      * @param two Second object to compare
61      * @return True if the arguments are either both null or equal in the
62      *     sense of their equals() method.
63      */
64     public static <T> boolean equals(T one, T two) {
65         return (one == null) ? (two == null) : one.equals(two);

78     /**
79      * Get the internal state of a random number generator as a
80      * string.  It would have been more convenient to simply return
81      * the current seed, but unfortunately it is private.
82      *
83      * @param random A pseudo-random number source.
84      * @return A <code>String</code> encapsulating the object state.
85      */
86     public static synchronized String getRandomState(Random random) {
87         ByteArrayOutputStream bos = new ByteArrayOutputStream();
88         try {
89             ObjectOutputStream oos = new ObjectOutputStream(bos);
90             oos.writeObject(random);
91             oos.flush();
92         } catch (IOException e) {
93             throw new IllegalStateException("IO exception in memory!?", e);
94         }
95         byte[] bytes = bos.toByteArray();
96         StringBuilder sb = new StringBuilder(bytes.length * 2);
97         for (byte b : bytes) {
98             sb.append(HEX_DIGITS.charAt((b >> 4) & 0x0F));
99             sb.append(HEX_DIGITS.charAt(b & 0x0F));
100         }
101         return sb.toString();
102     }

104     /**
105      * Restore a previously saved state.
106      *
107      * @param state The saved state (@see #getRandomState()).
108      * @return The restored <code>Random</code>.
109      */
110     public static synchronized Random restoreRandomState(String state) {
111         if (state == null || state.isEmpty()) return null;
112         byte[] bytes = new byte[state.length() / 2];
113         int pos = 0;
114         for (int i = 0; i < bytes.length; i++) {
115             bytes[i] = (byte) HEX_DIGITS.indexOf(state.charAt(pos++));
116             bytes[i] <<= 4;
117             bytes[i] |= (byte) HEX_DIGITS.indexOf(state.charAt(pos++));
118         }
119         ByteArrayInputStream bis = new ByteArrayInputStream(bytes);
120         try {
121             ObjectInputStream ois = new ObjectInputStream(bis);
122             return (Random) ois.readObject();
123         } catch (IOException | ClassNotFoundException e) {
124             logger.log(Level.WARNING, "Unable to restore random state.", e);
125         }
126         return null;
127     }

160     /**
161      * Helper to make an XML Transformer.
162      *
163      * @param declaration If true, include the XML declaration.
164      * @param indent If true, set up the transformer to indent.
165      * @return A suitable <code>Transformer</code>.
166      */
167     public static Transformer makeTransformer(boolean declaration,
168                                               boolean indent) {
169         Transformer tf = null;
170         try {
171             TransformerFactory factory = TransformerFactory.newInstance();
172             tf = factory.newTransformer();
173             if (!declaration) {
174                 tf.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
175             }
176             if (indent) {
177                 tf.setOutputProperty(OutputKeys.INDENT, "yes");
178                 tf.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
179             }
180         } catch (TransformerException e) {
181             logger.log(Level.WARNING, "Failed to install transformer!", e);
182         }
183         return tf;
