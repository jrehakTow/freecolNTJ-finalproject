42 /**
43  * The <code>WorkLocation</code> is a place in a {@link Colony} where
44  * <code>Units</code> can work.  The unit capacity of a WorkLocation
45  * is likely to be limited.  ColonyTiles can only hold a single
46  * worker, and Buildings can hold no more than three workers, for

568      * Gets the productivity of a unit working in this work location,
569      * considering *only* the contribution of the unit, exclusive of
570      * that of the work location.
571      *
572      * Used below, only public for the test suite.
573      *
574      * @param unit The <code>Unit</code> to check.
575      * @param goodsType The <code>GoodsType</code> to check the production of.
576      * @return The maximum return from this unit.
577      */
578     public int getUnitProduction(Unit unit, GoodsType goodsType) {
579         if (unit == null || unit.getWorkType() != goodsType) return 0;
580         final UnitType unitType = unit.getType();
581         final Turn turn = getGame().getTurn();
582         return max(getOutputs(),
583             ag -> ag.getType() == goodsType,
584             ag -> (int)applyModifiers(getBaseProduction(getProductionType(),
585                                                         goodsType, unitType),
586                 turn, getProductionModifiers(goodsType, unitType)));
587     }

592      * @param unit The unit to do the work.
593      * @param goodsType The type of goods to get the production of.

600     }

602     /**
603      * Get the potential production of a given goods type at this
604      * location.  An optional unit type to do the production may be
605      * specified, however if null the unattended production will be
606      * calculated.
607      *
608      * Usually if a unit type is specified and a unit can not be added
609      * to the work location, zero production is returned.  However,
610      * this routine is intended to be used for planning purposes, so
611      * some exceptions are allowed --- the calculation proceeds:
612      *
613      *   - for unclaimed tiles
614      *   - when the location is currently full of units
615      *
616      * which are conditions that an AI might plausibly be able and
617      * willing to change (a case could be made for including the
618      * OCCUPIED_BY_ENEMY condition).
619      *
620      * @param goodsType The <code>GoodsType</code> to produce.
621      * @param unitType The optional <code>UnitType</code> to do the work.
622      * @return The potential production with the given goods type and
623      *     unit type.
624      */
625     public int getPotentialProduction(GoodsType goodsType,
626                                       UnitType unitType) {
627         if (!canProduce(goodsType, unitType)) return 0;
628
629         if (unitType != null) {
630             switch (getNoWorkReason()) {
631             case NONE: case ALREADY_PRESENT: case CLAIM_REQUIRED:
632                 break;
633             case CAPACITY_EXCEEDED:
634                 if (getUnitCapacity() > 0) break; // Could work after reorg!
635                 // Fall through
636             case WRONG_TYPE: case OWNED_BY_ENEMY: case ANOTHER_COLONY:
637             case COLONY_CENTER: case MISSING_ABILITY: case MISSING_SKILL:
638             case MINIMUM_SKILL: case MAXIMUM_SKILL:
639             case OCCUPIED_BY_ENEMY: // Arguable!
640             default:
641                 // Non-transient or inapplicable conditions.  Production
642                 // is impossible.
643                 return 0;
644             }
645         }
646
647         int amount = getBaseProduction(null, goodsType, unitType);
648         amount = (int)applyModifiers(amount, getGame().getTurn(),
649             getProductionModifiers(goodsType, unitType));
650         return (amount < 0) ? 0 : amount;
651     }

682      * {@inheritDoc}
683      */
684     @Override
685     public boolean add(final Locatable locatable) {
686         NoAddReason reason = getNoAddReason(locatable);
687         switch (reason) {
688         case NONE:
689             break;
690         case ALREADY_PRESENT:
691             return true;
692         default:
693             throw new IllegalStateException("Can not add " + locatable
694                 + " to " + this + " because " + reason);
695         }
696         Unit unit = (Unit)locatable;
697         if (!super.add(unit)) return false;
698
699         unit.setState(Unit.UnitState.IN_COLONY);
700         unit.setMovesLeft(0);
701
702         // Choose a sensible work type, which should update production type.
703         setWorkFor(unit);
704
705         getColony().invalidateCache();
706         return true;
707     }

709     /**
710      * {@inheritDoc}
711      */
712     @Override
713     public boolean remove(final Locatable locatable) {
714         if (!(locatable instanceof Unit)) {
715             throw new IllegalStateException("Not a unit: " + locatable);
716         }
717         Unit unit = (Unit)locatable;
718         if (!contains(unit)) return true;
719         if (!super.remove(unit)) return false;
720
721         unit.setState(Unit.UnitState.ACTIVE);
722         unit.setMovesLeft(0);
723
724         // Switch to unattended production if possible.
725         if (isEmpty()) updateProductionType();
726
727         getColony().invalidateCache();
728         return true;
729     }
