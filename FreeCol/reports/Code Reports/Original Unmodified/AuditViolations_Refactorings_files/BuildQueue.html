29  * A queue of things for a colony to build.
30  */

33     public static enum CompletionAction {
34         /**
35          * Always remove the completed item. Not used by any build
36          * queue at the moment.
37          */
38         REMOVE,
39         /**
40          * Remove the completed item unless it is the last item and
41          * several instances of the item can co-exist (which is true
42          * for units, but not buildings). This is the strategy used by
43          * the colony build queue.
44          */
45         REMOVE_EXCEPT_LAST,
46         /**
47          * Shuffle the items rather than remove the completed
48          * item. This is the strategy used by the colony population
49          * queue.
50          */
51         SHUFFLE,
52         /**
53          * Remove the completed item and add a random new item. This
54          * is the strategy of the immigration queue (which is not
55          * implemented as a build queue at the moment, however).
56          */
57         ADD_RANDOM
58     };

149     /**
150      * {@inheritDoc}
151      */
152     @Override
153     public List<AbstractGoods> getConsumedGoods() {
154         T current = getCurrentlyBuilding();
155         return (current == null) ? new ArrayList<AbstractGoods>()
156             : current.getRequiredGoods();
157     }

159     /**
160      * Return the <code>ProductionInfo</code> for this BuildQueue.
161      *
162      * @param input A list of input <code>AbstractGoods</code>.
163      * @return The <code>ProductionInfo</code> for this BuildQueue.
164      */
165     public ProductionInfo getProductionInfo(List<AbstractGoods> input) {
166         ProductionInfo result = new ProductionInfo();
167         T current = getCurrentlyBuilding();
168         if (current != null) {
169             // ATTENTION: this code presupposes that we will consume
170             // all required goods at once
171             final boolean overflow = colony.getSpecification()
172                 .getBoolean(GameOptions.SAVE_PRODUCTION_OVERFLOW);
173             List<AbstractGoods> consumption = new ArrayList<>();
174             for (AbstractGoods ag : current.getRequiredGoods()) {
175                 AbstractGoods available
176                     = AbstractGoods.findByType(ag.getType(), input);
177                 if (available != null
178                     && ag.getAmount() <= available.getAmount()) {
179                     int amount = (overflow || ag.getType().isStorable())
180                         ? ag.getAmount()
181                         : available.getAmount();
182                     consumption.add(new AbstractGoods(ag.getType(), amount));
183                 } else { // don't build anything
184                     return result;
185                 }
186             }
187             result.setConsumption(consumption);
188         }
189         return result;
190     }

204     /**
205      * Does the consumer have the ability with the given identifier?
206      *
207      * The two abilities most relevant to consumers are
208      * "consumeAllOrNothing", which implies that the consumer will not
209      * consume any goods if its requirements can not be met (used by
210      * the Colony when building), as well as
211      * "consumeOnlySurplusProduction", which implies that the consumer
