705     private void updateUnitList() {
706         final Specification spec = getSpecification();
707         final Turn turn = getGame().getTurn();
708         StringTemplate tmpl;
709         DefaultListModel<UnitType> units
710             = (DefaultListModel<UnitType>)this.unitList.getModel();
711         units.clear();
712         loop: for (UnitType unitType : spec.getBuildableUnitTypes()) {
713             // compare colony.getNoBuildReason()
714             List<String> lockReason = new ArrayList<>();
715             if (unbuildableTypes.contains(unitType)) {
716                 continue;
717             }
718
719             if (unitType.getRequiredPopulation() > this.colony.getUnitCount()) {
720                 tmpl = StringTemplate.template("buildQueuePanel.populationTooSmall")
721                     .addAmount("%number%", unitType.getRequiredPopulation());
722                 lockReason.add(Messages.message(tmpl));
723             }
724
725             if (unitType.getLimits() != null) {
726                 for (Limit limit : unitType.getLimits()) {
727                     if (!limit.evaluate(this.colony)) {
728                         lockReason.add(Messages.getDescription(limit));
729                     }
730                 }
731             }
732
733             if (!(this.colony.hasAbility(Ability.BUILD, unitType, turn)
734                     || this.featureContainer.hasAbility(Ability.BUILD,
735                         unitType, null))) {
736                 boolean builderFound = false;
737                 for (Ability ability : spec.getAbilities(Ability.BUILD)) {
738                     FreeColObject source = ability.getSource();
739                     if (ability.appliesTo(unitType)
740                         && ability.getValue()
741                         && source != null
742                         && !unbuildableTypes.contains(source)) {
743                         builderFound = true;
744                         if (source instanceof Named) {
745                             lockReason.add(Messages.getName((Named)source));
746                         }
747                         break;
748                     }
749                 }
750                 if (!builderFound) {
751                     unbuildableTypes.add(unitType);
752                     continue;
753                 }
754             }
755
756             for (Entry<String, Boolean> entry
757                      : unitType.getRequiredAbilities().entrySet()) {
758                 if (this.colony.hasAbility(entry.getKey()) != entry.getValue()
759                     && this.featureContainer.hasAbility(entry.getKey(),
760                             null, null)
761                     != entry.getValue()) {
762                     List<FreeColSpecObjectType> sources
763                         = spec.getTypesProviding(entry.getKey(),
764                                                  entry.getValue());
765                     if (sources.isEmpty()) {
766                         // no type provides the required ability
767                         unbuildableTypes.add(unitType);
768                         continue loop;
769                     } else {
770                         lockReason.add(Messages.getName(sources.get(0)));
771                     }
772                 }
773             }
774             if (lockReason.isEmpty()) {
775                 lockReasons.put(unitType, null);
776             } else {
777                 tmpl = StringTemplate.template("buildQueuePanel.requires")
778                     .addName("%string%", join("/", lockReason));
779                 lockReasons.put(unitType, Messages.message(tmpl));
780             }
781             if (lockReason.isEmpty() || showAllBox.isSelected()) {
782                 units.addElement(unitType);
783             }
784         }
785     }

787     private void updateBuildingList() {
788         final Specification spec = getSpecification();
789         final DefaultListModel<BuildingType> buildings
790             = (DefaultListModel<BuildingType>)this.buildingList.getModel();
791         final DefaultListModel<BuildableType> current
792             = (DefaultListModel<BuildableType>)this.buildQueueList.getModel();
793         StringTemplate tmpl;
794
795         buildings.clear();
796         loop: for (BuildingType buildingType : spec.getBuildingTypeList()) {
797             // compare colony.getNoBuildReason()
798             List<String> lockReason = new ArrayList<>();
799             Building colonyBuilding = this.colony.getBuilding(buildingType);
800             if (current.contains(buildingType)
801                 || hasBuildingType(buildingType)) {
802                 // only one building of any kind
803                 continue;
804             } else if (unbuildableTypes.contains(buildingType)) {
805                 continue;
806             } else if (!buildingType.needsGoodsToBuild()) {
807                 // pre-built
808                 continue;
809             } else if (unbuildableTypes.contains(buildingType
810                     .getUpgradesFrom())) {
811                 unbuildableTypes.add(buildingType); // impossible upgrade path
812                 continue;
813             }
814
815             if (buildingType.hasAbility(Ability.COASTAL_ONLY)
816                 && !this.colony.getTile().isCoastland()) {
817                 tmpl = StringTemplate.template("buildQueuePanel.coastalOnly");
818                 lockReason.add(Messages.message(tmpl));
819             }
820                                                 
821             if (buildingType.getRequiredPopulation() > this.colony
822                     .getUnitCount()) {
823                 tmpl = StringTemplate.template("buildQueuePanel.populationTooSmall")
824                     .addAmount("%number%", buildingType.getRequiredPopulation());
825                 lockReason.add(Messages.message(tmpl));
826             }
827
828             for (Entry<String, Boolean> entry
829                      : buildingType.getRequiredAbilities().entrySet()) {
830                 if (this.colony.hasAbility(entry.getKey()) != entry.getValue()
831                     && this.featureContainer.hasAbility(entry.getKey(),
832                             null, null)
833                     != entry.getValue()) {
834                     List<FreeColSpecObjectType> sources = getSpecification()
835                         .getTypesProviding(entry.getKey(), entry.getValue());
836                     if (sources.isEmpty()) {
837                         // no type provides the required ability
838                         unbuildableTypes.add(buildingType);
839                         continue loop;
840                     } else {
841                         lockReason.add(Messages.getName(sources.get(0)));
842                     }
843                 }
844             }
845
846             if (buildingType.getLimits() != null) {
847                 for (Limit limit : buildingType.getLimits()) {
848                     if (!limit.evaluate(this.colony)) {
849                         lockReason.add(Messages.getDescription(limit));
850                     }
851                 }
852             }
853
854             if (buildingType.getUpgradesFrom() != null
855                 && !current.contains(buildingType.getUpgradesFrom())) {
856                 if (colonyBuilding == null
857                     || colonyBuilding.getType() != buildingType
858                             .getUpgradesFrom()) {
859                     lockReason.add(Messages.getName(buildingType
860                             .getUpgradesFrom()));
861                 }
862             }
863             if (lockReason.isEmpty()) {
864                 lockReasons.put(buildingType, null);
865             } else {
866                 tmpl = StringTemplate.template("buildQueuePanel.requires")
867                     .addName("%string%", join("/", lockReason));
868                 lockReasons.put(buildingType, Messages.message(tmpl));
869             }
870             if (lockReason.isEmpty() || showAllBox.isSelected()) {
871                 buildings.addElement(buildingType);
872             }
873         }
874     }

876     private void updateAllLists() {
877         final DefaultListModel<BuildableType> current
878             = (DefaultListModel<BuildableType>)this.buildQueueList.getModel();
879
880         this.featureContainer.clear();
881         for (Enumeration<BuildableType> e = current.elements();
882              e.hasMoreElements();) {
883             BuildableType type = e.nextElement();
884             if (getMinimumIndex(type) >= 0) {
885                 featureContainer.addFeatures(type);
886             } else {
887                 current.removeElement(type);
888             }
889         }
890         // ATTENTION: buildings must be updated first, since units
891         // might depend on the build ability of an unbuildable
892         // building
893         updateBuildingList();
894         updateUnitList();
895
896         // Update the buy button
897         final boolean pay = getSpecification()
898             .getBoolean(GameOptions.PAY_FOR_BUILDING);
899         BuildableType bt = (current.getSize() <= 0) ? null
900             : current.getElementAt(0);
901         this.buyBuildable.setEnabled(bt != null && pay
902             && this.colony.canPayToFinishBuilding(bt));
903         this.setBuyLabel(bt);
904
905         // Update the construction panel
906         if (current.getSize() > 0) {
907             this.constructionPanel.update(current.getElementAt(0));
908         } else if (current.getSize() == 0) {
909             this.constructionPanel.update(); // generates Building: Nothing
910         }
911     }

944     private int getMinimumIndex(BuildableType buildableType) {
945         ListModel<BuildableType> buildQueue = this.buildQueueList.getModel();
946         if (buildableType instanceof UnitType) {
947             if (this.colony.canBuild(buildableType)) return 0;
948             for (int index = 0; index < buildQueue.getSize(); index++) {
949                 if (buildQueue.getElementAt(index).hasAbility(Ability.BUILD,
950                         buildableType)) return index + 1;
951             }
952         } else if (buildableType instanceof BuildingType) {
953             BuildingType upgradesFrom = ((BuildingType)buildableType)
954                     .getUpgradesFrom();
955             if (upgradesFrom == null) return 0;
