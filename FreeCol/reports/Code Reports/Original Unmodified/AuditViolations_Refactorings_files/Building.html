211      * @param outputs The output goods already available in the colony,
212      *     necessary in order to avoid excess production.
213      * @return The production information.

216     public ProductionInfo getAdjustedProductionInfo(List<AbstractGoods> inputs,
217                                                     List<AbstractGoods> outputs) {
218         ProductionInfo result = new ProductionInfo();
219         if (!hasOutputs()) return result;
220         final Specification spec = getSpecification();
221         final Turn turn = getGame().getTurn();
222         final boolean avoidOverflow
223             = hasAbility(Ability.AVOID_EXCESS_PRODUCTION);
224         final int capacity = getColony().getWarehouseCapacity();
225         // Calculate two production ratios, the minimum (and actual)
226         // possible multiplier between the nominal input and output
227         // goods and the amount actually consumed and produced, and
228         // the maximum possible ratio that would apply but for
229         // circumstances such as limited input availability.
230         double maximumRatio = 0.0, minimumRatio = Double.MAX_VALUE;
231
232         // First, calculate the nominal production ratios.
233         if (canAutoProduce()) {
234             // Autoproducers are special
235             for (AbstractGoods output : getOutputs()) {
236                 if (output.getAmount() <= 0) continue;
237                 final GoodsType goodsType = output.getType();
238                 int available = getColony().getGoodsCount(goodsType);
239                 if (available >= capacity) {
240                     minimumRatio = maximumRatio = 0.0;
241                 } else {
242                     int divisor = (int)getType().applyModifiers(0f, turn,
243                         Modifier.BREEDING_DIVISOR);
244                     int factor = (int)getType().applyModifiers(0f, turn,
245                         Modifier.BREEDING_FACTOR);
246                     int production = (available < goodsType.getBreedingNumber()
247                         || divisor <= 0) ? 0
248                         // Deliberate use of integer division
249                         : ((available - 1) / divisor + 1) * factor;
250                     double newRatio = (double)production / output.getAmount();
251                     minimumRatio = Math.min(minimumRatio, newRatio);
252                     maximumRatio = Math.max(maximumRatio, newRatio);
253                 }
254             }
255         } else {
256             for (AbstractGoods output : getOutputs()) {
257                 final GoodsType goodsType = output.getType();
258                 float production = sum(getUnitList(),
259                                        u -> getUnitProduction(u, goodsType));
260                 // Unattended production always applies for buildings!
261                 production += getBaseProduction(null, goodsType, null);
262                 production = applyModifiers(production, turn,
263                     getProductionModifiers(goodsType, null));
264                 production = (int)Math.floor(production);
265                 // Beware!  If we ever unify this code with ColonyTile,
266                 // ColonyTiles have outputs with zero amount.
267                 double newRatio = production / output.getAmount();
268                 minimumRatio = Math.min(minimumRatio, newRatio);
269                 maximumRatio = Math.max(maximumRatio, newRatio);
270             }
271         }
272
273         // Then reduce the minimum ratio if some input is in short supply.
274         for (AbstractGoods input : getInputs()) {
275             long required = (long)Math.floor(input.getAmount() * minimumRatio);
276             long available = getAvailable(input.getType(), inputs);
277             // Do not allow auto-production to go negative.
278             if (canAutoProduce()) available = Math.max(0, available);
279             // Experts in factory level buildings may produce a
280             // certain amount of goods even when no input is available.
281             // Factories have the EXPERTS_USE_CONNECTIONS ability.
282             if (available < required
283                 && hasAbility(Ability.EXPERTS_USE_CONNECTIONS)
284                 && spec.getBoolean(GameOptions.EXPERTS_HAVE_CONNECTIONS)) {
285                 long minimumGoodsInput = 4 // FIXME: magic number
286                     * count(getUnitList(), u -> u.getType() == getExpertUnitType());
287                 if (minimumGoodsInput > available) {
288                     available = minimumGoodsInput;
289                 }
290             }
291             // Scale production by limitations on availability.
292             if (available < required) {
293                 minimumRatio *= (double)available / required;
294                 //maximumRatio = Math.max(maximumRatio, minimumRatio);
295             }
296         }
297
298         // Check whether there is space enough to store the goods
299         // produced in order to avoid excess production.
300         if (avoidOverflow) {
301             for (AbstractGoods output : getOutputs()) {
302                 double production = output.getAmount() * minimumRatio;
303                 if (production <= 0) continue;
304                 double headroom = (double)capacity
305                     - getAvailable(output.getType(), outputs);
306                 // Clamp production at warehouse capacity
307                 if (production > headroom) {
308                     minimumRatio = Math.min(minimumRatio,
309                         headroom / output.getAmount());
310                 }
311                 production = output.getAmount() * maximumRatio;
312                 if (production > headroom) {
313                     maximumRatio = Math.min(maximumRatio, 
314                         headroom / output.getAmount());
315                 }
316             }
317         }
318
319         final double epsilon = 0.0001;
320         for (AbstractGoods input : getInputs()) {
321             GoodsType type = input.getType();
322             // maximize consumption
323             int consumption = (int)Math.floor(input.getAmount()
324                 * minimumRatio + epsilon);
325             int maximumConsumption = (int)Math.floor(input.getAmount()
326                 * maximumRatio);
327             result.addConsumption(new AbstractGoods(type, consumption));
328             if (consumption < maximumConsumption) {
329                 result.addMaximumConsumption(new AbstractGoods(type, maximumConsumption));
330             }
331         }
332         for (AbstractGoods output : getOutputs()) {
333             GoodsType type = output.getType();
334             // minimize production, but add a magic little something
335             // to counter rounding errors
336             int production = (int)Math.floor(output.getAmount() * minimumRatio
337                 + epsilon);
338             int maximumProduction = (int)Math.floor(output.getAmount()
339                 * maximumRatio);
340             result.addProduction(new AbstractGoods(type, production));
341             if (production < maximumProduction) {
342                 result.addMaximumProduction(new AbstractGoods(type, maximumProduction));
343             }
344         }
345         return result;
346     }

513         final Player owner = getOwner();
514         final Turn turn = getGame().getTurn();
515
516         List<Modifier> mods = new ArrayList<>();
517         if (unitType == null) { // Add only the building-specific bonuses
518             mods.addAll(colony.getModifiers(id, type, turn));
519             if (owner != null) {
520                 mods.addAll(owner.getModifiers(id, type, turn));
521             }
522
523         } else { // If a unit is present add unit specific bonuses.
524             mods.addAll(this.getModifiers(id, unitType, turn));
525             mods.addAll(colony.getProductionModifiers(goodsType));
526             mods.addAll(unitType.getModifiers(id, goodsType, turn));
527             if (owner != null) {
528                 mods.addAll(owner.getModifiers(id, unitType, turn));
529             }
530         }
531         Collections.sort(mods);
532         return mods;
533     }
