72     /**
73      * Creates a new <code>ChoiceItem</code> with the
74      * given object.
75      *
76      * @param object The object contained by this choice.
77      */
78     public ChoiceItem(T object) {
79         this(Messages.message(object.toString()), object, true);
80         
81         // Check to see if we can improve upon object.toString()
82         if (object instanceof AbstractGoods) {
83             this.text = Messages.message(((AbstractGoods)object).getLabel());
84         } else if (object instanceof AbstractUnit) {
85             this.text = Messages.message(((AbstractUnit)object).getId());
86         } else if (object instanceof Player) {
87             this.text = Messages.message(((Player)object).getLabel());
88         }
89     }
