470     /**
471      * Chooses the two best raw materials, updating the production
472      * map and lists.
473      *
474      * FIXME: scale with colony size.
475      *
476      * @param production The production map.
477      */
478     private void updateRawMaterials(Map<GoodsType, Map<WorkLocation, Integer>> production) {
479         Player player = colony.getOwner();
480         Market market = player.getMarket();
481         NationType nationType = player.getNationType();
482         GoodsType primaryRawMaterial = null;
483         GoodsType secondaryRawMaterial = null;
484         int primaryValue = -1;
485         int secondaryValue = -1;
486
487         produce.clear();
488         List<GoodsType> rawMaterials = new ArrayList<>(rawLuxuryGoodsTypes);
489         rawMaterials.addAll(otherRawGoodsTypes);
490         for (GoodsType g : rawMaterials) {
491             int value = sum(production.get(g).entrySet(), Entry::getValue);
492             if (value <= LOW_PRODUCTION_THRESHOLD) {
493                 production.remove(g);
494                 continue;
495             }
496             if (market != null) {
497                 // If the market is available, weight by sale price of
498                 // the material, or if it is the raw material for a
499                 // refined goods type, the average of the raw and
500                 // refined goods prices.
501                 if (g.getOutputType() == null) {
502                     value *= market.getSalePrice(g, 1);
503                 } else if (production.containsKey(g.getOutputType())) {
504                     value *= (market.getSalePrice(g, 1)
505                         + market.getSalePrice(g.getOutputType(), 1)) / 2;
506                 }
507             }
508             if (nationType.hasModifier(g.getId())) {
509                 value = (value * 12) / 10; // Bonus for national advantages
510             }
511             if (value > secondaryValue && secondaryRawMaterial != null) {
512                 production.remove(secondaryRawMaterial);
513                 production.remove(secondaryRawMaterial.getOutputType());
514                 if (rawLuxuryGoodsTypes.contains(secondaryRawMaterial)) {
515                     rawLuxuryGoodsTypes.remove(secondaryRawMaterial);
516                     luxuryGoodsTypes.remove(secondaryRawMaterial.getOutputType());
517                 } else if (otherRawGoodsTypes.contains(secondaryRawMaterial)) {
518                     otherRawGoodsTypes.remove(secondaryRawMaterial);
519                 }
520             }
521             if (value > primaryValue) {
522                 secondaryRawMaterial = primaryRawMaterial;
523                 secondaryValue = primaryValue;
524                 primaryRawMaterial = g;
525                 primaryValue = value;
526             } else if (value > secondaryValue) {
527                 secondaryRawMaterial = g;
528                 secondaryValue = value;
529             }
530         }
531         if (primaryRawMaterial != null) {
532             produce.add(primaryRawMaterial);
533             if (primaryRawMaterial.getOutputType() != null) {
534                 produce.add(primaryRawMaterial.getOutputType());
535             }
536             if (secondaryRawMaterial != null) {
537                 produce.add(secondaryRawMaterial);
538                 if (secondaryRawMaterial.getOutputType() != null) {
539                     produce.add(secondaryRawMaterial.getOutputType());
540                 }
541             }
542         }
543     }

655     /**
656      * Updates the build plans for this colony.
657      */
658     private void updateBuildableTypes() {
659         final AIPlayer euaip = getAIMain()
660             .getAIPlayer(colony.getOwner());
661         String advantage = euaip.getAIAdvantage();
662         buildPlans.clear();
663
664         int maxLevel;
665         switch (profileType) {
666         case OUTPOST:
667         case SMALL:   maxLevel = 1; break;
668         case MEDIUM:  maxLevel = 2; break;
669         case LARGE:   maxLevel = 3; break;
670         case CAPITAL: maxLevel = 4; break;
671         default:
672             throw new IllegalStateException("Bogus profile type: "
673                 + profileType);
674         }
675
676         Player player = colony.getOwner();
677         for (BuildingType type : spec().getBuildingTypeList()) {
678             boolean expectFail = false;
679             if (!colony.canBuild(type)) continue;
680
681             // Exempt defence and export from the level check.
682             if (type.hasModifier(Modifier.DEFENCE)) {
683                 double factor = 1.0;
684                 if ("conquest".equals(advantage)) factor = 1.1;
685                 prioritize(type, FORTIFY_WEIGHT * factor,
686                     1.0/*FIXME: 0 if FF underway*/);
687             }
688             if (type.hasAbility(Ability.EXPORT)) {
689                 double factor = 1.0;
690                 if ("trade".equals(advantage)) factor = 1.1;
691                 prioritize(type, EXPORT_WEIGHT * factor,
692                     1.0/*FIXME: weigh production v transport*/);
693             }
694
695             // Skip later stage buildings for smaller settlements.
696             if (type.getLevel() > maxLevel) continue;
697
698             // Scale docks by the improvement available to the food supply.
699             if (type.hasAbility(Ability.PRODUCE_IN_WATER)) {
700                 double factor = 0.0;
701                 if (!colony.hasAbility(Ability.PRODUCE_IN_WATER)
702                     && colony.getTile().isShore()) {
703                     int landFood = 0, seaFood = 0;
704                     for (Tile t : colony.getTile().getSurroundingTiles(1)) {
705                         if (t.getOwningSettlement() == colony
706                             || player.canClaimForSettlement(t)) {
707                             for (AbstractGoods ag : t.getSortedPotential()) {
708                                 if (ag.getType().isFoodType()) {
709                                     if (t.isLand()) {
710                                         landFood += ag.getAmount();
711                                     } else {
712                                         seaFood += ag.getAmount();
713                                     }
714                                 }
715                             }
716                         }
717                     }
718                     factor = (seaFood + landFood == 0) ? 0.0
719                         : seaFood / (double)(seaFood + landFood);
720                 }
721                 prioritize(type, FISH_WEIGHT, factor);
722             }
723
724             if (type.hasAbility(Ability.BUILD)) {
725                 double factor = 1.0;
726                 if ("building".equals(advantage)) factor = 1.1;
727                 double support = 1.0;
728                 for (Ability a : type.getAbilities(Ability.BUILD)) {
729                     List<Scope> scopes = a.getScopes();
730                     if (scopes != null && !scopes.isEmpty()) support = 0.1;
731                 }
732                 prioritize(type, BUILDING_WEIGHT * factor,
733                     support/*FIXME: need for the thing now buildable*/);
734             }
735
736             if (type.hasAbility(Ability.TEACH)) {
737                 prioritize(type, TEACH_WEIGHT,
738                     1.0/*FIXME: #students, #specialists here, #wanted*/);
739             }
740
741             if (type.hasAbility(Ability.REPAIR_UNITS)) {
742                 double factor = 1.0;
743                 if ("naval".equals(advantage)) factor = 1.1;
744                 prioritize(type, REPAIR_WEIGHT * factor,
745                     1.0/*FIXME: #units-to-repair, has-Europe etc*/);
746             }

753                 }

756                     if (type.hasModifier(g.getId())) {
757                         if (!prioritizeProduction(type, g)) {
758                             expectFail = true;
759                         }
760                     }
761                 }
762                 // Hacks.  No good way to make this really generic.
763                 if (type.hasModifier(Modifier.WAREHOUSE_STORAGE)) {
764                     double factor = 1.0;
765                     if ("trade".equals(advantage)) factor = 1.1;
766                     prioritize(type, STORAGE_WEIGHT * factor,
767                         1.0/*FIXME: amount of goods*/);
768                 }
769                 if (type.hasModifier(Modifier.BREEDING_DIVISOR)) {
770                     prioritize(type, BREEDING_WEIGHT,
771                         1.0/*FIXME: horses present?*/);
772                 }
773             }

775             if (findBuildPlan(type) == null && !expectFail) {
776                 logger.warning("No building priority found for: " + type);
777             }
778         }
779
780         double wagonNeed = 0.0;
781         if (!colony.isConnectedPort()) { // Inland colonies need transportation
782             int wagons = euaip.getNeededWagons(colony.getTile());
783             wagonNeed = (wagons <= 0) ? 0.0 : (wagons > 3) ? 1.0
784                 : wagons / 3.0;
785         }
786         for (UnitType unitType : spec().getUnitTypeList()) {
787             if (!colony.canBuild(unitType)) continue;
788             if (unitType.hasAbility(Ability.NAVAL_UNIT)) {
789                 ; // FIXME: decide to build a ship
790             } else if (unitType.isDefensive()) {
791                 if (colony.isBadlyDefended()) {
792                     prioritize(unitType, DEFENCE_WEIGHT,
793                         1.0/*FIXME: how badly defended?*/);
794                 }
795             } else if (unitType.hasAbility(Ability.CARRY_GOODS)) {
796                 if (wagonNeed > 0.0) {
797                     double factor = 1.0;
798                     if ("trade".equals(advantage)) factor = 1.1;
799                     prioritize(unitType, TRANSPORT_WEIGHT * factor,
800                         wagonNeed/*FIXME: type.getSpace()*/);
801                 }
802             }
803         }
804
805         // Weight by lower required goods.
806         for (BuildPlan bp : buildPlans) {
807             int difficulty = sum(bp.type.getRequiredGoods(),
808                 ag -> ag.getAmount() > colony.getGoodsCount(ag.getType()),
809                 ag -> {
810                     final GoodsType type = ag.getType();
811                     return (ag.getAmount() - colony.getGoodsCount(type))
812                     // Penalize building with type that can not be
813                     // made locally.
814                         * ((produce.contains(type.getInputType())) ? 1 : 5);
815                 });
816             bp.difficulty = Math.max(1.0f, Math.sqrt((double)difficulty));
817         }
818
819         Collections.sort(buildPlans, buildPlanComparator);
820     }

822     /**
823      * Makes a plan for each type of possible production, that is
824      * those work locations that can use a unit or can auto-produce.
825      * Note that this will almost certainly include clashes over work
826      * locations.  That gets sorted out elsewhere as ColonyPlans do
827      * not examine the units present.
828      *
829      * With the complete list of work plans, finish creating the list
830      * of goods to produce.
831      *
832      * Then filter out the auto-production plans as they are not
833      * going to be helpful for unit allocation.
834      *
835      * Finally sort by desirability.
836      *
837      * @param production A map of the goods type and production.
838      */
839     private void updatePlans(Map<GoodsType, Map<WorkLocation, Integer>> production) {
840         workPlans.clear();
841         for (Entry<GoodsType, Map<WorkLocation, Integer>> entry
842                  : production.entrySet()) {
843             GoodsType g = entry.getKey();
844             // Do not make plans to produce into a full warehouse.
