50         DESTROY_COLONY,       // Winning natives burn a colony

167     /**
168      * Is this a combat between a rebel player and the REF at a colony?
169      *
170      * @param attacker The attacker object.
171      * @param defender The defender object.
172      * @return True if the attack is a war of independence battle for a colony.
173      */
174     public boolean combatIsWarOfIndependence(FreeColGameObject attacker,
175                                              FreeColGameObject defender) {
176         if (attacker instanceof Unit
177             && defender instanceof Ownable) {
178             Player aPlayer = ((Unit)attacker).getOwner();
179             Player dPlayer = ((Ownable)defender).getOwner();
180             return (aPlayer.isRebel() && aPlayer.getREFPlayer() == dPlayer)
181                 || (dPlayer.isRebel() && dPlayer.getREFPlayer() == aPlayer);
182         }
183         return false;
184     }
