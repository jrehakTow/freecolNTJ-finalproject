463             String str = xr.getAttribute(FreeColServer.OWNER_TAG,
464                                          (String)null);
465             if (str != null) FreeCol.setName(str);
466
467             defaultSinglePlayer
468                 = xr.getAttribute(FreeColServer.SINGLE_PLAYER_TAG, false);
469             defaultPublicServer
470                 = xr.getAttribute(FreeColServer.PUBLIC_SERVER_TAG, false);
471
472         } catch (FileNotFoundException e) {
473             SwingUtilities.invokeLater(new ErrorJob("server.fileNotFound"));
474             logger.log(Level.WARNING, "Can not find file: " + file.getName(),
475                 e);
476             return false;
477         } catch (XMLStreamException e) {
478             SwingUtilities.invokeLater(new ErrorJob(FreeCol.badLoad(file)));
479             logger.log(Level.WARNING, "Error reading game from: "
480                 + file.getName(), e);
481             return false;
482         } catch (Exception e) {
483             SwingUtilities.invokeLater(new ErrorJob(FreeCol.badLoad(file)));
484             logger.log(Level.WARNING, "Could not load game from: "
485                 + file.getName(), e);
486             return false;
487         }
488
489         // Reload the client options saved with this game.
490         final boolean singlePlayer;
491         final String name;
492         final int port;
493         final int sgo = options.getInteger(ClientOptions.SHOW_SAVEGAME_SETTINGS);
494         boolean show = sgo == ClientOptions.SHOW_SAVEGAME_SETTINGS_ALWAYS
495             || (!defaultSinglePlayer
496                 && sgo == ClientOptions.SHOW_SAVEGAME_SETTINGS_MULTIPLAYER);
497         if (show) {
498             if (!getGUI().showLoadingSavegameDialog(defaultPublicServer,
499                                                     defaultSinglePlayer))
500                 return false;
501             LoadingSavegameInfo lsd = getGUI().getLoadingSavegameInfo();
502             singlePlayer = lsd.isSinglePlayer();
503             name = lsd.getServerName();
504             port = lsd.getPort();
505         } else {
506             singlePlayer = defaultSinglePlayer;
507             name = null;
508             port = -1;
509         }
510         Messages.loadActiveModMessageBundle(options.getActiveMods(),
511                                             FreeCol.getLocale());
512         if (!unblockServer(port)) return false;
513         getGUI().showStatusPanel(Messages.message("status.loadingGame"));
514
515         final File theFile = file;
516         Runnable loadGameJob = () -> {
517             FreeColServer freeColServer = null;
518             StringTemplate err = null;
519             try {
520                 final FreeColSavegameFile saveGame
521                     = new FreeColSavegameFile(theFile);
522                 freeColServer = new FreeColServer(saveGame,
523                     (Specification)null, port, name);
524                 fcc.setFreeColServer(freeColServer);
525                 // Server might have bounced to another port.
526                 fcc.setSinglePlayer(singlePlayer);
527                 igc().setGameConnected();
528                 if (login(FreeCol.getName(), freeColServer.getHost(), 
529                           freeColServer.getPort())) {
530                     SwingUtilities.invokeLater(() -> {
531                             ResourceManager.setScenarioMapping(saveGame.getResourceMapping());
532                             if (userMsg != null) {
533                                 getGUI().showInformationMessage(userMsg);
534                             }
535                             getGUI().closeStatusPanel();
536                         });
537                     return; // Success!
538                 }
539                 err = StringTemplate.key("server.couldNotLogin");
540                 logger.warning("Could not log in.");
541             } catch (FileNotFoundException e) {
542                 err = StringTemplate.key("server.fileNotFound");
543                 logger.log(Level.WARNING, "Can not find file.", e);
544             } catch (IOException e) {
545                 err = StringTemplate.key("server.initialize");
546                 logger.log(Level.WARNING, "Error starting game.", e);
547             } catch (XMLStreamException e) {
548                 err = FreeCol.badLoad(theFile);
549                 logger.log(Level.WARNING, "Stream error.", e);
550             } catch (Exception e) {
551                 String msg = e.getMessage();
552                 err = StringTemplate.name((msg == null) ? "(null)" : msg);
553                 logger.log(Level.WARNING, "FreeCol error.", e);
554             }                
555             if (err != null) {
556                 // If this is a debug run, fail hard.
557                 if (fcc.isHeadless() || FreeColDebugger.getDebugRunTurns() >= 0) {
558                     FreeCol.fatal(Messages.message(err));
559                 }
560                 SwingUtilities.invokeLater(() -> {
561                         getGUI().closeMainPanel();
562                         getGUI().showMainPanel(null);
563                     });
564                 SwingUtilities.invokeLater(new ErrorJob(err));
565             }
566         };
567         fcc.setWork(loadGameJob);
568         return true;
569     }

605      * Quits the current game, optionally notifying and stopping the server.

617             finish();
618         } else {

620         }
621     }
