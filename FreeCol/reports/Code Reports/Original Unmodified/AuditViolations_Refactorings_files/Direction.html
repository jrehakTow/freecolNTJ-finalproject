32 /**
33  * The directions a Unit can move to. Includes deltas for moving
34  * to adjacent squares, which are required due to the isometric
35  * map. Starting north and going clockwise.
36 */
37 public enum Direction implements Named {
38     N  ( 0, -2,  0, -2),
39     NE ( 1, -1,  0, -1),
40     E  ( 1,  0,  1,  0),
41     SE ( 1,  1,  0,  1),
42     S  ( 0,  2,  0,  2),
43     SW ( 0,  1, -1,  1),
44     W  (-1,  0, -1,  0),
45     NW ( 0, -1, -1, -1);

88      *

90      * @param y The y coordinate.
91      * @return The map position after the step.

93     public Map.Position step(int x, int y) {

184     /**
185      * Creates an array of the directions in an order that favours
186      * a supplied one.  Entry 0 will be the supplied direction,
187      * entry 1+2 will be those immediately to the left and right
188      * of it (chosen randomly), and so on until the last entry
189      * will be the complete reverse of the supplied direction.
190      * 
191      * Useful if we want to travel in a particular direction, but
192      * if this fails to be able to try the closest other
193      * directions to the original one in order.
194      *
195      * @param logMe A string to log with the random results.
196      * @param logger A <code>Logger</code> to log to.
197      * @param random A <code>Random</code> number source.
198      * @return An array of the <code>Direction</code>s favouring this one.
199      */
200     public Direction[] getClosestDirections(String logMe, Logger logger,
201                                             Random random) {
202         // Will need 3 bits of randomness --- 2 directions are known,
203         // need one bit to randomize each remaining pair.
204         final int nbits = (NUMBER_OF_DIRECTIONS - 2) / 2;
205         final int r = randomInt(logger, logMe, random, 1 << nbits);
206
207         Direction[] ret = new Direction[NUMBER_OF_DIRECTIONS];
208         ret[0] = this;
209
210         int step = 1, mask = 1;
211         for (int i = 1; i < NUMBER_OF_DIRECTIONS - 1; i += 2) {
212             Direction dr = this.rotate(step);
213             Direction dl = this.rotate(NUMBER_OF_DIRECTIONS - step);
214             ret[i] = ((r & mask) == 0) ? dr : dl;
215             ret[i+1] = ((r & mask) == 0) ? dl : dr;
216             step += 1;
217             mask *= 2;
218         }
219
220         ret[NUMBER_OF_DIRECTIONS-1] = this.getReverseDirection();
221         return ret;
222     }
