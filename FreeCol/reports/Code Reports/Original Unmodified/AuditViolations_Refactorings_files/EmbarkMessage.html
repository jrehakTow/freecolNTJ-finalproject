35 /**
36  * The message sent when embarking.
37  */
38 public class EmbarkMessage extends DOMMessage {
39
40     public static final String TAG = "embark";
41     private static final String CARRIER_TAG = "carrier";
42     private static final String DIRECTION_TAG = "direction";
43     private static final String UNIT_TAG = "unit";
44
45     /** The identifier of the unit embarking. */
46     private final String unitId;
47
48     /** The identifier of the carrier to embark onto. */
49     private final String carrierId;
50
51     /** An optional direction for the unit to move to find the carrier. */
52     private final String directionString;
53
54
55     /**
56      * Create a new <code>EmbarkMessage</code> with the
57      * supplied unit, carrier and optional direction.
58      *
59      * @param unit The <code>Unit</code> to embark.
60      * @param carrier The carrier <code>Unit</code> to embark on.
61      * @param direction An option direction to embark in.
62      */
63     public EmbarkMessage(Unit unit, Unit carrier, Direction direction) {
64         super(getTagName());
65
66         this.unitId = unit.getId();
67         this.carrierId = carrier.getId();
68         this.directionString = (direction == null) ? null
69             : String.valueOf(direction);
70     }
71
72     /**
73      * Create a new <code>EmbarkMessage</code> from a
74      * supplied element.
75      *
76      * @param game The <code>Game</code> this message belongs to.
77      * @param element The <code>Element</code> to use to create the message.
78      */
79     public EmbarkMessage(Game game, Element element) {
80         super(getTagName());
81
82         this.unitId = getStringAttribute(element, UNIT_TAG);
83         this.carrierId = getStringAttribute(element, CARRIER_TAG);
84         this.directionString = getStringAttribute(element, DIRECTION_TAG);
85     }
86
87
88     /**
89      * Handle a "embark"-message.
90      *
91      * @param server The <code>FreeColServer</code> handling the message.
92      * @param player The <code>Player</code> the message applies to.
93      * @param connection The <code>Connection</code> message was received on.
94      * @return An update containing the embarked unit, or an error
95      *     <code>Element</code> on failure.
96      */
97     public Element handle(FreeColServer server, Player player,
98                           Connection connection) {
99         final ServerPlayer serverPlayer = server.getPlayer(connection);
100
101         ServerUnit unit;
102         try {
103             unit = player.getOurFreeColGameObject(this.unitId, ServerUnit.class);
104         } catch (Exception e) {
105             return serverPlayer.clientError(e.getMessage())
106                 .build(serverPlayer);
107         }
108
109         Unit carrier;
110         try {
111             carrier = player.getOurFreeColGameObject(this.carrierId, Unit.class);
112         } catch (Exception e) {
113             return serverPlayer.clientError(e.getMessage())
114                 .build(serverPlayer);
115         }
116
117         Location sourceLocation = unit.getLocation();
118         if (this.directionString == null) {
119             // Locations must be the same, or the source is also a
120             // carrier in the same location as the carrier, or they
121             // must be on the same tile.
122             if (!carrier.isAtLocation(sourceLocation)) {
123                 return serverPlayer.clientError("Unit " + this.unitId
124                     + " at " + sourceLocation.getId()
125                     + " and carrier " + this.carrierId
126                     + " at " + carrier.getLocation().getId()
127                     + " are not co-located.")
128                     .build(serverPlayer);
129             }
130         } else {
131             // Units have to be on the map and have moves left if a
132             // move is involved.
133             if (unit.getMovesLeft() <= 0) {
134                 return serverPlayer.clientError("Unit has no moves left: "
135                     + this.unitId)
136                     .build(serverPlayer);
137             }
138
139             Tile destinationTile = null;
140             try {
141                 destinationTile = unit.getNeighbourTile(this.directionString);
142             } catch (Exception e) {
143                 return serverPlayer.clientError(e.getMessage())
144                     .build(serverPlayer);
145             }
146             if (carrier.getTile() != destinationTile) {
147                 return serverPlayer.clientError("Carrier: " + this.carrierId
148                     + " is not at destination tile: " + destinationTile)
149                     .build(serverPlayer);
150             }
151         }
152
153         // Proceed to embark
154         return server.getInGameController()
155             .embarkUnit(serverPlayer, unit, carrier)
156             .build(serverPlayer);
157     }

159     /**
160      * Convert this EmbarkMessage to XML.
161      *
162      * @return The XML representation of this message.
163      */
164     @Override
165     public Element toXMLElement() {
166         return new DOMMessage(getTagName(),
167             UNIT_TAG, this.unitId,
168             CARRIER_TAG, this.carrierId,
169             DIRECTION_TAG, this.directionString).toXMLElement();
170     }

178         return TAG;
179     }
