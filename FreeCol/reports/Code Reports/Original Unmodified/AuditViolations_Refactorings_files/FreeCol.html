436      * Takes care to use the *last* instance.
437      *
438      * @param option The option to find.
439      * @param args The  command-line arguments.
440      * @return The option's parameter.
441      */
442     private static String findArg(String option, String[] args) {
443         for (int i = args.length - 2; i >= 0; i--) {
444             if (option.equals(args[i])) {
445                 return args[i+1];
446             }
447         }
448         return null;
449     }
