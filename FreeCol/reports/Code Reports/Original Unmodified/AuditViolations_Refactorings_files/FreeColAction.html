113     protected FreeColAction(FreeColClient freeColClient, String id) {
114         super(Messages.getName(id));
115
116         this.freeColClient = freeColClient;
117
118         putValue(ACTION_ID, id);
119
120         String shortDescription = Messages.getDescription(id);

123         }
124
125         String acceleratorKey = id + ".accelerator";
126         String accelerator = Messages.message(acceleratorKey);
127         if (!accelerator.equals(acceleratorKey)) {
128             setAccelerator(KeyStroke.getKeyStroke(accelerator));
129         }
130     }

237         orderButtonImageCount = (ResourceManager.hasImageResource(normalKey) ? 1 : 0)

240             + (ResourceManager.hasImageResource(disabledKey) ? 1 : 0);
241         if (hasOrderButtons()) {
242             putValue(BUTTON_IMAGE, normalKey);

244             putValue(BUTTON_PRESSED_IMAGE, pressedKey);
245             putValue(BUTTON_DISABLED_IMAGE, disabledKey);
246         } else {
247             logger.warning("Missing " + (4-orderButtonImageCount)
