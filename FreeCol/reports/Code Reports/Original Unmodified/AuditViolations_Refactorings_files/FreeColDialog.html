123     /**
124      * Select the default option from the supplied options.
125      *
126      * @param options A list of options to choose from.
127      * @return The option to select initially.
128      */
129     private int selectDefault(List<ChoiceItem<T>> options) {
130         int def = -1, can = -1, ok = -1, i = 0;
131         for (ChoiceItem<T> ci : options) {
132             if (ci.isDefault()) def = i;
133             else if (ci.isCancel()) can = i;
134             else if (ci.isOK()) ok = i;
135             i++;
136         }
137         return (def >= 0) ? def : (can >= 0) ? can : (ok >= 0) ? ok
138             : options.size() - 1;
139     }
