473     /**
474      * Set the derived fields for the goods types in a specification.
475      *
476      * The "derived" fields are: buildingMaterial + makes
477      * - buildingMaterial depends on whether a GoodsType is present on
478      *   a BuildableType requiredGoods list
479      * - makes depends on whether a GoodsType madeFrom field refers
480      *   to another
481      *   
482      * This is called from Specification.clean() when the
483      * specification is fully read.  We must wait until then as the
484      * made-from field can change in extended specifications and mods.
485      * The current example of which is horses, which is made-from food
486      * in the classic ruleset and made-from grain in the freecol
487      * ruleset.
488      *
489      * @param spec The <code>Specification</code> to operate on.
490      */
491     public static void setDerivedAttributes(Specification spec) {
492         // Reset to default state
493         for (GoodsType g : spec.getGoodsTypeList()) {
494             g.buildingMaterial = false;
495             g.makes = null;
496         }
497
498         // Set buildingMaterial attribute
499         List<BuildableType> buildableTypes = new ArrayList<>();
500         buildableTypes.addAll(spec.getBuildingTypeList());
501         buildableTypes.addAll(spec.getUnitTypeList());
502         buildableTypes.addAll(spec.getRoles());
503         for (BuildableType b : buildableTypes) {
504             for (AbstractGoods ag : b.getRequiredGoods()) {
505                 ag.getType().buildingMaterial = true;
506             }
507         }
508
509         // Set makes attribute
510         for (GoodsType g : spec.getGoodsTypeList()) {
511             if (g.madeFrom != null) g.madeFrom.makes = g;
512         }
513     }
