32 /**
33  * A class to wrap a StringBuilder for log generation purposes.
34  */
35 public class LogBuilder {
36
37     /** The string builder to use. */
38     private final StringBuilder sb;
39
40     /** The remembered buffer index. */
41     private final List<Integer> points = new ArrayList<>();
42
43
44     /**
45      * Create a new LogBuilder that can only be used as a buffer.
46      *
47      * @param size An initial size for the buffer.
48      */
49     public LogBuilder(int size) {
50         this.sb = (size <= 0) ? null : new StringBuilder(size);
51     }

54     /**
55      * Convert a simple object to a string suitable for a log buffer.
56      *
57      * @param o The <code>Object</code> to convert.
58      * @return The simple string result.
59      */
60     private static String o2s(Object o) {
61         return (o == null) ? "null"
62             : (o instanceof Class) ? ((Class)o).getName()
63             : (o instanceof String) ? (String)o
64             : (o instanceof Location) ? ((Location)o).toShortString()
65             : o.toString();
66     }
67
68     /**
69      * Add objects to a string builder.
70      *
71      * @param sb The <code>StringBuilder</code> to add to.
72      * @param objects The objects to add.
73      */
74     private static void add(StringBuilder sb, Object... objects) {
75         for (Object o : objects) {
76             if (o instanceof Object[]) {
77                 for (Object o2 : (Object[])o) {
78                     sb.append(o2s(o2));
79                 }
80             } else {
81                 sb.append(o2s(o));
82             }
83         }
84     }
85
86     /**
87      * Add objects to the buffer.
88      *
89      * @param objects The objects to add.
90      */
91     public void add(Object... objects) {
92         if (sb != null) add(sb, objects);
93     }
94
95     /**
96      * Add a delimited collection to the buffer.
97      *
98      * @param <T> The collection member type.
99      * @param delim An internal delimiter.
100      * @param c The <code>Collection</code> of objects to add.
101      */
102     public <T> void addCollection(String delim, Collection<T> c) {
103         if (sb != null) {
104             for (T t : c) add(sb, t, delim);
105             if (!c.isEmpty()) shrink(delim);
106         }
107     }
108
109     /**
110      * Add a stack trace to the buffer.
111      */
112     public void addStackTrace() {
113         if (sb != null) FreeColDebugger.addStackTrace(this);
114     }

116     /**
117      * Truncate the buffer to a given size.
118      *
119      * @param size The size to truncate to.
120      */
121     public void truncate(int size) {
122         if (sb != null && sb.length() > size) sb.setLength(size);
123     }

125     /**
126      * Remember a position in a buffer.
127      */
128     public void mark() {
129         if (sb != null) {
130             this.points.add(0, sb.length());
131         }
132     }
133
134     /**
135      * Check if a buffer has grown since marked, and optionally insert
136      * text at that point.
137      *
138      * @param objects Optional <code>Object</code>s to insert if the buffer has
139      *     grown.
140      * @return True if the buffer grew (before inserting).
141      */
142     public boolean grew(Object... objects) {
143         if (sb == null) return false;
144         int p = this.points.remove(0);
145         if (sb.length() <= p) return false;
146         StringBuilder sb2 = new StringBuilder(64);
147         add(sb2, objects);
148         this.sb.insert(p, sb2.toString());
149         return true;
150     }

152     /**
153      * Shorten a buffer by a trailing delimiter.
154      *
155      * (Cheats, does not really check that the delimiter is there)
156      *
157      * @param delim The delimiter to remove.
158      */
159     public void shrink(String delim) {
160         if (sb != null && delim != null) {
161             sb.setLength(sb.length() - delim.length());
162         }
163     }

165     /**
166      * Output to a logger.
167      *
168      * @param logger The <code>Logger</code> to write to.
169      * @param level The logging <code>Level</code>.
170      */
171     public void log(Logger logger, Level level) {
172         if (sb != null && logger != null && level != null
173             && logger.isLoggable(level)) {
174             logger.log(level, toString());
175         }
176     }

179      * Get the amount of accumulated data.
180      *
181      * @return The amount of data accumulated so far.
182      */
183     public int size() {
184         return (sb == null) ? 0 : sb.length();
185     }

187     /**
188      * Add a group of objects to the buffer at a particular width
189      *
190      * @param size The width to set.
191      * @param objects The <code>Object</code>s to add.
192      * @return The widened string.
193      */
194     public static String wide(int size, Object... objects) {
195         if (size == 0) return "";
196         boolean left = size > 0;
197         if (!left) size = -size;
198         StringBuilder s2 = new StringBuilder(size);
199         add(s2, objects);
200         int delta = size - s2.length();
201         if (left) {
202             for (; delta > 0; delta--) s2.append(" ");
203         } else {
204             for (; delta > 0; delta--) s2.insert(0, " ");
205         }
206         if (delta < 0) s2.setLength(size);
207         return s2.toString();
208     }
