51 /**
52  * Represents a collection of messages in a particular locale.
53  *
54  * The individual messages are read from property files in the
55  * <code>data/strings</code> directory. The property files are called
56  * "FreeColMessages[_LANGUAGE[_COUNTRY[_VARIANT]]].properties", where
57  * LANGUAGE should be an ISO 639-2 or ISO 639-3 language code, COUNTRY
58  * should be an ISO 3166-2 country code, and VARIANT is an arbitrary
59  * string. The encoding of the property files is UTF-8. Since the Java
60  * Properties class is unable to handle UTF-8 directly, this class
61  * uses its own implementation.
62  *
63  * The individual messages may include variables, which must be
64  * delimited by percent characters (e.g. "%nation%"), and will be
65  * replaced when the message is formatted. Furthermore, the messages
66  * may include choice formats consisting of a tag followed by a colon
67  * (":"), a selector and one or several choices separated from the
68  * selector and each other by pipe characters ("|"). The entire choice
69  * format must be enclosed in double brackets ("{{" and "}}",
70  * respectively).
71  *
72  * Each choice must consist of a key and a value separated by an
73  * equals character ("="), unless it is a variable, in which case the
74  * variable must resolve to another choice format. The selector may
75  * also be a variable. If the selector is omitted, then one of the
76  * choices should use the key "default". Choice formats may be
77  * nested.
78  *
79  * <pre>
80  *   key1=%colony% tuottaa tuotetta {{tag:acc|%goods%}}.
81  *   key2={{plural:%amount%|one=ruoka|other=ruokaa|default={{tag:|acc=viljaa|default=Vilja}}}}
82  *   key3={{tag:|acc=viljaa|default={{plural:%amount%|one=ruoka|other=ruokaa|default=Ruoka}}}}
83  * </pre>
84  *
85  * This class is NOT thread-safe. (CO: I cannot find any place that
86  * really has a problem)
87  */

450         String name = (containsKey(nameKey(id))) ? getName(id) : null;

674     /**
675      * Get the text mapping for a particular identifier in the
676      * default locale message bundle.  Returns the key as the value if
677      * there is no mapping found!
678      *
679      * @param messageId The key of the message to find.
680      * @return String text mapping or the key
681      */
682     public static String message(String messageId) {
683         // Check that all the values are correct.
684         if (messageId == null) {
685             throw new NullPointerException("Message id must not be null!");
686         }
687
688         // return key as value if there is no mapping found
689         String message = messageBundle.get(messageId);
690         if (message == null) return messageId;
691
692         // otherwise replace variables in the text
693         message = replaceChoices(message, null);
694         return message.trim();
695     }

697     /**
698      * Localizes a StringTemplate.
699      *
700      * @param template The <code>StringTemplate</code> to localize.
701      * @return The localized string.
702      */
703     public static String message(StringTemplate template) {
704         if (template == null) return null;
705         String result = "";
706         switch (template.getTemplateType()) {
707         case LABEL:
708             List<StringTemplate> replacements = template.getReplacements();
709             if (replacements.isEmpty()) {
710                 result = message(template.getId());
711             } else {
712                 for (StringTemplate other : replacements) {
713                     result += template.getId() + message(other);
714                 }
715                 if (result.length() >= template.getId().length()) {
716                     result = result.substring(template.getId().length());
717                 } else {
718                     logger.warning("incorrect use of template " + template);
719                 }
720             }
721             break;
722         case TEMPLATE:
723             if (containsKey(template.getId())) {
724                 result = messageBundle.get(template.getId());
725             } else if (template.getDefaultId() != null) {
726                 result = messageBundle.get(template.getDefaultId());
727             }
728             result = replaceChoices(result, template);
729             for (String key : template.getKeys()) {
730                 result = result.replace(key,
731                                         message(template.getReplacement(key)));
732             }
733             break;
734         case KEY:
735             String key = messageBundle.get(template.getId());
736             result = (key == null) ? template.getId()
737                 : replaceChoices(key, null);
738             break;
739         case NAME:
740         default:
741             result = template.getId();
742             break;
743         }
744         return result;
745     }

747     /**
748      * Replace all choice formats in the given string, using keys and
749      * replacement values from the given template, which may be null.
750      *
751      * A choice format is enclosed in double brackets and consists of
752      * a tag, followed by a colon, followed by an optional selector,
753      * followed by a pipe character, followed by one or several
754      * choices separated by pipe characters. If there is only one
755      * choice, it must be a message identifier or a
756      * variable. Otherwise, each choice consists of a key and a value
757      * separated by an assignment character. Example:
758      * "{{tag:selector|key1=val1|key2=val2}}".
759      *
760      * @param input a <code>String</code> value
761      * @param template a <code>StringTemplate</code> value
762      * @return a <code>String</code> value
763      */
764     private static String replaceChoices(String input, StringTemplate template) {
765         int openChoice = 0;
766         int closeChoice = 0;
767         int highWaterMark = 0;
768         StringBuilder result = new StringBuilder();
769         while ((openChoice = input.indexOf("{{", highWaterMark)) >= 0) {
770             result.append(input.substring(highWaterMark, openChoice));
771             closeChoice = findMatchingBracket(input, openChoice + 2);
772             if (closeChoice < 0) {
773                 // no closing brackets found
774                 logger.warning("Mismatched brackets: " + input);
775                 return result.toString();
776             }
777             highWaterMark = closeChoice + 2;
778             int colonIndex = input.indexOf(':', openChoice + 2);
779             if (colonIndex < 0 || colonIndex > closeChoice) {
780                 logger.warning("No tag found: " + input);
781                 continue;
782             }
783             String tag = input.substring(openChoice + 2, colonIndex);
784             int pipeIndex = input.indexOf('|', colonIndex + 1);
785             if (pipeIndex < 0 || pipeIndex > closeChoice) {
786                 logger.warning("No choices found: " + input);
787                 continue;
788             }
789             String selector = input.substring(colonIndex + 1, pipeIndex);
790             if (selector.isEmpty()) {
791                 selector = "default";
792             } else if (selector.startsWith("%") && selector.endsWith("%")) {
793                 if (template == null) {
794                     selector = "default";
795                 } else {
796                     StringTemplate replacement = template.getReplacement(selector);
797                     if (replacement == null) {
798                         logger.warning("Failed to find replacement for " + selector);
799                         continue;
800                     } else {
801                         selector = message(replacement);
802                         Selector taggedSelector = getSelector(tag);
803                         if (taggedSelector != null) {
804                             selector = taggedSelector.getKey(selector, input);
805                         }
806                     }
807                 }
808             } else {
809                 Selector taggedSelector = getSelector(tag);
810                 if (taggedSelector != null) {
811                     selector = taggedSelector.getKey(selector, input);
812                 }
813             }
814             int keyIndex = input.indexOf(selector, pipeIndex + 1);
815             if (keyIndex < 0 || keyIndex > closeChoice) {
816                 // key not found, choice might be a key itself
817                 String otherKey = input.substring(pipeIndex + 1, closeChoice);
818                 if (otherKey.startsWith("%") && otherKey.endsWith("%")
819                     && template != null) {
820                     StringTemplate replacement = template.getReplacement(otherKey);
821                     if (replacement == null) {
822                         logger.warning("Failed to find replacement for " + otherKey);
823                         continue;
824                     } else if (replacement.getTemplateType() == TemplateType.KEY) {
825                         otherKey = messageBundle.get(replacement.getId());
826                         keyIndex = otherKey.indexOf("{{");
827                         if (keyIndex < 0) {
828                             // not a choice format
829                             result.append(otherKey);
830                         } else {
831                             keyIndex = otherKey.indexOf(selector, keyIndex);
832                             if (keyIndex < 0) {
833                                 logger.warning("Failed to find key " + selector + " in replacement "
834                                                + replacement.getId());
835                                 continue;
836                             } else {
837                                 result.append(getChoice(otherKey, selector));
838                             }
839                         }
840                     } else {
841                         logger.warning("Choice substitution attempted, but template type was "
842                                        + replacement.getTemplateType());
843                         continue;
844                     }
845                 } else if (containsKey(otherKey)) {
846                     otherKey = getChoice(messageBundle.get(otherKey), selector);
847                     result.append(otherKey);
848                 } else {
849                     logger.warning("Unknown key or untagged choice: '" + otherKey
850                                    + "', selector was '" + selector
851                                    + "', trying 'default' instead");
852                     int defaultStart = otherKey.indexOf("default=");
853                     if (defaultStart >= 0) {
854                         defaultStart += 8;
855                         int defaultEnd = otherKey.indexOf('|', defaultStart);
856                         String defaultChoice;
857                         if (defaultEnd < 0) {
858                             defaultChoice = otherKey.substring(defaultStart);
859                         } else {
860                             defaultChoice = otherKey.substring(defaultStart, defaultEnd);
861                         }
862                         result.append(defaultChoice);
863                     } else {
864                         logger.warning("No default choice found.");
865                         continue;
866                     }
867                 }
868             } else {
869                 int start = keyIndex + selector.length() + 1;
870                 int replacementIndex = input.indexOf('|', start);
871                 int nextOpenIndex = input.indexOf("{{", start);
872                 if (nextOpenIndex >= 0 && nextOpenIndex < replacementIndex) {
873                     replacementIndex = input.indexOf('|',
874                         findMatchingBracket(input, nextOpenIndex + 2) + 2);
875                 }
876                 int end = (replacementIndex < 0
877                     || replacementIndex > closeChoice) ? closeChoice
878                     : replacementIndex;
879                 String replacement = input.substring(start, end);
880                 if (!replacement.contains("{{")) {
881                     result.append(replacement);
882                 } else {
883                     result.append(replaceChoices(replacement, template));
884                 }
885             }
886         }
887         result.append(input.substring(highWaterMark));
888         return result.toString();
889     }

891     /**
892      * Return the choice tagged with the given key, or null, if the
893      * given input string does not contain the key.
894      *
895      * @param input a <code>String</code> value
896      * @param key a <code>String</code> value
897      * @return a <code>String</code> value
898      */
899     private static String getChoice(String input, String key) {
900         int keyIndex = input.indexOf(key);
901         if (keyIndex < 0) {
902             return null;
903         } else {
904             int start = keyIndex + key.length() + 1;
905             int end = input.indexOf('|', start);
906             if (end < 0) {
907                 end = input.indexOf("}}", start);

928         int level = 0;
