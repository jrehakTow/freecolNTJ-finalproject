324     /**
325      * Split a message into a list of text and link objects.
326      *
327      * @param player The <code>Player</code> who will see the result.
328      * @return A list of strings and buttons.
329      */
330     public List<Object> splitLinks(Player player) {
331         final FreeColGameObject source = player.getGame()
332             .getMessageSource(this);
333
334         // Build a list of objects, initially containing just the plain
335         // text of the message.
336         List<Object> result = new ArrayList<>();
337         result.add(Messages.message(this));
338
339         for (String key : getKeys()) {
340             // Then for each key, check if it can be made into a link.
341             // If not, ignore it.
342             String val = Messages.message(getReplacement(key));
343             Object b = Utility.getMessageButton(key, val, player, source);
344             if (b == null) continue;
345
346             // ...if so, find all instances of the replacement of the key
347             // in the object list texts, and replace them with buttons.
348             List<Object> next = new ArrayList<>();
349             for (Object o : result) {
350                 if (o instanceof String) {
351                     String str = (String)o;
352                     int index, start = 0;
353                     while ((index = str.indexOf(val, start)) >= 0) {
354                         if (index > start) {
355                             next.add(str.substring(start, index));
356                         }
357                         next.add(b);
358                         start = index + val.length();
359                     }
360                     next.add(str.substring(start, str.length()));
361                 } else {
362                     next.add(o);
363                 }
364             }
365             result = next;
366         }
367         return result;
368     }
