299         for (UnitType unitType : spec.getUnitTypeList()) {
300             if (unitType.hasAbility(Ability.SUPPORT_UNIT)) {
301                 if (unitType.hasAbility(Ability.NAVAL_UNIT)) {
302                     navalTypes.add(unitType);
303                 } else if (unitType.hasAbility(Ability.BOMBARD)) {
304                     bombardTypes.add(unitType);
305                 } else if (unitType.hasAbility(Ability.CAN_BE_EQUIPPED)) {
306                     landTypes.add(unitType);
307                 }
308             }
309             if (unitType.hasAbility(Ability.MERCENARY_UNIT)) {
310                 mercenaryTypes.add(unitType);
311             }
312         }
313         for (Role r : spec.getMilitaryRoles()) {
314             boolean ok = r.isAvailableTo(player, landTypes.get(0));
315             boolean armed = r.hasAbility(Ability.ARMED);
316             boolean mounted = r.hasAbility(Ability.MOUNTED);
317             boolean ref = r.requiresAbility(Ability.REF_UNIT);
318             if (armed && mounted) {

321                 } else if (!ok && ref && refMountedRole == null) {
322                     refMountedRole = r;
323                 }
324             } else if (armed && !mounted) {
325                 if (ok && !ref && armedRole == null) {
326                     armedRole = r;
327                 } else if (!ok && ref && refArmedRole == null) {
328                     refArmedRole = r;
329                 }
330             }
331         }

360     /**
361      * Checks if a specified action is valid at present.
362      *
363      * @param action The <code>MonarchAction</code> to check.
364      * @return True if the action is valid.
365      */
366     public boolean actionIsValid(MonarchAction action) {
367         initializeCaches();
368
369         switch (action) {
370         case NO_ACTION:
371             return true;
372         case RAISE_TAX_ACT: case RAISE_TAX_WAR:
373             return player.getTax() < taxMaximum();
374         case FORCE_TAX:
375             return false;
376         case LOWER_TAX_WAR: case LOWER_TAX_OTHER:
377             return player.getTax() > MINIMUM_TAX_RATE + 10;
378         case WAIVE_TAX:
379             return true;
380         case ADD_TO_REF:
381             return !(navalREFUnitTypes.isEmpty() || landREFUnitTypes.isEmpty());
382         case DECLARE_PEACE:
383             return !collectPotentialFriends().isEmpty();
384         case DECLARE_WAR:
385             return !collectPotentialEnemies().isEmpty();
386         case SUPPORT_SEA:
387             return player.getAttackedByPrivateers() && !getSupportSea()
388                 && !getDispleasure();
389         case SUPPORT_LAND: case MONARCH_MERCENARIES:
390             return player.isAtWar() && !getDispleasure()
391                 && player.hasSettlements();
392         case HESSIAN_MERCENARIES:
393             return player.checkGold(HESSIAN_MINIMUM_PRICE)
394                 && player.hasSettlements();
395         case DISPLEASURE:
396             return false;
397         default:
398             throw new IllegalArgumentException("Bogus monarch action: "

411         int dx = 1 + spec.getInteger(GameOptions.MONARCH_MEDDLING);

426         addIfValid(choices, MonarchAction.RAISE_TAX_ACT, 5 + dx);
427         addIfValid(choices, MonarchAction.RAISE_TAX_WAR, 5 + dx);
428         addIfValid(choices, MonarchAction.LOWER_TAX_WAR, 5 - dx);

430         addIfValid(choices, MonarchAction.ADD_TO_REF, 10 + dx);
431         addIfValid(choices, MonarchAction.DECLARE_PEACE, 6 - dx);
432         addIfValid(choices, MonarchAction.DECLARE_WAR, 5 + dx);

437         }
438         addIfValid(choices, MonarchAction.SUPPORT_SEA, 6 - dx);
439         addIfValid(choices, MonarchAction.HESSIAN_MERCENARIES, 6-dx);

441         return choices;

643     /**
644      * Check if the monarch provides support for a war.
645      *
646      * @param enemy The enemy <code>Player</code>.
647      * @param random A pseudo-random number source.
648      * @return A list of <code>AbstractUnit</code>s provided as support.
649      */
650     public List<AbstractUnit> getWarSupport(Player enemy, Random random) {
651         final Specification spec = getSpecification();
652         final double baseStrength = player.calculateStrength(false);
653         final double enemyStrength = enemy.calculateStrength(false);
654         final double strengthRatio
655             = Player.strengthRatio(baseStrength, enemyStrength);
656         List<AbstractUnit> result = new ArrayList<AbstractUnit>();
657         // We do not really know what Col1 did to decide whether to
658         // provide war support, so we have made something up.
659         //
660         // Strength ratios are in [0, 1].  Support is granted in negative
661         // proportion to the base strength ratio, such that we always
662         // support if the enemy is stronger (ratio < 0.5), and never
663         // support if the player is more than 50% stronger (ratio >
664         // 0.6).  However if war support force sufficiently large with
665         // respect to the current player and enemy forces it will be
666         // reduced.
667         // The principle at work is that the Crown is cautious/stingy.
668         final double NOSUPPORT = 0.6;
669         double p = 10.0 * (NOSUPPORT - strengthRatio);
670         if (p >= 1.0 // Avoid calling randomDouble if unnecessary
671             || (p > 0.0 && p > randomDouble(logger, "War support?", random))) {
672             Force wsf = getWarSupportForce();
673             result.addAll(wsf.getUnits());
674             double supportStrength, fullRatio, strength, ratio;
675             supportStrength = wsf.calculateStrength(false);
676             fullRatio = Player.strengthRatio(baseStrength + supportStrength,
677                                              enemyStrength);
678             if (fullRatio < NOSUPPORT) { // Full support, some randomization
679                 for (AbstractUnit au : result) {
680                     int amount = au.getNumber();
681                     amount += randomInt(logger, "Vary war force " + au.getId(),
682                                         random, 3) - 1;
683                     au.setNumber(amount);
684                 }
685             } else if (enemyStrength <= 0.0) { // Enemy is defenceless: 1 unit
686                 while (result.size() > 1) result.remove(0);
687                 result.get(0).setNumber(1);
688             } else { // Reduce force until below NOSUPPORT or single unit/s
689                 outer: for (AbstractUnit au : result) {
690                     for (int n = au.getNumber() - 1; n >= 1; n--) {
691                         au.setNumber(n);
692                         strength = AbstractUnit.calculateStrength(spec, result);
693                         ratio = Player.strengthRatio(baseStrength + strength,
694                                                      enemyStrength);
695                         if (ratio < NOSUPPORT) break outer;
696                     }
697                 }
698             }
699             strength = AbstractUnit.calculateStrength(spec, result);
700             ratio = Player.strengthRatio(baseStrength+strength, enemyStrength);
701             logger.finest("War support:"
702                 + " initially=" + supportStrength + "/" + fullRatio
703                 + " finally=" + strength + "/" + ratio);
704         }
705         return result;
706     }

710      *
711      * @param random The <code>Random</code> number source to use.
712      * @return A troop of mercenaries.

715         initializeCaches();
716
717         final Specification spec = getSpecification();
718         final Role defaultRole = spec.getDefaultRole();
719         final int mercPrice = spec.getInteger(GameOptions.MERCENARY_PRICE);

721         landRoles.add(armedRole);
722         landRoles.add(mountedRole);
723
724         // FIXME: magic numbers for 2-4 mercs
725         List<AbstractUnit> mercs = new ArrayList<>();
726         int count = randomInt(logger, "Mercenary count", random, 2) + 2;
727         int price = 0;
728         UnitType unitType = null;
729         List<UnitType> unitTypes = new ArrayList<>(mercenaryTypes);
730         while (!unitTypes.isEmpty() && count > 0) {

758                 ? armedRole : defaultRole;
759             mercs.add(new AbstractUnit(unitType, r.getId(), 1));
760         }
