69     /**
70      * Creates a nation summary for the specified player.
71      *
72      * @param player The <code>Player</code> to create the summary for.
73      * @param requester The <code>Player</code> making the request.
74      */
75     public NationSummary(Player player, Player requester) {
76         this();
77
78         stance = player.getStance(requester);
79         if (stance == Stance.UNCONTACTED) stance = Stance.PEACE;
80
81         numberOfSettlements = player.getSettlements().size();
82
83         if (player.isEuropean()) {
84             numberOfUnits = player.getUnits().size();
85             militaryStrength = player.calculateStrength(false);
86             navalStrength = player.calculateStrength(true);
87             gold = player.getGold();
88             if (player == requester || requester
89                 .hasAbility(Ability.BETTER_FOREIGN_AFFAIRS_REPORT)) {
90                 soL = player.getSoL();
91                 foundingFathers = player.getFatherCount();
92                 tax = player.getTax();
93             } else {
94                 soL = foundingFathers = tax = -1;
95             }
96         } else {
97             numberOfUnits = militaryStrength = navalStrength = gold = soL
98                 = foundingFathers = tax = -1;
99         }
100     }
