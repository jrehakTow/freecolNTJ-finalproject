344      * Update specification and difficulty as needed.
345      */
346     private void update() {
347         // If the TC box changed, update the specification.
348         boolean changed = !this.specification.getId().equals(getSelectedTC().getId());
349         if (changed) this.specification = getSpecification();
350             
351         // If the difficulty box changed, update the difficulty.
352         if (!this.difficulty.getId().equals(getSelectedDifficulty().getId())) {
353             this.difficulty = getSelectedDifficulty();
354             changed = true;
355         }
356
357         // Harmonize the new spec and difficulty if something changed.
358         if (changed) {
359             // If using a custom difficulty, update it within the
360             // spec, otherwise read the difficulty from the new spec.
361             if (this.difficulty.isEditable()) {
362                 this.specification.prepare(null, this.difficulty);
363             } else {
364                 String oldId = this.difficulty.getId();
365                 this.difficulty = this.specification
366                     .getDifficultyOptionGroup(oldId);
367             }
368             updateDifficultyBox();
369         }
370     }

511
512     /**
513      * {@inheritDoc}
514      */
515     @Override
516     public void actionPerformed(ActionEvent ae) {
517         final ConnectController cc = getFreeColClient().getConnectController();
518         final SwingGUI gui = getGUI();
519         final String command = ae.getActionCommand();
520
521         switch (Enum.valueOf(NewPanelAction.class, command)) {
522         case OK:
523             FreeCol.setName(getSelectedName());
524             FreeCol.setAdvantages(getSelectedAdvantages());
525             FreeCol.setTC(getSelectedTC().getId());
526
527             NewPanelAction action = Enum.valueOf(NewPanelAction.class,
528                 buttonGroup.getSelection().getActionCommand());
529             switch (action) {
530             case SINGLE:
531                 this.specification.prepare(getSelectedAdvantages(),
532                                            this.difficulty);
533                 if (cc.startSinglePlayerGame(this.specification, false)) return;
534                 break;
535             case JOIN:
536                 int joinPort = getSelectedPort(this.joinPortField);
537                 if (joinPort < 0) break;
538                 if (cc.joinMultiplayerGame(this.joinNameField.getText(),
539                                            joinPort)) return;
540                 break;
541             case START:
542                 int serverPort = getSelectedPort(this.serverPortField);
543                 if (serverPort < 0) break;
544                 this.specification.prepare(getSelectedAdvantages(),
545                                            this.difficulty);
546                 if (cc.startMultiplayerGame(this.specification,
547                         this.publicServer.isSelected(), serverPort)) return;
548                 break;
549             case META_SERVER:
550                 List<ServerInfo> servers = cc.getServerList();
551                 if (servers != null) gui.showServerListPanel(servers);
552                 break;
553             default:
554                 break;
555             }
556             break;
557         case CANCEL:
558             gui.removeFromCanvas(this);
559             gui.showMainPanel(null);
560             break;
561         case SHOW_DIFFICULTY:
562             this.difficulty = gui.showDifficultyDialog(this.specification,
563                                                        this.difficulty);
564             update(); // Brings in new difficulty if edited
565             break;
566         case SINGLE: case JOIN: case START: case META_SERVER:
567             enableComponents();
568             break;
569         default:
570             super.actionPerformed(ae);
571             break;
572         }
573     }
