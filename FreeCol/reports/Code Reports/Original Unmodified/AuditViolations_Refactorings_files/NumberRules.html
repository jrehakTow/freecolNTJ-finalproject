123     private static void readFromXML(FreeColXMLReader xr) throws XMLStreamException {
124         while (xr.nextTag() != XMLStreamConstants.END_ELEMENT) {
125             String tag = xr.getLocalName();
126             if (null != tag) switch (tag) {
127                 case VERSION_TAG:
128                     xr.nextTag();
129                     break;
130                 case GENERATION_TAG:
131                     xr.nextTag();
132                     break;
133                 case PLURALS_TAG:
134                     while (xr.nextTag() != XMLStreamConstants.END_ELEMENT) {
135                         tag = xr.getLocalName();
136                         if (PLURAL_RULES_TAG.equals(tag)) {
137                             readChild(xr);
138                         }
139                     }   break;
140             }
141         }
142     }

144     private static void readChild(FreeColXMLReader xr) throws XMLStreamException {
145         String loc = xr.getAttribute(LOCALES_TAG, (String)null);
146         String[] locales = (loc == null) ? null : loc.split(" ");
147         if (locales != null) {
148             DefaultNumberRule numberRule = new DefaultNumberRule();
149             while (xr.nextTag() != XMLStreamConstants.END_ELEMENT) {
150                 if (PLURAL_RULE_TAG.equals(xr.getLocalName())) {
151                     String plu = xr.getAttribute(COUNT_TAG, (String)null);
152                     Category category = Category.valueOf(plu);
153                     Rule rule = new Rule(xr.getElementText());
154                     numberRule.addRule(category, rule);
155                 }
156             }
157             Number number = null;
158             switch(numberRule.countRules()) {
159             case 0:
160                 number = OTHER_NUMBER_RULE;
161                 break;
162             case 1:
163                 Rule rule = numberRule.getRule(Category.one);
164                 if (rule != null) {
165                     if (null != rule.toString()) switch (rule.toString()) {
166                     case "n is 1":
167                         number = PLURAL_NUMBER_RULE;
168                         break;
169                     case "n in 0..1":
170                         number = ZERO_ONE_NUMBER_RULE;
171                         break;
172                 }
173                 }
174                 break;
175             case 2:
176                 Rule oneRule = numberRule.getRule(Category.one);
177                 Rule twoRule = numberRule.getRule(Category.two);
178                 if (oneRule != null
179                     && "n is 1".equals(oneRule.toString())
180                     && twoRule != null
181                     && "n is 2".equals(twoRule.toString())) {
182                     number = DUAL_NUMBER_RULE;
183                 }
184                 break;
185             default:
186                 number = numberRule;
187             }
188             for (String locale : locales) {
189                 numberMap.put(locale, number);
190             }
191         }
