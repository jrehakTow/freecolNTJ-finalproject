552          * @param row The specified row.
553          * @param column The specified column.
554          * @return True if the specified cell is editable.
555          */
556         @Override
557         public boolean isCellEditable(int row, int column) {
558             if (row >= 0 && row < getRowCount()) {
559                 Nation nation = nations.get(row);
560                 boolean ownRow = thisPlayer == players.get(nation)
561                     && !thisPlayer.isReady();
562                 switch (column) {
563                 case AVAILABILITY_COLUMN:
564                     return !ownRow && thisPlayer.isAdmin();
565                 case ADVANTAGE_COLUMN:
566                     return nation.getType() instanceof EuropeanNationType
567                         && ownRow;
568                 case COLOR_COLUMN:
569                     // Allow a player to change all the colors. 
570                     // This is an accessibility issue for users with a
571                     // colour vision deficiency.  Better to support them
572                     // and just admit that if someone wants to be a pain
573                     // and set all the colours the same, they can.
574                     return nation.getType() instanceof EuropeanNationType;
575                 case PLAYER_COLUMN:
576                     return nation.getType() instanceof EuropeanNationType
577                         && players.get(nation) == null;
578                 }
579             }
580             return false;
581         }
582
583         /**
584          * Sets the value at the specified location.
585          *
586          * @param value The new value.
587          * @param row The specified row.
588          * @param column The specified column.
589          */
590         @Override
591         public void setValueAt(Object value, int row, int column) {
592             if (row >= 0 && row < getRowCount()
593                 && column > 0 && column < getColumnCount()) {
594                 // Column 0 can't be updated.
595                 Nation nation = nations.get(row);
596                 switch (column) {
597                 case ADVANTAGE_COLUMN:
598                     preGameController.setNationType((NationType)value);
599                     break;
600                 case AVAILABILITY_COLUMN:
601                     preGameController.setAvailable(nations.get(row),
602                         (NationState)value);
603                     update();
604                     break;
605                 case COLOR_COLUMN:
606                     preGameController.setColor(nation, (Color)value);
607                     break;
608                 case PLAYER_COLUMN:
609                     if (nationOptions.getNationState(nation)
610                         == NationState.AVAILABLE) {
611                         preGameController.setNation(nation);
612                         preGameController.setNationType(nation.getType());
613                         update();
614                     }
615                     break;
616                 default:
617                     break;
618                 }
619                 fireTableCellUpdated(row, column);
620             }
621         }
622     }
