32 /**
33  * The <code>ProductionCache</code> is contains all relevant
34  * information about the production and consumption of the
35  * colony. This includes the production of all colony tiles and
36  * buildings, as well as the consumption of all units, buildings and
37  * build queues.
38  *
39  */
40 public class ProductionCache {
41
42     /**
43      * The colony whose production is being cached. The goods stored
44      * in the colony may need to be considered in order to prevent
45      * excess production.
46      */
47     private final Colony colony;
48
49     /** A map of net production by goods type. */
50     private final TypeCountMap<GoodsType> netProduction
51         = new TypeCountMap<>();
52
53     /** A map of production info for various producers and consumers. */
54     private final Map<Object, ProductionInfo> productionAndConsumption
55         = new HashMap<>();
56
57     /** A set of the goods used by the colony. */
58     private final Set<GoodsType> goodsUsed = new HashSet<>();
59
60     /**

77     /**
78      * Updates all data structures.  The method has no side effects.
79      *
80      * For now, there is a hard assumption that ColonyTiles do not
81      * consume but Buildings do.  One day we may want to generalize
82      * this, which will require processing the goods types in an order
83      * sorted by the requirement dependencies.  But not yet.  This
84      * assumption is made explicit by getting the ProductionInfo from
85      * ColonyTiles with the simple getBasicProductionInfo() routine,
86      * but from Buildings with getAdjustedProductionInfo() which takes
87      * account of the input and output goods levels.
88      *
89      * FIXME: Ideally these should be unified into a
90      * WorkLocation.getProductionInfo with the Building-form
91      * arguments.
92      */
93     private synchronized void update() {
94         if (upToDate) return; // nothing to do
95         final Specification spec = colony.getSpecification();
96         final GoodsType bells = spec.getGoodsType("model.goods.bells");
97
98         productionAndConsumption.clear();
99         netProduction.clear();
100         goodsUsed.clear();
101         ProductionMap production = new ProductionMap();
102
103         for (ColonyTile colonyTile : colony.getColonyTiles()) {
104             ProductionInfo info = colonyTile.getBasicProductionInfo();
105             production.add(info.getProduction());
106             productionAndConsumption.put(colonyTile, info);
107             for (AbstractGoods goods : info.getProduction()) {
108                 goodsUsed.add(goods.getType());
109                 netProduction.incrementCount(goods.getType().getStoredAs(),
110                                              goods.getAmount());
111             }
112         }
113
114         // Add bell production to compensate for the units-that-use-no-bells
115         // as this is not handled by the unit conumption.
116         int unitsThatUseNoBells
117             = spec.getInteger(GameOptions.UNITS_THAT_USE_NO_BELLS);
118         int amount = Math.min(unitsThatUseNoBells, colony.getUnitCount());
119         ProductionInfo bellsInfo = new ProductionInfo();
120         bellsInfo.addProduction(new AbstractGoods(bells, amount));
121         productionAndConsumption.put(this, bellsInfo);
122         netProduction.incrementCount(bells, amount);
123
124         List<AbstractGoods> goods = new ArrayList<>();
125         for (Consumer consumer : colony.getConsumers()) {
126             Set<Modifier> modifiers = consumer
127                 .getModifiers(Modifier.CONSUME_ONLY_SURPLUS_PRODUCTION);
128             goods.clear();
129             for (AbstractGoods g : consumer.getConsumedGoods()) {
130                 goodsUsed.add(g.getType());
131                 AbstractGoods surplus
132                     = new AbstractGoods(production.get(g.getType()));
133                 if (modifiers.isEmpty()) {
134                     surplus.setAmount(surplus.getAmount()
135                         + getGoodsCount(g.getType()));
136                 } else {
137                     surplus.setAmount((int)FeatureContainer
138                         .applyModifiers(surplus.getAmount(), null, modifiers));
139                 }
140                 goods.add(surplus);
141             }
142             ProductionInfo info = null;
143             if (consumer instanceof Building) {
144                 Building building = (Building)consumer;
145                 List<AbstractGoods> outputs = new ArrayList<>();
146                 for (AbstractGoods output : building.getOutputs()) {
147                     GoodsType outputType = output.getType();
148                     goodsUsed.add(outputType);
149                     AbstractGoods newOutput
150                         = new AbstractGoods(production.get(outputType));
151                     newOutput.setAmount(newOutput.getAmount()
152                         + getGoodsCount(outputType));
153                     outputs.add(newOutput);
154                 }
155                 info = building.getAdjustedProductionInfo(goods, outputs);
156             } else if (consumer instanceof Unit) {
157                 info = ((Unit)consumer).getProductionInfo(goods);
158             } else if (consumer instanceof BuildQueue) {
159                 info = ((BuildQueue<?>)consumer).getProductionInfo(goods);
160             }
161             if (info != null) {
162                 production.add(info.getProduction());
163                 production.remove(info.getConsumption());
164                 for (AbstractGoods g : info.getProduction()) {
165                     netProduction.incrementCount(g.getType().getStoredAs(),
166                                                  g.getAmount());
167                 }
168                 for (AbstractGoods g : info.getConsumption()) {
169                     netProduction.incrementCount(g.getType().getStoredAs(),
170                                                  -g.getAmount());
171                 }
172                 productionAndConsumption.put(consumer, info);
173             }
174         }
175         upToDate = true;
176     }

190     /**
191      * Invalidates the production cache. This method needs to be
192      * called whenever global production modifiers change. This might
193      * be the case when a new {@link FoundingFather} is added, or when

195      *

198         upToDate = false;
199     }
200
201     /**
202      * Invalidates the production cache if it produces or consumes the
203      * given GoodsType. This method needs to be called whenever goods
204      * are added to or removed from the colony.
205      *
206      * @param goodsType a <code>GoodsType</code> value
207      */
208     public synchronized void invalidate(GoodsType goodsType) {
209         if (goodsUsed.contains(goodsType)) {
210             upToDate = false;
211         }
212     }

214     /**
215      * Does this production cache contain production of a goods type?
216      *
217      * @param goodsType The <code>GoodsType</code> to check.
218      * @return True if there is a production entry for the given type.
219      */
220     public boolean isProducing(GoodsType goodsType) {
221         update();
222         return any(productionAndConsumption.values(),
223             pi -> AbstractGoods.containsType(goodsType, pi.getProduction()));
224     }
225
226     /**

228      *

230      * @return True if there is a consumption entry for the given type.
231      */
232     public boolean isConsuming(GoodsType goodsType) {
233         update();
234         return any(productionAndConsumption.values(),
235             pi -> AbstractGoods.containsType(goodsType, pi.getConsumption()));
236     }

262     /**
