143     /**
144      * Reduce the available quantity by the bonus output of
145      * <code>GoodsType</code>.
146      *
147      * @param goodsType The <code>GoodsType</code> to check.
148      * @param unitType The producing <code>UnitType</code>.
149      * @param potential The base potential of the tile.
150      * @return The new quantity of resource.
151      */
152     public int useQuantity(GoodsType goodsType, UnitType unitType,
153                            int potential) {
154         // Return UNLIMITED here if not limited resource?
155         return useQuantity(applyBonus(goodsType, unitType, potential)
156             - potential);
157     }
158
159     /**
160      * Reduces the available quantity.
161      *
162      * @param usedQuantity The quantity that was used up.
163      * @return The final value of quantity.
164      */
165     public int useQuantity(int usedQuantity) {
166         if (quantity == UNLIMITED) {
167             ; // No change
168         } else if (quantity >= usedQuantity) {
169             quantity -= usedQuantity;
170         } else {
171             // Shouldn't generally happen.  Do something more drastic here?
172             logger.severe("Insufficient quantity in " + this);
173             quantity = 0;
174         }
175         return quantity;
176     }
