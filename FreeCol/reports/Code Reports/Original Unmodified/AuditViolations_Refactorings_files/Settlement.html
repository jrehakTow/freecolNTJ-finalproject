584      * {@inheritDoc}
585      */
586     @Override
587     public boolean equipForRole(Unit unit, Role role, int roleCount) {
588         if (!unit.roleIsAvailable(role)) return false;
589
590         // Get the change in goods
591         List<AbstractGoods> required = unit.getGoodsDifference(role, roleCount);
592
593         // Check if the required goods are available
594         if (priceGoods(required) < 0) return false;
595
596         // Make the change
597         for (AbstractGoods ag : required) {
598             addGoods(ag.getType(), -ag.getAmount());
599         }
600
601         unit.changeRole(role, roleCount);
602         return true;
603     }
