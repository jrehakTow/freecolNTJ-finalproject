493     /**
494      * Creates mountain ranges on the given map.  The number and size
495      * of mountain ranges depends on the map size.
496      *
497      * @param map The map to use.
498      * @param lb A <code>LogBuilder</code> to log to.
499      * @return A list of created <code>ServerRegion</code>s.
500      */
501     private List<ServerRegion> createMountains(Map map, LogBuilder lb) {
502         List<ServerRegion> result = new ArrayList<>();
503         float randomHillsRatio = 0.5f;
504         // 50% of user settings will be allocated for random hills
505         // here and there the rest will be allocated for large
506         // mountain ranges
507         int maximumLength
508             = Math.max(mapOptions.getInteger(MapGeneratorOptions.MAP_WIDTH),
509                 mapOptions.getInteger(MapGeneratorOptions.MAP_HEIGHT)) / 10;
510         int number = (int)((1.0f - randomHillsRatio)
511             * (getApproximateLandCount()
512                 / mapOptions.getInteger(MapGeneratorOptions.MOUNTAIN_NUMBER)));
513         lb.add("Number of mountain tiles is ", number, "\n",
514             "Maximum length of mountain ranges is ", maximumLength, "\n");
515
516         // lookup the resources from specification
517         final TileType hills = spec.getTileType("model.tile.hills");
518         final TileType mountains = spec.getTileType("model.tile.mountains");
519         if (hills == null || mountains == null) {
520             throw new RuntimeException("Both Hills and Mountains TileTypes must be defined");
521         }
522
523         // Generate the mountain ranges
524         int counter = 0;
525         for (int tries = 0; tries < 100; tries++) {
526             Tile startTile = getGoodMountainTile(map);
527             if (startTile == null) break;
528
529             ServerRegion mountainRegion
530                 = new ServerRegion(game, RegionType.MOUNTAIN);
531             startTile.setType(mountains);
532             mountainRegion.addTile(startTile);
533             Direction direction = Direction.getRandomDirection("getLand",
534                 logger, random);
535             int length = maximumLength
536                 - randomInt(logger, "MLen", random, maximumLength/2);
537             for (int index = 0; index < length; index++) {
538                 Tile nextTile = startTile.getNeighbourOrNull(direction);
539                 if (nextTile == null || !nextTile.isLand()) continue;
540                 nextTile.setType(mountains);
541                 mountainRegion.addTile(nextTile);
542                 counter++;
543                 for (Tile neighbour : nextTile.getSurroundingTiles(1)) {
544                     if (!neighbour.isLand()
545                         || neighbour.getType() == mountains) continue;
546                     int r = randomInt(logger, "MSiz", random, 8);
547                     if (r == 0) {
548                         neighbour.setType(mountains);
549                         mountainRegion.addTile(neighbour);
550                         counter++;
551                     } else if (r > 2) {
552                         neighbour.setType(hills);
553                         mountainRegion.addTile(neighbour);
554                     }
555                 }
556             }
557             int scoreValue = 2 * mountainRegion.getSize();
558             mountainRegion.setScoreValue(scoreValue);
559             result.add(mountainRegion);
560             lb.add("Created mountain region (direction ", direction,
561                 ", length ", length,
562                 ", size ", mountainRegion.getSize(),
563                 ", score value ", scoreValue, ").\n");
564             if (counter >= number) break;
565         }
566         lb.add("Added ", counter, " mountain range tiles.\n");
567
568         // and sprinkle a few random hills/mountains here and there
569         number = (int) (getApproximateLandCount() * randomHillsRatio)
570             / mapOptions.getInteger(MapGeneratorOptions.MOUNTAIN_NUMBER);
571         counter = 0;
572         for (int tries = 0; tries < 1000; tries++) {
573             Tile t = getGoodMountainTile(map);
574             if (t == null) break;
575
576             // 25% mountains, 75% hills
577             boolean m = randomInt(logger, "MorH", random, 4) == 0;
578             t.setType((m) ? mountains : hills);
579             if (++counter >= number) break;
580         }
581         lb.add("Added ", counter, " random hilly tiles.\n");
582         return result;
583     }

585     /**
586      * Creates rivers on the given map. The number of rivers depends
587      * on the map size.
588      *
589      * @param map The <code>Map</code> to create rivers on.
590      * @param lb A <code>LogBuilder</code> to log to.
591      * @return A list of created <code>ServerRegion</code>s.
592      */
593     private List<ServerRegion> createRivers(Map map, LogBuilder lb) {
594         List<ServerRegion> result = new ArrayList<>();
595         final TileImprovementType riverType
596             = spec.getTileImprovementType("model.improvement.river");
597         final int number = getApproximateLandCount()
598             / mapOptions.getInteger(MapGeneratorOptions.RIVER_NUMBER);
599         int counter = 0;
600         HashMap<Tile, River> riverMap = new HashMap<>();
601         List<River> rivers = new ArrayList<>();
602
603         outer: for (int i = 0; i < number; i++) {
604             for (int tries = 0; tries < 100; tries++) {
605                 Tile tile = map.getRandomLandTile(random);
606                 if (tile == null) break outer;
607
608                 if (!riverType.isTileTypeAllowed(tile.getType())) continue;
609
610                 // check the river source/spring is not too close to the ocean
611                 if (!all(tile.getSurroundingTiles(1, 2),
612                         Tile::isLand)) continue;
613
614                 if (riverMap.get(tile) == null) {
615                     // no river here yet
616                     ServerRegion riverRegion = new ServerRegion(game, RegionType.RIVER);
617                     River river = new River(map, riverMap, riverRegion, random);
618                     if (river.flowFromSource(tile)) {
619                         lb.add("Created new river with length ",
620                             river.getLength(), "\n");
621                         result.add(riverRegion);
622                         rivers.add(river);
623                         if (++counter >= number) break;
624                     } else {
625                         lb.add("Failed to generate river.\n");
626                     }
627                     break;
628                 }
629             }
630         }
631         lb.add("Created ", counter, " rivers of maximum ", number, "\n");
632
633         for (River river : rivers) {
634             ServerRegion region = river.getRegion();
635             int score = 2 * sum(river.getSections(), RiverSection::getSize);
636             region.setScoreValue(score);
637             lb.add("Created river region (length ", river.getLength(),
638                 ", score value ", score, ").\n");
639         }
640         return result;
641     }

688             .getTileType("model.tile.lake");

691         int lakeCount = 0;

722     /**
723      * Adds a terrain bonus with a probability determined by the
724      * <code>MapGeneratorOptions</code>.
725      *
726      * @param t The <code>Tile</code> to add bonuses to.
727      * @param generateBonus Generate the bonus or not.
728      */
729     private void perhapsAddBonus(Tile t, boolean generateBonus) {
730         final Specification spec = t.getSpecification();
731         TileImprovementType fishBonusLandType
732             = spec.getTileImprovementType("model.improvement.fishBonusLand");
733         TileImprovementType fishBonusRiverType
734             = spec.getTileImprovementType("model.improvement.fishBonusRiver");
735         final int bonusNumber
736             = mapOptions.getInteger(MapGeneratorOptions.BONUS_NUMBER);
737         if (t.isLand()) {
738             if (generateBonus
739                 && randomInt(logger, "Land Resource", random, 100) < bonusNumber) {
740                 // Create random Bonus Resource
741                 t.addResource(createResource(t));
742             }
743         } else {
744             int adjacentLand = 0;
745             boolean adjacentRiver = false;
746             for (Direction direction : Direction.values()) {
747                 Tile otherTile = t.getNeighbourOrNull(direction);
748                 if (otherTile != null && otherTile.isLand()) {
749                     adjacentLand++;
750                     if (otherTile.hasRiver()) {
751                         adjacentRiver = true;
752                     }
753                 }
754             }
755
756             // In Col1, ocean tiles with less than 3 land neighbours
757             // produce 2 fish, all others produce 4 fish
758             if (adjacentLand > 2) {
759                 t.add(new TileImprovement(game, t, fishBonusLandType));
760             }
761
762             // In Col1, the ocean tile in front of a river mouth would
763             // get an additional +1 bonus
764             // FIXME: This probably has some false positives, means
765             // river tiles that are NOT a river mouth next to this tile!
766             if (!t.hasRiver() && adjacentRiver) {
767                 t.add(new TileImprovement(game, t, fishBonusRiverType));
768             }
769
770             if (t.getType().isHighSeasConnected()) {
771                 if (generateBonus && adjacentLand > 1
772                     && randomInt(logger, "Sea resource", random,
773                                  10 - adjacentLand) == 0) {
774                     t.addResource(createResource(t));
775                 }
776             } else {
777                 if (randomInt(logger, "Water resource", random, 100) < bonusNumber) {
778                     // Create random Bonus Resource
779                     t.addResource(createResource(t));
780                 }
781             }
782         }
783     }

808     /**
809      * Sets the style of the tiles.
810      * Only relevant to water tiles for now.
811      * Public because it is used in the river generator.
812      *
813      * @param tile The <code>Tile</code> to set the style of.
814      */
815     public static void encodeStyle(Tile tile) {
816         EnumMap<Direction, Boolean> connections
817             = new EnumMap<>(Direction.class);
818
819         // corners
820         for (Direction d : Direction.corners) {
821             Tile t = tile.getNeighbourOrNull(d);
822             connections.put(d, t != null && t.isLand());
823         }
824         // edges
825         for (Direction d : Direction.longSides) {
826             Tile t = tile.getNeighbourOrNull(d);
827             if (t != null && t.isLand()) {
828                 connections.put(d, Boolean.TRUE);
829                 // ignore adjacent corners
830                 connections.put(d.getNextDirection(), Boolean.FALSE);
831                 connections.put(d.getPreviousDirection(), Boolean.FALSE);
832             } else {
833                 connections.put(d, Boolean.FALSE);
834             }
835         }
836         int result = 0;
837         int index = 0;
838         for (Direction d : Direction.corners) {
839             if (connections.get(d)) result += (int)Math.pow(2, index);
840             index++;
841         }
842         for (Direction d : Direction.longSides) {
843             if (connections.get(d)) result += (int)Math.pow(2, index);
844             index++;
845         }
846         tile.setStyle(result);
847     }

860         final int height = landMap.getHeight();
861         final boolean importTerrain = (importGame != null)
862             && mapOptions.getBoolean(MapGeneratorOptions.IMPORT_TERRAIN);
863         final boolean importBonuses = (importGame != null)
864             && mapOptions.getBoolean(MapGeneratorOptions.IMPORT_BONUSES);
865
866         boolean mapHasLand = false;
867         Map map = new Map(game, width, height);
868         int minimumLatitude = mapOptions
869             .getInteger(MapGeneratorOptions.MINIMUM_LATITUDE);
870         int maximumLatitude = mapOptions
871             .getInteger(MapGeneratorOptions.MAXIMUM_LATITUDE);
872         // make sure the values are in range
873         minimumLatitude = limitToRange(minimumLatitude, -90, 90);
874         maximumLatitude = limitToRange(maximumLatitude, -90, 90);
875         map.setMinimumLatitude(Math.min(minimumLatitude, maximumLatitude));
876         map.setMaximumLatitude(Math.max(minimumLatitude, maximumLatitude));
877
878         java.util.Map<String, ServerRegion> regionMap = new HashMap<>();
879         if (importTerrain) { // Import the regions
880             lb.add("Imported regions: ");
881             for (Region r : importGame.getMap().getRegions()) {
882                 ServerRegion region = new ServerRegion(game, r);
883                 map.addRegion(region);
884                 regionMap.put(r.getId(), region);
885                 lb.add(" ", region.toString());
886             }
887             for (Region r : importGame.getMap().getRegions()) {
888                 ServerRegion region = regionMap.get(r.getId());
889                 Region x = r.getParent();
890                 if (x != null) x = regionMap.get(x.getId());
891                 region.setParent(x);
892                 for (Region c : r.getChildren()) {
893                     x = regionMap.get(c.getId());
894                     if (x != null) region.addChild(x);
895                 }
896             }
897             lb.add("\n");
898         }
899
900         List<Tile> fixRegions = new ArrayList<>();
901         for (int y = 0; y < height; y++) {
902             int latitude = map.getLatitude(y);
903             for (int x = 0; x < width; x++) {
904                 if (landMap.isLand(x, y)) mapHasLand = true;
905                 Tile t, importTile = null;
906                 if (importTerrain
907                     && importGame.getMap().isValid(x, y)
908                     && (importTile = importGame.getMap().getTile(x, y)) != null
909                     && importTile.isLand() == landMap.isLand(x, y)) {
910                     String id = importTile.getType().getId();
911                     t = new Tile(game, spec.getTileType(id), x, y);
912                     if (importTile.getMoveToEurope() != null) {
913                         t.setMoveToEurope(importTile.getMoveToEurope());
914                     }
915                     if (importTile.getTileItemContainer() != null) {
916                         TileItemContainer container
917                             = new TileItemContainer(game, t);
918                         // TileItemContainer copies every natural item
919                         // including Resource if importBonuses is true.
920                         // Rumors and roads are not copied.
921                         container.copyFrom(importTile.getTileItemContainer(),
922                             (importBonuses) ? Map.Layer.RESOURCES
923                             : Map.Layer.RIVERS);
924                         t.setTileItemContainer(container);
925                     }
926                     Region r = importTile.getRegion();
927                     if (r == null) {
928                         fixRegions.add(t);
929                     } else {
930                         ServerRegion ours = regionMap.get(r.getId());
931                         if (ours == null) {
932                             lb.add("Could not set tile region ", r.getId(),
933                                 " for tile: ", t, "\n");
934                             fixRegions.add(t);
935                         } else {
936                             ours.addTile(t);
937                         }
938                     }
939                 } else {
940                     t = createTile(x, y, landMap, latitude);
941                 }
942                 map.setTile(t, x, y);
943             }
944         }
945         game.setMap(map);
946
947         // Build the regions.
948         List<ServerRegion> fixed = ServerRegion.requireFixedRegions(map, lb);
949         List<ServerRegion> newRegions = new ArrayList<>();
950         if (importTerrain) {
951             if (!fixRegions.isEmpty()) { // Fix the tiles missing regions.
952                 newRegions.addAll(createLakeRegions(map, lb));
953                 newRegions.addAll(createLandRegions(map, lb));
954             }
955         } else {
956             map.resetHighSeas(
957                 mapOptions.getInteger(MapGeneratorOptions.DISTANCE_TO_HIGH_SEA),
958                 mapOptions.getInteger(MapGeneratorOptions.MAXIMUM_DISTANCE_TO_EDGE));
959             if (mapHasLand) {
960                 newRegions.addAll(createMountains(map, lb));
961                 newRegions.addAll(createRivers(map, lb));
962                 newRegions.addAll(createLakeRegions(map, lb));
963                 newRegions.addAll(createLandRegions(map, lb));
964             }
965         }
966         lb.shrink("\n");
967
968         // Connect all new regions to their geographic parent and add to
969         // the map.
970         List<ServerRegion> geographic = new ArrayList<>();
971         for (ServerRegion sr : fixed) {
972             if (sr.isGeographic()) geographic.add(sr);
973         }
974         for (ServerRegion sr : newRegions) {
975             for (ServerRegion gr : geographic) {
976                 if (gr.containsCenter(sr)) {
977                     sr.setParent(gr);
978                     gr.addChild(sr);
979                     gr.setSize(gr.getSize() + sr.getSize());
980                     break;
981                 }
982             }
983             map.addRegion(sr);
984         }
985
986         // Probably only needed on import of old maps.
987         map.fixupRegions();
988
989         // Add the bonuses only after the map is completed.
990         // Otherwise we risk creating resources on fields where they
991         // do not belong (like sugar in large rivers or tobacco on hills).
992         for (Tile tile : map.getAllTiles()) {
993             perhapsAddBonus(tile, !importBonuses);
994             if (!tile.isLand()) {
995                 encodeStyle(tile);
996             }
997         }
998
999         // Final cleanups
1,000         map.resetContiguity();
1,001         map.resetHighSeasCount();
1,002         return map;
1,003     }
