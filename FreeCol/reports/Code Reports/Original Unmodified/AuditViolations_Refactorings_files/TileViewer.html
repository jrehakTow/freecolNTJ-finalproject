439                     int edgeStyle = tile.getStyle() >> 4;
440                     if (edgeStyle > 0) {
441                         g.drawImage(lib.getBeachEdgeImage(edgeStyle, x, y),
442                                     0, 0, null);
443                     }
444                     int cornerStyle = tile.getStyle() & 15;
445                     if (cornerStyle > 0) {
446                         g.drawImage(lib.getBeachCornerImage(cornerStyle, x, y),
447                                     0, 0, null);
448                     }
449                 }
450
451                 List<SortableImage> imageBorders = new ArrayList<>(8);
452                 SortableImage si;
453                 for (Direction direction : Direction.values()) {
454                     Tile borderingTile = tile.getNeighbourOrNull(direction);
455                     if (borderingTile != null && borderingTile.isExplored()) {
456                         TileType borderingTileType = borderingTile.getType();
457                         if (borderingTileType != tileType) {
458                             if (!tile.isLand() && borderingTile.isLand()) {
459                                 // If there is a Coast image (eg. beach) defined, use it, otherwise skip
460                                 // Draw the grass from the neighboring tile, spilling over on the side of this tile
461                                 si = new SortableImage(
462                                     lib.getBorderImage(borderingTileType, direction, x, y),
463                                     borderingTileType.getIndex());
464                                 imageBorders.add(si);
465                                 TileImprovement river = borderingTile.getRiver();
466                                 if (river != null && river.isConnectedTo(direction.getReverseDirection())) {
467                                     si = new SortableImage(
468                                         lib.getRiverMouthImage(direction,
469                                             borderingTile.getRiver().getMagnitude(), x, y),
470                                         -1);
471                                     imageBorders.add(si);
472                                 }
473                             } else if (!tile.isLand() || borderingTile.isLand()) {
474                                 if (borderingTileType.getIndex() < tileType.getIndex() &&
475                                         !lib.getTerrainImage(tileType, 0, 0).equals(lib.getTerrainImage(borderingTileType, 0, 0))) {
476                                     // Draw land terrain with bordering land type, or ocean/high seas limit,
477                                     // if the tiles do not share same graphics (ocean & great river)
478                                     si = new SortableImage(
479                                             lib.getBorderImage(borderingTileType, direction, x, y),
480                                             borderingTileType.getIndex());
481                                     imageBorders.add(si);
482                                 }
483                             }
484                         }
485                     }
486                 }
487                 Collections.sort(imageBorders);
488                 for (SortableImage sorted : imageBorders) {
489                     g.drawImage(sorted.image, 0, 0, null);
490                 }
491             }
492         }

529     /**
530      * Displays the Tile text for a Tile.
531      * Shows tile names, coordinates and colony values.
532      *
533      * @param g The Graphics2D object on which the text gets drawn.
534      * @param tile The Tile to draw the text on.
535      */
536     void displayOptionalTileText(Graphics2D g, Tile tile) {
537         String text = null;
538         int op = getClientOptions().getInteger(ClientOptions.DISPLAY_TILE_TEXT);
539         switch (op) {
540         case ClientOptions.DISPLAY_TILE_TEXT_NAMES:
541             text = Messages.getName(tile);
542             break;
543         case ClientOptions.DISPLAY_TILE_TEXT_OWNERS:
544             if (tile.getOwner() != null) {
545                 text = Messages.message(tile.getOwner().getNationLabel());
546             }
547             break;
548         case ClientOptions.DISPLAY_TILE_TEXT_REGIONS:
549             if (tile.getRegion() != null) {
550                 if (FreeColDebugger.isInDebugMode(FreeColDebugger.DebugMode.MENUS)
551                     && tile.getRegion().getName() == null) {
552                     text = tile.getRegion().getSuffix();
553                 } else {
554                     text = Messages.message(tile.getRegion().getLabel());
555                 }
556             }
557             break;
558         case ClientOptions.DISPLAY_TILE_TEXT_EMPTY:
559             break;
560         default:
561             logger.warning("displayTileText option " + op + " out of range");
562             break;
563         }
564
565         g.setColor(Color.BLACK);
566         g.setFont(FontLibrary.createFont(FontLibrary.FontType.NORMAL,
567             FontLibrary.FontSize.TINY, lib.getScaleFactor()));
568         if (text != null) {
569             int b = getBreakingPoint(text);
570             if (b == -1) {
571                 g.drawString(text,
572                     (tileWidth - g.getFontMetrics().stringWidth(text)) / 2,
573                     (tileHeight - g.getFontMetrics().getAscent()) / 2);
574             } else {
575                 g.drawString(text.substring(0, b),
576                     (tileWidth - g.getFontMetrics().stringWidth(text.substring(0, b)))/2,
577                     halfHeight - (g.getFontMetrics().getAscent()*2)/3);
578                 g.drawString(text.substring(b+1),
579                     (tileWidth - g.getFontMetrics().stringWidth(text.substring(b+1)))/2,
580                     halfHeight + (g.getFontMetrics().getAscent()*2)/3);
581             }
582         }
583
584         if (FreeColDebugger.debugDisplayCoordinates()) {
585             String posString = tile.getX() + ", " + tile.getY();
586             if (tile.getHighSeasCount() >= 0) {
587                 posString += "/" + Integer.toString(tile.getHighSeasCount());
588             }
589             g.drawString(posString,
590                 (tileWidth - g.getFontMetrics().stringWidth(posString)) / 2,
591                 (tileHeight - g.getFontMetrics().getAscent()) / 2);
592         }
593         String value = DebugUtils.getColonyValue(tile);
594         if (value != null) {
595             g.drawString(value,
596                 (tileWidth - g.getFontMetrics().stringWidth(value)) / 2,
597                 (tileHeight - g.getFontMetrics().getAscent()) / 2);
598         }
599     }

614
615         if (settlement != null) {
616             if (settlement instanceof Colony) {
617                 Colony colony = (Colony)settlement;
618
619                 // Draw image of colony in center of the tile.

683     /**
684      * Displays the given tile's items onto the given Graphics2D object.
685      * Additions and improvements to Tile will be drawn.
686      * Only works for explored tiles.
687      *
688      * @param g The Graphics2D object on which to draw the Tile.
689      * @param tile The Tile to draw.
690      * @param overlayImage The BufferedImage for the tile overlay.
691      */
692     void displayTileItems(Graphics2D g, Tile tile, BufferedImage overlayImage) {
693         // ATTENTION: we assume that only overlays and forests
694         // might be taller than a tile.
695
696         // layer additions and improvements according to zIndex
697         List<TileItem> tileItems = tile.getCompleteItems();
698         int startIndex = 0;
699         for (int index = startIndex; index < tileItems.size(); index++) {
700             if (tileItems.get(index).getZIndex() < Tile.OVERLAY_ZINDEX) {
701                 displayTileItem(g, tile, tileItems.get(index));
702                 startIndex = index + 1;
703             } else {
704                 startIndex = index;
705                 break;
706             }
707         }
708         // Tile Overlays (eg. hills and mountains)
709         if (overlayImage != null) {
710             g.drawImage(overlayImage,
711                 0, (tileHeight - overlayImage.getHeight()), null);
712         }
713         for (int index = startIndex; index < tileItems.size(); index++) {
714             if (tileItems.get(index).getZIndex() < Tile.FOREST_ZINDEX) {
715                 displayTileItem(g, tile, tileItems.get(index));
716                 startIndex = index + 1;
717             } else {
718                 startIndex = index;
719                 break;
720             }
721         }
722         // Forest
723         if (tile.isForested()) {
724             BufferedImage forestImage = lib.getForestImage(
725                 tile.getType(), tile.getRiverStyle());
726             g.drawImage(forestImage,
727                 0, (tileHeight - forestImage.getHeight()), null);
728         }
729
730         // draw all remaining items
731         for (TileItem ti : tileItems.subList(startIndex, tileItems.size())) {
732             displayTileItem(g, tile, ti);
733         }
734     }
