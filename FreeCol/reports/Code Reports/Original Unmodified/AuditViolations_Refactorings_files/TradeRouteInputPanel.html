635     /**
636      * Add any stops selected in the destination selector.
637      */
638     private void addSelectedStops() {
639         int startIndex = -1;
640         int endIndex = -1;
641         int sel = this.destinationSelector.getSelectedIndex();
642         if (sel == 0) { // All colonies + Europe
643             startIndex = 1;
644             endIndex = this.destinationSelector.getItemCount();
645         } else { // just one place
646             startIndex = sel;
647             endIndex = startIndex+1;
648         }
649         List<GoodsType> cargo = transform(cargoPanel.getComponents(),
650             c -> c instanceof CargoLabel, c -> ((CargoLabel)c).getType(),
651             Collectors.toList());
652         int maxIndex = this.stopList.getMaxSelectionIndex();
653         for (int i = startIndex; i < endIndex; i++) {
654             String id = this.destinationSelector.getItemAt(i);
655             FreeColGameObject fcgo = getGame().getFreeColGameObject(id);
656             if (fcgo instanceof Location) {
657                 TradeRouteStop stop
658                     = new TradeRouteStop(getGame(), (Location)fcgo);
659                 stop.setCargo(cargo);
660                 if (maxIndex < 0) {
661                     this.stopListModel.addElement(stop);
662                 } else {
663                     maxIndex++;
664                     this.stopListModel.add(maxIndex, stop);
665                 }
666             }
667         }
668     }

670     /**
671      * Delete any stops currently selected in the stop list.
672      */
673     private void deleteCurrentlySelectedStops() {
674         int count = 0;
675         int lastIndex = 0;
676         for (int index : this.stopList.getSelectedIndices()) {
677             this.stopListModel.remove(index - count);
678             count++;
679             lastIndex = index;
680         }
681
682         // If the remaining list is non-empty, make sure that the
683         // element beneath the last of the previously selected is
684         // selected (ie, delete one of many, the one -under- the
685         // deleted is selected) the user can then click in the list
686         // once, and continue deleting without having to click in the
687         // list again.
688         if (this.stopListModel.getSize() > 0) {
689             this.stopList.setSelectedIndex(lastIndex - count + 1);
690         }
691     }

716     /**
717      * Check that the trade route is valid.
718      *
719      * @return True if the trade route is valid.
720      */
721     private boolean verifyNewTradeRoute() {
722         final Player player = getFreeColClient().getMyPlayer();
723
724         // Update the trade route with the current settings
725         this.newRoute.setName(tradeRouteName.getText());
726         this.newRoute.clearStops();
727         for (int index = 0; index < this.stopListModel.getSize(); index++) {
728             this.newRoute.addStop(this.stopListModel.get(index));
729         }
730         this.newRoute.setSilent(this.messagesBox.isSelected());
731
732         StringTemplate err = this.newRoute.verify();
733         if (err != null) {
734             getGUI().showInformationMessage(err);
735             this.newRoute.setName(null); // Mark as unacceptable
736             return false;
737         }
738         return true;
739     }

744     /**
745      * {@inheritDoc}
746      */
747     @Override
748     public void actionPerformed(ActionEvent ae) {
749         final String command = ae.getActionCommand();
750         if (command == null) return;
751         switch (command) {
752         case OK:
753             if (!verifyNewTradeRoute()) return;
754             // Return to TradeRoutePanel, which will add the route
755             // if needed, and it is valid.
756             super.actionPerformed(ae);
757             break;
758         case CANCEL:
759             cancelTradeRoute();
760             break;
761         default:
762             super.actionPerformed(ae);
763             break;
764         }
765     }
