477     /**
478      * For a given transportable, work out where the carrier has to go to
479      * advance the cargo (target), and what to do there (mode), allowing
480      * a new <code>Cargo</code> to be defined.
481      *
482      * AIUnit cargo is harder than AIGoods, because AIUnits might have their
483      * own inland paths, and thus we need to consider drop nodes.
484      *
485      * @param t The <code>TransportableAIObject</code> to consider.
486      * @param lb A <code>LogBuilder</code> to log to.
487      * @return A new <code>Cargo</code> defining the action to take
488      *     with the <code>TransportableAIObject</code>, or null if impossible.
489      */
490     public Cargo makeCargo(TransportableAIObject t, LogBuilder lb) {
491         final Unit carrier = getUnit();
492         String reason;
493         Cargo cargo = null;
494         if (t.getTransportDestination() == null) {
495             if (!isCarrying(t)) {
496                 reason = "null transport destination";
497             } else {
498                 // Can happen with carriers with units transferred in
499                 // Spanish succession.
500                 PathNode path = carrier.getTrivialPath();
501                 if (path == null) {
502                     reason = "null transport destination";
503                 } else {
504                     try {
505                         reason = null;
506                         cargo = Cargo.newCargo(t, carrier,
507                             path.getLastNode().getLocation(), true);
508                     } catch (FreeColException fce) {
509                         reason = fce.getMessage();
510                         cargo = null;
511                     }
512                 }
513             }
514         } else if (!isCarrying(t) && !t.carriableBy(carrier)) {
515             reason = "carrier " + carrier.toShortString() + " can not carry";
516         } else {
517             try {
518                 reason = null;
519                 cargo = Cargo.newCargo(t, carrier);
520             } catch (FreeColException fce) {
521                 reason = fce.getMessage();
522                 cargo = null;
523             }
524         }
525         if (reason == null) {
526             lb.add(", made ", cargo.toShortString());
527             return cargo;
528         } else {
529             lb.add(", failed to make cargo for ", t, " (", reason, ")");
530             return null;
531         }
532     }

594     /**
595      * Incrementally queue a cargo to the cargoes list.
596      *
597      * If the carrier is at the collection point favour immediate
598      * collection.  Otherwise try to place it with other cargoes with
599      * the same target, but do not break the space restrictions.  If
600      * this does not work, it has to go at the end.
601      *
602      * @param cargo The new <code>Cargo</code> to add.
603      * @param requireMatch Fail if an existing destination is not matched.
604      * @param lb A <code>LogBuilder</code> to log to.
605      * @return True if the cargo was queued.
606      */
607     private boolean queueCargo(Cargo cargo, boolean requireMatch,
608                                LogBuilder lb) {
609         final Unit carrier = getUnit();
610         final List<Cargo> ts = tCopy();
611         int candidate = -1;
612
613         if (ts.isEmpty() // Trivial case
614             || (Map.isSameLocation(carrier.getLocation(), // Carrier here?
615                                    cargo.getCarrierTarget())
616                 && cargo.canQueueAt(carrier, 0, ts))) {
617             candidate = 0;
618         }
619
620         if (candidate < 0) { // Match an existing target?
621             for (int i = 0; i < ts.size(); i++) {
622                 Cargo tr = ts.get(i);
623                 if (Map.isSameLocation(tr.getCarrierTarget(),
624                                        cargo.getCarrierTarget())) {
625                     if (!cargo.canQueueAt(carrier, i, ts)) continue;
626                     candidate = i;
627                     break;
628                 }
629             }
630         }
631         
632         if (candidate < 0) { // Queue at end unless match required
633             if (requireMatch) return false;
634             candidate = ts.size();
635         }
636         return addCargo(cargo, candidate, lb);
637     }

641      *
642      * @param cargo The <code>Cargo</code> to dump.
643      * @param lb A <code>LogBuilder</code> to log to.
644      * @return True if the cargo is no longer on board and not on the

653             if (reason != null) {
654                 lb.add(", dump failed(", reason, ")");
655                 return false;
656             } else {
657                 lb.add(", dumping");
658             }
659         }

663     /**
664      * Requeue an existing cargo.  Typically done when the target changes.
665      *
666      * @param cargo The <code>Cargo</code> to requeue.
667      * @param lb A <code>LogBuilder</code> to log to.
668      * @return True if the queuing succeeded.
669      */
670     public boolean requeueCargo(Cargo cargo, LogBuilder lb) {
671         final TransportableAIObject t = cargo.getTransportable();
672         boolean ret = false;
673         assert tFind(t) == cargo;
674         String reason = cargo.update();
675         if (reason != null) {
676             lb.add(" requeue/update fail(", reason, ") ",
677                    cargo.toShortString());
678             dumpCargo(cargo, lb);
679         } else if (!tRemove(cargo)) {
680             lb.add(" requeue/remove fail ", cargo.toShortString());
681         } else if (!queueCargo(cargo, false, lb)) {
682             lb.add(" requeue/queue fail ", cargo.toShortString());
683             dropTransportable(t);
684         } else {
685             lb.add(" requeued(", cargo.getTransportTarget(), ") ",
686                    cargo.toShortString());
687             takeTransportable(t);
688             ret = true;
689         }
690         return ret;
691     }

693     /**
694      * Checks for invalid cargoes, and units and goods on board but
695      * not in the cargoes list.  On exit from this routine, every
696      * cargo on board should be on the cargoes list but the list is
697      * not necessarily going to be in a sensible order.
698      *
699      * @param lb A <code>LogBuilder</code> to log to.
700      */
701     private void checkCargoes(LogBuilder lb) {
702         final Unit carrier = getUnit();
703         if (carrier.isAtSea()) return; // Let it emerge.
704         final AIUnit aiCarrier = getAIUnit();
705
706         List<Unit> unitsPresent = carrier.getUnitList();
707         List<Goods> goodsPresent = carrier.getCompactGoods();
708         List<TransportableAIObject> todo = new ArrayList<>();
709         List<TransportableAIObject> drop = new ArrayList<>();
710
711         String reason;
712         PathNode path;
713         boolean dump;
714         lb.add(" [check");
715         for (Cargo cargo : tCopy()) {
716             dump = false;
717             TransportableAIObject t = cargo.getTransportable();
718             reason = invalidReason(aiCarrier, cargo.getCarrierTarget());
719             if (reason != null || (reason = cargo.check(aiCarrier)) != null) {
720                 // Just remove, it is invalid
721                 removeCargo(cargo);
722                 lb.add(", INVALID(", reason, ") ", cargo.toShortString());
723             } else if (cargo.isDelivered()) {
724                 removeCargo(cargo);
725                 lb.add(", COMPLETED ", cargo.toShortString());
726             } else if (!cargo.hasPath() && !cargo.retry()) {
727                 reason = " no-path";
728                 dump = true;
729             } else if (carrier.hasTile() && (reason = cargo.update()) != null) {
730                 if (reason.startsWith("invalid")) {
731                     removeCargo(cargo);
732                     lb.add(", FAIL(", reason, ") ", cargo.toShortString());
733                 } else if (cargo.retry()) {
734                     lb.add(", retry-", cargo.getTries(), "(", reason, ")");
735                 } else {
736                     dump = true;
737                 }
738             } else if (cargo.isCollectable()) {
739                 lb.add(", collect ", cargo.toShortString());
740             } else if (cargo.isDeliverable()) {
741                 lb.add(", deliver ", cargo.toShortString());
742             } else {
743                 lb.add(", ok ", cargo.toShortString()); // Good
744                 cargo.resetTries();
745             }
746             if (dump) {
747                 if (cargo.isCarried()) {
748                     dumpCargo(cargo, lb); // FIXME: can fail
749                 } else {
750                     removeCargo(cargo);
751                     lb.add(", dropped(", reason, ") ", cargo.toShortString());
752                 }
753             }
754             if (t instanceof AIUnit) {
755                 unitsPresent.remove(((AIUnit)t).getUnit());
756             } else if (t instanceof AIGoods) {
757                 Goods goods = ((AIGoods)t).getGoods();
758                 if (goods != null) {
759                     Iterator<Goods> gi = goodsPresent.iterator();
760                     while (gi.hasNext()) {
761                         Goods g = gi.next();
762                         if (g.getType() == goods.getType()) {
763                             gi.remove();
764                             break;
765                         }
766                     }
767                 }
768             }
769         }
770
771         // Find anything that was not on the cargoes list
772         if (!unitsPresent.isEmpty()) {
773             lb.add(", found unexpected units");
774             for (Unit u : unitsPresent) {
775                 AIUnit aiu = getAIMain().getAIUnit(u);
776                 if (aiu == null) throw new IllegalStateException("Bogus:" + u);
777                 todo.add(aiu);
778             }
779         }
780         if (!goodsPresent.isEmpty()) {
781             lb.add(", found unexpected goods");
782             for (Goods g : goodsPresent) {
783                 AIGoods aig = new AIGoods(getAIMain(), carrier, g.getType(),
784                                           g.getAmount(), null);
785                 todo.add(aig);
786             }
787         }
788
789         // Try to queue the surprise transportables.
790         while (!todo.isEmpty()) {
791             TransportableAIObject t = todo.remove(0);
792             if (!queueTransportable(t, false, lb)) drop.add(t);
793         }
794
795         // Drop transportables on the drop list, or queue them to be
796         // dropped at the next port.
797         if (!drop.isEmpty()) {
798             path = carrier.getTrivialPath();
799             Location end = (path == null) ? null
800                 : path.getLastNode().getLocation();
801             
802             while (!drop.isEmpty()) {
803                 TransportableAIObject t = drop.remove(0);
804                 if (t.leaveTransport()) {
805                     lb.add(" ", t, " left");
806                 } else if (end != null) {
807                     try {
808                         Cargo cargo = Cargo.newCargo(t, carrier, end, false);
809                         boolean result = queueCargo(cargo, false, lb);
810                         lb.add(" to drop at ", Location.upLoc(end),
811                             "=", result);
812                     } catch (FreeColException fce) {
813                         lb.add(" ", t, " drop-fail(", fce.getMessage(), ")");
814                     }
815                 } else {
816                     lb.add(" ", t, " stuck");
817                 }
818             }
819         }
820
821         lb.add("]");
822     }
823
824     /**
825      * Check a <code>Cargo</code> for continued validity and
826      * whether action is needed at the current location.
827      *
828      * @param cargo The <code>Cargo</code> to check.
829      * @param lb A <code>LogBuilder</code> to log to.
830      * @return TCONTINUE if the <code>Cargo</code> should continue,
831      *     TDONE if it has completed,
832      *     TFAIL if it has failed,
833      *     TNEXT if it has progressed to the next stage,
834      *     TRETRY if a blockage has occurred and it should be retried,
835      */
836     private CargoResult tryCargo(Cargo cargo, LogBuilder lb) {
837         final Unit carrier = getUnit();
838         final Location here = carrier.getLocation();
839         final TransportableAIObject t = cargo.getTransportable();
840         final Locatable l = t.getTransportLocatable();
841         if (l == null) {
842             logger.warning("Null-locatable: " + cargo);
843             return CargoResult.TDONE;
844         }
845         if (!Map.isSameLocation(here, cargo.getCarrierTarget())) {
846             lb.add(", ", t, " unready");
847             return CargoResult.TCONTINUE;
848         }
849         Direction d = null;
850         Location tloc = here;
851
852         switch (cargo.getMode()) {
853         case PICKUP:
854             if (!t.canMove()) {
855                 lb.add(", ", t, " out of moves");
856                 return CargoResult.TCONTINUE;
857             }
858             if ((d = cargo.getJoinDirection()) == null) {
859                 logger.warning("Null pickup direction"
860                     + " for " + cargo.toShortString()
861                     + " at " + t.getLocation().toString()
862                     + " to " + carrier);
863                 return CargoResult.TFAIL;
864             }                    
865             tloc = tloc.getTile().getNeighbourOrNull(d.getReverseDirection());
866             // Fall through
867         case LOAD:
868             if (!Map.isSameLocation(tloc, t.getLocation())) {
869                 lb.add(", ", t, " at ", t.getLocation(), " not ", tloc,
870                     " # ", cargo.toShortString());
871                 return CargoResult.TCONTINUE;
872             }
873             switch (carrier.getNoAddReason(l)) {
874             case NONE:
875                 if (!t.joinTransport(carrier, d)) {
876                     lb.add(", ", t, " NO-JOIN");
877                     return CargoResult.TFAIL;
878                 }
879                 break;
880             case ALREADY_PRESENT:
881                 break;
882             case CAPACITY_EXCEEDED:
883                 lb.add(", ", t, " NO-ROOM on ", carrier);
884                 return CargoResult.TFAIL;
885             default:
886                 lb.add(", ", t, " retry-", carrier.getNoAddReason(l));
887                 return CargoResult.TRETRY;
888             }
889             
890             String reason = cargo.update();
891             if (reason != null) {
892                 lb.add(", ", t, " NO-UPDATE(", reason, ")");
893                 return CargoResult.TFAIL;
894             }
895             lb.add(", ", t, " collected");
896             return CargoResult.TNEXT;
897
898         case DROPOFF:
899             if (!t.canMove()) {
900                 lb.add(", ", t, " about to leave");
901                 return CargoResult.TCONTINUE;
902             }
903             if ((d = cargo.getLeaveDirection()) == null) {
904                 Unit.MoveType mt = ((AIUnit)t).getUnit()
905                     .getSimpleMoveType(t.getLocation().getTile(),
906                                        cargo.getTransportTarget().getTile());
907                 switch (mt) {
908                 case ATTACK_UNIT: case MOVE_NO_ATTACK_CIVILIAN:
909                     return CargoResult.TRETRY;
910                 default:
911                     PathNode path = t.getDeliveryPath(carrier,
912                         cargo.getTransportTarget());
