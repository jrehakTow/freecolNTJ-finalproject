62     /**
63      * Connects a client to host:port (or more).
64      *
65      * @param threadName The name for the thread.
66      * @param host The name of the machine running the
67      *     <code>FreeColServer</code>.
68      * @param port The port to use when connecting to the host.
69      * @param messageHandler The <code>MessageHandler</code> to use on this
70      *     connection.
71      * @return True if the connection succeeded.
72      * @exception IOException on connection failure.
73      */
74     public boolean connect(String threadName, String host, int port,
75                            MessageHandler messageHandler) 
76         throws IOException {
77         int tries;
78         if (port < 0) {
79             port = FreeCol.getServerPort();
80             tries = 10;
81         } else {
82             tries = 1;
83         }
84         for (int i = tries; i > 0; i--) {
85             try {
86                 this.clientConnection = new ClientConnection(host, port,
87                     messageHandler, threadName);
88                 if (this.clientConnection != null) break;
89             } catch (IOException e) {
90                 if (i <= 1) throw e;
91             }
92         }
93         return this.clientConnection != null;
94     }
95
96     /**
97      * Disconnect the client connection..
98      *
99      * @exception IOException on failure to send the required messages.
100      */
101     public void disconnect() throws IOException {
102         if (this.clientConnection != null) {
103             this.clientConnection.disconnect();
104             reset();

106     }
