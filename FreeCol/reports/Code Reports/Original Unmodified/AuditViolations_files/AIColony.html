447      * This is always needed even when there is no customs house, because
448      * updateExportGoods needs to know what to export by transport.
449      *
450      * FIXME: consider market prices?
451      */
452     private void resetExports() {
453         final Specification spec = getSpecification();
454         final Player player = colony.getOwner();
455
456         fullExport.clear();
457         partExport.clear();
458         // Initialize the exportable sets.
459         // Luxury goods, non-raw materials (silver), and raw materials
460         // we do not produce (might have been gifted) should always be
461         // fully exported.  Other raw and manufactured goods should be
462         // exported only to the extent of not filling the warehouse.
463         for (GoodsType g : spec.getStorableGoodsTypeList()) {
464             if (!g.isFoodType()
465                 && !g.isBuildingMaterial()
466                 && !g.isMilitaryGoods()
467                 && !g.isTradeGoods()) {
468                 if (g.isRawMaterial()) {
469                     partExport.add(g);
470                 } else {
471                     fullExport.add(g);
472                 }
473             }
474         }
475         for (Role role : spec.getRoles()) {
476             if (role.isAvailableTo(player, spec.getDefaultUnitType(player))) {
477                 for (AbstractGoods ag : role.getRequiredGoods()) {
478                     if (fullExport.contains(ag.getType())) {
479                         fullExport.remove(ag.getType());
480                         partExport.add(ag.getType());
481                     }
482                 }
483             }
484         }
485
486         if (colony.getOwner().getMarket() == null) {
487             // Do not export when there is no market!
488             for (GoodsType g : spec.getGoodsTypeList()) {
489                 colony.getExportData(g).setExported(false);
490             }
491         } else {
492             int exportLevel = 4 * colony.getWarehouseCapacity() / 5;
493             for (GoodsType g : spec.getGoodsTypeList()) {
494                 if (fullExport.contains(g)) {
495                     colony.getExportData(g).setExportLevel(0);
496                     colony.getExportData(g).setExported(true);
497                 } else if (partExport.contains(g)) {
498                     colony.getExportData(g).setExportLevel(exportLevel);
499                     colony.getExportData(g).setExported(true);
500                 } else {
501                     colony.getExportData(g).setExported(false);
502                 }
503             }
504         }
505     }

534     /**
535      * Steals neighbouring tiles but only if the colony has some
536      * defence.  Grab everything if at war with the owner, otherwise
537      * just take the tile that best helps with the currently required
538      * raw building materials, with a lesser interest in food.
539      *
540      * @param lb A <code>LogBuilder</code> to log to.
541      */
542     private void stealTiles(LogBuilder lb) {
543         final Specification spec = getSpecification();
544         final Tile tile = colony.getTile();
545         final Player player = colony.getOwner();
546         boolean hasDefender = any(tile.getUnitList(),
547             u -> u.isDefensiveUnit()
548                 && getAIUnit(u).hasDefendSettlementMission());
549         if (!hasDefender) return;
550
551         // What goods are really needed?
552         List<GoodsType> needed = transform(spec.getRawBuildingGoodsTypeList(),
553             gt -> colony.getTotalProductionOf(gt) <= 0, Collectors.toList());
554
555         // If a tile can be stolen, do so if already at war with the
556         // owner or if it is the best one available.
557         UnitType unitType = spec.getDefaultUnitType(player);
558         Tile steal = null;
559         double score = 1.0;
560         for (Tile t : tile.getSurroundingTiles(1)) {
561             Player owner = t.getOwner();
562             if (owner == null || owner == player
563                 || owner.isEuropean()
564                 || !player.canClaimForSettlement(t)) continue;
565             if (owner.atWarWith(player)) {
566                 if (AIMessage.askClaimLand(t, this, Constants.STEAL_LAND)
567                     && player.owns(t)) {
568                     lb.add(", stole tile ", t,
569                           " from hostile ", owner.getName());
570                 }
571             } else {
572                 // Pick the best tile to steal, considering mainly the
573                 // building goods needed, but including food at a lower
574                 // weight.
575                 double s = sumDouble(needed,
576                         gt -> t.getPotentialProduction(gt, unitType))
577                     + sumDouble(spec.getFoodGoodsTypeList(),
578                         ft -> 0.1 * t.getPotentialProduction(ft, unitType));
579                 if (s > score) {
580                     score = s;
581                     steal = t;
582                 }
583             }
584         }
585         if (steal != null) {
586             Player owner = steal.getOwner();
587             if (AIMessage.askClaimLand(steal, this, Constants.STEAL_LAND)
588                 && player.owns(steal)) {
589                 lb.add(", stole tile ", steal, " (score = ", score,
590                       ") from ", owner.getName());
591             }
592         }
593     }

623                     }
624                 }
625             }

747     /**
748      * Creates a list of the goods which should be shipped out of this colony.
749      *
750      * @param lb A <code>LogBuilder</code> to log to.
751      */
752     private void updateExportGoods(LogBuilder lb) {
753         if (colony.hasAbility(Ability.EXPORT)) {
754             for (AIGoods aig : getExportGoods()) {
755                 goodsLog(aig, "customizes", lb);
756                 dropExportGoods(aig);
757             }
758
759         } else { // does not have a customs house
760             for (AIGoods aig : getExportGoods()) {
761                 if (aig == null) {
762                     removeExportGoods(aig);
763                 } else if (aig.checkIntegrity(false) < 0) {
764                     goodsLog(aig, "reaps", lb);
765                     dropExportGoods(aig);
766                 } else if (aig.getGoods().getLocation() != colony) {
767                     // On its way, no longer of interest here, but do
768                     // not dispose as that will happen when delivered.
769                     goodsLog(aig, "sends", lb);
770                     removeExportGoods(aig);
771                 } else if (colony.getAdjustedNetProductionOf(aig.getGoods()
772                         .getType()) < 0) {
773                     goodsLog(aig, "needs", lb);
774                     dropExportGoods(aig);
775                 }
776             }
777
778             // Create a new batch of exported goods.
779             final Europe europe = colony.getOwner().getEurope();
780             final int capacity = colony.getWarehouseCapacity();
781             List<AIGoods> newAIGoods = new ArrayList<>();
782             outer: for (GoodsType gt : getSpecification().getGoodsTypeList()) {
783                 if (colony.getAdjustedNetProductionOf(gt) < 0) continue;
784                 int count = colony.getGoodsCount(gt);
785                 int exportAmount = (fullExport.contains(gt))
786                     ? count
787                     : (partExport.contains(gt))
788                     ? count - colony.getExportData(gt).getExportLevel()
789                     : -1;
790                 if (exportAmount <= 0) continue;
791                 int priority = (exportAmount >= capacity)
792                     ? TransportableAIObject.IMPORTANT_DELIVERY
793                     : (exportAmount >= GoodsContainer.CARGO_SIZE)
794                     ? TransportableAIObject.FULL_DELIVERY
795                     : 0;
796
797                 // Find all existing AI goods of type gt, update
798                 // amount of goods to export to that present in the
799                 // colony, and drop exports of trivial amounts.  If
800                 // existing goods are found we do not continue with
801                 // this goods type but add the updated goods to the
802                 // new goods batch.
803                 for (AIGoods aig : getExportGoods()) {
804                     Goods goods = aig.getGoods();
805                     if (goods.getType() == gt) {
806                         int amount = goods.getAmount();
807                         if (amount <= exportAmount) {
808                             goods.setAmount(exportAmount);
809                             goodsLog(aig, "updates", lb);
810                             newAIGoods.add(aig);
811                         } else {
812                             if (exportAmount >= EXPORT_MINIMUM) {
813                                 goods.setAmount(exportAmount);
814                                 goodsLog(aig, "clamps", lb);
815                                 newAIGoods.add(aig);
816                             } else {
817                                 goodsLog(aig, "unexports", lb);
818                                 dropExportGoods(aig);
819                             }
820                         }
821                         continue outer;
822                     }
823                 }
824                 // Export new goods, to Europe if possible.
825                 Location destination = (colony.getOwner().canTrade(gt))
826                     ? europe : null;
827                 if (exportAmount >= EXPORT_MINIMUM) {
828                     AIGoods newGoods = new AIGoods(getAIMain(), colony, gt,
829                                                    exportAmount, destination);
830                     newGoods.setTransportPriority(priority);
831                     newAIGoods.add(newGoods);
832                     goodsLog(newGoods, "makes", lb);
833                 }
834             }
835             setExportGoods(newAIGoods);
836         }
837     }
