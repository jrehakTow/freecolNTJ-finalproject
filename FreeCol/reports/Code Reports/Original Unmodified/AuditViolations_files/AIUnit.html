467     /**
468      * Equips this AI unit for a particular role.
469      *
470      * The unit must be at a location where the required goods are available
471      * (possibly requiring a purchase, which may fail due to lack of gold
472      * or boycotts in effect).
473      *
474      * @param role The <code>Role</code> to equip for identifier.
475      * @return True if the role change was successful.
476      */
477     public boolean equipForRole(Role role) {
478         final Specification spec = getSpecification();
479         final Player player = unit.getOwner();
480         Location loc = Location.upLoc(unit.getLocation());
481         if (!(loc instanceof UnitLocation)) return false;
482         int count = role.getMaximumCount();
483         if (count > 0) {
484             for (; count > 0; count--) {
485                 List<AbstractGoods> req = unit.getGoodsDifference(role, count);
486                 int price = ((UnitLocation)loc).priceGoods(req);
487                 if (price < 0) continue;
488                 if (player.checkGold(price)) break;
489             }
490             if (count <= 0) return false;
491         }
492         return AIMessage.askEquipForRole(this, role, count)
493             && unit.getRole() == role && unit.getRoleCount() == count;
494     }

498      *
499      * Favour unequipped freeColonists, and other unskilled over experts.

501      *
502      * @return An integer score.
503      */
504     public int getBuilderScore() {
505         Unit unit = getUnit();
506         if (unit == null || BuildColonyMission.invalidReason(this) != null)
507             return -1000;

636     /**
637      * {@inheritDoc}
638      */
639     @Override
640     public boolean leaveTransport() {
641         if (!unit.isOnCarrier()) return true; // Harmless error
642
643         // Just leave at once if in Europe
644         if (unit.isInEurope()) return leaveTransport(null);
645
646         // Otherwise if not on the map, do nothing.
647         final Tile tile = unit.getTile();
648         if (tile == null) return false;
649
650         // Try to go to the target location.
651         final Mission mission = getMission();
652         final Location target = (mission == null || !mission.isValid()) ? null
653             : mission.getTarget();
654         Direction direction;
655         if (target != null) {
656             if (Map.isSameLocation(target, tile)) return leaveTransport(null);
657             if (target.getTile() != null
658                 && (direction = tile.getDirection(target.getTile())) != null) {
659                 return leaveTransport(direction);
660             }
661             PathNode path = unit.findPath(target); // Not using carrier!
662             if (path != null
663                 && (direction = tile.getDirection(path.next.getTile())) != null) {
664                 try {
665                     return leaveTransport(direction);
666                 } catch (Exception e) {
667                     logger.log(Level.WARNING, "Leave transport crash for "
668                         + this + "/" + unit.getMovesLeft(), e);
669                 }
670             }
671         }
672
673         // Just get off here if possible.
674         if (tile.isLand()) return leaveTransport(null);
675
676         // Collect neighbouring land tiles, get off if one has our settlement
677         List<Tile> tiles = new ArrayList<>();
678         for (Tile t : tile.getSurroundingTiles(1)) {
679             if (!t.isBlocked(unit)) {
680                 if (t.getSettlement() != null) {
681                     return leaveTransport(tile.getDirection(t));
682                 }
683                 tiles.add(t);
684             }
685         }
686
687         // No adjacent unblocked tile, fail.
688         if (tiles.isEmpty()) return false;
689
690         // Pick the available tile with the shortest path to one of our
691         // settlements, or the tile with the highest defence value.
692         final Player player = unit.getOwner();
693         Tile safe = tiles.get(0);
694         Tile best = null;
695         int bestTurns = Unit.MANY_TURNS;
696         Settlement settlement = null;
697         for (Tile t : tiles) {
698             if (settlement == null || t.isConnectedTo(settlement.getTile())) {
699                 settlement = t.getNearestSettlement(player, 10, true);
700             }
701             if (settlement != null) {
702                 int turns = unit.getTurnsToReach(t, settlement);
703                 if (bestTurns > turns) {
704                     bestTurns = turns;
705                     best = t;
706                 }
707             }
708             if (safe.getDefenceValue() < t.getDefenceValue()) {
709                 safe = t;
710             }
711         }
712         return leaveTransport(tile.getDirection((best != null) ? best : safe));
713     }

717      */
718     @Override
719     public boolean leaveTransport(Direction direction) {
720         if (!unit.isOnCarrier()) return false;
721         final Unit carrier = unit.getCarrier();
722         boolean result = (direction == null)
723             ? (AIMessage.askDisembark(this)
724                 && unit.getLocation() == carrier.getLocation())
725             : move(direction);
726
727         if (result) {
728             requestLocalRearrange();
729             dropTransport();
730         }
731         return result;
732     }
