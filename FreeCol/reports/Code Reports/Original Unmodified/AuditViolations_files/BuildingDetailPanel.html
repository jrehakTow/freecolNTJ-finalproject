77     /**
78      * {@inheritDoc}
79      */
80     @Override
81     public void addSubTrees(DefaultMutableTreeNode root) {
82         DefaultMutableTreeNode parent
83             = new DefaultMutableTreeNode(new ColopediaTreeItem(this, getId(),
84                     getName(), null));
85
86         List<BuildingType> buildingTypes = new ArrayList<>();
87         Map<BuildingType, DefaultMutableTreeNode> buildingHash
88             = new HashMap<>();
89         for (BuildingType buildingType
90                  : getSpecification().getBuildingTypeList()) {
91             if (buildingType.getUpgradesFrom() == null) {
92                 String name = Messages.getName(buildingType);
93                 DefaultMutableTreeNode item =
94                     new DefaultMutableTreeNode(new ColopediaTreeItem(
95                         this, buildingType.getId(), name,
96                         new ImageIcon(ImageLibrary.getBuildingImage(buildingType,
97                             new Dimension(-1, ImageLibrary.ICON_SIZE.height)))));
98                 buildingHash.put(buildingType, item);
99                 parent.add(item);
100             } else {
101                 buildingTypes.add(buildingType);
102             }
103         }
104
105         while (!buildingTypes.isEmpty()) {
106             for (Iterator<BuildingType> iterator = buildingTypes.iterator(); iterator.hasNext();) {
107                 BuildingType buildingType = iterator.next();
108                 DefaultMutableTreeNode node = buildingHash.get(buildingType.getUpgradesFrom());
109                 if (node != null) {
110                     String name = Messages.getName(buildingType);
111                     DefaultMutableTreeNode item =
112                         new DefaultMutableTreeNode(new ColopediaTreeItem(
113                             this, buildingType.getId(), name,
114                             new ImageIcon(ImageLibrary.getBuildingImage(buildingType,
115                                 new Dimension(-1, ImageLibrary.ICON_SIZE.height)))));
116                     node.add(item);
117                     buildingHash.put(buildingType, item);
118                     iterator.remove();
119                 }
120             }
121         }
122         root.add(parent);
123     }
