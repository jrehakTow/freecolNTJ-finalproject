609      * @param width The component width to use.
610      * @param height The component height to use.
611      * @param popupPosition An optional <code>PopupPosition</code> hint.
612      * @return A suitable <code>Point</code> to place the component.
613      */
614     private Point chooseLocation(Component comp, int width, int height, 
615                                  PopupPosition popupPosition) {
616         Point p = null;
617         if ((comp instanceof FreeColPanel)
618             && (p = getSavedPosition(comp)) != null) {
619             // Sanity check stuff coming out of client options.
620             if (p.getX() < 0
621                 || p.getX() >= getWidth() - width
622                 || p.getY() < 0
623                 || p.getY() >= getHeight() - height) {
624                 p = null;
625             }
626         }
627         int x = 0, y = 0;
628         if (p != null) {
629             x = (int)p.getX();
630             y = (int)p.getY();
631         } else if (popupPosition != null) {
632             switch (popupPosition) {
633             case CENTERED:
634                 x = (getWidth() - width) / 2;
635                 y = (getHeight() - height) / 2;
636                 break;
637             case CENTERED_LEFT:
638                 x = (getWidth() - width) / 4;
639                 y = (getHeight() - height) / 2;
640                 break;
641             case CENTERED_RIGHT:
642                 x = ((getWidth() - width) * 3) / 4;
643                 y = (getHeight() - height) / 2;
644                 break;
645             case ORIGIN:
646                 x = y = 0;
647                 break;
648             }
649         }
650         if ((p = getClearSpace(x, y, width, height, MAXTRY)) != null
651             && p.x >= 0 && p.x < getWidth()
652             && p.y >= 0 && p.y < getHeight()) {
653             x = p.x;
654             y = p.y;
655         }
656         return new Point(x, y);
657     }

670     /**
671      * Try to find some free space on the canvas for a component,
672      * starting at x,y.
673      *
674      * @param x A starting x coordinate.
675      * @param y A starting y coordinate.
676      * @param w The component width to use.
677      * @param h The component height to use.
678      * @param tries The number of attempts to find a clear space.
679      * @return A <code>Point</code> to place the component at or null
680      *     on failure.
681      */
682     private Point getClearSpace(final int x, final int y,
683                                 final int w, final int h, int tries) {
684         final Rectangle bounds = this.getBounds();
685         if (!bounds.contains(x, y)) return null;
686
687         tries = 3 * tries + 1; // 3 new candidates per level
688         List<Point> todo = new ArrayList<>();
689         Point p = new Point(x, y);
690         todo.add(p);
691
692         List<Component> allComponents = transform(this.getComponents(),
693             c -> !(c instanceof GrayLayer) && c.isValid(), Collectors.toList());
694         for (FreeColDialog<?> fcd : dialogs) allComponents.add(fcd);
695
696         // Find the position with the least overlap
697         int bestScore = Integer.MAX_VALUE;
698         Point best = p;
699         while (!todo.isEmpty()) {
700             p = todo.remove(0);
701             Rectangle r = new Rectangle(p.x, p.y, w, h);
702             if (!bounds.contains(r)) {
703                 continue;
704             }
705
706             // Find the most overlapping component at this position,
707             // as well as the globally least.
708             int foundScore = 0;
709             Component found = null;
710             for (Component c : allComponents) {
711                 if (c.getBounds().intersects(r)) {
712                     Rectangle rr = c.getBounds().intersection(r);
713                     int score = (int)Math.round(rr.getWidth() * rr.getHeight());
714                     if (foundScore < score) {
715                         foundScore = score;
716                         found = c;
717                     }
718                 }
719             }
720             if (found == null) { // Can not improve on no overlap, return now
721                 return p;
722             }
723             if (bestScore > foundScore) {
724                 bestScore = foundScore;
725                 best = p;
726             }
727             // Guarantee eventual completion
728             if (--tries <= 0) break;
729
730             int n = todo.size(),
731                 // Some alternative new positions
732                 //   0: move right/down to avoid the collision
733                 //   1: move as far as possible right/down
734                 //   2: wrap back to the far left
735                 x0 = found.getX() + found.getWidth() + 1,
736                 y0 = found.getY() + found.getHeight() + 1,
737                 x1 = bounds.x + bounds.width - w - 1,
738                 y1 = bounds.y + bounds.height - h - 1,
739                 x2 = bounds.x,
740                 y2 = bounds.y;
741             boolean x0ok = bounds.contains(x0 + w, y),
742                 y0ok = bounds.contains(x, y0 + h),
743                 x1ok = bounds.contains(x1, y),
744                 y1ok = bounds.contains(x, y1);
745             todo.add(n, new Point((x0ok) ? x0 : (x1ok) ? x1 : x2,
746                                   (y0ok) ? y0 : (y1ok) ? y1 : y2));
747             todo.add(n, new Point(x, (y0ok) ? y0 : (y1ok) ? y1 : y2));
748             todo.add(n, new Point((x0ok) ? x0 : (x1ok) ? x1 : x2, y));
749         }
750         return best;
751     }
