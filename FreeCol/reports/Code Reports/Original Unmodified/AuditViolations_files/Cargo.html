91         /**
92          * Plan the transport of a transportable with a given carrier.
93          *
94          * @param t The <code>TransportableAIObject</code> to deliver.
95          * @param carrier The carrier <code>Unit</code> to use.
96          * @param destination The destination <code>Location</code> to take the
97          *     transportable to, using the transport destination if null.
98          * @param allowFallback Allow a fallback plan that does not complete
99          *     the transport but does at least improve matters.
100          * @return Null on success, an error message on failure.
101          */
102         public String initialize(TransportableAIObject t, Unit carrier,
103                                  Location destination, boolean allowFallback) {
104             // Do some sanity checking
105             if (t.isDisposed()) {
106                 return "invalid-disposed";
107             } else if (carrier == null) {
108                 return "invalid-null-carrier";
109             } else if (destination == null
110                 && (destination = t.getTransportDestination()) == null) {
111                 return "invalid-null-destination";
112             }
113             this.tdst = Location.upLoc(destination);
114             final Location src = t.getLocation();
115             final boolean carrying = src == carrier;
116             if (!carrying && src instanceof Unit) {
117                 // FIXME: drop this and allow multi-stage plans?
118                 return "invalid-collected-elsewhere";
119             }
120
121             // Get the path to the destination, possibly allowing fallback
122             // to a destination that at least improves matters.
123             PathNode deliver = t.getDeliveryPath(carrier, tdst);
124             fallback = false;
125             if (deliver == null && allowFallback) {
126                 deliver = t.getIntermediatePath(carrier, tdst);
127                 fallback = true;
128             }
129             if (deliver == null) {
130                 return "no-deliver " + t
131                     + "/" + carrier.toShortString()
132                     + " -> " + tdst.toShortString();
133             }
134
135             // Where is the transportable collected?  At the first
136             // path node where it is on the carrier.
137             PathNode pick = deliver.getCarrierMove();
138             if (pick == null) return "invalid-transport-not-needed";
139             // The pickup node determines the c/twait locations.
140             if (carrying) {
141                 this.twait = this.cwait = null;
142             } else {
143                 this.cwait = Location.upLoc(pick.getLocation());
144                 // If there is a previous non-carrier move on the delivery
145                 // path, that is where the transportable should wait.
146                 // This will be true for units moving directly from land
147                 // to a naval carrier, but usually false when collection
148                 // occurs in a colony (as for goods).
149                 PathNode prev = (pick.previous == null) ? pick : pick.previous;
150                 this.twait = Location.upLoc(prev.getLocation());
151             }
152
153             // Can the carrier reach the pickup point?  If already
154             // carrying this is obviously moot.
155             PathNode collect = null;
156             if (!carrying
157                 && (collect = carrier.findPath(this.cwait)) == null) {
158                 return "no-collect " + t
159                     + "/" + carrier.toShortString()
160                     + " at " + this.cwait.toShortString();
161             }
162                 
163             // Where is the transportable dropped?  At the drop node,
164             // or at its predecessor from the carrier point of view.
165             PathNode drop = pick.getTransportDropNode();
166             if (drop == null || drop.previous == null) {
167                 throw new IllegalStateException("Cargo failure " + t
168                     + " " + deliver.fullPathToString()
169                     + " " + pick.fullPathToString()
170                     + " " + drop);
171             }
172             this.cdst = Location.upLoc(drop.previous.getLocation());
173
174             // The transportable ends up at the end of the delivery path.
175             this.tdst = Location.upLoc(deliver.getLastNode().getLocation());
176
177             // Total turns is just that of the delivery path if the
178             // transportable has been collected.  Otherwise, it is the
179             // maximum of the turns for the transportable and carrier to
180             // reach the collection point, plus the turns from there to
181             // the destination.
182             //
183             // The mode depends whether the carrier and transportable
184             // have the same terminal points.
185             if (carrying) {
186                 this.turns = deliver.getTotalTurns();
187                 this.mode = (this.cdst instanceof Europe
188                     || this.cdst == this.tdst) ? CargoMode.UNLOAD
189                     : CargoMode.DROPOFF;
190             } else {
191                 this.turns = Math.max(pick.getTurns(), collect.getTotalTurns())
192                     + pick.getTotalTurns();
193                 this.mode = (this.cwait instanceof Europe
194                     || this.cwait == this.twait) ? CargoMode.LOAD
195                     : CargoMode.PICKUP;
196             }
197             return null;
198         }
199     }

560      * Check the integrity of this cargo.
561      *
562      * @param aiCarrier The <code>AIUnit</code> version of the carrier.
563      * @return A reason for integrity failure, or null if none.
564      */
565     public String check(AIUnit aiCarrier) {
566         if (transportable == null) {
567             return "null transportable";
568         } else if (transportable.isDisposed()) {
569             return "disposed transportable";
570         }
571             
572         Locatable l = transportable.getTransportLocatable();
573         if (l == null) {
574             return "null locatable: " + transportable;
575         } else if (l instanceof FreeColGameObject
576             && ((FreeColGameObject)l).isDisposed()) {
577             return "locatable disposed";
578         }
579             
580         Location tLoc = l.getLocation();
581         if (tLoc instanceof Unit && tLoc != carrier) {
582             return "carrier usurped"; // On another carrier!
583         }
584         return null;
585     }

587     /**
588      * Can this cargo be queued at the given index in a list of cargoes?
589      *
590      * TODO: be smarter and break out of the loop if the cargo reaches
591      * its delivery point.
592      *
593      * @param carrier The <code>Unit</code> to queue to.
594      * @param index The queuing position to test.
595      * @param cargoes A list of <code>Cargo</code>s.
596      * @return True if there is space to add the cargo.
597      */
598     public boolean canQueueAt(Unit carrier, int index, List<Cargo> cargoes) {
599         final int maxHolds = carrier.getCargoCapacity();
600         final int newSpace = this.getNewSpace();
601         Cargo tr = cargoes.get(index);
602         for (int j = index; j < cargoes.size(); j++) {
603             int holds = (j == 0) ? carrier.getCargoSpaceTaken()
604                 : maxHolds - cargoes.get(j-1).getSpaceLeft();
605             holds += newSpace;
606             if (holds < 0 || holds > maxHolds) return false;
607         }
608         return true;
609     }

706         final Game game = aiMain.getGame();
707
708         String tid = xr.readId();
709         TransportableAIObject tao = null;
710         if (tid != null) {
711             AIObject aio = aiMain.getAIObject(tid);
712             if (aio == null) {

714                     tao = new AIUnit(aiMain, tid);
715                 } else if (tid.startsWith(AIGoods.getTagName())) {
716                     tao = new AIGoods(aiMain, tid);
717                 }
718             } else {
719                 tao = (TransportableAIObject)aio;
720             }
721         }
722         if (tao == null) {
723             throw new XMLStreamException("Transportable expected: " + tid);
724         }
725         this.transportable = tao;
726
727         this.carrier = xr.getAttribute(game, CARRIER_TAG,
728                                        Unit.class, (Unit)null);
729
730         this.tries = xr.getAttribute(TRIES_TAG, 0);
731         
732         this.spaceLeft = xr.getAttribute(SPACELEFT_TAG, -1);
733             
734         this.wrapped = null;

741
