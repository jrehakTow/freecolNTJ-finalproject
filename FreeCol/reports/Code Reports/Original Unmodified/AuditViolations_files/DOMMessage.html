622     public static Document readDocument(InputSource inputSource)
623         throws SAXException, IOException {
624         Document tempDocument = null;
625         boolean dumpMsgOnError = true;
626         if (dumpMsgOnError) {
627             inputSource.setByteStream(new BufferedInputStream(inputSource.getByteStream()));
628
629             inputSource.getByteStream().mark(1000000);
630         }
631         try {
632             synchronized (parser) {
633                 tempDocument = parser.parse(inputSource);
634             }
635         } catch (IOException ex) {
636             //} catch (IOException|SAXException ex) {
637             throw ex;
638         } catch (Exception ex) {
639             // Xerces throws ArrayIndexOutOfBoundsException when it barfs on
640             // some FreeCol messages. I'd like to see the messages upon which
641             // it barfs.
642             // Its also throwing SAXParseException in BR#2925
643             if (dumpMsgOnError) {
644                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
645                 inputSource.getByteStream().reset();
646                 while (true) {

648                     if (-1 == i) {

650                     }
651                     baos.write(i);
652                 }
653                 logger.log(Level.SEVERE, baos.toString("UTF-8"), ex);
654             } else {
655                 logger.log(Level.WARNING, "Parse error", ex);
656             }

658         }
