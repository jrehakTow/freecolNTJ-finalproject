532     /**
533      * Debug action to check for client-server desynchronization.
534      *
535      * Called from the debug menu and client controller.
536      * TODO: This is still fairly new and messy.  Defer i18n for a while.
537      *
538      * @param freeColClient The <code>FreeColClient</code> for the game.
539      * @return True if desynchronization found.
540      */
541     public static boolean checkDesyncAction(final FreeColClient freeColClient) {
542         final FreeColServer server = freeColClient.getFreeColServer();
543         final Game game = freeColClient.getGame();
544         final Map map = game.getMap();
545         final Player player = freeColClient.getMyPlayer();
546         final Game sGame = server.getGame();
547         final Map sMap = sGame.getMap();
548         final ServerPlayer sPlayer = sGame.getFreeColGameObject(player.getId(),
549                                                                 ServerPlayer.class);
550
551         boolean problemDetected = false;
552         LogBuilder lb = new LogBuilder(256);
553         lb.add("Desynchronization detected\n");
554         for (Tile t : sMap.getAllTiles()) {
555             if (!sPlayer.canSee(t)) continue;
556             for (Unit u : t.getUnitList()) {
557                 if (!sPlayer.owns(u)
558                     && (t.hasSettlement() || u.isOnCarrier())) continue;
559                 if (game.getFreeColGameObject(u.getId(), Unit.class) == null) {
560                     lb.add("Unit missing on client-side.\n", "  Server: ",
561                            u.getDescription(Unit.UnitLabelType.NATIONAL),
562                            "(", u.getId(), ") from: ", 
563                            u.getLocation().getId(), ".\n");
564                     try {
565                         lb.add("  Client: ", map.getTile(u.getTile().getX(),
566                                u.getTile().getY()).getFirstUnit().getId(),
567                                "\n");
568                     } catch (NullPointerException npe) {}
569                     problemDetected = true;
570                 } else {
571                     Unit cUnit = game.getFreeColGameObject(u.getId(),
572                                                            Unit.class);
573                     if (cUnit.hasTile()
574                         && !cUnit.getTile().getId().equals(u.getTile().getId())) {
575                         lb.add("Unit located on different tiles.\n",
576                             "  Server: ", u.getDescription(Unit.UnitLabelType.NATIONAL),
577                             "(", u.getId(), ") from: ",
578                             u.getLocation().getId(), "\n",
579                             "  Client: ", cUnit.getDescription(Unit.UnitLabelType.NATIONAL),
580                             "(", cUnit.getId(), ") at: ",
581                             cUnit.getLocation().getId(), "\n");
582                         problemDetected = true;
583                     }
584                 }
585             }
586             Tile ct = game.getFreeColGameObject(t.getId(), Tile.class);
587             Settlement sSettlement = t.getSettlement();
588             Settlement cSettlement = ct.getSettlement();
589             if (sSettlement == null) {
590                 if (cSettlement == null) {
591                     ;// OK
592                 } else {
593                     lb.add("Settlement still present in client: ", cSettlement);
594                     problemDetected = true;
595                 }
596             } else {
597                 if (cSettlement == null) {
598                     lb.add("Settlement not present in client: ", sSettlement);
599                     problemDetected = true;
600                 } else if (sSettlement.getId().equals(cSettlement.getId())) {
601                     ;// OK
602                 } else {
603                     lb.add("Settlements differ.\n  Server: ",
604                         sSettlement.toString(), "\n  Client: ", 
605                         cSettlement.toString(), "\n");
606                 }
607             }
608         }
609
610         boolean goodsProblemDetected = false;
611         for (GoodsType sg : sGame.getSpecification().getGoodsTypeList()) {
612             int sPrice = sPlayer.getMarket().getBidPrice(sg, 1);
613             GoodsType cg = game.getSpecification().getGoodsType(sg.getId());
614             int cPrice = player.getMarket().getBidPrice(cg, 1);
615             if (sPrice != cPrice) {
616                 lb.add("Goods prices for ", sg, " differ.\n");
617                 goodsProblemDetected = true;
618             }
619         }
620         if (goodsProblemDetected) {
621             lb.add("  Server:\n", sPlayer.getMarket(), "\n",
622                 "  Client:\n", player.getMarket(), "\n");
623             problemDetected = true;
624         }
625
626         if (problemDetected) {
627             lb.shrink("\n");
628             String err = lb.toString();
629             freeColClient.getGUI().showInformationMessage(err);
630             logger.severe(err);
631         }
632         return problemDetected;
633     }
