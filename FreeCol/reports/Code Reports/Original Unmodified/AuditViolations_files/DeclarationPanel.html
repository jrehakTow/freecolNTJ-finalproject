138     /**
139      * A panel for displaying an animated signature.
140      * 
141      * The panel should be {@link #initialize(String) initialized} with a name
142      * before the animation is {@link #startAnimation() started}.
143      */
144     private class SignaturePanel extends JPanel {
145
146         private final FAFile faFile;
147
148         private final ArrayList<ActionListener> actionListeners
149             = new ArrayList<>();
150
151         private Point[] points = null;
152
153         private int counter = 0;
154
155
156         SignaturePanel() {
157             faFile = ResourceManager.getFAFile("animatedfont.signature");
158             setOpaque(false);
159         }
160
161         /**
162          * Gets an abbreviated version of the given name.
163          * 
164          * @param name The name to be abbreviated
165          * @return The abbreviated version of the given name.  The name
166          *     is made small enough to fit within the bounds the
167          *     <code>DeclarationPanel</code>.
168          */
169         private String getAbbreviatedName(String name) {
170             if (!isTooLarge(name)) return name;
171
172             String[] partNames = name.split(" ");
173
174             // Abbreviate middle names:
175             for (int i = 1; i < partNames.length - 1
176                      && isTooLarge(join(" ", partNames)); i++) {
177                 partNames[i] = partNames[i].charAt(0) + ".";
178             }
179
180             // Remove middle names:
181             while (partNames.length > 2
182                 && isTooLarge(join(" ", partNames))) {
183                 String[] newPartNames = new String[partNames.length - 1];
184                 newPartNames[0] = partNames[0];
185                 for (int i = 1; i < newPartNames.length; i++) {
186                     newPartNames[i] = partNames[i + 1];
187                 }
188                 partNames = newPartNames;
189             }
190
191             String first = partNames[0], second = partNames[1];
192             String s = join(" ", partNames);
193             if (!isTooLarge(s)) return s;
194             s = first.charAt(0) + ". " + second;
195             if (!isTooLarge(s)) return s;
196             s = first + " " + second.charAt(0) + ".";
197             if (!isTooLarge(s)) return s;
198             return first.charAt(0) + ". " + second.charAt(0) + ".";
199         }
200
201         /**
202          * Checks if the given string is to large to be displayed within this
203          * panel.
204          * 
205          * @param name The string to be tested.
206          * @return <code>true</code> if the given string was to large to be
207          *     fully displayed.
208          */
209         private boolean isTooLarge(String name) {
210             Dimension d = faFile.getDimension(name);
211             return (d.width > DeclarationPanel.this.getWidth() - 10);
212         }
213
214         /**
215          * Initializes this panel with the given name.
216          * 
217          * @param name The name to be used when making the signature.
218          */
219         public void initialize(String name) {
220             name = getAbbreviatedName(name);
221
222             points = faFile.getPoints(name);
223             counter = 0;
224             setSize(faFile.getDimension(name));
225         }
226
227         /**
228          * Adds an <code>ActionListener</code> to this
229          * <code>SignaturePanel</code>.  An event gets fired when the
230          * animation is stopped.
231          * 
232          * @param al The <code>ActionListener</code>.
233          * @see #startAnimation()
234          */
235         public void addActionListener(ActionListener al) {
236             if (!actionListeners.contains(al)) {
237                 actionListeners.add(al);
238             }
239         }
240
241         private void notifyStopped() {
242             for (ActionListener actionListener : actionListeners) {
243                 actionListener.actionPerformed(new ActionEvent(this,
244                         ActionEvent.ACTION_PERFORMED, ANIMATION_STOPPED));
245             }
246         }
247
248         /**
249          * Starts the animation of the signature.  An
250          * <code>ActionEvent</code> gets sent to the registered
251          * listeners when the animation has stopped.
252          * 
253          * @see #addActionListener(ActionListener)
254          */
255         public void startAnimation() {
256             ActionListener taskPerformer = (ActionEvent ae) -> {
257                 if (counter < points.length - 1) {
258                     counter += 20;
259                     if (counter > points.length) {
260                         counter = points.length - 1;
261                         ((Timer)ae.getSource()).stop();
262                         notifyStopped();
263                     }
264                     validate();
265                     repaint();
266                 } else {
267                     ((Timer)ae.getSource()).stop();
268                     notifyStopped();
269                 }
270             };
271             new Timer(ANIMATION_DELAY, taskPerformer).start();
272         }

284             }

290             ((Graphics2D)g).setComposite(AlphaComposite
291                 .getInstance(AlphaComposite.SRC_OVER, 0.75f));
292
293             for (int i = 0; i < counter-1; i++) {
294                 Point p1 = points[i];
295                 Point p2 = points[i+1];
296                 g.drawLine((int) p1.getX(), (int) p1.getY(),
297                     (int) p2.getX(), (int) p2.getY());
298             }
299         }
