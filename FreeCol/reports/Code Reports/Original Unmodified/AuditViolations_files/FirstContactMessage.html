31 /**
32  * The message sent when making first contact between players.
33  */
34 public class FirstContactMessage extends DOMMessage {
35
36     public static final String TAG = "firstContact";
37     private static final String CAMPS_TAG = "camps";
38     private static final String OTHER_TAG = "other";
39     private static final String PLAYER_TAG = "player";
40     private static final String RESULT_TAG = "result";
41     private static final String TILE_TAG = "tile";
42
43     /** The identifier for the player making contact. */
44     private final String playerId;
45
46     /** The identifier for the player being contacted. */
47     private final String otherId;
48
49     /**
50      * The identifier for a tile to offer the contacting player if this is a
51      * first landing and the contacted player is a friendly native.
52      */
53     private final String tileId;
54
55     /** The number of settlements the contacted player has, if native. */
56     private final String settlementCount;
57
58     /** The result of the contact. */
59     private String result;

61
62     /**
63      * Create a new <code>FirstContactMessage</code>.
64      *
65      * @param player The <code>Player</code> making contact.
66      * @param other The <code>Player</code> being contacted.
67      * @param tile An optional <code>Tile</code> to offer.
68      */
69     public FirstContactMessage(Player player, Player other, Tile tile) {
70         super(getTagName());
71
72         this.playerId = player.getId();
73         this.otherId = other.getId();
74         this.tileId = (tile == null) ? null : tile.getId();
75         this.settlementCount = (other.isEuropean()) ? null
76             : Integer.toString(other.getSettlements().size());
77         this.result = null;
78     }
79
80     /**
81      * Create a new <code>FirstContactMessage</code> from a
82      * supplied element.
83      *
84      * @param game The <code>Game</code> this message belongs to.
85      * @param element The <code>Element</code> to use to create the message.
86      */
87     public FirstContactMessage(Game game, Element element) {
88         super(getTagName());
89
90         this.playerId = getStringAttribute(element, PLAYER_TAG);
91         this.otherId = getStringAttribute(element, OTHER_TAG);
92         this.tileId = getStringAttribute(element, TILE_TAG);
93         this.settlementCount = getStringAttribute(element, CAMPS_TAG);
94         this.result = getStringAttribute(element, RESULT_TAG);
95     }

129     /**
130      * Handle a "firstContact"-message.
131      *
132      * @param server The <code>FreeColServer</code> handling the message.
133      * @param connection The <code>Connection</code> message was received on.
134      * @return An update containing the firstContactd unit, or an error
135      *     <code>Element</code> on failure.
136      */
137     public Element handle(FreeColServer server, Connection connection) {
138         final ServerPlayer serverPlayer = server.getPlayer(connection);
139         final Game game = serverPlayer.getGame();
140
141         Player first = getPlayer(game);
142         if (first == null) {
143             return serverPlayer.clientError("Invalid player: " + this.playerId)
144                 .build(serverPlayer);
145         } else if (serverPlayer.getId().equals(this.playerId)) {
146             ; // OK
147         } else {
148             return serverPlayer.clientError("Not our player: " + this.playerId)
149                 .build(serverPlayer);
150         }
151
152         ServerPlayer otherPlayer = (ServerPlayer)getOtherPlayer(game);
153         if (otherPlayer == null) {
154             return serverPlayer.clientError("Invalid other player: "
155                 + this.otherId)
156                 .build(serverPlayer);
157         } else if (otherPlayer == serverPlayer) {
158             return serverPlayer.clientError("First contact with self!?!")
159                 .build(serverPlayer);
160         }
161
162         // Proceed to contact.
163         return server.getInGameController()
164             .nativeFirstContact(serverPlayer, otherPlayer,
165                                 getTile(game), getResult())
166             .build(serverPlayer);
167     }
