119     /**
120      * Pass the source directory as first argument.
121      *
122      * @param args The program arguments.
123      * @exception IOException if the resources can not be read or written.
124      */
125     public static void main(String[] args) throws IOException {
126
127         if (args.length == 0) {
128             System.out.println("Usage: ForestMaker <directory>...");
129             System.out.println("Directory name should match a directory in");
130             System.out.println("   " + DESTDIR);
131             System.exit(1);
132         }
133
134         String riverName = "data/rules/classic/resources/images/terrain/"
135             + "ocean/center0.png";
136         BufferedImage river = ImageIO.read(new File(riverName));
137         // grab a rectangle completely filled with water
138         river = river.getSubimage(44, 22, 40, 20);
139         Rectangle2D rectangle = new Rectangle(0, 0, river.getWidth(), river.getHeight());
140         TexturePaint texture = new TexturePaint(river, rectangle);
141
142         for (String arg : args) {
143             File sourceDirectory = new File(arg);
144             if (!sourceDirectory.exists()) {
145                 System.out.println("Source directory " + arg + " does not exist.");
146                 continue;
147             }
148             String baseName = sourceDirectory.getName();
149             File destinationDirectory = new File(DESTDIR, baseName);
150             if (!destinationDirectory.exists()) {
151                 System.out.println("Destination directory " + destinationDirectory.getPath()
152                                    + " does not exist.");
153                 continue;
154             }
155             File[] imageFiles = sourceDirectory.listFiles();
156             if (imageFiles == null) {
157                 System.out.println("No images found in source directory " + arg + ".");
158                 continue;
159             } else {
160                 System.out.println(imageFiles.length + " images found in source directory "
161                                    + arg + ".");
162             }
163             List<BufferedImage> images = new ArrayList<>(imageFiles.length);
164             int maximumHeight = 0;
165             for (File imageFile : imageFiles) {
166                 if (imageFile.isFile() && imageFile.canRead()) {
167                     try {
168                         BufferedImage image = ImageIO.read(imageFile);
169                         images.add(image);
170                         if (image.getHeight() > maximumHeight) {
171                             maximumHeight = image.getHeight();
172                         }
173                     } catch(IOException e) {
174                         System.out.println("Unable to load image " + imageFile.getName() + ":\n");
175                         e.printStackTrace();
176                     }
177                 }
178             }
179             int numberOfImages = images.size();
180             Random random = new Random(1492);
181
182             /**
183              * In order to ensure that trees do not occlude the rivers
184              * on other tiles, we must move the rivers to the top NE
185              * and NW edges of the tile.
186              *
187              * If we consider two adjoining edges of the diamond to be
188              * vectors, any point within the diamond can be generated
189              * as the addition of these two vectors, suitably scaled.
190              * For the sake of convenience, we choose the two edges
191              * that will be shortened if a river is present, i.e. the
192              * SE and SW edges, and move the origin to their
193              * intersection.
194              */
195             for (int index = 0; index < 16; index++) {
196                 BufferedImage base = new BufferedImage(BASE_WIDTH, BASE_HEIGHT + MARGIN,
197                                                        BufferedImage.TYPE_INT_ARGB);
198                 Graphics2D g = base.createGraphics();
199                 g.translate(HALF_WIDTH, BASE_HEIGHT + MARGIN);
200
201                 g.setPaint(texture);
202                 String counter = "";
203                 boolean[] branches = new boolean[4];
204                 if (index > 0) {
205                     for (int i = 0; i < POWERS_OF_TWO.length; i++) {
206                         if ((index & POWERS_OF_TWO[i]) == POWERS_OF_TWO[i]) {
207                             branches[i] = true;
208                             counter += "1";
209                         } else {
210                             counter += "0";
211                         }
212                     }
213                 }
214
215                 // the two vectors that describe the diamond
216                 Point right = new Point(HALF_WIDTH, -HALF_HEIGHT);
217                 Point left = new Point(-HALF_WIDTH, -HALF_HEIGHT);
218                 int treeCount = 0;
219                 // make space for rivers
220                 if (branches[0] || branches[2]) {
221                     left.translate(RIVER_WIDTH, RIVER_HEIGHT);
222                     treeCount++;
223                 }
224                 if (branches[1] || branches[3]) {
225                     right.translate(-RIVER_WIDTH, RIVER_HEIGHT);
226                     treeCount++;
227                 }
228
229                 if (drawBorders) {
230                     int dx = right.x + left.x;
231                     int dy = right.y + left.y;
232                     g.setColor(Color.RED);
233                     g.drawLine(0, 0, right.x, right.y);
234                     g.drawLine(right.x, right.y, dx, dy);
235                     g.drawLine(dx, dy, left.x, left.y);
236                     g.drawLine(left.x, left.y, 0, 0);
237                 }
238
239                 if (drawTrees) {
240                     List<ImageLocation> trees = new ArrayList<>(TREES);
241                     // reduce number of trees if river branches are present
242                     int numberOfTrees = (6 - treeCount) * TREES / 6;
243
244                     int count = 0;
245                     while (count < numberOfTrees) {
246                         BufferedImage image = images.get(random.nextInt(numberOfImages));
247                         int width = image.getWidth();
248                         int height = image.getHeight();
249                         int halfWidth = width / 2;
250                         /**
251                          * Find a point for the root of the tree, that
252                          * is the center of the lower edge of the tree
253                          * image.
254                          */
255                         float a = random.nextFloat();
256                         float b = random.nextFloat();
257                         int x = (int) (a * right.x + b * left.x);
258                         int y = (int) (a * right.y + b * left.y);
259                         /**
260                          * Additional constraint: the left and right
261                          * edges of the tree image must be within the
262                          * tile bounds (this will fail if the tree
263                          * image is too large).
264                          */
265                         if (x - halfWidth < - HALF_WIDTH) {
266                             x = -HALF_WIDTH + halfWidth; // left
267                         }
268                         if (x + halfWidth > HALF_WIDTH) {
269                             x = HALF_WIDTH - halfWidth; // right
270                         }
271                         /**
272                          * Additional constraint: the top edge of the
273                          * tree image must be within the tile bounds.
274                          */
275                         int crown = Math.max(y - height, -(BASE_HEIGHT + MARGIN));
276                         /**
277                          * Additional constraint: if there is a river
278                          * along the top right edge of the diamond,
279                          * the top right corner of the tree most not be
280                          * "above" the line defined by that edge.
281                          */
282                         if ((branches[1] || branches[3])
283                             && crown < -BASE_HEIGHT + RIVER_HEIGHT + (x + halfWidth) / 2) {
284                             continue;
285                         }
286                         /**
287                          * Additional constraint: if there is a river
288                          * along the top left edge of the diamond,
289                          * the top left corner of the tree most not be
290                          * "above" the line defined by that edge.
291                          */
292                         if ((branches[0] || branches[2])
293                             && crown < -BASE_HEIGHT + RIVER_HEIGHT - (x - halfWidth) / 2) {
294                             continue;
295                         }
296                         //System.out.println("x=" + x + ", y=" + (y - height));
297                         trees.add(new ImageLocation(image, x - halfWidth, crown));
298                         count++;
299                     }
300
301                     // sort by y, x coordinate
302                     Collections.sort(trees);
303                     for (ImageLocation imageLocation : trees) {
304                         g.drawImage(imageLocation.image, imageLocation.x, imageLocation.y, null);
305                     }
306
307                 }
308                 g.dispose();
309
310                 ImageIO.write(base, "png", new File(destinationDirectory,
311                                                     sourceDirectory.getName() + counter + ".png"));
312
313             }
314         }
315     }
