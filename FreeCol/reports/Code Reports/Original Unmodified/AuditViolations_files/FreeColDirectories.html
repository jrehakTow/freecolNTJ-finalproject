220     /**
221      * Get directories for XDG compliant systems.
222      *
223      * Result is:
224      * - Negative if a non-XDG OS is detected or there is insufficient
225      *   XDG structure to merit migrating, or what structure there is is
226      *   broken in some way.
227      * - Zero if there is at least one relevant XDG environment
228      *   variable in use and it points to a valid writable directory,
229      *   or the default exists and is writable.
230      * - Positive if there are a full set of suitable XDG directories and
231      *   there are freecol directories therein.
232      * - Otherwise negative, including non-directories in the wrong place
233      *   and unwritable directories.
234      *
235      * The intent is to ignore XDG on negative, migrate on zero, and use
236      * on positive.
237      *
238      * @param dirs An array of <code>File</code> to be filled in with the
239      *     XDG directory if it is present or created.
240      * @return The XDG compliance state.
241      */
242     private static int getXDGDirs(File[] dirs) {
243         if (onMacOSX() || onWindows() || !onUnix()) return -1;
244
245         int ret = -1;
246         File home = getUserDefaultDirectory();
247         if (home == null) return -1; // Fail badly
248         String[][] xdg = { { XDG_CONFIG_HOME_ENV, XDG_CONFIG_HOME_DEFAULT },
249                            { XDG_DATA_HOME_ENV,   XDG_DATA_HOME_DEFAULT },
250                            { XDG_CACHE_HOME_ENV,  XDG_CACHE_HOME_DEFAULT } };
251         File[] todo = new File[xdg.length];
252         for (int i = 0; i < xdg.length; i++) {
253             String env = System.getenv(xdg[i][0]);
254             File d = (env != null) ? new File(env) : new File(home, xdg[i][1]);
255             if (d.exists()) {
256                 if (!d.isDirectory() || !d.canWrite()) {
257                     return -1; // Fail hard if something is broken
258                 }
259                 ret = Math.max(ret, 0);
260                 File f = new File(d, FREECOL_DIRECTORY);
261                 if (f.exists()) {
262                     if (!f.isDirectory() || !f.canWrite()) {
263                         return -1; // Again, fail hard
264                     }
265                     dirs[i] = f;
266                     todo[i] = null;
267                     ret++;
268                 } else {
269                     dirs[i] = d;
270                     todo[i] = f;
271                 }
272             } else {
273                 dirs[i] = null;
274                 todo[i] = d;
275             }
276         }
277         if (ret < 0) return -1; // No evidence of interest in XDG standard
278         if (ret == xdg.length) return 1; // Already fully XDG compliant
279
280         // Create the directories for migration
281         for (int i = 0; i < xdg.length; i++) {
282             if (todo[i] != null) {
283                 if (!todo[i].getPath().endsWith(FREECOL_DIRECTORY)) {
284                     if (!todo[i].mkdir()) return -1;
285                     todo[i] = new File(todo[i], FREECOL_DIRECTORY);
286                 }
287                 if (!todo[i].mkdir()) return -1;
288                 dirs[i] = todo[i];
289             }
290         }
291         return 0;
292     }

524     /**
525      * Checks/creates the freecol directory structure for the current
526      * user.
527      *
528      * The main user directory used to be in the current user's home
529      * directory, and called ".freecol" (UNIXes including Mac in
530      * 0.9.x) or "freecol" or even FreeCol.  Now we use:
531      *
532      * - on XDG standard compliant Unixes:
533      *   - config:  ~/.config/freecol
534      *   - data:    ~/.local/share/freecol
535      *   - logging: ~/.cache/freecol
536      * - on Mac:
537      *   - config:  ~/Library/Preferences/freecol
538      *   - else:    ~/Library/Application Support/freecol
539      * - on Windows:
540      *   - everything in <em>default directory</em>/freecol
541      * - otherwise use what was there
542      *
543      * Note: the freecol data directory is set independently and earlier
544      * in initialization than this routine.
545      *
546      * FIXME: Should the default location of the main user and data
547      * directories be determined by the installer?
548      *
549      * @return A message key to use to create a message to the user
550      *     possibly describing any directory migration, or null if
551      *     nothing to say.
552      */
553     public static synchronized String setUserDirectories() {
554         if (userConfigDirectory != null
555             && !isGoodDirectory(userConfigDirectory))
556             userConfigDirectory = null;
557         if (userDataDirectory != null
558             && !isGoodDirectory(userDataDirectory))
559             userDataDirectory = null;
560         if (userCacheDirectory != null
561             && !isGoodDirectory(userCacheDirectory))
562             userCacheDirectory = null;
563         File dirs[] = { userConfigDirectory, userDataDirectory,
564                         userCacheDirectory };
565
566         // If the CL-specified directories are valid, all is well.
567         // Check for OSX next because it is a Unix.
568         int migrate = (dirs[0] != null && isGoodDirectory(dirs[0])
569             && dirs[1] != null && isGoodDirectory(dirs[1])
570             && dirs[2] != null && isGoodDirectory(dirs[2])) ? 1
571             : (onMacOSX()) ? getMacOSXDirs(dirs)
572             : (onUnix()) ? getXDGDirs(dirs)
573             : (onWindows()) ? getWindowsDirs(dirs)
574             : -1;
575         File oldDir = getOldUserDirectory();
576         if (migrate < 0) {
577             if (oldDir == null) return "main.userDir.fail";
578             dirs[0] = dirs[1] = dirs[2] = oldDir; // Do not migrate.
579             migrate = 1;
580         }
581
582         // Only set user directories if not already overridden at the
583         // command line, and do not migrate in such cases.
584         if (userConfigDirectory == null) {
585             userConfigDirectory = dirs[0];
586         } else migrate = 1;
587         if (userDataDirectory == null) {
588             userDataDirectory = dirs[1];
589         } else migrate = 1;
590         if (userCacheDirectory == null) {
591             userCacheDirectory = dirs[2];
592         } else migrate = 1;
593         if (migrate == 0 && oldDir != null) {
594             copyIfFound(oldDir, "classic", userConfigDirectory);
595             copyIfFound(oldDir, "freecol", userConfigDirectory);
596             copyIfFound(oldDir, "save",    userDataDirectory);
597             copyIfFound(oldDir, "mods",    userDataDirectory);
598         }
599
600         if (logFilePath == null) {
601             logFilePath = getUserCacheDirectory() + SEPARATOR + LOG_FILE;
602         }
603
604         if (saveDirectory == null) {
605             saveDirectory = new File(getUserDataDirectory(), SAVE_DIRECTORY);
606             if (!insistDirectory(saveDirectory)) return "main.userDir.fail";
607         }
608         deriveAutosaveDirectory();
609
610         userModsDirectory = new File(getUserDataDirectory(), MODS_DIRECTORY);
611         if (!insistDirectory(userModsDirectory)) userModsDirectory = null;
612
613         return (migrate > 0) ? null
614             : (onMacOSX())  ? "main.userDir.macosx"
615             : (onUnix())    ? "main.userDir.unix"
616             : (onWindows()) ? "main.userDir.windows"
617             : null;
618     }
