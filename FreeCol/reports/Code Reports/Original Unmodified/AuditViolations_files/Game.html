422         return new Iterator<FreeColGameObject>() {
423
424             /** An iterator over the freeColGameObjects map. */
425             private final Iterator<Entry<String,
426                                          WeakReference<FreeColGameObject>>> it
427                 = freeColGameObjects.entrySet().iterator();
428
429             /** Read ahead to this next entry. */
430             private Entry<String, WeakReference<FreeColGameObject>> readAhead
431                 = null;
432
433             /** State of the readahead value. */
434             private FcgoState fcgoState = FcgoState.INVALID;
435
436
437             @Override
438             public boolean hasNext() {
439                 if (this.fcgoState == FcgoState.VALID) return true;
440                 while (this.it.hasNext()) {
441                     this.readAhead = this.it.next();
442                     if (this.readAhead.getValue().get() != null) {
443                         this.fcgoState = FcgoState.VALID;
444                         return true;
445                     }
446                     this.fcgoState = FcgoState.CONSUMED;
447                     remove();
448                 }
449                 return false;
450             }
451
452             @Override
453             public FreeColGameObject next() {
454                 if (!hasNext()) throw new NoSuchElementException();
455                 FreeColGameObject fcgo = this.readAhead.getValue().get();
456                 this.fcgoState = FcgoState.CONSUMED;
457                 return fcgo;
458             }
459
460             @Override

466                 this.fcgoState = FcgoState.INVALID;
467                 this.it.remove();
468                 logger.finest("removeFCGO/expire: " + key);
469                 notifyRemoveFreeColGameObject(key);
470             }
471         };

586     /**
587      * Gets the live player after the given player.
588      *
589      * @param beforePlayer The <code>Player</code> before the
590      *     <code>Player</code> to be returned.
591      * @return The <code>Player</code> after the <code>beforePlayer</code>
592      *     in the list which determines the order each player becomes the
593      *     current player.
594      * @see #getNextPlayer
595      */
596     public Player getPlayerAfter(Player beforePlayer) {
597         if (players.isEmpty()) return null;
598
599         final int start = players.indexOf(beforePlayer);
600         int index = start;
601         do {
602             if (++index >= players.size()) index = 0;
603             Player player = players.get(index);
604             if (!player.isUnknownEnemy() && !player.isDead()) return player;
605         } while (index != start);
606         return null;
607     }
