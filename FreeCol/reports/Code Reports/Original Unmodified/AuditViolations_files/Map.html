159         /**
160          * Gets the distance in tiles between two map positions.
161          * With an isometric map this is a non-trivial task.
162          * The formula below has been developed largely through trial and
163          * error.  It should cover all cases, but I wouldn't bet my
164          * life on it.
165          *
166          * @param ax The x-coordinate of the first position.
167          * @param ay The y-coordinate of the first position.
168          * @param bx The x-coordinate of the second position.
169          * @param by The y-coordinate of the second position.
170          * @return The distance in tiles between the positions.
171          */
172         public static int getDistance(int ax, int ay, int bx, int by) {
173             int r = (bx - ax) - (ay - by) / 2;
174
175             if (by > ay && ay % 2 == 0 && by % 2 != 0) {
176                 r++;
177             } else if (by < ay && ay % 2 != 0 && by % 2 == 0) {
178                 r--;
179             }
180             return Math.max(Math.abs(ay - by + r), Math.abs(r));
181         }

704      * Unified argument tests for full path searches, which then finds
705      * the actual starting location for the path.  Deals with special
706      * cases like starting on a carrier and/or high seas.
707      *
708      * @param unit The <code>Unit</code> to find the path for.
709      * @param start The <code>Location</code> in which the path starts from.
710      * @param carrier An optional naval carrier <code>Unit</code> to use.
711      * @return The actual starting location.
712      * @throws IllegalArgumentException If there are any argument problems.
713      */
714     private Location findRealStart(final Unit unit, final Location start,
715                                    final Unit carrier) {
716         // Unit checks.
717         if (unit == null) {
718             throw new IllegalArgumentException("Null unit.");
719         } else if (carrier != null && !carrier.canCarryUnits()) {
720             throw new IllegalArgumentException("Non-carrier carrier: "
721                 + carrier);
722         } else if (carrier != null && !carrier.couldCarry(unit)) {
723             throw new IllegalArgumentException("Carrier could not carry unit: "
724                 + carrier + "/" + unit);
725         }
726
727         Location entry;
728         if (start == null) {
729             throw new IllegalArgumentException("Null start: " + unit);
730         } else if (start instanceof Unit) {
731             Location unitLoc = ((Unit)start).getLocation();
732             if (unitLoc == null) {
733                 throw new IllegalArgumentException("Null on-carrier start: "
734                     + unit + "/" + start);
735             } else if (unitLoc instanceof HighSeas) {
736                 if (carrier == null) {
737                     throw new IllegalArgumentException("Null carrier when"
738                         + " starting on high seas: " + unit);
739                 } else if (carrier != start) {
740                     throw new IllegalArgumentException("Wrong carrier when"
741                         + " starting on high seas: " + unit
742                         + "/" + carrier + " != " + start);
743                 }
744                 entry = carrier.resolveDestination();
745             } else {
746                 entry = unitLoc;
747             }
748             
749         } else if (start instanceof HighSeas) {
750             if (unit.isOnCarrier()) {
751                 entry = unit.getCarrier().resolveDestination();
752             } else if (unit.isNaval()) {
753                 entry = unit.resolveDestination();
754             } else {
755                 throw new IllegalArgumentException("No carrier when"
756                     + " starting on high seas: " + unit
757                     + "/" + unit.getLocation());
758             }
759         } else if (start instanceof Europe || start.getTile() != null) {
760             entry = start; // OK
761         } else {
762             throw new IllegalArgumentException("Invalid start: " + start);
763         }
764         // Valid result, reduce to tile if possible.
765         return (entry.getTile() != null) ? entry.getTile() : entry;
766     }

829     /**
830      * Find the quickest path for a unit (with optional carrier) from
831      * a start tile to an end tile.
832      *
833      * @param unit The <code>Unit</code> to find the path for.
834      * @param start The <code>Tile</code> in which the path starts from.
835      * @param end The <code>Tile</code> at the end of the path.
836      * @param carrier An optional naval carrier <code>Unit</code> to use.
837      * @param costDecider An optional <code>CostDecider</code> for
838      *     determining the movement costs (uses default cost deciders
839      *     for the unit/s if not provided).
840      * @param lb An optional <code>LogBuilder</code> to log to.
841      * @return A path starting at the start tile and ending at the end
842      *     tile, or null if none found.
843      */
844     private PathNode findMapPath(Unit unit, Tile start, Tile end, Unit carrier,
845                                  CostDecider costDecider, LogBuilder lb) {
846         final Unit offMapUnit = (carrier != null) ? carrier
847             : (unit != null && unit.isNaval()) ? unit
848             : null;
849         final GoalDecider gd = GoalDeciders.getLocationGoalDecider(end);
850         final SearchHeuristic sh = getManhattenHeuristic(end);
851         Unit embarkTo;
852
853         PathNode path;
854         if (start.getContiguity() == end.getContiguity()) {
855             // If the unit potentially could get to the destination
856             // without a carrier, compare both with-carrier and
857             // without-carrier paths.  The latter will usually be
858             // faster, but not always, e.g. mounted units on a good
859             // road system.
860             path = searchMap(unit, start, gd, costDecider,
861                              INFINITY, null, sh, lb);
862             PathNode carrierPath = (carrier == null) ? null
863                 : searchMap(unit, start, gd, costDecider,
864                             INFINITY, carrier, sh, lb);
865             if (carrierPath != null
866                 && (path == null
867                     || (path.getLastNode().getCost()
868                         > carrierPath.getLastNode().getCost()))) {
869                 path = carrierPath;
870             }
871
872         } else if (offMapUnit != null) {
873             // If there is an off-map unit then complex paths which
874             // use settlements and inland lakes are possible, but hard
875             // to capture with the contiguity test, so just allow the
876             // search to proceed.
877             path = searchMap(unit, start, gd, costDecider,
878                              INFINITY, carrier, sh, lb);
879
880         } else if (unit != null && unit.isOnCarrier()
881             && !start.isLand() && end.isLand()
882             && !start.getContiguityAdjacent(end.getContiguity()).isEmpty()) {
883             // Special case where a land unit is trying to move off a
884             // ship to adjacent land.
885             path = searchMap(unit, start, gd, costDecider, INFINITY,
886                              carrier, sh, lb);
887
888         } else if (start.isLand() && !end.isLand()
889             && end.getFirstUnit() != null
890             && !end.getContiguityAdjacent(start.getContiguity()).isEmpty()
891             && unit != null && unit.getOwner().owns(end.getFirstUnit())
892             && (embarkTo = end.getCarrierForUnit(unit)) != null) {
893             // Special case where a land unit is trying to move from
894             // land to an adjacent ship.
895             path = searchMap(unit, start,
896                 GoalDeciders.getAdjacentLocationGoalDecider(end), costDecider,
897                 INFINITY, null, null, lb);
898             if (path != null) {
899                 PathNode last = path.getLastNode();
900                 last.next = new PathNode(embarkTo, 0, last.getTurns()+1, true,
901                                          last, null);
902             }
903         } else { // Otherwise, there is a connectivity failure.
904             path = null;
905         }
906         return path;
907     }

918             // Add the turns remaining on the high seas.
