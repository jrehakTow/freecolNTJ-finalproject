262     private class RiverTransform extends MapTransform {
263         private final int magnitude;
264
265         private RiverTransform(int magnitude) {
266             this.magnitude = magnitude;
267         }
268
269         @Override
270         public void transform(Tile tile) {
271             TileImprovementType riverType =
272                 tile.getSpecification().getTileImprovementType("model.improvement.river");
273
274             if (riverType.isTileTypeAllowed(tile.getType())
275                 && !tile.hasRiver()) {
276                 String conns = "";
277                 for (Direction direction : Direction.longSides) {
278                     Tile t = tile.getNeighbourOrNull(direction);
279                     TileImprovement otherRiver = (t == null) ? null
280                         : t.getRiver();
281                     conns += (t == null
282                         || (t.isLand() && otherRiver == null)) ? "0"
283                         : Integer.toString(magnitude);
284                 }
285                 tile.addRiver(magnitude, conns);
286             }
287         }
288     }

290     /**
291      * Adds, removes or cycles through the available resources for
292      * this Tile.  Cycles through the ResourceTypeList and picks the
293      * next valid, or removes if end of list.
294      */
295     private class ResourceTransform extends MapTransform {
296         @Override
297         public void transform(Tile t) {
298             // Check if there is a resource already
299             Resource resource = null;
300             if (t.getTileItemContainer() != null) {
301                 resource = t.getTileItemContainer().getResource();
302             }
303             if (resource != null) {
304                 t.getTileItemContainer().removeTileItem(resource);
305             } else {
306                 List<ResourceType> resList = t.getType().getResourceTypes();
307                 switch (resList.size()) {
308                 case 0:
309                     return;
310                 case 1:
311                     ResourceType resourceType = resList.get(0);
312                     // FIXME: create GUI for setting the quantity
313                     t.addResource(new Resource(t.getGame(), t, resourceType,
314                                   resourceType.getMaxValue()));
315                     return;
316                 default:
317                     ResourceType choice = getGUI().getChoice(null, 
318                         Messages.message("mapEditorTransformPanel.chooseResource"),
319                         "cancel",
320                         toList(map(resList, rt ->
321                                 new ChoiceItem<>(Messages.getName(rt), rt))));
322                     if (choice != null) {
323                         t.addResource(new Resource(t.getGame(), t, choice,
324                                       choice.getMaxValue()));
325                     }
326                 }
327             }
328         }

345     private class SettlementTransform extends MapTransform {
346         @Override
347         public void transform(Tile t) {
348             if (!t.isLand()
349                 || t.hasSettlement()
350                 || nativeNation == null) return;
351             UnitType skill = ((IndianNationType)nativeNation.getType())
352                 .getSkills().get(0).getObject();
353             Player nativePlayer = getGame().getPlayerByNation(nativeNation);
354             if (nativePlayer == null) return;
355             String name = nativePlayer.getSettlementName(null);
356             ServerIndianSettlement settlement
357                 = new ServerIndianSettlement(t.getGame(),
358                     nativePlayer, name, t, false, skill, null);
359             nativePlayer.addSettlement(settlement);
360             settlement.placeSettlement(true);
361             settlement.addUnits(null);
362             logger.info("Add settlement " + settlement.getName()
363                 + " to tile " + t);
364         }
