326     /**
327      * Gets a list of goods required to change from the first role to
328      * the second.  The first role may be <code>null</code> implying
329      * the default role, the second must not.  Note that excess goods
330      * that are left over after the change will appear on the list
331      * with negative amounts.
332      *
333      * @param from The current <code>Role</code>.
334      * @param fromCount The role count for the current role.
335      * @param to The new <code>Role</code> to assume.
336      * @param toCount The role count for the new role.
337      * @return A list of <code>AbstractGoods</code> required to
338      *     make the change.
339      */
340     public static List<AbstractGoods> getGoodsDifference(Role from,
341         int fromCount, Role to, int toCount) {
342         List<AbstractGoods> result = new ArrayList<>();
343         if (from != to && !(from == null && to.isDefaultRole())) {
344             List<AbstractGoods> fromGoods = (from == null)
345                 ? new ArrayList<AbstractGoods>()
346                 : from.getRequiredGoods(fromCount);
347             List<AbstractGoods> toGoods = to.getRequiredGoods(toCount);
348             for (AbstractGoods ag : toGoods) {
349                 int amount = ag.getAmount()
350                     - AbstractGoods.getCount(ag.getType(), fromGoods);
351                 if (amount != 0) {
352                     result.add(new AbstractGoods(ag.getType(), amount));
353                 }
354             }
355             result.addAll(transform(fromGoods,
356                     ag -> !AbstractGoods.containsType(ag.getType(), toGoods),
357                     ag -> new AbstractGoods(ag.getType(), -ag.getAmount()),
358                     Collectors.toList()));
359         }
360         return result;
361     }
