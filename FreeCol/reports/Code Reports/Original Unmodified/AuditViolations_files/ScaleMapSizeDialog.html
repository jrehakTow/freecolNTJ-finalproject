114     /**
115      * Force the text fields to contain non-negative integers.
116      */
117     private void checkFields() {
118         try {
119             int w = Integer.parseInt(inputWidth.getText());
120             if (w <= 0) throw new NumberFormatException();
121         } catch (NumberFormatException nfe) {
122             inputWidth.setText(Integer.toString(oldMap.getWidth()));
123         } 
124         try {
125             int h = Integer.parseInt(inputHeight.getText());
126             if (h <= 0) throw new NumberFormatException();
127         } catch (NumberFormatException nfe) {
128             inputHeight.setText(Integer.toString(oldMap.getHeight()));
129         } 
130     }
