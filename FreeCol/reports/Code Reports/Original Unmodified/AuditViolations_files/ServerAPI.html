156     /**

163         final Connection c = getConnection();

167             } catch (IOException ioe) {
168                 logger.log(Level.WARNING, "Could not resolve: " + e.getTagName(),
169                     ioe);
170                 break;
171             }
172         }

194     /**
195      * Sends the specified message to the server and returns the reply,
196      * if it has the specified tag.
197      *
198      * Handle error replies if they have a messageId or when in debug mode.
199      * This routine allows code simplification in much of the following
200      * client-server communication.
201      *
202      * In following routines we follow the convention that server I/O
203      * is confined to the ask<em>foo</em>() routine, which typically
204      * returns true if the server interaction succeeded, which does
205      * *not* necessarily imply that the actual substance of the
206      * request was allowed (e.g. a move may result in the death of a
207      * unit rather than actually moving).
208      *
209      * @param game The current <code>Game</code>.
210      * @param message A <code>DOMMessage</code> to send.
211      * @param tag The expected tag.
212      * @return The answer from the server if it has the specified tag,
213      *     otherwise <code>null</code>.
214      */
215     private Element askExpecting(Game game, DOMMessage message, String tag) {
216         Element reply = ask(message);
217         if (reply == null) return null;
218
219         if (MultipleMessage.TAG.equals(reply.getTagName())) {
220             // Process multiple returns, pick out expected element if
221             // present and continue processing the reply.
222             MultipleMessage mm = new MultipleMessage(game, reply);
223             Element e = mm.extract(tag);
224             resolve(handle(e));
225             if (mm == null) return null;
226             reply = mm.toXMLElement();
227         }
228
229         if (ErrorMessage.TAG.equals(reply.getTagName())) {
230             // Shortcut error processing
231             handle(reply);
232             return null;
233         }
234
235         if (tag == null || tag.equals(reply.getTagName())) {
236             // Success.  Do the standard processing.
237             doClientProcessingFor(reply);
238             return reply;
239         }
240
241         // Unexpected reply.  Whine and fail.
242         logger.warning("Received reply with tag " + reply.getTagName()
243             + " which should have been " + tag
244             + " to message " + message);
245         return null;
246     }

248     /**
249      * Extends askExpecting to also handle returns from the server.
250      *
251      * @param game The current <code>Game</code>.
252      * @param message A <code>DOMMessage</code> to send.
253      * @param tag The expected tag
254      * @return True if the server interaction succeeded and an element
255      *     with the expected tag was found in the reply, else false.
256      */
257     private boolean askHandling(Game game, DOMMessage message, String tag) {
258         Element reply = askExpecting(game, message, tag);
259         if (reply == null) return false;
260         resolve(handle(reply));
261         return true;
262     }
