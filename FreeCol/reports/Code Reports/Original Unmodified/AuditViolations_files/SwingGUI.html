640             try {
641                 final String panelName = "net.sf.freecol.client.gui.panel."
642                     + lastPart(className, ".");
643                 Class<?> controls = Class.forName(panelName);
644                 mapControls = (MapControls)controls
645                     .getConstructor(FreeColClient.class)
646                     .newInstance(getFreeColClient());
647                 logger.info("Instantiated " + panelName);
648             } catch (Exception e) {
649                 logger.log(Level.INFO, "Fallback to CornerMapControls from "
650                     + className, e);
651                 mapControls = new CornerMapControls(getFreeColClient());
652             }
653             if (mapControls != null) {
654                 mapControls.addToComponent(canvas);
655                 mapControls.update();
656             }
657         } else if (!enable && mapControls != null) {
658             mapControls.removeFromComponent(canvas);
659             mapControls = null;
660         }
661     }
