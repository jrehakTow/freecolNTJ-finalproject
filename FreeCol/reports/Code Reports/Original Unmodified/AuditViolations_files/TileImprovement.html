535      */
536     @Override
537     public int checkIntegrity(boolean fix) {
538         int result = super.checkIntegrity(fix);
539         // @compat 0.10.x
540         // We check only if this improvement is not connected to a
541         // neighbour that *is* connected to this one, and connect this
542         // one.
543         //
544         // FIXME: drop this one day when we never have style
545         // discontinuities.  This alas is not the case in 0.10.x.
546         String curr = (style == null) ? null : style.getString();
547         String found = (isRiver()) ? updateRiverConnections(curr)
548             : (isRoad() && isComplete()) ? updateRoadConnections(true)
549             : null;
550         if ((found == null && curr == null)
551             || (found != null && curr != null && found.equals(curr))) {
552             result = Math.min(1, result);
553         } else if (fix) {
554             this.style = TileImprovementStyle.getInstance(found);
555             if ((this.style != null)
556                 != (isRiver() || (isRoad() && isComplete()))) {
557                 logger.warning("Bad style for improvement: " + this);
558                 result = -1;
559             } else {
560                 logger.warning("Fixing improvement style from "
561                     + curr + " to " + found + " at " + tile);
562                 result = Math.min(0, result);
563             }
564         } else {
565             logger.warning("Broken improvement style " + curr
566                 + " should be " + found + " at " + tile);
567             result = -1;
568         }
569         // end @compat 0.10.x
570         return result;
571     }
