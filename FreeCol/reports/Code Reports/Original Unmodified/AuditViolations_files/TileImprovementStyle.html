26 /**
27  * Represents the style of a tile improvement, such as a river or
28  * road. Since TileImprovementStyles are immutable and some styles are
29  * far more common than others (e.g. rivers with two branches are far
30  * more common than rivers with a single branch, or three or four
31  * branches), the class caches all styles actually used. This should
32  * also speed up reading maps with old base-three styles.
33  *
34  * This is the documentation for the old style (&le; 0.10.5):
35  *   Tile improvement styles should be encoded as a string of base-36
36  *   values, one for each of the directions the improvement might be
37  *   connected to (counting clockwise from N). In the case of rivers,
38  *   which can only be connected via the "long sides" (NE, SE, SW, NW),
39  *   possible values range from "0000" to "zzzz". Improvement styles
40  *   that use only four directions can add up to three characters of
41  *   arbitrary style information (such as "%$#", for example).
42  *   Improvement styles that use eight directions can add an arbitrary
43  *   amount of style information.
44  * As of 0.10.6 we use:
45  *   - Four character encoded strings for rivers: a "0" for no connection,
46  *     otherwise the string value of the integer magnitude of the river
47  *     for each of Direction.longSides.
48  *   - Eight character binary encoded strings for roads: a "0" or "1" for
49  *     each of Direction.values()
50  *   These are distinct so that the overlays can vary.
51  */

65      * Private constructor, only called in getInstance() below.
66      *
67      * @param style The (decoded) style.
68      */
69     private TileImprovementStyle(String style) {
70         this.style = style;
71
72         StringBuilder sb = new StringBuilder();
73         for (int i = 0; i < style.length(); i++) {
74             char c = style.charAt(i);
75             if (Character.digit(c, Character.MAX_RADIX) < 0) break;
76             sb.append((c == '0') ? "0" : "1");
77         }
78         this.mask = sb.toString();
79     }

103     /**
104      * Decode the old base-3 encoded style format.
105      *
106      * @param input An old style string.
107      * @param pad Pad the result to this length.
108      * @return The style in the new format.
109      */
110     public static String decodeOldStyle(String input, int pad) {
111         if (pad <= 0) return null;
112         boolean isZero = true;
113         StringBuilder sb = new StringBuilder();
114         try {
115             int value = Integer.parseInt(input);
116             for (int index = 0; index < 4; index++) {
117                 int magnitude = value % 3;
118                 isZero &= magnitude == 0;
119                 sb.append(Integer.toString(magnitude, Character.MAX_RADIX));
120                 value /= 3;
121             }
122         } catch (NumberFormatException nfe) {
123             return null;
124         }
125         if (isZero) return null;
126         while (sb.length() < pad) {
127             sb.append("0");
128         }
129         return sb.toString();
130     }
131     // end @compat

134      * Gets the style corresponding to the given string.
135      *
136      * @param key The key to look up.
137      * @return The corresponding <code>TileImprovementStyle</code>.
138      */
139     public static TileImprovementStyle getInstance(String key) {
140         if (key == null || key.isEmpty() || "0".equals(key)) return null;
141
142         TileImprovementStyle result = cache.get(key);
143         if (result == null) {
144             result = new TileImprovementStyle(key);
145             cache.put(key, result);
146             if (!result.getString().equals(key)) {
147                 cache.put(result.getString(), result);
148             }
149         }
150         return result;
151     }
