40 /**
41  * Listens to mouse buttons being pressed at the level of the Canvas.
42  */
43 public final class CanvasMouseListener extends FreeColClientHolder
44     implements ActionListener, MouseListener {
45
46     private static final Logger logger = Logger.getLogger(CanvasMouseListener.class.getName());
47
48     private static final int doubleClickDelay = 200; // Milliseconds
49
50     private final Canvas canvas;
51
52     private final Timer doubleClickTimer = new Timer(doubleClickDelay,this);
53
54     private int centerX, centerY;

57     /**
58      * Create a new canvas mouse listener.
59      *
60      * @param freeColClient The enclosing <code>FreeColClient</code>.
61      * @param canvas The component this object gets created for.
62      */
63     public CanvasMouseListener(FreeColClient freeColClient, Canvas canvas) {
64         super(freeColClient);
65
66         this.canvas = canvas;
67     }

69     /**
70      * Invoked when a mouse button was clicked.
71      *
72      * @param e The MouseEvent that holds all the information.
73      */
74     @Override
75     public void mouseClicked(MouseEvent e) {
76         try {
77             if (e.getClickCount() > 1) {
78                 Tile tile = canvas.convertToMapTile(e.getX(), e.getY());
79                 Colony colony = tile.getColony();
80                 if (colony != null) {
81                     if (FreeColDebugger.isInDebugMode(FreeColDebugger.DebugMode.MENUS)) {
82                         canvas.showForeignColony(colony);
83                     } else {
84                         canvas.showColonyPanel(colony, null);
85                     }
86                 }
87             } else {
88                 canvas.requestFocus();
89             }
90         } catch (Exception ex) {
91             logger.log(Level.WARNING, "Error in mouseClicked!", ex);
92         }
93     }
94
95     /**
96      * Invoked when the mouse enters the component.
97      *
98      * @param e The MouseEvent that holds all the information.
99      */
100     @Override
101     public void mouseEntered(MouseEvent e) {
102         // Ignore for now.
103     }

105     /**
106      * Invoked when the mouse exits the component.
107      *
108      * @param e The MouseEvent that holds all the information.
109      */
110     @Override
111     public void mouseExited(MouseEvent e) {
112         // Ignore for now.
113     }

115     /**
116      * Invoked when a mouse button was pressed.
117      *
118      * @param e The MouseEvent that holds all the information.
119      */
120     @Override
121     public void mousePressed(MouseEvent e) {
122         if (!e.getComponent().isEnabled()) return;
123
124         int me = e.getButton();
125         if (e.isPopupTrigger()) me = MouseEvent.BUTTON3;
126         Tile tile = canvas.convertToMapTile(e.getX(), e.getY());
127
128         switch (me) {
129         case MouseEvent.BUTTON1:
130             // Record initial click point for purposes of dragging
131             canvas.setDragPoint(e.getX(), e.getY());
132             if (canvas.isGotoStarted()) {
133                 PathNode path = canvas.getGotoPath();
134                 if (path != null) {
135                     canvas.stopGoto();
136                     // Move the unit
137                     igc().goToTile(canvas.getActiveUnit(),
138                                    path.getLastNode().getTile());
139                 }
140             } else if (doubleClickTimer.isRunning()) {
141                 doubleClickTimer.stop();
142             } else {
143                 centerX = e.getX();
144                 centerY = e.getY();
145                 doubleClickTimer.start();
146             }
147             canvas.requestFocus();
148             break;
149         case MouseEvent.BUTTON2:
150             if (tile != null) {
151                 Unit unit = canvas.getActiveUnit();
152                 if (unit != null && unit.getTile() != tile) {
153                     PathNode dragPath = unit.findPath(tile);
154                     canvas.startGoto();
155                     canvas.setGotoPath(dragPath);
156                 }
157             }
158             break;
159         case MouseEvent.BUTTON3:
160             // Cancel goto if one is active
161             if (canvas.isGotoStarted()) canvas.stopGoto();
162             canvas.showTilePopup(tile, e.getX(), e.getY());
163             break;
164         default:
165             break;
166         }
167     }
168
169     /**
170      * Invoked when a mouse button was released.
171      *
172      * @param e The MouseEvent that holds all the information.
173      */
174     @Override
175     public void mouseReleased(MouseEvent e) {
176         try {
177             if (canvas.getGotoPath() != null) { // A mouse drag has ended.
178                 PathNode temp = canvas.getGotoPath();
179                 canvas.stopGoto();
180
181                 igc().goToTile(canvas.getActiveUnit(),
182                                temp.getLastNode().getTile());
183
184             } else if (canvas.isGotoStarted()) {
185                 canvas.stopGoto();
186             }
187         } catch (Exception ex) {
188             logger.log(Level.WARNING, "Error in mouseReleased!", ex);
189         }
190     }

195     /**
196      * {@inheritDoc}
197      */
198     @Override
199     public void actionPerformed(ActionEvent ae) {
200         doubleClickTimer.stop();
201         Tile tile=canvas.convertToMapTile(centerX, centerY);
202         if(canvas.getViewMode() == GUI.MOVE_UNITS_MODE) {
203             // Clear goto order when active unit is on the tile
204             Unit unit=canvas.getActiveUnit();
205             if(unit != null && unit.getTile() == tile) {
206                 igc().clearGotoOrders(unit);
207                 canvas.updateCurrentPathForActiveUnit();
208             } else {
209                 if (tile != null && tile.hasSettlement()) {
210                     getGUI().showSettlement(tile.getSettlement());
211                     return;
212                 }
213             }
214         }
215         getGUI().setSelectedTile(tile);
216     }
