550      * Update all the production-related panels.
551      *
552      * This has to be very broad as a change at one work location can
553      * have a secondary effect on another, and especially if the
554      * population hits a bonus boundary.  A simple example is that
555      * adding extra lumber production may improve the production of the
556      * lumber mill.  These changes can then flow on to production and
557      * construction displays.

559     private void updateProduction() {

561         if (colony == null) return;

564         for (WorkLocation wl : colony.getCurrentWorkLocations()) {
565             boolean change = false, check = wl.getProductionType() == null;
566             for (Unit u : wl.getUnitList()) {
567                 if (check || !wl.produces(u.getWorkType())) {
568                     GoodsType workType = wl.getWorkFor(u);
569                     if (workType != null && workType != u.getWorkType()) {
570                         igc().changeWorkType(u, workType);
571                     }
572                     change = true;
573                 }
574             }
575             if (change) wl.updateProductionType();
576         }
577
578         updateTilesPanel();
579         updateBuildingsPanel();
580         updateNetProductionPanel();
581         updateConstructionPanel();
