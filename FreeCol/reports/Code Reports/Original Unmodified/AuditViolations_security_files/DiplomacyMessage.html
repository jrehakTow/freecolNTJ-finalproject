190      * Handle a "diplomacy"-message.
191      *
192      * @param server The <code>FreeColServer</code> that handles the message.
193      * @param connection The <code>Connection</code> the message is from.
194      * @return An <code>Element</code> describing the trade with
195      *     either "accept" or "reject" status, null on trade failure,
196      *     or an error <code>Element</code> on outright error.
197      */
198     public Element handle(FreeColServer server, Connection connection) {
199         final ServerPlayer serverPlayer = server.getPlayer(connection);
200         final Game game = serverPlayer.getGame();
201
202         if (this.agreement == null) {
203             return serverPlayer.clientError("Null diplomatic agreement.")
204                 .build(serverPlayer);
205         }
206
207         Unit ourUnit = null;
208         Colony ourColony = null;
209         FreeColGameObject our = getOurFCGO(game);
210         if (our == null) {
211             return serverPlayer.clientError("Missing our object: "
212                 + this.ourId)
213                 .build(serverPlayer);
214         } if (our instanceof Unit) {
215             ourUnit = (Unit)our;
216             if (!serverPlayer.owns(ourUnit)) {
217                 return serverPlayer.clientError("Not our unit: " + this.ourId)
218                     .build(serverPlayer);
219             } else if (!ourUnit.hasTile()) {
220                 return serverPlayer.clientError("Our unit is not on the map: "
221                     + this.ourId)
222                     .build(serverPlayer);
223             }
224         } else if (our instanceof Colony) {
225             ourColony = (Colony)our;
226             if (!serverPlayer.owns(ourColony)) {
227                 return serverPlayer.clientError("Not our settlement: "
228                     + this.ourId)
229                     .build(serverPlayer);
230             }
231         } else {
232             return serverPlayer.clientError("Our object is bogus: " + our)
233                 .build(serverPlayer);
234         }
235
236         Unit otherUnit = null;
237         Colony otherColony = null;
238         Player otherPlayer = null;
239         FreeColGameObject other = getOtherFCGO(game);
240         if (other == null) {
241             return serverPlayer.clientError("Missing other object: "
242                 + this.otherId)
243                 .build(serverPlayer);
244         } else if (other instanceof Unit) {
245             otherUnit = (Unit)other;
246             if (serverPlayer.owns(otherUnit)) {
247                 return serverPlayer.clientError("Contacting our unit? "
248                     + this.otherId)
249                     .build(serverPlayer);
250             } else if (!otherUnit.hasTile()) {
251                 return serverPlayer.clientError("Other unit is not on the map: "
252                     + this.otherId)
253                     .build(serverPlayer);
254             } else if (ourUnit != null
255                 && !ourUnit.getTile().isAdjacent(otherUnit.getTile())) {
256                 return serverPlayer.clientError("Our unit " + this.ourId
257                     + " is not adjacent to other unit " + this.otherId)
258                     .build(serverPlayer);
259             } else if (ourColony != null
260                 && !ourColony.getTile().isAdjacent(otherUnit.getTile())) {
261                 return serverPlayer.clientError("Our colony " + this.ourId
262                     + " is not adjacent to other unit " + this.otherId)
263                     .build(serverPlayer);
264             }
265             otherPlayer = otherUnit.getOwner();
266         } else if (other instanceof Colony) {
267             otherColony = (Colony)other;
268             if (serverPlayer.owns(otherColony)) {
269                 return serverPlayer.clientError("Contacting our colony? "
270                     + this.otherId)
271                     .build(serverPlayer);
272             } else if (ourUnit != null
273                 && !ourUnit.getTile().isAdjacent(otherColony.getTile())) {
274                 return serverPlayer.clientError("Our unit " + this.ourId
275                     + " is not adjacent to other colony " + this.otherId)
276                     .build(serverPlayer);
277             } else if (ourColony != null
278                 && !ourColony.getTile().isAdjacent(otherColony.getTile())) {
279                 return serverPlayer.clientError("Our colony " + this.ourId
280                     + " is not adjacent to other colony " + this.otherId)
281                     .build(serverPlayer);
282             }
283             otherPlayer = otherColony.getOwner();
284         } else {
285             return serverPlayer.clientError("Other object is bogus: " + other)
286                 .build(serverPlayer);
287         }
288         if (ourUnit == null && otherUnit == null) {
289             return serverPlayer.clientError("Both units null")
290                 .build(serverPlayer);
291         }
292
293         Player senderPlayer = this.agreement.getSender();
294         Player recipientPlayer = this.agreement.getRecipient();
295         Player refPlayer = serverPlayer.getREFPlayer();
296         if (senderPlayer == null) {
297             return serverPlayer.clientError("Null sender in agreement.")
298                 .build(serverPlayer);
299         } else if (recipientPlayer == null) {
300             return serverPlayer.clientError("Null recipient in agreement.")
301                 .build(serverPlayer);
302         } else if (senderPlayer != (Player)serverPlayer) {
303             return serverPlayer.clientError("Sender is not our player: "
304                 + senderPlayer.getId())
305                 .build(serverPlayer);
306         } else if (recipientPlayer != otherPlayer) {
307             return serverPlayer.clientError("Recipient is not other player: "
308                 + recipientPlayer.getId())
309                 .build(serverPlayer);
310         } else if (senderPlayer == refPlayer || recipientPlayer == refPlayer) {
311             return serverPlayer.clientError("The REF does not negotiate: "
312                 + refPlayer.getId())
313                 .build(serverPlayer);
314         }
315
316         final InGameController igc = server.getInGameController();
317         ChangeSet cs = null;
318         switch (this.agreement.getContext()) {
319         case CONTACT:
320             cs = (ourColony != null)
321                 ? igc.europeanFirstContact(serverPlayer, ourColony, otherUnit,
322                                            this.agreement)
323                 : (otherUnit != null)
324                 ? igc.europeanFirstContact(serverPlayer, ourUnit, otherUnit,
325                                            this.agreement)
326                 : igc.europeanFirstContact(serverPlayer, ourUnit, otherColony,
327                                            this.agreement);
328             break;
329         case DIPLOMATIC:
330             cs = (ourUnit != null) 
331                 ? ((!ourUnit.hasAbility(Ability.NEGOTIATE))
332                     ? serverPlayer.clientError("Unit lacks ability"
333                         + " to negotiate: " + ourUnit)
334                     : (otherColony == null)
335                     ? serverPlayer.clientError("Null other settlement")
336                     : igc.diplomacy(serverPlayer, ourUnit, otherColony,
337                                     this.agreement))
338                 : ((!otherUnit.hasAbility(Ability.NEGOTIATE))
339                     ? serverPlayer.clientError("Unit lacks ability"
340                         + " to negotiate: " + otherUnit)
341                     : igc.diplomacy(serverPlayer, ourColony, otherUnit,
342                                     this.agreement));
343             break;
344         case TRADE:
345             cs = (ourUnit != null)
346                 ? ((!ourUnit.isCarrier())
347                     ? serverPlayer.clientError("Unit is not a carrier: "
348                         + ourUnit)
349                     : (!serverPlayer.hasAbility(Ability.TRADE_WITH_FOREIGN_COLONIES))
350                     ? serverPlayer.clientError("Player lacks ability"
351                         + " to trade with other Europeans: " + serverPlayer)
352                     : (otherColony == null)
353                     ? serverPlayer.clientError("Null other settlement")
354                     : igc.diplomacy(serverPlayer, ourUnit, otherColony,
355                                     this.agreement))
356                 : ((!otherUnit.isCarrier())
357                     ? serverPlayer.clientError("Unit is not a carrier: "
358                         + otherUnit)
359                     : (!otherPlayer.hasAbility(Ability.TRADE_WITH_FOREIGN_COLONIES))
360                     ? serverPlayer.clientError("Player lacks ability"
361                         + " to trade with other Europeans: " + otherPlayer)
362                     : igc.diplomacy(serverPlayer, ourColony, otherUnit,
363                                     this.agreement));
364             break;
365         case TRIBUTE:
366             cs = (ourUnit != null)
367                 ? ((!ourUnit.isOffensiveUnit() || ourUnit.isNaval())
368                     ? serverPlayer.clientError("Unit is not an offensive"
369                         + " land unit: " + ourUnit)
370                     : (otherColony == null)
371                     ? serverPlayer.clientError("Null other settlement")
372                     : igc.diplomacy(serverPlayer, ourUnit, otherColony,
373                                     this.agreement))
374                 : ((!otherUnit.isOffensiveUnit() || otherUnit.isNaval())
375                     ? serverPlayer.clientError("Unit is not an offensive"
376                         + " land unit: " + otherUnit)
377                     : igc.diplomacy(serverPlayer, ourColony, otherUnit,
378                                     this.agreement));
379             break;
380         default:
381             break;
382         }
383         if (cs == null) cs = serverPlayer.clientError("Invalid diplomacy for "
384             + this.agreement.getContext());
385         return cs.build(serverPlayer);
386     }
