515         case PER_SALTIRE:
516             drawPerSaltire(g);
517             break;
518         default:
519             break;
520         }
521
522         // draw decoration
523         GeneralPath decorationShape = null;
524         switch(decoration) {
525         case GREEK_CROSS:
526         case CROSS:
527         case SCANDINAVIAN_CROSS:
528             decorationShape = getCross(decoration);
529             break;
530         case CHEVRON:
531             decorationShape = getTriangle(UnionShape.CHEVRON, false);
532             break;
533         case PALL:
534             decorationShape = getPall();
535             break;
536         case BEND:
537             decorationShape = getBend(false);
538             break;
539         case BEND_SINISTER:
540             decorationShape = getBend(true);
541             break;
542         case SALTIRE:
543             decorationShape = getBend(true);
544             decorationShape.append(getBend(false), false);
545             break;
546         case SALTIRE_AND_CROSS:
547             decorationShape = getBend(true);
548             decorationShape.append(getBend(false), false);
549             decorationShape.append(getCross(Decoration.CROSS), false);
550             break;
551         default:
552             break;
553         }
554         if (decorationShape != null) {
555             g.setColor(decorationColor);
556             g.fill(decorationShape);
557         }
558
559         if (unionPosition == null
560             || unionPosition == UnionPosition.NONE) {
561             return image;
562         }

569         }

574             starShape = getUnionRectangle(rectangle);

576         case CHEVRON:
577             union = getTriangle(unionShape, decoration == Decoration.PALL);
578             starShape = getUnionTriangle(true);
579             break;
580         case BEND:

582                                 || (decoration == Decoration.BEND_SINISTER));
583             starShape = getUnionTriangle(false);

590             break;
591         case TRIANGLE:
592             union = getTriangle(unionShape, decoration == Decoration.SALTIRE);
593             starShape = getUnionTriangle(true);
594             transformTriangle(union);
595             transformTriangle(starShape);
596             break;

599         }

603         }

606             g.fill(starShape);
607         }

637     private GeneralPath getUnionTriangle(boolean isosceles) {
638         boolean small = (decoration == Decoration.PALL
639                          || decoration == Decoration.BEND
640                          || decoration == Decoration.BEND_SINISTER);
641
642         double x = 0;
643         double y = 0;
644         double r = 0;
645         if (unionShape == UnionShape.CHEVRON) {
646             x = CHEVRON_X;
647             y = HEIGHT;
648             if (small) {
649                 x -= BEND_X;
650                 y -= 2 * BEND_Y;
651             }
652             r = SQRT_3 * y / 6;
653         } else if (unionShape == UnionShape.TRIANGLE) {
654             if (unionPosition == UnionPosition.LEFT
655                 || unionPosition == UnionPosition.RIGHT) {
656                 x = WIDTH / 2;
657                 y = HEIGHT;
658                 if (small) {
659                     x -= BEND_X;
660                     y -= 2 * BEND_Y;
661                 }
662                 r = SQRT_3 * y / 6;
663             } else {
664                 x = HEIGHT / 2;
665                 y = WIDTH;
666                 if (small) {
667                     x -= BEND_Y;
668                     y -= 2 * BEND_X;
669                 }
670                 double h = y / 2;
671                 double c = Math.sqrt(h * h + x * x);
672                 r = x * y / (y + 2 * c);
673             }
674         } else {
675             // union shape is bend
676             x = WIDTH;
677             y = HEIGHT;
678             if (small) {
679                 x -= BEND_X;
680                 y -= BEND_Y;
681             }
682             double c = Math.sqrt(x * x + y * y);
683             double A = x * y / 2;
684             r = 2 * A / (x + y + c);
685         }
686         // leave a margin
687         double radius = r * 0.6;
688
689         GeneralPath unionPath = new GeneralPath();
690         if (stars < 14) {
691             unionPath = getCircleOfStars(radius);
692             if (isosceles) {
693                 if (unionPosition == UnionPosition.LEFT
694                     || unionPosition == UnionPosition.RIGHT) {
695                     center(unionPath, r, HEIGHT / 2);
696                 } else {
697                     center(unionPath, r, WIDTH / 2);
698                 }
699             } else {
700                 center(unionPath, r, r);
701             }
702         } else {
703
704             int sum = 0;
705             int rows = 1;
706             while (sum < stars) {
707                 sum += rows;
708                 rows++;
709             }
710             int missing = sum - stars;
711             double slope = y / x;
712             double dx = x / rows;
713             double xx = dx / 2;
714             double height = y - xx * slope;
715             double offset = 0;
716             rows--;
717             double dy = height / rows;
718             for (int index = rows; index > 0; index--) {
719                 if (isosceles) {
720                     height = y - xx * slope;
721                     dy = height / index;
722                     offset = (HEIGHT - height) / 2;
723                 }
724                 double yy = dy / 2;
725                 int count = index;
726                 if (missing > 0) {
727                     count = index - missing;
728                     yy += missing / 2 * dy;
729                     missing = 0;
730                 }
731                 for (int star = 0; star < count; star++) {
732                     unionPath.append(getStar(xx, yy + offset), false);
733                     yy += dy;
734                 }
735                 xx += dx;
736             }
737         }
738         return unionPath;
739     }

794     private GeneralPath getUnionRhombus() {
795         GeneralPath unionPath = new GeneralPath();
796
797         int count = 1;
798         int square = 1;
799         while (square < stars) {
800             count++;
801             square = count * count;
802         }
803         int rows = stars / count;
804         if (rows * count < stars) rows++;
805         int starCount = 0;
806         double a = WIDTH / 2 - BEND_X;
807         double b = HEIGHT / 2 - BEND_Y;
808
809         if (stars < 14) {
810             double c = Math.sqrt(a * a + b * b);
811             double r = (a * b) / c;
812             double radius = 0.6 * r;
813             unionPath = getCircleOfStars(radius);
814             center(unionPath, WIDTH / 2, HEIGHT / 2);
815         } else {
816             double dx = a / count;
817             double dy = b / count;
818             double dx1 = a / rows;
819             double dy1 = b / rows;
820             outer: for (int index = 0; index < rows; index++) {
821                 double x = BEND_X + dx + index * dx1;
822                 double y = HEIGHT / 2 - index * dy1;
823                 for (int star = 0; star < count; star++) {
824                     unionPath.append(getStar(x, y), false);
825                     starCount++;
826                     if (starCount == stars) {
827                         break outer;
828                     } else {
829                         x += dx;
830                         y += dy;
831                     }
832                 }
833             }
834         }
835
836         return unionPath;
837     }
