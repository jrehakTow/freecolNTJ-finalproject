43 /**

92     /**
93      * Returns a <code>ListCellRenderer</code> for the given
94      * <code>JList</code>.
95      *
96      * @param list The <code>JList</code>.
97      * @param value The list cell.
98      * @param index The index in the list.
99      * @param isSelected <code>true</code> if the given list cell is selected.
100      * @param hasFocus <code>false</code> if the given list cell has the focus.
101      * @return The <code>ListCellRenderer</code>
102      */
103     @Override
104     public Component getListCellRendererComponent(JList<? extends T> list,
105                                                   T value, int index,
106                                                   boolean isSelected,
107                                                   boolean hasFocus) {
108         JLabel c = isSelected ? SELECTED_COMPONENT : NORMAL_COMPONENT;
109         c.setForeground(list.getForeground());
110         c.setFont(list.getFont());
111         setLabelValues(c, value);
112         return c;
113     }

115     /**
116      * Set the text to place in a label.
117      *
118      * @param c The <code>JLabel</code> to assign.
119      * @param value The object to derive a text value from.
120      */
121     public void setLabelValues(JLabel c, T value) {
122         if (value == null) {
123             c.setText(null);
124         } else if (value instanceof Integer) {
125             // partial load values from SelectAmountDialog are Integers
126             c.setText(value.toString());
127         } else if (value instanceof Language) {
128             c.setText(value.toString());
129         } else if (value instanceof String) {
130             String string = (String)value;
131             if (localize) {
132                 String[] nd = Messages.getBestNameAndDescription(string);
133                 c.setText(nd[0]);
134                 if (nd[1] != null) c.setToolTipText(nd[1]);
135             } else {
136                 c.setText(string);
137             }
138         } else if (value instanceof Named) {
139             Named named = (Named)value;
140             String[] nad = Messages.getBestNameAndDescription(named);
141             c.setText(nad[0]);
142             if (nad[1] != null) c.setToolTipText(nad[1]);
143         } else if (value instanceof ObjectWithId) {
144             String id = ((prefix == null) ? "" : prefix)
145                 + ((ObjectWithId)value).getId();
146             String[] nd = Messages.getBestNameAndDescription(id);
147             if (value instanceof Nameable) {
148                 String realname = ((Nameable)value).getName();
149                 if (realname != null) nd[0] = realname;
150             }
151             c.setText(nd[0]);
152             if (nd[1] != null) c.setToolTipText(nd[1]);
153         } else {
154             logger.warning("What is this?: " + value
155                 + " (" + value.getClass() + ")");
156         }
157     }
