61      * all of the submenus and items.

64      */
65     protected FreeColMenuBar(FreeColClient f) {
66         // FIXME: FreeColClient should not have to be passed in to
67         // this class.  This is only a menu bar, it doesn't need a
68         // reference to the main controller. The only reason it has
69         // one now is because DebugMenu needs it. And DebugMenu needs
70         // it because it is using inner classes for ActionListeners
71         // and those inner classes use the reference.  If those inner
72         // classes were in seperate classes, when they were created,
73         // they could use the FreeColClient reference of the
74         // ActionManger. So DebugMenu needs to be refactored to remove
75         // inner classes so that this MenuBar can lose its unnecessary
76         // reference to the main controller.  See FreeColMenuTest.
77         //
78         // Okay, I lied.. the update() and paintComponent() methods in
79         // this MenuBar use freeColClient, too. But so what.  Move
80         // those to another class too. :)
81         super();
82

84
85         this.freeColClient = f;

110     protected JMenuItem getMenuItem(String actionId) {
111         JMenuItem rtn = null;
112         FreeColAction action = am.getFreeColAction(actionId);
113
114         if (action != null) {
115             rtn = new JMenuItem();
116             rtn.setAction(action);
117             rtn.setOpaque(false);
118
119             if (action.getMnemonic() != null) {
120                 rtn.addMenuKeyListener(action.getMenuKeyListener());
121             }
122         } else {
123             logger.finest("Could not create menu item. [" + actionId
124                 + "] not found.");
125         }
126         return rtn;
127     }
