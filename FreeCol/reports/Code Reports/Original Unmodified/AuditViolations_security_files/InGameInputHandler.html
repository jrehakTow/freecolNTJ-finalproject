633     /**
634      * Handles a "logout"-message.
635      * 
636      * @param connection The <code>Connection</code> the message was received
637      *            on.
638      * @param logoutElement The element (root element in a DOM-parsed XML tree)
639      *            that holds all the information.
640      * @return The reply.
641      */
642     @Override
643     protected Element logout(Connection connection, Element logoutElement) {
644         ServerPlayer serverPlayer = getFreeColServer().getPlayer(connection);
645         if (serverPlayer == null) return null;
646         logger.info("Logout by: " + connection
647                     + ((serverPlayer != null) ? " (" + serverPlayer.getName() + ") " : ""));
648
649         /*
650          * FIXME: Setting the player dead directly should be a server
651          * option, but for now - allow the player to reconnect:
652          */
653         ChangeSet cs = null;
654         serverPlayer.setConnected(false);
655         if (getFreeColServer().getGame().getCurrentPlayer() == serverPlayer
656                 && !getFreeColServer().getSinglePlayer()) {
657             cs = getFreeColServer().getInGameController()
658                 .endTurn(serverPlayer);
659         }
660         getFreeColServer().updateMetaServer();
661         return (cs == null) ? null : cs.build(serverPlayer);
662     }
