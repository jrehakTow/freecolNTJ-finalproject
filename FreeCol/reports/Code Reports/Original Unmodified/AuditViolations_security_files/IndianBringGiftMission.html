119         IndianSettlement home;

122             : ((home = aiUnit.getUnit().getHomeIndianSettlement()) == null

128     /**
129      * Why would an IndianBringGiftMission be invalid with the given
130      * unit and colony.
131      *
132      * @param aiUnit The <code>AIUnit</code> to test.
133      * @param colony The <code>Colony</code> to test.
134      * @return A reason why the mission would be invalid with the unit
135      *     and colony or null if none found.
136      */
137     private static String invalidColonyReason(AIUnit aiUnit, Colony colony) {
138         String reason = invalidTargetReason(colony);
139         if (reason != null) return reason;
140         final Unit unit = aiUnit.getUnit();
141         final Player owner = unit.getOwner();
142         Player targetPlayer = colony.getOwner();
143         switch (owner.getStance(targetPlayer)) {
144         case UNCONTACTED: case WAR: case CEASE_FIRE:
145             return "bad-stance";
146         case PEACE: case ALLIANCE:
147             Tension tension = unit.getHomeIndianSettlement()
148                 .getAlarm(targetPlayer);
149             if (tension != null && tension.getLevel()
150                 .compareTo(Tension.Level.HAPPY) > 0) return "unhappy";
151         }
152         return null;
153     }

229     /**
230      * {@inheritDoc}
231      */
232     @Override
233     public Mission doMission(LogBuilder lb) {
234         lb.add(tag);
235         String reason = invalidReason();
236         if (reason != null) return lbFail(lb, false, reason);
237
238         final AIUnit aiUnit = getAIUnit();
239         final Unit unit = getUnit();
240         final IndianSettlement is = unit.getHomeIndianSettlement();
241
242         while (!this.collected) {
243             Unit.MoveType mt = travelToTarget(getTarget(), null, lb);
244             switch (mt) {
245             case MOVE: // Arrived
246                 break;
247
248             case MOVE_HIGH_SEAS: case MOVE_NO_MOVES: case MOVE_ILLEGAL:
249                 return lbWait(lb);
250
251             case MOVE_NO_REPAIR:
252                 return lbFail(lb, false, AIUNITDIED);
253
254             case MOVE_NO_TILE:
255                 return this;
256             
257             case ATTACK_SETTLEMENT: case ATTACK_UNIT: // A blockage!
258                 Location blocker = resolveBlockage(aiUnit, getTarget());
259                 if (blocker != null
260                     && AIMessage.askAttack(aiUnit, unit.getTile()
261                         .getDirection(blocker.getTile()))) {
262                     return lbAttack(lb, blocker);
263                 }
264                 moveRandomly(tag, null);
265                 continue;
266
267             default:
268                 return lbMove(lb, mt);
269             }
270
271             // Load the goods.
272             lbAt(lb);
273             Goods gift = is.getRandomGift(getAIRandom());
274             if (gift == null) return lbFail(lb, false, "found no gift");
275             if (!AIMessage.askLoadGoods(is, gift.getType(), gift.getAmount(),
276                                         aiUnit) || !hasGift()) {
277                 return lbFail(lb, false, "failed to collect gift");
278             }
279             this.collected = true;
280             lb.add(", collected gift");
281             return lbRetarget(lb);
282         }
283
284         // Move to the target's colony and deliver, avoiding trouble
285         // by choice of cost decider.
286         for (;;) {
287             Unit.MoveType mt = travelToTarget(getTarget(),
288                 CostDeciders.avoidSettlementsAndBlockingUnits(), lb);
289             switch (mt) {
290             case MOVE_HIGH_SEAS: case MOVE_NO_MOVES: case MOVE_ILLEGAL:
291                 return lbWait(lb);
292
293             case MOVE_NO_REPAIR:
294                 return lbFail(lb, false, AIUNITDIED);
295
296             case MOVE_NO_TILE:
297                 return this;
298             
299             case MOVE: case ATTACK_SETTLEMENT: // Arrived (do not attack!)
300                 break;
301
302             case ATTACK_UNIT:
303                 Location blocker = resolveBlockage(aiUnit, getTarget());
304                 if (blocker != null
305                     && AIMessage.askAttack(aiUnit, unit.getTile()
306                         .getDirection(blocker.getTile()))) {
307                     return lbAttack(lb, blocker);
308                 }
309                 moveRandomly(tag, null);
310                 continue;
311             
312             default:
313                 return lbMove(lb, mt);
314             }
315         
316             // Deliver the goods.
317             lbAt(lb);
318             Settlement settlement = (Settlement)getTarget();
319             boolean result = false;
320             if (AIMessage.askGetTransaction(aiUnit, settlement)) {
321                 result = AIMessage.askDeliverGift(aiUnit, settlement,
322                     unit.getGoodsList().get(0));
323                 AIMessage.askCloseTransaction(aiUnit, settlement);
324             }
325             return (result)
326                 ? lbDone(lb, false, "delivered")
327                 : lbFail(lb, false, "delivery");
328         }
329     }
