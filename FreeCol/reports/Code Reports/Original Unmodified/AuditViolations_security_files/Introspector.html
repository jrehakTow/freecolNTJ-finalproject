69      * Get a get-method for this Introspector.
70      *

73      */
74     private Method getGetMethod() throws IntrospectorException {
75         String methodName = "get" + field.substring(0, 1).toUpperCase()
76             + field.substring(1);
77
78         try {

80         } catch (NoSuchMethodException | SecurityException e) {

82                 + "." + methodName, e);

127      * Get a function that converts to String from a given class.
128      * We use Enum.name() for enums, and String.valueOf(argType) for the rest.
129      *
130      * @param argType A <code>Class</code> to find a converter for.
131      * @return A conversion function, or null on error.
132      * @exception NoSuchMethodException if no converter is found.
133      */
134     private Method getToStringConverter(Class<?> argType)
135         throws NoSuchMethodException {
136         return (argType.isEnum()) ? argType.getMethod("name")
137             : String.class.getMethod("valueOf", argType);
138     }

140     /**
141      * Get a function that converts from String to a given class.
142      * We use Enum.valueOf(Class, String) for enums, and
143      * argType.valueOf(String) for the rest, having first dodged
144      * the primitive types.
145      *
146      * @param argType A <code>Class</code> to find a converter for.
147      * @return A conversion function, or null on error.
148      */
149     private Method getFromStringConverter(Class<?> argType) {
150         Method method;
151
152         if (argType.isEnum()) {
153             try {
154                 method = Enum.class.getMethod("valueOf", Class.class, String.class);
155             } catch (NoSuchMethodException | SecurityException e) {
156                 throw new IllegalArgumentException("Enum.getMethod(valueOf(Class, String))", e);
157             }
158         } else {
159             if (argType.isPrimitive()) {
160                 if (argType == Integer.TYPE) {
161                     argType = Integer.class;
162                 } else if (argType == Boolean.TYPE) {
163                     argType = Boolean.class;
164                 } else if (argType == Float.TYPE) {
165                     argType = Float.class;
166                 } else if (argType == Double.TYPE) {
167                     argType = Double.class;
168                 } else if (argType == Character.TYPE) {
169                     argType = Character.class;
170                 } else {
171                     throw new IllegalArgumentException("Need compatible class for primitive " + argType.getName());
172                 }
173             }
174             try {
175                 method = argType.getMethod("valueOf", String.class);
176             } catch (NoSuchMethodException | SecurityException e) {
177                 throw new IllegalArgumentException(argType.getName()
178                                                    + ".getMethod(valueOf(String))", e);
179             }
180         }
181         return method;
182     }
