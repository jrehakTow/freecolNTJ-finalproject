306     /**
307      * Adjust the prices.
308      *
309      * Sets the costToBuy and paidForSale fields from the amount in
310      * the market, initial price and goods-type specific information.
311      * Ensures that prices change incrementally with a clamping
312      * mechanism.
313      *
314      * @return True if the price changes.
315      */
316     public boolean price() {
317         if (!goodsType.isStorable()) return false;
318         int diff = goodsType.getPriceDifference();
319         float amountPrice = initialPrice * (goodsType.getInitialAmount()
320             / (float) amountInMarket);
321         int newSalePrice = Math.round(amountPrice);
322         int newPrice = newSalePrice + diff;
323
324         // Work-around to limit prices of new world goods
325         // and related manufactured goods.
326         if ((goodsType.isNewWorldGoodsType()
327              || (goodsType.getInputType() != null
328                  && goodsType.getInputType().isNewWorldGoodsType()))
329             && newSalePrice > initialPrice + 2) {
330             newSalePrice = initialPrice + 2;
331             newPrice = newSalePrice + diff;
332         }
333
334         // Another hack to prevent price changing too fast in one hit.
335         // Push the amount in market back as well to keep this stable.
336         //
337         // Prices that change by more than the buy/sell difference
338         // allow big traders to exploit the market and extract free
339         // money... not sure I want to be fighting economic reality
340         // but game balance demands it here.
341         if (costToBuy > 0) {
342             if (newPrice > costToBuy + diff) {
343                 amountPrice -= newPrice - (costToBuy + diff);
344                 amountInMarket = Math.round(goodsType.getInitialAmount()
345                     * (initialPrice / amountPrice));
346                 logger.info("Clamped price rise for " + getId()
347                     + " from " + newPrice
348                     + " to " + (costToBuy + diff));
349                 newPrice = costToBuy + diff;
350             } else if (newPrice < costToBuy - diff) {
351                 amountPrice += (costToBuy - diff) - newPrice;
352                 amountInMarket = Math.round(goodsType.getInitialAmount()
353                     * (initialPrice / amountPrice));
354                 logger.info("Clamped price fall for " + getId()
355                     + " from " + newPrice
356                     + " to " + (costToBuy - diff));
357                 newPrice = costToBuy - diff;
358             }
359             newSalePrice = newPrice - diff;
360         }
361
362         // Clamp extremes.
363         if (newPrice > MAXIMUM_PRICE) {
364             newPrice = MAXIMUM_PRICE;
365             newSalePrice = newPrice - diff;
366         } else if (newSalePrice < MINIMUM_PRICE) {
367             newSalePrice = MINIMUM_PRICE;
368             newPrice = newSalePrice + diff;
369         }
370
371         int oldCostToBuy = costToBuy, oldPaidForSale = paidForSale;
372         costToBuy = newPrice;
373         paidForSale = newSalePrice;
374         if (costToBuy != oldCostToBuy) {
375             firePropertyChange(goodsType.getId(), oldCostToBuy, costToBuy);
376         } else if (paidForSale != oldPaidForSale) {
377             firePropertyChange(goodsType.getId(), oldPaidForSale, paidForSale);
378         }            
379         return costToBuy != oldCostToBuy || paidForSale != oldPaidForSale;
380     }
