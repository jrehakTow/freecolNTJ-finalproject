625     /**
626      * Tries to move this mission's unit to a target location.
627      *
628      * First check for units in transit, that is units on a carrier that
629      * are going to but not yet in Europe, or going to but not yet at
630      * a Tile and whose path still requires the carrier.  These need to
631      * be handled by the carrier's TransportMission, not by the unit's
632      * Mission.
633      *
634      * Similarly check for units not in transit but should be, that
635      * is units not on a carrier but can not get to their target
636      * without one.  These must just wait.
637      *
638      * If there is no impediment to the unit moving towards the target,
639      * do so.  Return an indicative MoveType for the result of the travel.
640      * - MOVE if the unit has arrived at the target, although it may
641      *     have exhausted its moves
642      * - MOVE_HIGH_SEAS if the unit has set sail to/from Europe
643      * - MOVE_NO_MOVES is underway but ran out of moves
644      * - MOVE_NO_ACCESS_EMBARK if progress depends on a carrier, either
645      *     currently boarded or due to collect the unit
646      * - MOVE_NO_REPAIR if the unit died for whatever reason
647      * - MOVE_NO_TILE if there is no path (usually transitory on rivers)
648      * - MOVE_ILLEGAL if there is an error or permanent restriction
649      * - other legal results (e.g. ENTER_INDIAN_SETTLEMENT*) if that would
650      *   occur if the unit proceeded.  Such moves require special handling
651      *   and are not performed here, the calling mission code must
652      *   handle them.
653      *
654      * @param target The destination <code>Location</code>.
655      * @param costDecider The <code>CostDecider</code> to use in any path
656      *     finding.
657      * @param lb A <code>LogBuilder</code> to log to.
658      * @return The type of move the unit stopped at.
659      */
660     protected MoveType travelToTarget(Location target, CostDecider costDecider,
661                                       LogBuilder lb) {
662         if (target == null) return MoveType.MOVE_ILLEGAL;
663         final Tile targetTile = target.getTile();
664         if (!(target instanceof Europe) && targetTile == null) {
665             throw new IllegalStateException("Target neither Europe nor Tile");
666         }
667         final Unit unit = getUnit();
668         AIUnit aiCarrier = aiUnit.getTransport();
669         final Map map = unit.getGame().getMap();
670         PathNode path = null;
671         boolean useTransport = false;
672         target = Location.upLoc(target);
673
674         // Consider where the unit is starting.
675         if (unit.isAtSea()) {
676             // Wait for carrier to arrive on the map or in Europe.
677             lb.add(", at sea");
678             return MoveType.MOVE_HIGH_SEAS;
679
680         } else if (unit.isOnCarrier()) {
681             // Transport mission will disembark the unit when it
682             // arrives at the drop point.
683             lb.add(", on carrier");
684             return MoveType.MOVE_NO_ACCESS_EMBARK;
685
686         } else if (unit.isAtLocation(target)) {
687             // Arrived!
688             return MoveType.MOVE;
689
690         } else if (unit.isInEurope()) {
691             // Leave, or require transport.
692             if (!unit.getOwner().canMoveToEurope()) {
693                 lb.add(", impossible move from Europe");
694                 return MoveType.MOVE_ILLEGAL;
695             }
696             if (unit.getType().canMoveToHighSeas()) {
697                 unit.setDestination(target);
698                 if (AIMessage.askMoveTo(aiUnit, map)) {
699                     lb.add(", sailed for ", target);
700                     return MoveType.MOVE_HIGH_SEAS;
701                 } else {
702                     lb.add(", failed to sail for ", target);
703                     return MoveType.MOVE_ILLEGAL;
704                 }
705             }
706             useTransport = true;
707
708         } else if (!unit.hasTile()) {
709             // Fail!
710             return MoveType.MOVE_ILLEGAL;
711
712         } else {
713             // On map.  Either find a path or decide to use transport.
714             if (target instanceof Europe) {
715                 // Going to Europe.
716                 if (!unit.getOwner().canMoveToEurope()) {
717                     lb.add(", impossible move to Europe");
718                     return MoveType.MOVE_ILLEGAL;
719                 }
720                 if (!unit.getType().canMoveToHighSeas()
721                     || aiCarrier != null) {
722                     useTransport = true;
723                 } else {
724                     path = unit.findPath(unit.getLocation(), target,
725                                          null, costDecider);
726                 }
727             } else if (aiCarrier != null) {
728                 // Transport already allocated.
729                 useTransport = true;
730
731             } else if (!unit.getType().canMoveToHighSeas()
732                 && !Map.isSameContiguity(target, unit.getLocation())) {
733                 // Transport necessary.
734                 useTransport = true;
735
736             } else {
737                 // Should not need transport within the same contiguity.
738                 path = unit.findPath(unit.getLocation(), target,
739                                      null, costDecider);
740             }
741         }
742
743         if (useTransport) {
744             if (aiCarrier != null) {
745                 // A carrier has been assigned.  Try to go to the
746                 // collection point.
747                 Location pick;
748                 TransportMission tm;
749                 boolean waiting = false;
750                 PathNode ownPath;
751                 int pathTurns, ownTurns;
752
753                 if ((tm = aiCarrier.getMission(TransportMission.class)) == null) {
754                     // Carrier has no transport mission?!?  Bogus.
755                     lb.add(", had bogus carrier ", aiCarrier.getUnit());
756                     logger.warning(unit + " has transport " + aiCarrier
757                         + " without transport mission");
758                     aiUnit.dropTransport();
759                     aiCarrier = null;
760
761                 } else if ((pick = tm.getTransportTarget(aiUnit)) == null) {
762                     // No collection point for this unit?  Bogus.
763                     lb.add(", had bogus transport on ", aiCarrier.getUnit());
764                     logger.warning(unit + " has transport " + aiCarrier
765                         + " with transport mission but null transport target\n"
766                         + tm.toFullString());
767                     aiUnit.dropTransport();
768                     aiCarrier = null;
769
770                 } else if (Map.isSameLocation(pick, unit.getLocation())) {
771                     // Waiting for the carrier at the collection point.
772                     waiting = true;
773
774                 } else if ((path = unit.findPath(unit.getLocation(), pick,
775                                                  null, costDecider)) == null) {
776                     // No path to the collection point.
777                     lbAt(lb);
778                     lb.add(", no path to meet ", aiCarrier.getUnit(),
779                            " at ", pick);
780                     path = unit.findPath(unit.getLocation(), target,
781                                          null, costDecider);
782                     if (path == null) {
783                         // Unable to fall back to going direct.
784                         // Return failure in the hope that it is a
785                         // transient blockage.
786                         return MoveType.MOVE_NO_TILE;
787                     }
788                     // Fall back to going direct to the target.
789                     lb.add(", dropped carrier");
790                     aiUnit.dropTransport();
791                     aiCarrier = null;
792                     useTransport = false;
793
794                 } else if ((ownPath = unit.findPath(unit.getLocation(), 
795                             target, null, costDecider)) == null
796                     || (ownTurns = ownPath.getTotalTurns())
797                     > (pathTurns = path.getTotalTurns())) {
798                     // Either there is no direct path to the target or
799                     // a path exists but takes longer than using the
800                     // carrier.  This confirms that it is not only
801                     // possible to travel to the collection point, it
802                     // is also the best plan.
803                     MoveType ret = followMapPath(path.next, lb);
804                     if (ret != MoveType.MOVE) return ret;
805                     waiting = true; // Arrived for collection.
806
807                 } else {
808                     // It is quicker to cancel the transport and go to
809                     // the target directly.
810                     lb.add(", dropping carrier", aiCarrier.getUnit(),
811                         " as it is faster (", ownTurns, "<", pathTurns,
812                         " without it");
813                     aiUnit.dropTransport();
814                     aiCarrier = null;
815                     path = ownPath;
816                     useTransport = false;
817                 }
818
819                 if (waiting) {
820                     // If waiting for the carrier, signal that this
821                     // unit can be reexamined if the carrier is still
822                     // moving.
823                     lbAt(lb);
824                     lb.add(", wait for ", aiCarrier.getUnit());
825                     return (aiCarrier.getUnit().getMovesLeft() > 0)
826                         ? MoveType.MOVE_NO_ACCESS_EMBARK
827                         : MoveType.MOVE_NO_MOVES;
828                 }
829             }
830
831             if (useTransport && aiCarrier == null) {
832                 // Still interested in transport but no carrier.
833                 lb.add(", needs transport to ", target);
834                 return MoveType.MOVE_NO_ACCESS_EMBARK;
835             }
836         }
837
838         // Follow the path to the target.  If there is one.
839         if (path == null) {
840             lbAt(lb);
841             lb.add(", no path to ", target);
842             return MoveType.MOVE_NO_TILE;
843         }
844         if (path.next == null) {
845             // This should not happen, the isAtLocation() test above
846             // should have succeeded.
847             throw new IllegalStateException("Trivial path found "
848                 + path.fullPathToString()
849                 + " from " + unit.getLocation() + " to target " + target
850                 + " result=" +  unit.isAtLocation(target));
851         }            
852         return followMapPath(path.next, lb);
853     }
854
855     /**
856      * Follow a path that is on the map (except perhaps the last node)
857      * and does not use a carrier.
858      *
859      * @param path The <code>PathNode</code> to follow.
860      * @param lb A <code>LogBuilder</code> to log to.
861      * @return The type of move the unit stopped at.
862      */
863     private MoveType followMapPath(PathNode path, LogBuilder lb) {
864         final Unit unit = getUnit();
865         final Location target = path.getLastNode().getLocation();
866
867         for (; path != null; path = path.next) {
868             // Check for immediate failure
869             if (unit.isDisposed()) {
870                 lb.add(", died going to ", Location.upLoc(path.getLocation()));
871                 return MoveType.MOVE_NO_REPAIR;
872             } else if (unit.getMovesLeft() <= 0) {
873                 lbAt(lb);
874                 lb.add(", en route to ", Location.upLoc(target));
875                 return MoveType.MOVE_NO_MOVES;
876             }
877
878             if (path.getLocation() instanceof Europe) {
879                 if (AIMessage.askMoveTo(aiUnit, path.getLocation())) {
880                     lb.add(", sailed to Europe");
881                     return MoveType.MOVE_HIGH_SEAS;
882                 } else {
883                     lb.add(", failed to sail for Europe");
884                     return MoveType.MOVE_ILLEGAL;
885                 }
886             }
887             MoveType mt = unit.getMoveType(path.getDirection());
