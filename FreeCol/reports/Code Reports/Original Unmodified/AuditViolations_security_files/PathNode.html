268     /**
269      * Standard function to get the cost of moving to a <code>PathNode</code>.
270      *
271      * Static version provided for path calculation comparisons.  Some
272      * care is taken to avoid overflow as test paths with infinite
273      * moves are created in the path planning process.
274      *
275      * @param turns The number of turns taken.
276      * @param movesLeft The number of moves left for the moving unit.
277      * @return The cost of moving to a <code>PathNode</code>.
278      */
279     public static int getCost(int turns, int movesLeft) {
280         return (turns >= FreeColObject.INFINITY / (TURN_FACTOR + 1))
281             ? FreeColObject.INFINITY
282             : TURN_FACTOR * turns + (TURN_FACTOR - movesLeft);
283     }
