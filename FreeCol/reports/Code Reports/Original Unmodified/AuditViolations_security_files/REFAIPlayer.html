469     /**
470      * Require more transport missions, recruiting from the privateering
471      * missions.
472      *
473      * @param nt The number of transport missions required.
474      * @param transports The list of <code>AIUnit</code>s with a transport
475      *     mission.
476      * @param privateers The list of <code>AIUnit</code>s with a privateer
477      *     mission.
478      * @param lb A <code>LogBuilder</code> to log to.
479      * @return A list of new <code>AIUnit</code>s with transport missions.
480      */
481     private List<AIUnit> requireTransports(int nt, List<AIUnit> transports,
482                                            List<AIUnit> privateers,
483                                            LogBuilder lb) {
484         Mission m;
485         List<AIUnit> naval = new ArrayList<>();
486         List<AIUnit> result = new ArrayList<>();
487         if (transports.size() < nt) {
488             // Recruit privateers not currently chasing a unit.
489             // Collect privateers that are on the map.
490             for (AIUnit aiu : privateers) {
491                 Location target = aiu.getMission().getTarget();
492                 if (target instanceof Unit && aiu.getUnit().hasTile()) {
493                     naval.add(aiu);
494                 } else if ((m = getTransportMission(aiu)) != null) {
495                     lb.add(" notarget ", m);
496                     result.add(aiu);
497                 }
498             }
499         }
500         if (transports.size() < nt) {
501             // Sort by longest distance to target
502             final ToIntFunction<AIUnit> targetDistance
503                 = cacheInt(aiu -> {
504                         Unit target = (Unit)aiu.getMission().getTarget();
505                         Tile tile = aiu.getUnit().getTile();
506                         return tile.getDistanceTo(target.getTile());
507                     });
508             Collections.sort(naval, Comparator.comparingInt(targetDistance));
509             while (!naval.isEmpty()) {
510                 AIUnit aiu = naval.remove(0);
511                 int distance = targetDistance.applyAsInt(aiu);
512                 if ((m = getTransportMission(aiu)) != null) {
513                     lb.add(" REQUIRED ", distance, " ", m);
514                     result.add(aiu);
515                     if (result.size() + transports.size() >= nt) break;
516                 }
517             }
518         }
519         privateers.removeAll(result);
520         transports.addAll(result);
521         return result;
522     }

689             for (AIUnit aiu : transports) {
690                 TransportMission tm = aiu.getMission(TransportMission.class);
691                 if (!tm.isEmpty()) continue;

693                 Location key;

695                     && idlers.containsKey(key = player.getEurope())) {

794                         - aiu.getUnit().getCargoSpaceTaken();
795                 }

799                     requireTransports(nt, todo, privateers, lb);
800                 }

802                 // Send transports to the idle ports, preferring the ones
803                 // with the most units.
804                 final Comparator<Location> portUnitComparator

807                 Collections.sort(idlePorts, portUnitComparator);
808                 boolean bad = false;
809                 while (!bad && !todo.isEmpty()) {

812                         AIUnit best = null;
813                         for (AIUnit aiu : todo) {
814                             int value = aiu.getUnit().getTurnsToReach(l);
815                             if (bestValue > value) {
816                                 bestValue = value;
817                                 best = aiu;
818                             }
819                         }
820                         if (best == null) {
821                             bad = true;
822                             continue;
823                         }
824                         todo.remove(best);
825                         best.getMission().setTarget(l);
826                         lb.add(" retarget ", best, " to ", l,
827                             "(", idlers.get(l).size(), ")");
828                     }
829                 }
830             }
831         }
832
833         // Fall back to the normal EuropeanAI behaviour for remaining units.
834         super.giveNormalMissions(lb);
835     }

886                 }

895                     } else {

898                         }
899                     }
900                 } else if (target instanceof Unit) {

902                     if (!getPlayer().hasSettlements()) {
903                         return Integer.MIN_VALUE;
904                     }
905                     // Do not chase the same unit!
906                     if (any(getAIUnits().stream()
907                             .filter(aiu -> aiu != aiUnit)

909                             m -> m != null
910                                 && m.getTarget() instanceof Unit
911                                 && (Unit)m.getTarget() == target))
912                         return Integer.MIN_VALUE;
913                     // The REF is more interested in colonies.
914                     value /= 2;
915                 }

917         }
