168      * Does this scope apply to a given object?
169      *
170      * @param object The <code>FreeColSpecObjectType</code> to test.
171      * @return True if the scope is applicable.
172      */
173     public boolean appliesTo(FreeColObject object) {
174         if (object == null) {
175             return matchesNull;
176         }
177         if (type != null) {
178             if (object instanceof FreeColSpecObjectType) {
179                 if (!type.equals(object.getId())) {
180                     return matchNegated;
181                 }
182             } else if (object instanceof FreeColObject) {
183                 FreeColSpecObjectType fcgot = object.invokeMethod("getType",
184                     FreeColSpecObjectType.class, (FreeColSpecObjectType)null);
185                 if (fcgot == null || !type.equals(fcgot.getId())) {
186                     return matchNegated;
187                 }
188             } else {
189                 return matchNegated;
190             }
191         }
192         if (abilityId != null && object.hasAbility(abilityId) != abilityValue) {
193             return matchNegated;
194         }
195         if (methodName != null) {
196             Object ret = object.invokeMethod(methodName, Object.class, null);
197             if (!String.valueOf(ret).equals(methodValue)) return matchNegated;
198         }
199         return !matchNegated;
200     }
