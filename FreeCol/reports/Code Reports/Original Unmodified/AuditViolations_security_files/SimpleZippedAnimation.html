200      *
201      * @param zipStream An <code>ZipInputStream</code> to a zip-file. 
202      * @return A list of <code>AnimationEvent</code>s.
203      * @exception IOException if there is an error with the stream.
204      */
205     private static List<AnimationEvent> loadEvents(ZipInputStream zipStream)
206         throws IOException {
207         List<AnimationEvent> events = new ArrayList<>();
208
209         // Preload all files from the archive since we cannot use a
210         // ZipFile for reading (as we should support an arbitrary stream).
211         final Map<String, BufferedImage> loadingImages = new HashMap<>();
212         final List<String> loadingDescriptor = new ArrayList<>();
213         try {
214             BufferedReader in;
215             ZipEntry ze;
216             while ((ze = zipStream.getNextEntry()) != null) {
217                 if (ANIMATION_DESCRIPTOR_FILE.equals(ze.getName())) {
218                     in = new BufferedReader(new InputStreamReader(zipStream,
219                                                                   "UTF-8"));
220                     String line;
221                     while ((line = in.readLine()) != null) {
222                         loadingDescriptor.add(line);
223                     }
224                 } else {
225                     loadingImages.put(ze.getName(), ImageIO.read(zipStream));
226                 }
227                 zipStream.closeEntry();
228             }
229         } finally {
230             try { zipStream.close(); } catch (IOException e) {}
231         }
232         
233         if (loadingDescriptor.isEmpty()) {
234             throw new IOException(ANIMATION_DESCRIPTOR_FILE
235                 + " is missing from the SZA.");
236         }
237         
238         for (String line : loadingDescriptor) {
239             final int idx = line.indexOf('(');
240             final int idx2 = line.indexOf("ms)");
241             if (idx < 0 || idx2 <= idx) {
242                 throw new IOException(ANIMATION_DESCRIPTOR_FILE
243                     + " should use the format: FILNAME (TIMEms)");
244             }
245             final String imageName = line.substring(0, idx).trim();
246             final int ms = Integer.parseInt(line.substring(idx+1, idx2));
247             final BufferedImage image = loadingImages.get(imageName);
248             if (image == null) {
249                 throw new IOException("Could not find referenced image: "
250                     + imageName);
251             }
252             events.add(new ImageAnimationEventImpl(image, ms));
253         }
254         return events;
255     }
