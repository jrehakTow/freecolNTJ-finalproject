224         private void setVolume(SourceDataLine line, int vol) {
225             FloatControl.Type type
226                 = (line.isControlSupported(FloatControl.Type.VOLUME))
227                 ? FloatControl.Type.VOLUME
228                 : (line.isControlSupported(FloatControl.Type.MASTER_GAIN))
229                 ? FloatControl.Type.MASTER_GAIN
230                 : null;
231             if (type == null) {
232                 logger.warning("No volume or master gain controls.");
233                 return;
234             }
235             FloatControl control;
236             try {
237                 control = (FloatControl)line.getControl(type);
238             } catch (IllegalArgumentException e) {
239                 return; // Should not happen
240             }
241             //
242             // The units of MASTER_GAIN seem to consistently be dB, but
243             // in the case of VOLUME this is unclear (there is even a query
244             // to that effect in the source).  getUnits() says "pulseaudio
245             // units" on my boxen, and the PulseAudio doco talks about dB
246             // so for now we are assuming that the controls we are using
247             // are both logarithmic:
248             //
249             //   gain = A.log_10(k.vol)
250             // So scale vol <= 1 to gain_min and vol >= 100 to gain_max
251             //   gain_min = A.log_10(k.1)
252             //   gain_max = A.log_10(k.100)
253             // Solving for A,k yields:
254             //   A = (gain_max - gain_min)/2
255             //   k = 10^(gain_min/A)
256             // =>
257             //   gain = gain_min + (gain_max - gain_min)/2 * log_10(vol)
258             //
259             float min = control.getMinimum();
260             float max = control.getMaximum();
261             float gain = (vol <= 0) ? min
262                 : (vol >= 100) ? max
263                 : min + 0.5f * (max - min) * (float)Math.log10(vol);
264             try {
265                 control.setValue(gain);
266                 logger.finest("Using volume " + vol + "%, "
267                     + control.getUnits() + "=" + gain
268                     + " control=" + type);
269             } catch (Exception e) {
270                 logger.log(Level.WARNING, "Could not set volume "
271                     + " (control=" + type + " in [" + min + "," + max + "])"
272                     + " to " + gain + control.getUnits(), e);
273             }
274         }

276         private SourceDataLine openLine(AudioFormat audioFormat) {
277             SourceDataLine line = null;
278             DataLine.Info info = new DataLine.Info(SourceDataLine.class,
279                                                    audioFormat);
280             if (!mixer.isLineSupported(info)) {
281                 logger.log(Level.WARNING, "Mixer does not support " + info);
282                 return null;
283             }
284             try {
285                 line = (SourceDataLine)mixer.getLine(info);
286                 line.open(audioFormat, BUFSIZ);
287                 line.start();
288                 setVolume(line, volume);
289             } catch (Exception e) {
290                 logger.log(Level.WARNING, "Can not open SourceDataLine", e);
291                 return null;
292             }
293             return line;
294         }

297          * Play a sound.
298          *
299          * @param in The <code>AudioInputStream</code> to play.
300          * @return True if the stream was played without incident.
301          * @exception IOException if unable to read or write the sound data.
302          */
303         private boolean playSound(AudioInputStream in) throws IOException {
304             boolean ret = false;
305
306             SourceDataLine line = openLine(in.getFormat());
307             if (line == null) return false;
308             try {
309                 startPlaying();
310                 int rd;
311                 while (keepPlaying() && (rd = in.read(data)) > 0) {
312                     line.write(data, 0, rd);
313                 }
314                 ret = true;
315             } finally {
316                 stopPlaying();
317                 line.drain();
318                 line.stop();
319                 line.close();
320             }
321             return ret;
322         }
