741     private class TypeReader<T extends FreeColSpecObjectType> implements ChildReader {
742
743         private final Class<T> type;
744         private final List<T> result;
745         private int index = 0;
746
747         // Is there really no easy way to capture T?
748         public TypeReader(Class<T> type, List<T> listToFill) {
749             result = listToFill;
750             this.type = type;
751         }
752
753         @Override
754         public void readChildren(FreeColXMLReader xr) throws XMLStreamException {
755             while (xr.nextTag() != XMLStreamConstants.END_ELEMENT) {
756                 final String tag = xr.getLocalName();
757                 String id = xr.readId();
758                 if (id == null) {
759                     logger.warning("Null identifier, tag: " + tag);
760
761                 } else if (FreeColSpecObjectType.DELETE_TAG.equals(tag)) {
762                     FreeColSpecObjectType object = allTypes.remove(id);
763                     if (object != null) result.remove(object);
764
765                 } else {
766                     T object = getType(id, type);
767                     allTypes.put(id, object);
768
769                     // If this an existing object (with id) and the
770                     // PRESERVE tag is present, then leave the
771                     // attributes intact and only read the child
772                     // elements, otherwise do a full attribute
773                     // inclusive read.  This allows mods and spec
774                     // extensions to not have to re-specify all the
775                     // attributes when just changing the children.
776                     if (object.getId() != null
777                         && xr.getAttribute(FreeColSpecObjectType.PRESERVE_TAG,
778                                            (String)null) != null) {
779                         object.readChildren(xr);
780                     } else {
781                         object.readFromXML(xr);
782                     }
783                     if (!object.isAbstractType() && !result.contains(object)) {
784                         result.add(object);
785                         object.setIndex(index);
786                         index++;
787                     }
788                 }
789             }
790         }
791     }

795      * collection, which has its own particular semantics.  So they

810             final String tag = xr.getLocalName();

812             boolean recursive = xr.getAttribute(RECURSIVE_TAG, true);
813

815                 String id = xr.readId();

819                     allOptionGroups.put(id, group);
820                 }
821                 group.readFromXML(xr);
822                 Specification.this.addOptionGroup(group, recursive);
823
824             } else {
825                 logger.warning(OptionGroup.getTagName()
826                     + " expected in OptionReader, not: " + tag);
827                 xr.nextTag();
828             }
829         }
