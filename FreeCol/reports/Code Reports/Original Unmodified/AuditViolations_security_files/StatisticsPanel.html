188     private JPanel displayStatsMessage(String title,
189                                        Map<String, String> stats) {
190         JPanel panel = new JPanel();
191         panel.setBorder(Utility.localizedBorder(title));

193         panel.add(b);
194         Map<String, String> memory = new HashMap<>();

196         for (String k : memoryKeys) {

198                        stats.remove(k));

201             if (k.startsWith("AI")) { // FIXME: AIMain.aiStatisticsPrefix
202                 ai.put(k, stats.remove(k));
203             }
204         }
205         b.add(createStatsTable("Memory", memory));
206         b.add(createStatsTable("Game", stats));
207         if (ai.isEmpty()) {
208             b.add(new JLabel());
209         } else {
210             b.add(createStatsTable("AI", ai));
211         }
212         return panel;
213     }
