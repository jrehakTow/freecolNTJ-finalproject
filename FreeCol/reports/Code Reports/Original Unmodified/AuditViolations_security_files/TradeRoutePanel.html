256     /**
257      * Update the list of routes displayed.
258      *
259      * @param selectRoute An optional <code>TradeRoute</code> to select.
260      */
261     private void updateList(TradeRoute selectRoute) {
262         final Player player = getMyPlayer();
263
264         // Create a sorted list of routes.
265         // We are deliberately *not* sorting the player's list.
266         List<TradeRoute> routes = new ArrayList<>();
267         for (TradeRoute tr : player.getTradeRoutes()) {
268             StringTemplate st = tr.verify();
269             if (st == null) {
270                 routes.add(tr);
271             } else {
272                 igc().deleteTradeRoute(tr);
273                 logger.warning("Dropped trade route: " + Messages.message(st));
274             }
275         }
276         Collections.sort(routes, tradeRouteComparator);
277
278         // Update the counts
279         this.counts.clear();
280         for (Unit u : player.getUnits()) {
281             TradeRoute tradeRoute = u.getTradeRoute();
282             if (tradeRoute != null && routes.contains(tradeRoute)) {
283                 Integer i = counts.get(tradeRoute);
284                 int value = (i == null) ? 0 : i;
285                 counts.put(tradeRoute, value + 1);
286             }
287         }
288
289         // Then add the routes to the list model.
290         this.listModel.clear();
291         for (TradeRoute route : routes) {
292             this.listModel.addElement(route);
293         }
294
295         // Select the route if given.
296         if (selectRoute != null && this.tradeRoutes != null) {
297             this.tradeRoutes.setSelectedValue(selectRoute, true);
298         }
299     }
